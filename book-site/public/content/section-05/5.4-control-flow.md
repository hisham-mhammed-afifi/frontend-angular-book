# 5.4 Control Flow

> Control flow determines execution order. Beyond basic if/else and for loops, JavaScript has labeled statements, iterator protocols, switch scoping traps, and finally-block quirks that trip up even experienced developers.

---

## Conditionals

### `if`/`else if`/`else`

The condition is evaluated via ToBoolean. Remember the 9 falsy values: `false`, `0`, `-0`, `0n`, `""`, `null`, `undefined`, `NaN`, and `document.all`. Everything else is truthy, including empty arrays and empty objects.

```javascript
const arr = [];
if (arr) {
  console.log("truthy"); // Runs! [] is truthy
}

if (arr.length) {
  console.log("has items"); // Does NOT run. 0 is falsy
}
```

### Accidental Semicolon Trap

A semicolon after the condition silently terminates the if-statement:

```javascript
// BAD: The semicolon ends the if-statement; the block always runs
if (x === 10);
{
  console.log("always runs");
}

// BAD: Assignment instead of comparison
if (x = 5) {
  // x is now 5, and 5 is truthy, so this always runs
}
```

### Ternary Operator

An expression, not a statement. Returns one of two values.

```javascript
const status = score >= 90 ? "pass" : "fail";

// Nested ternaries are legal but hard to read
// BAD: Deeply nested ternary
const grade = score >= 90 ? "A" : score >= 80 ? "B" : score >= 70 ? "C" : "F";

// GOOD: Use if/else or a lookup for complex branching
function getGrade(score) {
  if (score >= 90) return "A";
  if (score >= 80) return "B";
  if (score >= 70) return "C";
  return "F";
}
```

### Optional Chaining in Conditionals

Combine `?.` with `??` for clean, null-safe branching:

```javascript
const color = user?.settings?.theme?.color ?? "blue";
const label = response?.data?.items?.[0]?.name ?? "Unknown";

// Short-circuiting stops side effects
let i = 0;
const val = null?.prop[i++]; // i is still 0 (the [i++] never executed)
```

## Switch Statements

### Strict Equality

`switch` uses `===` for case matching. No type coercion occurs.

```javascript
const input = "3"; // string from user input

switch (input) {
  case 3: // Never matches! "3" !== 3
    console.log("number three");
    break;
  case "3": // Matches
    console.log("string three");
    break;
}
```

### Fall-Through

Without `break`, execution continues into the next case:

```javascript
// Intentional fall-through: grouping cases
switch (day) {
  case "Monday":
  case "Tuesday":
  case "Wednesday":
  case "Thursday":
  case "Friday":
    console.log("Weekday");
    break;
  case "Saturday":
  case "Sunday":
    console.log("Weekend");
    break;
}
```

### Scope Trap

All `case` clauses share the same block scope. `let`/`const` declarations in one case are visible in others, causing conflicts.

```javascript
// BAD: SyntaxError because message is declared in shared scope
switch (action) {
  case "create":
    let message = "Created";
    break;
  case "delete":
    let message = "Deleted"; // SyntaxError: Identifier 'message' already declared
    break;
}

// GOOD: Wrap each case in its own block
switch (action) {
  case "create": {
    const message = "Created";
    console.log(message);
    break;
  }
  case "delete": {
    const message = "Deleted";
    console.log(message);
    break;
  }
}
```

### `default` Placement

`default` can appear anywhere in the switch, not only at the end. If no case matches, execution jumps to `default`, then continues downward (fall-through still applies).

```javascript
switch (value) {
  default:
    console.log("default");
    // Falls through to case 1 if no break!
  case 1:
    console.log("one");
    break;
  case 2:
    console.log("two");
    break;
}

// If value is 5: prints "default", then "one" (fall-through)
// If value is 1: prints "one" only
```

Place `default` last with a `break` to avoid this surprise.

## Loops

### Classic `for` Loop

```javascript
for (let i = 0; i < items.length; i++) {
  process(items[i]);
}

// Length caching (marginal benefit on modern engines, but good habit)
for (let i = 0, len = items.length; i < len; i++) {
  process(items[i]);
}

// Multiple variables with comma operator
for (let i = 0, j = 10; i < j; i++, j--) {
  console.log(i, j); // 0 10, 1 9, 2 8, 3 7, 4 6
}
```

### `while` and `do...while`

`while` checks the condition before the first iteration. `do...while` executes the body at least once.

```javascript
// while: might never execute
let attempts = 0;
while (attempts < maxAttempts && !isConnected()) {
  attempts++;
  tryConnect();
}

// do...while: always executes at least once
let input;
do {
  input = prompt("Enter a positive number:");
} while (Number(input) <= 0 || Number.isNaN(Number(input)));
```

**Retry pattern using `do...while`:**

```javascript
async function fetchWithRetry(url, maxRetries = 3) {
  let attempt = 0;
  let lastError;

  do {
    try {
      return await fetch(url);
    } catch (error) {
      lastError = error;
      attempt++;
      await delay(attempt * 1000); // backoff
    }
  } while (attempt < maxRetries);

  throw lastError;
}
```

### `for...in` (Enumerable Keys)

Iterates over all **enumerable string property keys**, including inherited ones from the prototype chain.

```javascript
const obj = { a: 1, b: 2, c: 3 };
for (const key in obj) {
  console.log(key, obj[key]); // "a" 1, "b" 2, "c" 3
}
```

**Prototype chain leakage:**

```javascript
// BAD: for...in picks up inherited properties
Object.prototype.polluted = "oops";
const data = { name: "Alice" };

for (const key in data) {
  console.log(key); // "name", "polluted"
}

// Guard with Object.hasOwn
for (const key in data) {
  if (Object.hasOwn(data, key)) {
    console.log(key); // "name" only
  }
}

// Clean up
delete Object.prototype.polluted;
```

**Never use `for...in` on arrays:**

```javascript
const arr = [10, 20, 30];
arr.custom = "extra";

for (const key in arr) {
  console.log(key); // "0", "1", "2", "custom" (keys are strings, includes non-index)
}

// Use for...of or a classic for loop instead
```

### `for...of` (Iterable Values)

Iterates over **values** from any object that implements `[Symbol.iterator]`. Works with Array, String, Map, Set, NodeList, TypedArray, generators, and custom iterables.

```javascript
const arr = [10, 20, 30];
for (const value of arr) {
  console.log(value); // 10, 20, 30
}

// Does NOT work on plain objects
const obj = { a: 1 };
// for (const val of obj) {} // TypeError: obj is not iterable

// Use Object.entries() to iterate objects
for (const [key, value] of Object.entries(obj)) {
  console.log(key, value); // "a" 1
}
```

**Unicode-aware string iteration:**

```javascript
const emoji = "Hello 🇺🇸";

// BAD: Classic for loop splits surrogate pairs
for (let i = 0; i < emoji.length; i++) {
  // emoji[6] and emoji[7] are broken halves of the flag emoji
}

// GOOD: for...of iterates by Unicode code points
for (const char of emoji) {
  console.log(char); // H, e, l, l, o, " ", 🇺🇸 (correct)
}
```

### `for...of` with Map and Set

```javascript
// Map: default iterator yields [key, value] pairs
const map = new Map([["a", 1], ["b", 2]]);
for (const [key, value] of map) {
  console.log(key, value); // "a" 1, "b" 2
}

// No need to call .entries() explicitly; it is the default

// Set: default iterator yields values
const set = new Set([10, 20, 30]);
for (const value of set) {
  console.log(value); // 10, 20, 30
}

// Set.entries() returns [value, value] pairs (for API parity with Map)
for (const [a, b] of set.entries()) {
  console.log(a === b); // true (both are the same value)
}
```

### Sparse Array Behavior

Different iteration methods handle array holes differently:

```javascript
const sparse = [1, , 3]; // Index 1 is a hole

// for...of visits holes as undefined
for (const v of sparse) {
  console.log(v); // 1, undefined, 3
}

// Classic for loop also visits holes
for (let i = 0; i < sparse.length; i++) {
  console.log(sparse[i]); // 1, undefined, 3
}

// forEach and for...in SKIP holes
sparse.forEach(v => console.log(v)); // 1, 3
for (const k in sparse) {
  console.log(k); // "0", "2" (skips index 1)
}
```

## Labeled Statements

Labels provide a way to break out of or continue specific loops in nested structures. They are the only mechanism in JavaScript for multi-level break/continue.

### Breaking Out of Nested Loops

```javascript
search: for (let row = 0; row < matrix.length; row++) {
  for (let col = 0; col < matrix[row].length; col++) {
    if (matrix[row][col] === target) {
      console.log(`Found at ${row}, ${col}`);
      break search; // Exits BOTH loops
    }
  }
}
```

### Continuing an Outer Loop

```javascript
outer: for (let i = 0; i < 5; i++) {
  for (let j = 0; j < 5; j++) {
    if (j === 2) continue outer; // Skips to next iteration of i
    console.log(i, j);
  }
  // This line never runs when j hits 2
}
// Output: 0,0  0,1  1,0  1,1  2,0  2,1  3,0  3,1  4,0  4,1
```

### Labels on Block Statements

`break` (but not `continue`) can reference a labeled block for early exit without a function:

```javascript
validate: {
  if (!user) break validate;
  if (!user.email) break validate;
  if (!user.email.includes("@")) break validate;
  sendVerification(user.email);
}
// Execution continues here after any break
```

### ASI Gotcha

A line break between `break`/`continue` and the label is treated as a bare `break;` due to automatic semicolon insertion:

```javascript
// BAD: ASI inserts a semicolon after break
outer: for (let i = 0; i < 5; i++) {
  for (let j = 0; j < 5; j++) {
    break    // ASI inserts semicolon here
      outer; // This is a standalone expression statement (label reference), not reached
  }
}

// GOOD: Keep on the same line
break outer;
```

## The Iteration Protocol

Any object can be made iterable by implementing `[Symbol.iterator]()`, which must return an iterator (an object with a `next()` method).

### Custom Iterable

```javascript
class Range {
  constructor(start, end, step = 1) {
    this.start = start;
    this.end = end;
    this.step = step;
  }

  [Symbol.iterator]() {
    let current = this.start;
    const end = this.end;
    const step = this.step;

    return {
      next() {
        if (current <= end) {
          const value = current;
          current += step;
          return { value, done: false };
        }
        return { done: true };
      }
    };
  }
}

const range = new Range(1, 10, 2);

for (const n of range) {
  console.log(n); // 1, 3, 5, 7, 9
}

// Works with spread and destructuring
console.log([...range]);        // [1, 3, 5, 7, 9]
const [first, second] = range;  // first=1, second=3
```

### Iterator Cleanup with `return()`

When iteration ends early (via `break`, `throw`, or `return`), the iterator's `return()` method is called if it exists. This enables resource cleanup.

```javascript
function createFileReader(filename) {
  const handle = openFile(filename);

  return {
    [Symbol.iterator]() {
      return {
        next() {
          const line = handle.readLine();
          if (line === null) {
            handle.close();
            return { done: true };
          }
          return { value: line, done: false };
        },
        return() {
          handle.close(); // Cleanup on early exit
          return { done: true };
        }
      };
    }
  };
}

// The file handle is closed even with an early break
for (const line of createFileReader("data.txt")) {
  if (line.startsWith("END")) break; // return() called, file closed
  process(line);
}
```

### One-Shot vs Reusable Iterables

Generators produce one-shot iterators. Once consumed, they yield nothing on re-iteration.

```javascript
function* count(n) {
  for (let i = 1; i <= n; i++) yield i;
}

const gen = count(3);
console.log([...gen]); // [1, 2, 3]
console.log([...gen]); // [] (exhausted)

// Arrays are reusable: [Symbol.iterator] creates a new iterator each time
const arr = [1, 2, 3];
console.log([...arr]); // [1, 2, 3]
console.log([...arr]); // [1, 2, 3]
```

### Making Plain Objects Iterable

```javascript
const config = {
  host: "localhost",
  port: 3000,
  debug: true,

  [Symbol.iterator]() {
    const entries = Object.entries(this).filter(
      ([key]) => key !== Symbol.iterator.toString()
    );
    let index = 0;

    return {
      next() {
        if (index < entries.length) {
          return { value: entries[index++], done: false };
        }
        return { done: true };
      }
    };
  }
};

for (const [key, value] of config) {
  console.log(key, value); // "host" "localhost", "port" 3000, "debug" true
}
```

## Async Iteration (`for await...of`)

The async iteration protocol uses `Symbol.asyncIterator`. The iterator's `next()` returns a Promise.

```javascript
async function* fetchPages(baseUrl) {
  let page = 1;
  while (true) {
    const response = await fetch(`${baseUrl}?page=${page}`);
    const data = await response.json();
    if (data.items.length === 0) return;
    yield data.items;
    page++;
  }
}

// Consume with for await...of
async function getAllItems(url) {
  const allItems = [];
  for await (const items of fetchPages(url)) {
    allItems.push(...items);
  }
  return allItems;
}
```

`for await...of` also works with sync iterables containing promises, awaiting each value sequentially:

```javascript
const promises = [
  fetch("/api/users"),
  fetch("/api/posts"),
  fetch("/api/comments")
];

// Processes each promise sequentially (waits for each before moving to the next)
for await (const response of promises) {
  console.log(response.status);
}
```

## `try`/`catch`/`finally`

### `finally` Always Runs

```javascript
function readFile(path) {
  const handle = openFile(path);
  try {
    return parseContent(handle.read());
  } catch (error) {
    logError(error);
    return null;
  } finally {
    handle.close(); // Always runs, even after return
  }
}
```

### `finally` Overrides Return Values

If `finally` has its own `return`, it replaces the return value from `try` or `catch`:

```javascript
function test() {
  try {
    return "from try";
  } finally {
    return "from finally"; // This wins
  }
}
test(); // "from finally"
```

### `finally` Swallows Errors

A `return` in `finally` silently swallows exceptions:

```javascript
function dangerous() {
  try {
    throw new Error("critical failure");
  } catch (error) {
    throw error; // Re-throw
  } finally {
    return "everything is fine"; // Error is swallowed!
  }
}
dangerous(); // "everything is fine" (error gone)
```

**Never use `return`, `throw`, `break`, or `continue` inside `finally` blocks.** Use ESLint's `no-unsafe-finally` rule to enforce this.

### `try...finally` Without `catch`

Valid for guaranteed cleanup without error handling:

```javascript
function withLock(lock, fn) {
  lock.acquire();
  try {
    return fn();
  } finally {
    lock.release(); // Always releases, error or not
  }
}
```

### Optional Catch Binding

Since ES2019, the catch parameter is optional if you do not need the error object:

```javascript
try {
  JSON.parse(input);
} catch {
  // No (error) parameter needed
  console.log("Invalid JSON");
}
```

## Loop Performance

### Relative Performance (Hot Path, Large Datasets)

| Method | Relative Speed | Notes |
|---|---|---|
| Classic `for` / `while` | Fastest | Direct index access, no overhead |
| `for...of` | Slightly slower | Iterator protocol overhead |
| `for...in` | Slow | Prototype chain traversal |
| `.forEach()` / `.map()` | Slowest | Function call per iteration |

### When Performance Matters

In most applications, the difference is negligible. Optimize only when profiling confirms a bottleneck.

```javascript
// When processing 100k+ items and every millisecond counts:

// 1. Cache the length
for (let i = 0, len = arr.length; i < len; i++) { /* ... */ }

// 2. Avoid creating objects/closures inside hot loops
// BAD:
for (let i = 0; i < items.length; i++) {
  const handler = () => process(items[i]); // New closure each iteration
  register(handler);
}

// GOOD:
function handler(item) { process(item); }
for (let i = 0; i < items.length; i++) {
  register(() => handler(items[i]));
}

// 3. Avoid DOM access inside loops
// BAD:
for (let i = 0; i < 1000; i++) {
  document.getElementById("list").innerHTML += `<li>${i}</li>`;
}

// GOOD:
const fragments = [];
for (let i = 0; i < 1000; i++) {
  fragments.push(`<li>${i}</li>`);
}
document.getElementById("list").innerHTML = fragments.join("");
```

### `forEach` Cannot Break Early

```javascript
// BAD: Cannot exit early from forEach
[1, 2, 3, 4, 5].forEach(n => {
  if (n === 3) return; // This only skips the current callback, not the loop
  console.log(n); // 1, 2, 4, 5
});

// GOOD: for...of supports break
for (const n of [1, 2, 3, 4, 5]) {
  if (n === 3) break;
  console.log(n); // 1, 2
}

// Alternative: .some() or .every() for early exit with array methods
[1, 2, 3, 4, 5].some(n => {
  if (n === 3) return true; // Stops iteration
  console.log(n); // 1, 2
  return false;
});
```

## Common Mistakes

### 1. Semicolon After `if`/`for`/`while`

```javascript
// BAD: Semicolon silently terminates the statement
for (let i = 0; i < 5; i++);
{
  console.log(i); // ReferenceError (let is block-scoped, and the ; ended the for's block)
}

// With var: prints 5 once (loop body is the empty semicolon statement)
for (var i = 0; i < 5; i++);
console.log(i); // 5
```

### 2. Using `for...in` on Arrays

```javascript
// BAD: Gets string keys, includes prototype and non-index properties
const arr = [10, 20, 30];
Array.prototype.custom = "extra";

for (const key in arr) {
  console.log(typeof key, key); // "string" "0", "string" "1", "string" "2", "string" "custom"
}

// GOOD: for...of or classic for
for (const value of arr) {
  console.log(value); // 10, 20, 30
}
```

### 3. Returning Inside `finally`

```javascript
// BAD: Swallows the error
function process() {
  try {
    throw new Error("fail");
  } finally {
    return "ok"; // Error silently disappears
  }
}
```

### 4. Forgetting `break` in `switch`

```javascript
// BAD: Falls through to "medium" even when size is "small"
switch (size) {
  case "small":
    price = 5;
  case "medium":
    price = 10; // Overwrites price for "small"!
    break;
  case "large":
    price = 15;
    break;
}
```

### 5. Comparing with `==` in Conditionals

```javascript
// BAD: Coercion produces unexpected matches
if (value == "") { /* matches 0, false, null? No: 0 yes, false yes, null no */ }

// GOOD: Strict equality
if (value === "") { /* only matches empty string */ }
```

### 6. Modifying an Array While Iterating

```javascript
// BAD: Skips elements when splicing during forward iteration
const items = [1, 2, 3, 4, 5];
for (let i = 0; i < items.length; i++) {
  if (items[i] % 2 === 0) {
    items.splice(i, 1); // Shifts elements left, next element skipped
  }
}
console.log(items); // [1, 3, 5] (works by accident here, but fragile)

// GOOD: Iterate backward when splicing
for (let i = items.length - 1; i >= 0; i--) {
  if (items[i] % 2 === 0) {
    items.splice(i, 1);
  }
}

// BETTER: Use filter (creates new array)
const odds = items.filter(n => n % 2 !== 0);
```

## Best Practices

1. **Use `for...of` for arrays and iterables.** It gives values directly, handles Unicode correctly, and supports `break`/`continue`.

2. **Use `for...in` only for plain objects**, and always guard with `Object.hasOwn()`. Better yet, use `Object.keys()`, `Object.values()`, or `Object.entries()` with `for...of`.

3. **Always wrap `switch` cases in blocks** (`{}`) when using `let`/`const` to avoid shared-scope issues.

4. **Never `return`/`throw` in `finally`.** Use `finally` exclusively for cleanup (closing handles, clearing timers, releasing locks).

5. **Prefer early returns over deep nesting.** Invert conditions and return early to flatten the code.

6. **Use `===` in all comparisons.** The only reasonable exception is `value == null` as shorthand for checking both `null` and `undefined`.

7. **Use labeled breaks for nested loops** when extraction into a separate function is overkill. Document the intent with a clear label name.

8. **Use `.some()`/`.every()` when you need early exit from array method chains** instead of forcing `forEach` to simulate `break`.

## Real-World Use Cases

### Matrix Search with Labeled Break

```javascript
function findInMatrix(matrix, target) {
  let result = null;

  search: for (let row = 0; row < matrix.length; row++) {
    for (let col = 0; col < matrix[row].length; col++) {
      if (matrix[row][col] === target) {
        result = { row, col };
        break search;
      }
    }
  }

  return result;
}

const grid = [
  [1, 2, 3],
  [4, 5, 6],
  [7, 8, 9]
];
findInMatrix(grid, 5); // { row: 1, col: 1 }
```

### Paginated API Consumer with Async Iteration

```javascript
async function* paginate(url) {
  let cursor = null;

  do {
    const params = cursor ? `?cursor=${cursor}` : "";
    const response = await fetch(`${url}${params}`);
    const data = await response.json();

    for (const item of data.items) {
      yield item;
    }

    cursor = data.nextCursor;
  } while (cursor);
}

// Process all items across all pages
async function processAllUsers() {
  for await (const user of paginate("/api/users")) {
    if (user.role === "admin") {
      await notifyAdmin(user);
    }
  }
}
```

### State Machine with Switch

```javascript
function processOrder(order) {
  let state = "pending";

  while (state !== "completed" && state !== "cancelled") {
    switch (state) {
      case "pending": {
        if (!validateOrder(order)) {
          state = "cancelled";
          break;
        }
        state = "validated";
        break;
      }
      case "validated": {
        const paymentOk = chargePayment(order);
        state = paymentOk ? "paid" : "cancelled";
        break;
      }
      case "paid": {
        shipOrder(order);
        state = "completed";
        break;
      }
      default: {
        throw new Error(`Unknown state: ${state}`);
      }
    }
  }

  return state;
}
```

---

> **Key Takeaway:** Use `for...of` for iterables (never `for...in` on arrays), wrap `switch` cases in blocks, never `return` inside `finally`, and use labeled breaks when you need multi-level loop exit. The iteration protocol (`Symbol.iterator`) is the foundation of `for...of`, spread, and destructuring, and understanding it unlocks custom iterable patterns.
