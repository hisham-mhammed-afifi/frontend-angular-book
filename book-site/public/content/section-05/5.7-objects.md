# 5.7 Objects

> Objects are the fundamental building block of JavaScript. Nearly everything in the language is an object or behaves like one, and understanding how they work at a deep level is essential for writing predictable, performant code.

---

## Object Creation Patterns

There are several ways to create objects in JavaScript, each with different implications.

### Object Literal

The most common and concise approach.

```js
const user = {
  name: "Alice",
  age: 30,
  greet() {
    return `Hello, I'm ${this.name}`;
  }
};
```

### `Object.create()`

Creates a new object with a specified prototype. Useful for explicit prototype chain control.

```js
const personProto = {
  greet() {
    return `Hello, I'm ${this.name}`;
  }
};

const user = Object.create(personProto);
user.name = "Alice";

console.log(user.greet()); // "Hello, I'm Alice"
console.log(Object.getPrototypeOf(user) === personProto); // true
```

Creating an object with no prototype at all:

```js
const dict = Object.create(null);
dict.key = "value";

// No inherited properties from Object.prototype
console.log(dict.toString); // undefined
console.log(dict.hasOwnProperty); // undefined

// Useful for dictionaries where you don't want prototype pollution
```

### Constructor Functions

The pre-class pattern for creating multiple objects of the same shape.

```js
function Person(name, age) {
  this.name = name;
  this.age = age;
}

Person.prototype.greet = function () {
  return `Hello, I'm ${this.name}`;
};

const alice = new Person("Alice", 30);
```

### `new Object()` (Avoid)

```js
// Bad Practice: verbose and unnecessary
const obj = new Object();
obj.name = "Alice";

// Good: use a literal
const obj2 = { name: "Alice" };
```

## Property Shorthand and Computed Properties

### Shorthand Properties

When the variable name matches the property name, you can use shorthand syntax.

```js
const name = "Alice";
const age = 30;

// Without shorthand
const user1 = { name: name, age: age };

// With shorthand
const user2 = { name, age };
```

### Shorthand Methods

```js
// Old syntax
const obj1 = {
  greet: function () {
    return "hello";
  }
};

// Shorthand method syntax
const obj2 = {
  greet() {
    return "hello";
  }
};
```

Note: shorthand methods are NOT the same as arrow functions. They have their own `this` binding and can be used as constructors (though you shouldn't).

### Computed Property Names

Property names can be dynamically computed using bracket notation in the literal.

```js
const key = "status";

const obj = {
  [key]: "active",
  [`${key}Code`]: 200,
  [Symbol.iterator]: function* () {
    yield 1;
  }
};

console.log(obj.status);     // "active"
console.log(obj.statusCode); // 200
```

A practical example with computed keys for building lookup maps:

```js
function createLookup(items, keyProp) {
  return items.reduce((acc, item) => {
    acc[item[keyProp]] = item;
    return acc;
  }, {});
}

const users = [
  { id: 1, name: "Alice" },
  { id: 2, name: "Bob" }
];

const byId = createLookup(users, "id");
// { 1: { id: 1, name: "Alice" }, 2: { id: 2, name: "Bob" } }
```

## Property Access

### Dot Notation vs Bracket Notation

```js
const user = { name: "Alice", "favorite-color": "blue" };

// Dot notation: clean, but limited to valid identifier names
console.log(user.name); // "Alice"

// Bracket notation: works with any string or Symbol
console.log(user["name"]);           // "Alice"
console.log(user["favorite-color"]); // "blue"

// Dynamic access requires brackets
const prop = "name";
console.log(user[prop]); // "Alice"
```

### Optional Chaining (`?.`)

Safely access deeply nested properties without checking each level.

```js
const response = {
  data: {
    user: {
      address: {
        city: "Portland"
      }
    }
  }
};

// Without optional chaining
const city1 = response && response.data && response.data.user
  && response.data.user.address && response.data.user.address.city;

// With optional chaining
const city2 = response?.data?.user?.address?.city;

// Works with methods
const length = response?.data?.items?.length; // undefined (no error)

// Works with bracket notation
const key = "city";
const city3 = response?.data?.user?.address?.[key];

// Works with function calls
const result = response?.data?.process?.(); // undefined if process doesn't exist
```

### `in` Operator

Checks whether a property exists in an object or its prototype chain.

```js
const user = { name: "Alice", age: 30 };

console.log("name" in user);     // true
console.log("toString" in user); // true (inherited from Object.prototype)

// To check only own properties:
console.log(user.hasOwnProperty("name"));     // true
console.log(user.hasOwnProperty("toString")); // false

// Safer alternative (doesn't break with Object.create(null)):
console.log(Object.hasOwn(user, "name"));     // true
console.log(Object.hasOwn(user, "toString")); // false
```

`Object.hasOwn()` was introduced in ES2022 and is the preferred way to check own properties. It works even when `hasOwnProperty` has been overridden or when the object has no prototype.

## Property Descriptors

Every property has a descriptor that controls its behavior. Understanding descriptors is key to understanding why objects behave the way they do.

### The Three Descriptor Flags

```js
const user = { name: "Alice" };

const descriptor = Object.getOwnPropertyDescriptor(user, "name");
console.log(descriptor);
// {
//   value: "Alice",
//   writable: true,       // Can the value be changed?
//   enumerable: true,     // Does it show up in for...in / Object.keys?
//   configurable: true    // Can the descriptor be modified or property deleted?
// }
```

### `Object.defineProperty()`

Define a property with explicit descriptor control.

```js
const user = {};

Object.defineProperty(user, "id", {
  value: 42,
  writable: false,       // Cannot reassign
  enumerable: true,      // Shows in for...in
  configurable: false    // Cannot delete or reconfigure
});

user.id = 100; // Silently fails (throws in strict mode)
console.log(user.id); // 42

delete user.id; // Silently fails
console.log(user.id); // 42
```

Important: properties created with `defineProperty` default all flags to `false` if not specified. Properties created with literal syntax default all flags to `true`.

```js
const obj = {};

// Literal assignment: all flags true
obj.a = 1;

// defineProperty without explicit flags: all flags false
Object.defineProperty(obj, "b", { value: 2 });

console.log(Object.getOwnPropertyDescriptor(obj, "a"));
// { value: 1, writable: true, enumerable: true, configurable: true }

console.log(Object.getOwnPropertyDescriptor(obj, "b"));
// { value: 2, writable: false, enumerable: false, configurable: false }
```

### `Object.defineProperties()`

Define multiple properties at once.

```js
const user = {};

Object.defineProperties(user, {
  firstName: {
    value: "Alice",
    writable: true,
    enumerable: true,
    configurable: true
  },
  lastName: {
    value: "Smith",
    writable: true,
    enumerable: true,
    configurable: true
  },
  fullName: {
    get() {
      return `${this.firstName} ${this.lastName}`;
    },
    enumerable: true,
    configurable: true
  }
});

console.log(user.fullName); // "Alice Smith"
```

### Non-Enumerable Properties

```js
const user = { name: "Alice" };

Object.defineProperty(user, "secret", {
  value: "hidden",
  enumerable: false
});

console.log(user.secret);           // "hidden"
console.log(Object.keys(user));     // ["name"]
console.log(JSON.stringify(user));  // '{"name":"Alice"}'

// for...in skips non-enumerable properties
for (const key in user) {
  console.log(key); // only "name"
}

// But these still find it
console.log(Object.getOwnPropertyNames(user)); // ["name", "secret"]
```

## Getters and Setters

Accessor properties let you define computed properties and add validation logic.

```js
const user = {
  firstName: "Alice",
  lastName: "Smith",

  get fullName() {
    return `${this.firstName} ${this.lastName}`;
  },

  set fullName(value) {
    const parts = value.split(" ");
    if (parts.length < 2) {
      throw new Error("Full name must include first and last name");
    }
    this.firstName = parts[0];
    this.lastName = parts.slice(1).join(" ");
  }
};

console.log(user.fullName); // "Alice Smith"
user.fullName = "Bob Jones";
console.log(user.firstName); // "Bob"
console.log(user.lastName);  // "Jones"
```

### Getters for Lazy Computation

```js
const config = {
  _cachedRegex: null,
  pattern: "\\d{3}-\\d{4}",

  get regex() {
    if (!this._cachedRegex) {
      this._cachedRegex = new RegExp(this.pattern);
    }
    return this._cachedRegex;
  }
};
```

### Getters with `defineProperty` for "Private" Backing Fields

```js
function createCounter(initial = 0) {
  const obj = {};
  let count = initial; // truly private via closure

  Object.defineProperty(obj, "count", {
    get() { return count; },
    set(value) {
      if (typeof value !== "number" || value < 0) {
        throw new TypeError("Count must be a non-negative number");
      }
      count = value;
    },
    enumerable: true,
    configurable: false
  });

  obj.increment = function () { count++; };
  obj.decrement = function () { if (count > 0) count--; };

  return obj;
}

const counter = createCounter(0);
counter.increment();
console.log(counter.count); // 1
counter.count = 10;         // Works, passes validation
counter.count = -5;         // TypeError
```

## Object Immutability

JavaScript provides three levels of immutability, each progressively more restrictive.

### `Object.preventExtensions()`

Prevents new properties from being added. Existing properties can still be modified or deleted.

```js
const user = { name: "Alice" };
Object.preventExtensions(user);

user.name = "Bob";  // Works
delete user.name;   // Works
user.age = 30;      // Silently fails (throws in strict mode)

console.log(Object.isExtensible(user)); // false
```

### `Object.seal()`

Prevents adding or removing properties. Existing properties can still be modified (if writable).

```js
const user = { name: "Alice", age: 30 };
Object.seal(user);

user.name = "Bob";  // Works (property is still writable)
delete user.name;   // Fails
user.email = "a@b"; // Fails

console.log(Object.isSealed(user)); // true
```

### `Object.freeze()`

Most restrictive. No adding, removing, or modifying properties.

```js
const user = { name: "Alice", age: 30 };
Object.freeze(user);

user.name = "Bob";  // Fails silently
delete user.name;   // Fails silently
user.email = "a@b"; // Fails silently

console.log(Object.isFrozen(user)); // true
```

### Shallow Freeze Gotcha

All three operations are **shallow**. Nested objects are not affected.

```js
const config = {
  db: {
    host: "localhost",
    port: 5432
  }
};

Object.freeze(config);

config.db.host = "production-server"; // This WORKS!
console.log(config.db.host); // "production-server"
```

Deep freeze utility:

```js
function deepFreeze(obj) {
  Object.freeze(obj);
  for (const key of Object.getOwnPropertyNames(obj)) {
    const value = obj[key];
    if (value !== null && typeof value === "object" && !Object.isFrozen(value)) {
      deepFreeze(value);
    }
  }
  return obj;
}

const config = deepFreeze({
  db: { host: "localhost", port: 5432 }
});

config.db.host = "hacked"; // Fails silently
console.log(config.db.host); // "localhost"
```

### Comparison Table

| Feature | `preventExtensions` | `seal` | `freeze` |
|---|---|---|---|
| Add new properties | No | No | No |
| Delete properties | Yes | No | No |
| Modify values | Yes | Yes | No |
| Reconfigure descriptors | Yes | No | No |
| Check method | `isExtensible()` | `isSealed()` | `isFrozen()` |
| Depth | Shallow | Shallow | Shallow |

## Object Static Methods

### `Object.keys()`, `Object.values()`, `Object.entries()`

These return only **own, enumerable, string-keyed** properties.

```js
const user = { name: "Alice", age: 30 };

Object.keys(user);    // ["name", "age"]
Object.values(user);  // ["Alice", 30]
Object.entries(user); // [["name", "Alice"], ["age", 30]]
```

### `Object.fromEntries()`

The inverse of `Object.entries()`. Creates an object from key-value pairs.

```js
const entries = [["name", "Alice"], ["age", 30]];
const user = Object.fromEntries(entries);
// { name: "Alice", age: 30 }

// Works with Map
const map = new Map([["a", 1], ["b", 2]]);
const obj = Object.fromEntries(map);
// { a: 1, b: 2 }

// Transform object values
const prices = { apple: 1.5, banana: 0.75, cherry: 2.0 };
const doubled = Object.fromEntries(
  Object.entries(prices).map(([key, val]) => [key, val * 2])
);
// { apple: 3, banana: 1.5, cherry: 4 }
```

### `Object.assign()`

Copies own, enumerable properties from one or more source objects to a target.

```js
const defaults = { theme: "light", lang: "en", debug: false };
const userPrefs = { theme: "dark", lang: "fr" };

const config = Object.assign({}, defaults, userPrefs);
// { theme: "dark", lang: "fr", debug: false }
```

Key behaviors of `Object.assign()`:
- It performs a **shallow** copy.
- It invokes getters on source and setters on target.
- Properties from later sources overwrite earlier ones.
- It skips `null` and `undefined` sources.
- It only copies enumerable, own properties.

```js
// Shallow copy problem
const original = { data: { count: 1 } };
const copy = Object.assign({}, original);
copy.data.count = 99;
console.log(original.data.count); // 99 (shared reference!)
```

### `Object.assign()` vs Spread

```js
const a = { x: 1 };
const b = { y: 2 };

// These are nearly equivalent
const result1 = Object.assign({}, a, b);
const result2 = { ...a, ...b };

// Key differences:
// 1. Object.assign invokes setters on the target; spread does not
// 2. Spread is syntax, not a method call (cannot be polyfilled as easily)
// 3. Object.assign returns the target object
```

### `Object.groupBy()` (ES2024)

Groups elements of an iterable based on a callback.

```js
const people = [
  { name: "Alice", dept: "eng" },
  { name: "Bob", dept: "sales" },
  { name: "Carol", dept: "eng" },
  { name: "Dave", dept: "sales" },
  { name: "Eve", dept: "eng" }
];

const byDept = Object.groupBy(people, (person) => person.dept);
// {
//   eng: [{ name: "Alice", ... }, { name: "Carol", ... }, { name: "Eve", ... }],
//   sales: [{ name: "Bob", ... }, { name: "Dave", ... }]
// }
```

Note: `Object.groupBy()` returns a null-prototype object (like `Object.create(null)`), so there are no inherited properties.

## Cloning and Copying Objects

### Shallow Copy Methods

```js
const original = { a: 1, b: { c: 2 } };

// Method 1: spread
const copy1 = { ...original };

// Method 2: Object.assign
const copy2 = Object.assign({}, original);

// Both are shallow: nested objects are shared references
copy1.b.c = 99;
console.log(original.b.c); // 99
```

### `structuredClone()` for Deep Copy

`structuredClone()` is the modern, built-in way to deep copy objects. Available in all modern runtimes.

```js
const original = {
  name: "Alice",
  scores: [10, 20, 30],
  metadata: {
    created: new Date("2024-01-01"),
    tags: new Set(["admin", "user"])
  }
};

const clone = structuredClone(original);

clone.scores.push(40);
clone.metadata.tags.add("superadmin");

console.log(original.scores);        // [10, 20, 30] (unchanged)
console.log(original.metadata.tags); // Set(2) {"admin", "user"} (unchanged)
```

What `structuredClone` supports:
- Plain objects, arrays
- `Date`, `RegExp`, `Map`, `Set`
- `ArrayBuffer`, `TypedArray`, `Blob`, `File`
- `ImageData`, `ImageBitmap`

What `structuredClone` does NOT support:
- Functions (throws `DataCloneError`)
- DOM nodes
- Symbols as property keys (skipped)
- Getters/setters (evaluated, only values copied)
- Prototype chain (not preserved)

```js
// Bad Practice: using JSON for deep clone
const clone = JSON.parse(JSON.stringify(original));
// Problems:
// - Loses Date objects (becomes strings)
// - Loses undefined values
// - Loses Map, Set, RegExp
// - Fails on circular references
// - Ignores Symbol keys and non-enumerable properties
```

### Comparison Table: Cloning Methods

| Method | Deep? | Functions | Date | Map/Set | Circular | Prototype |
|---|---|---|---|---|---|---|
| Spread / `Object.assign` | No | Ref copy | Ref copy | Ref copy | N/A | No |
| `JSON.parse(JSON.stringify())` | Yes | Lost | String | Lost | Throws | No |
| `structuredClone()` | Yes | Throws | Preserved | Preserved | Handled | No |

## Property Enumeration Order

Since ES2015, JavaScript guarantees a specific order for property enumeration in `Object.keys()`, `Object.values()`, `Object.entries()`, `for...in`, `Object.getOwnPropertyNames()`, and `Reflect.ownKeys()`.

The order is:
1. Integer-like keys (array indices) in ascending numeric order
2. String keys in insertion order
3. Symbol keys in insertion order

```js
const obj = {};
obj.b = "b";
obj[2] = "two";
obj.a = "a";
obj[1] = "one";
obj[Symbol("s")] = "sym";

// Integer-like keys first (sorted), then string keys (insertion order)
console.log(Object.keys(obj));
// ["1", "2", "b", "a"]

// Reflect.ownKeys includes Symbols
console.log(Reflect.ownKeys(obj));
// ["1", "2", "b", "a", Symbol(s)]
```

What counts as an "integer-like" key:

```js
// These are integer-like (canonical numeric strings for array indices):
// "0", "1", "2", ..., "4294967294"

// These are NOT integer-like (treated as regular string keys):
// "-1", "1.5", "01", "Infinity", "4294967295"

const obj = {
  "01": "not integer-like",
  "1": "integer-like",
  b: "string key"
};

console.log(Object.keys(obj)); // ["1", "01", "b"]
```

## Iterating Over Objects

### `for...in`

Iterates over **own and inherited** enumerable string-keyed properties.

```js
const proto = { inherited: true };
const obj = Object.create(proto);
obj.own = true;

for (const key in obj) {
  console.log(key); // "own", "inherited"
}

// Filter to own properties only
for (const key in obj) {
  if (Object.hasOwn(obj, key)) {
    console.log(key); // "own"
  }
}
```

### Prefer `Object.keys/values/entries` Over `for...in`

```js
const user = { name: "Alice", age: 30, role: "admin" };

// Clean and predictable (own properties only)
for (const [key, value] of Object.entries(user)) {
  console.log(`${key}: ${value}`);
}

// Functional style
Object.entries(user)
  .filter(([_, value]) => typeof value === "string")
  .forEach(([key, value]) => {
    console.log(`${key}: ${value}`);
  });
```

### `Object.getOwnPropertyNames()` and `Reflect.ownKeys()`

```js
const obj = {};
Object.defineProperty(obj, "hidden", { value: 1, enumerable: false });
obj[Symbol("sym")] = 2;
obj.visible = 3;

Object.keys(obj);                    // ["visible"]
Object.getOwnPropertyNames(obj);     // ["hidden", "visible"]
Object.getOwnPropertySymbols(obj);   // [Symbol(sym)]
Reflect.ownKeys(obj);                // ["hidden", "visible", Symbol(sym)]
```

| Method | Own only | Enumerable only | String keys | Symbol keys |
|---|---|---|---|---|
| `for...in` | No | Yes | Yes | No |
| `Object.keys()` | Yes | Yes | Yes | No |
| `Object.getOwnPropertyNames()` | Yes | No | Yes | No |
| `Object.getOwnPropertySymbols()` | Yes | No | No | Yes |
| `Reflect.ownKeys()` | Yes | No | Yes | Yes |

## Symbol Keys

Symbols are unique identifiers that can serve as property keys. They do not collide with string keys and are not enumerable by default iteration methods.

```js
const ID = Symbol("id");
const ROLE = Symbol("role");

const user = {
  [ID]: 42,
  [ROLE]: "admin",
  name: "Alice"
};

console.log(user[ID]);   // 42
console.log(user[ROLE]); // "admin"

// Not visible to standard enumeration
console.log(Object.keys(user));           // ["name"]
console.log(JSON.stringify(user));        // '{"name":"Alice"}'

// Must use symbol-aware methods
console.log(Object.getOwnPropertySymbols(user)); // [Symbol(id), Symbol(role)]
console.log(Reflect.ownKeys(user));               // ["name", Symbol(id), Symbol(role)]
```

### Well-Known Symbols

JavaScript defines built-in symbols that customize object behavior.

```js
// Symbol.toPrimitive: controls type coercion
const money = {
  amount: 100,
  currency: "USD",

  [Symbol.toPrimitive](hint) {
    if (hint === "number") return this.amount;
    if (hint === "string") return `${this.amount} ${this.currency}`;
    return this.amount; // default
  }
};

console.log(+money);          // 100
console.log(`${money}`);      // "100 USD"
console.log(money + 50);      // 150

// Symbol.iterator: make an object iterable
const range = {
  start: 1,
  end: 5,

  [Symbol.iterator]() {
    let current = this.start;
    const end = this.end;
    return {
      next() {
        return current <= end
          ? { value: current++, done: false }
          : { done: true };
      }
    };
  }
};

console.log([...range]); // [1, 2, 3, 4, 5]
```

## Prototype Chain

Every object in JavaScript has an internal `[[Prototype]]` link. When you access a property that doesn't exist on the object itself, the engine walks up the prototype chain to find it.

```js
const animal = {
  type: "animal",
  speak() {
    return `${this.name} makes a sound`;
  }
};

const dog = Object.create(animal);
dog.name = "Rex";
dog.bark = function () {
  return `${this.name} barks`;
};

console.log(dog.bark());    // "Rex barks" (own property)
console.log(dog.speak());   // "Rex makes a sound" (inherited from animal)
console.log(dog.type);      // "animal" (inherited)
console.log(dog.toString); // [Function: toString] (from Object.prototype)
```

### Inspecting the Prototype Chain

```js
console.log(Object.getPrototypeOf(dog) === animal);                 // true
console.log(Object.getPrototypeOf(animal) === Object.prototype);    // true
console.log(Object.getPrototypeOf(Object.prototype) === null);      // true

// The chain: dog -> animal -> Object.prototype -> null
```

### Prototype Lookup Performance

Property lookup walks the prototype chain until it finds the property or reaches `null`. For hot code paths, prefer own properties to avoid the lookup overhead.

```js
// If performance matters and you're accessing a property millions of times:
// Cache the value in a local variable
const method = dog.speak; // Resolve once
for (let i = 0; i < 1000000; i++) {
  method.call(dog); // No prototype lookup per iteration
}
```

## Merging and Combining Objects

### Spread Operator for Merging

```js
const defaults = { theme: "light", lang: "en", notifications: true };
const userPrefs = { theme: "dark" };

const settings = { ...defaults, ...userPrefs };
// { theme: "dark", lang: "en", notifications: true }
```

### Deep Merge

Neither spread nor `Object.assign` handles nested objects. You need a custom deep merge or a library.

```js
function deepMerge(target, ...sources) {
  for (const source of sources) {
    for (const key of Object.keys(source)) {
      const targetVal = target[key];
      const sourceVal = source[key];

      if (
        sourceVal !== null &&
        typeof sourceVal === "object" &&
        !Array.isArray(sourceVal) &&
        targetVal !== null &&
        typeof targetVal === "object" &&
        !Array.isArray(targetVal)
      ) {
        target[key] = deepMerge({ ...targetVal }, sourceVal);
      } else {
        target[key] = sourceVal;
      }
    }
  }
  return target;
}

const config = deepMerge(
  { db: { host: "localhost", port: 5432 }, debug: false },
  { db: { host: "production" }, logging: true }
);
// { db: { host: "production", port: 5432 }, debug: false, logging: true }
```

## Object Equality and Comparison

### Reference Equality

Objects are compared by reference, not by value.

```js
const a = { x: 1 };
const b = { x: 1 };
const c = a;

console.log(a === b);  // false (different references)
console.log(a === c);  // true (same reference)
```

### `Object.is()`

Similar to `===` but handles two edge cases differently.

```js
console.log(NaN === NaN);          // false
console.log(Object.is(NaN, NaN));  // true

console.log(0 === -0);             // true
console.log(Object.is(0, -0));     // false
```

### Shallow Equality Check

Commonly used in UI frameworks for change detection.

```js
function shallowEqual(obj1, obj2) {
  if (obj1 === obj2) return true;
  if (!obj1 || !obj2) return false;

  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);

  if (keys1.length !== keys2.length) return false;

  return keys1.every((key) => Object.is(obj1[key], obj2[key]));
}

console.log(shallowEqual({ a: 1, b: 2 }, { a: 1, b: 2 })); // true
console.log(shallowEqual({ a: 1, b: 2 }, { a: 1, b: 3 })); // false
```

## `WeakRef` and `FinalizationRegistry`

These are advanced features for managing references to objects without preventing garbage collection.

### `WeakRef`

Holds a weak reference to an object. The referenced object can be garbage collected.

```js
let target = { data: "important" };
const weakRef = new WeakRef(target);

console.log(weakRef.deref()); // { data: "important" }

target = null; // Remove the strong reference

// At some point after GC runs:
// weakRef.deref() will return undefined
```

### `FinalizationRegistry`

Registers a callback that runs when a registered object is garbage collected.

```js
const registry = new FinalizationRegistry((heldValue) => {
  console.log(`Object with key "${heldValue}" was garbage collected`);
});

function createTrackedObject(key) {
  const obj = { key };
  registry.register(obj, key);
  return obj;
}

let tracked = createTrackedObject("user-session");
tracked = null; // Eventually: "Object with key "user-session" was garbage collected"
```

Use these sparingly. They are non-deterministic and should never be relied upon for critical cleanup. Their main use case is caching and resource management where losing data is acceptable.

## Common Mistakes

- **Mutating shared objects**: Passing an object to a function and modifying it changes the original. If you need isolation, clone first.

```js
// Bad Practice
function addDefaults(config) {
  config.timeout = config.timeout || 5000; // Mutates the caller's object!
  return config;
}

// Good Practice
function addDefaults(config) {
  return { timeout: 5000, ...config }; // Returns a new object
}
```

- **Confusing shallow and deep copy**: Assuming spread or `Object.assign` creates a deep copy.

```js
// Bad Practice
const copy = { ...original }; // Only top-level properties are copied
copy.nested.value = "changed"; // Mutates original.nested too!

// Good Practice
const copy = structuredClone(original);
```

- **Using `for...in` without own-property check**: Inherited properties leak into iteration.

```js
// Bad Practice
for (const key in obj) {
  doSomething(obj[key]); // May include inherited properties
}

// Good Practice
for (const key of Object.keys(obj)) {
  doSomething(obj[key]);
}
```

- **Relying on property order for non-integer keys before understanding the spec**: While modern engines follow the spec, some developers still assume insertion order without understanding that integer-like keys come first.

- **Using `delete` for performance-sensitive code**: `delete` makes objects transition to "dictionary mode" in V8, losing optimized hidden class shapes. If you need to "remove" a key, consider setting it to `undefined` instead, or use `Map`.

```js
// Potentially slow in hot paths
delete obj.key;

// Alternatives
obj.key = undefined; // Keeps the shape
// Or use a Map for dynamic key addition/removal
```

- **Freezing objects and expecting deep immutability**: `Object.freeze` is shallow.

## Best Practices

- Use `Object.hasOwn()` instead of `hasOwnProperty()`. It is safer and works with `Object.create(null)`.
- Prefer `structuredClone()` over `JSON.parse(JSON.stringify())` for deep copies. It handles more types and circular references.
- Use `Object.freeze()` for configuration objects that should never change. Combine with a deep freeze utility if the object has nested structures.
- Use `Object.fromEntries()` with `Object.entries()` for functional object transformations (filter, map).
- Prefer `Map` over plain objects when keys are dynamic, especially if keys are added and removed frequently. `Map` maintains insertion order for all key types and has better performance for frequent additions/deletions.
- Keep object shapes consistent. Creating objects with the same properties in the same order allows V8 to optimize with hidden classes.

```js
// Good: consistent shape
function createUser(name, age) {
  return { name, age, email: null }; // Always the same shape
}

// Bad: inconsistent shape
function createUser(name, age, email) {
  const user = { name, age };
  if (email) user.email = email; // Different shape depending on input
  return user;
}
```

- Use computed property names to build objects dynamically instead of mutation.

```js
// Good: declarative
const field = "email";
const value = "alice@example.com";
const update = { [field]: value };

// Avoid: imperative mutation
const update2 = {};
update2[field] = value;
```

## Real-World Use Cases

### Configuration Object Pattern

```js
// Merging layered configurations with defaults
function createApiClient(options = {}) {
  const defaults = {
    baseUrl: "https://api.example.com",
    timeout: 10000,
    headers: {
      "Content-Type": "application/json"
    },
    retry: {
      attempts: 3,
      delay: 1000
    }
  };

  // Deep merge user options over defaults
  const config = {
    ...defaults,
    ...options,
    headers: { ...defaults.headers, ...options.headers },
    retry: { ...defaults.retry, ...options.retry }
  };

  return {
    get(path) {
      return fetch(`${config.baseUrl}${path}`, {
        headers: config.headers,
        signal: AbortSignal.timeout(config.timeout)
      });
    }
  };
}

const client = createApiClient({
  baseUrl: "https://api.staging.example.com",
  headers: { Authorization: "Bearer token123" }
});
```

### Object as a Lookup / Registry

```js
// Permission system using object lookups
const PERMISSIONS = Object.freeze({
  READ: 1,
  WRITE: 2,
  DELETE: 4,
  ADMIN: 8
});

function createRole(name, ...perms) {
  const bitmask = perms.reduce((mask, p) => mask | p, 0);
  return Object.freeze({ name, bitmask });
}

const roles = {
  viewer: createRole("viewer", PERMISSIONS.READ),
  editor: createRole("editor", PERMISSIONS.READ, PERMISSIONS.WRITE),
  admin: createRole("admin",
    PERMISSIONS.READ, PERMISSIONS.WRITE,
    PERMISSIONS.DELETE, PERMISSIONS.ADMIN
  )
};

function hasPermission(role, permission) {
  return (role.bitmask & permission) !== 0;
}

console.log(hasPermission(roles.editor, PERMISSIONS.WRITE));  // true
console.log(hasPermission(roles.editor, PERMISSIONS.DELETE)); // false
```

### State Management with Immutable Updates

```js
// Functional state updates using object spread
function appReducer(state, action) {
  switch (action.type) {
    case "SET_USER":
      return {
        ...state,
        user: action.payload
      };

    case "UPDATE_PREFERENCES":
      return {
        ...state,
        preferences: {
          ...state.preferences,
          ...action.payload
        }
      };

    case "ADD_NOTIFICATION":
      return {
        ...state,
        notifications: [...state.notifications, action.payload]
      };

    case "REMOVE_NOTIFICATION":
      return {
        ...state,
        notifications: state.notifications.filter(
          (n) => n.id !== action.payload
        )
      };

    default:
      return state;
  }
}

let state = {
  user: null,
  preferences: { theme: "light", lang: "en" },
  notifications: []
};

state = appReducer(state, {
  type: "UPDATE_PREFERENCES",
  payload: { theme: "dark" }
});

// state.preferences is now { theme: "dark", lang: "en" }
// Original preferences object was not mutated
```
