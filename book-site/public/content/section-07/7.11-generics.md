# 7.11 Generics

> Generics let you write functions, classes, and types that work with any type while preserving the relationship between inputs and outputs. They are the mechanism for building reusable, type-safe abstractions without duplicating code.

---

## What Are Generics?

A generic is a type parameter: a placeholder that gets filled in when the function, class, or type is used. Instead of hardcoding a specific type, you parameterize it.

```typescript
// Without generics: one function per type
function identityString(value: string): string {
  return value;
}
function identityNumber(value: number): number {
  return value;
}

// With generics: one function for all types
function identity<T>(value: T): T {
  return value;
}

identity("hello"); // T inferred as string, returns string
identity(42);      // T inferred as number, returns number
identity(true);    // T inferred as boolean, returns boolean
```

The `<T>` after the function name declares a type parameter `T`. When the function is called, TypeScript infers what `T` is from the argument, or you can specify it explicitly.

## Generic Functions

### Basic Syntax

```typescript
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}

const a = first([1, 2, 3]);       // T = number, returns number | undefined
const b = first(["a", "b"]);      // T = string, returns string | undefined
const c = first<boolean>([true]);  // T = boolean explicitly
```

### Multiple Type Parameters

```typescript
function pair<A, B>(first: A, second: B): [A, B] {
  return [first, second];
}

const p = pair("hello", 42); // [string, number]
```

```typescript
function map<T, U>(arr: T[], fn: (item: T) => U): U[] {
  return arr.map(fn);
}

const lengths = map(["hello", "world"], (s) => s.length);
// T = string, U = number, result: number[]
```

### Arrow Function Generics

```typescript
const identity = <T>(value: T): T => value;

// In .tsx files, <T> looks like JSX. Add a trailing comma to disambiguate:
const identity = <T,>(value: T): T => value;
```

### Type Inference in Practice

TypeScript infers generic types from function arguments. You rarely need to specify them explicitly.

```typescript
function merge<T extends object, U extends object>(a: T, b: U): T & U {
  return { ...a, ...b };
}

// T and U are inferred from the arguments
const result = merge({ name: "Alice" }, { age: 30 });
// result: { name: string } & { age: number }
// TypeScript knows result.name is string and result.age is number
```

### When Explicit Type Arguments Are Needed

Sometimes inference fails or produces a type that is too wide.

```typescript
// Inference fails: no argument to infer from
function createArray<T>(): T[] {
  return [];
}

const nums = createArray<number>(); // must specify T
// const bad = createArray(); // T defaults to unknown

// Inference is too wide
function parseJSON<T>(json: string): T {
  return JSON.parse(json);
}

const user = parseJSON<User>('{"name":"Alice"}'); // must specify T
```

## Generic Constraints

Constraints restrict what types a generic parameter can accept using `extends`.

### Basic Constraints

```typescript
// T must have a length property
function logLength<T extends { length: number }>(value: T): T {
  console.log(value.length);
  return value;
}

logLength("hello");     // string has length
logLength([1, 2, 3]);   // array has length
logLength({ length: 5, custom: true }); // object with length
// logLength(42);        // Error: number does not have length
```

### `keyof` Constraint

The `keyof` constraint ensures a type parameter is a valid property name of another type.

```typescript
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const user = { name: "Alice", age: 30, email: "alice@example.com" };

const name = getProperty(user, "name");  // string
const age = getProperty(user, "age");    // number
// getProperty(user, "phone");           // Error: "phone" is not in keyof typeof user
```

This is one of the most important generic patterns. `K extends keyof T` guarantees type-safe property access.

### Multiple Constraints

Use intersection (`&`) to combine constraints.

```typescript
function processEntity<T extends { id: string } & { name: string }>(entity: T): string {
  return `${entity.id}: ${entity.name}`;
}

// Or use an interface
interface Identifiable {
  id: string;
}

interface Nameable {
  name: string;
}

function processEntity<T extends Identifiable & Nameable>(entity: T): string {
  return `${entity.id}: ${entity.name}`;
}
```

### Constraint with Another Type Parameter

```typescript
function assign<T extends object, U extends Partial<T>>(target: T, source: U): T {
  return Object.assign(target, source);
}

const user = { name: "Alice", age: 30 };
assign(user, { age: 31 });        // OK: { age: number } is Partial<typeof user>
// assign(user, { role: "admin" }); // Error: 'role' does not exist on Partial<typeof user>
```

## Default Type Parameters

Like function default values, generic parameters can have default types.

```typescript
type ApiResponse<T = unknown> = {
  data: T;
  status: number;
  message: string;
};

// Without specifying T, it defaults to unknown
const res1: ApiResponse = { data: "anything", status: 200, message: "OK" };

// With explicit T
const res2: ApiResponse<User[]> = { data: [], status: 200, message: "OK" };
```

### Rules for Defaults

1. Default parameters must come after non-default parameters.
2. You can only omit type arguments from the right.

```typescript
// Valid: default comes after required
type Result<T, E = Error> =
  | { ok: true; value: T }
  | { ok: false; error: E };

const r1: Result<string> = { ok: true, value: "hello" }; // E defaults to Error
const r2: Result<string, string> = { ok: false, error: "oops" }; // E is string

// Invalid: required after default
// type Bad<T = string, U> = { a: T; b: U }; // Error
```

### Defaults with Constraints

```typescript
type Container<T extends object = Record<string, unknown>> = {
  data: T;
  timestamp: Date;
};

const c1: Container = { data: { any: "thing" }, timestamp: new Date() };
const c2: Container<{ name: string }> = { data: { name: "Alice" }, timestamp: new Date() };
// const c3: Container<string> = ...; // Error: string does not extend object
```

## Generic Interfaces and Type Aliases

### Generic Interfaces

```typescript
interface Repository<T> {
  getById(id: string): Promise<T | null>;
  getAll(): Promise<T[]>;
  create(item: Omit<T, "id">): Promise<T>;
  update(id: string, item: Partial<T>): Promise<T>;
  delete(id: string): Promise<boolean>;
}
```

### Generic Type Aliases

```typescript
type Nullable<T> = T | null;
type AsyncResult<T, E = Error> = Promise<Result<T, E>>;
type Pair<T> = [T, T];
type Mapper<In, Out> = (input: In) => Out;
```

### Generic Interface vs Generic Method

There is a key difference between placing the type parameter on the interface vs on individual methods.

```typescript
// Type parameter on the interface: T is fixed for the entire interface
interface Box<T> {
  value: T;
  map(fn: (v: T) => T): Box<T>;
}

// Type parameter on the method: each method call can use a different type
interface Converter {
  convert<T>(input: string): T;
}
```

## Generic Classes

```typescript
class Stack<T> {
  private items: T[] = [];

  push(item: T): void {
    this.items.push(item);
  }

  pop(): T | undefined {
    return this.items.pop();
  }

  peek(): T | undefined {
    return this.items[this.items.length - 1];
  }

  get size(): number {
    return this.items.length;
  }
}

const numbers = new Stack<number>();
numbers.push(1);
numbers.push(2);
const top = numbers.pop(); // number | undefined

const strings = new Stack<string>();
strings.push("hello");
// strings.push(42); // Error: number is not assignable to string
```

### Generic Classes with Constraints

```typescript
class KeyValueStore<K extends string | number, V> {
  private store = new Map<K, V>();

  set(key: K, value: V): void {
    this.store.set(key, value);
  }

  get(key: K): V | undefined {
    return this.store.get(key);
  }

  has(key: K): boolean {
    return this.store.has(key);
  }

  keys(): K[] {
    return Array.from(this.store.keys());
  }
}

const settings = new KeyValueStore<string, string | number | boolean>();
settings.set("theme", "dark");
settings.set("fontSize", 14);
settings.set("notifications", true);
```

### Static Members Cannot Use Class Type Parameters

```typescript
class Container<T> {
  value: T;

  constructor(value: T) {
    this.value = value;
  }

  // Error: Static members cannot reference class type parameters
  // static defaultValue: T;

  // Static methods can have their OWN generics
  static create<U>(value: U): Container<U> {
    return new Container(value);
  }
}

const c = Container.create("hello"); // Container<string>
```

## Generic Utility Patterns

### The Identity Pattern

The simplest generic: preserves the exact input type.

```typescript
function identity<T>(value: T): T {
  return value;
}

// Useful for type inference in complex scenarios
const result = identity({ name: "Alice", age: 30 });
// result: { name: string; age: number } (exact shape preserved)
```

### The Lookup Pattern (`keyof` + indexed access)

```typescript
function pick<T, K extends keyof T>(obj: T, keys: K[]): Pick<T, K> {
  const result = {} as Pick<T, K>;
  for (const key of keys) {
    result[key] = obj[key];
  }
  return result;
}

const user = { id: 1, name: "Alice", email: "alice@example.com", password: "secret" };
const publicUser = pick(user, ["id", "name", "email"]);
// { id: number; name: string; email: string }
// password is excluded, and TypeScript knows the exact shape
```

### The Factory Pattern

```typescript
function create<T>(ctor: new () => T): T {
  return new ctor();
}

class User {
  name = "default";
}

const user = create(User); // User
```

With constructor arguments:

```typescript
function createWith<T, A extends any[]>(
  ctor: new (...args: A) => T,
  ...args: A
): T {
  return new ctor(...args);
}

class User {
  constructor(public name: string, public age: number) {}
}

const user = createWith(User, "Alice", 30);
// T = User, A = [string, number]
```

### The Builder Pattern

```typescript
class QueryBuilder<T extends object> {
  private conditions: string[] = [];
  private selectedFields: (keyof T)[] = [];

  where<K extends keyof T>(field: K, value: T[K]): this {
    this.conditions.push(`${String(field)} = ${JSON.stringify(value)}`);
    return this;
  }

  select<K extends keyof T>(...fields: K[]): QueryBuilder<Pick<T, K>> {
    this.selectedFields = fields as any;
    return this as any;
  }

  build(): string {
    const fields = this.selectedFields.length
      ? this.selectedFields.join(", ")
      : "*";
    const where = this.conditions.length
      ? ` WHERE ${this.conditions.join(" AND ")}`
      : "";
    return `SELECT ${fields}${where}`;
  }
}

type User = { id: number; name: string; email: string; age: number };

const query = new QueryBuilder<User>()
  .where("age", 30)           // field must be keyof User, value must match type
  .where("name", "Alice")
  .select("id", "name")       // only valid User keys
  .build();
// "SELECT id, name WHERE age = 30 AND name = "Alice""
```

## Inference with `infer`

The `infer` keyword declares a type variable inside a conditional type, letting TypeScript "extract" part of a type.

### Extract Return Type

```typescript
type ReturnOf<T> = T extends (...args: any[]) => infer R ? R : never;

type A = ReturnOf<() => string>;            // string
type B = ReturnOf<(x: number) => boolean>;  // boolean
type C = ReturnOf<string>;                  // never (not a function)
```

### Extract Element Type

```typescript
type ElementOf<T> = T extends (infer E)[] ? E : never;

type A = ElementOf<string[]>;       // string
type B = ElementOf<(number | boolean)[]>; // number | boolean
```

### Extract Promise Type

```typescript
type UnwrapPromise<T> = T extends Promise<infer U> ? UnwrapPromise<U> : T;

type A = UnwrapPromise<Promise<string>>;           // string
type B = UnwrapPromise<Promise<Promise<number>>>;  // number (recursive)
type C = UnwrapPromise<boolean>;                   // boolean (not a promise)
```

### Extract Constructor Instance Type

```typescript
type InstanceOf<T> = T extends new (...args: any[]) => infer I ? I : never;

class Foo { x = 1; }
type A = InstanceOf<typeof Foo>; // Foo
```

### Multiple `infer` Sites

```typescript
type FirstAndLast<T> = T extends [infer F, ...any[], infer L] ? [F, L] : never;

type A = FirstAndLast<[1, 2, 3, 4]>;  // [1, 4]
type B = FirstAndLast<["a", "b"]>;     // ["a", "b"]
type C = FirstAndLast<[string]>;       // never (needs at least 2 elements)
```

## Variance

Variance describes how the subtype relationship of a generic type relates to the subtype relationship of its type parameter. This is an advanced topic, but understanding the basics prevents subtle bugs.

### Covariance (Output Position)

A generic type is **covariant** in `T` when `Container<Cat>` is assignable to `Container<Animal>` (if `Cat extends Animal`). This happens when `T` appears in output (return) positions.

```typescript
interface Producer<T> {
  get(): T;
}

declare let catProducer: Producer<Cat>;
declare let animalProducer: Producer<Animal>;

// Cat extends Animal, so Producer<Cat> is assignable to Producer<Animal>
animalProducer = catProducer; // OK (covariant)
// catProducer = animalProducer; // Error (a Producer<Animal> might produce a Dog)
```

### Contravariance (Input Position)

A generic type is **contravariant** in `T` when `Handler<Animal>` is assignable to `Handler<Cat>` (reversed direction). This happens when `T` appears in input (parameter) positions.

```typescript
interface Consumer<T> {
  accept(value: T): void;
}

declare let catConsumer: Consumer<Cat>;
declare let animalConsumer: Consumer<Animal>;

// Consumer<Animal> can accept any Animal, including Cats
catConsumer = animalConsumer; // OK (contravariant)
// animalConsumer = catConsumer; // Error (Cat consumer cannot handle Dog)
```

### Variance Annotations (`in` / `out`)

TypeScript 4.7 introduced explicit variance annotations. You rarely need them, but they exist for performance optimization in large codebases.

```typescript
interface Producer<out T> {  // covariant
  get(): T;
}

interface Consumer<in T> {   // contravariant
  accept(value: T): void;
}

interface Processor<in out T> { // invariant (both input and output)
  process(value: T): T;
}
```

In practice, TypeScript infers variance correctly. Use annotations only if the compiler reports performance issues with complex generic types.

## The Golden Rule of Generics

From "Effective TypeScript": **Type parameters are for relating the types of multiple values.** If a type parameter appears only once in the function signature, you probably do not need it.

```typescript
// Bad: T is used only once, it does not relate anything
function greet<T extends string>(name: T): void {
  console.log(`Hello, ${name}`);
}
// Simpler:
function greet(name: string): void {
  console.log(`Hello, ${name}`);
}

// Good: T relates the input to the output
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}
// T connects the array element type to the return type

// Good: T and K relate to each other
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}
// T, K, and the return type are all connected
```

## Common Mistakes

### Mistake 1: Unnecessary Type Parameters

```typescript
// Bad: T serves no purpose
function logValue<T>(value: T): void {
  console.log(value);
}

// Good: just use the concrete type
function logValue(value: unknown): void {
  console.log(value);
}

// Bad: T is used once and does not relate values
function parse<T>(json: string): T {
  return JSON.parse(json); // This is actually unsafe
}

// Better: use a type guard or schema validation
function parse(json: string): unknown {
  return JSON.parse(json);
}
```

### Mistake 2: Over-Constraining

```typescript
// Bad: the constraint is the exact type you want
function processUser<T extends User>(user: T): string {
  return user.name;
}

// Good: just accept User directly
function processUser(user: User): string {
  return user.name;
}

// Exception: if you need to preserve the SPECIFIC subtype
function clone<T extends User>(user: T): T {
  return { ...user };
  // Returns the same subtype, not just User
}
```

### Mistake 3: Not Using Constraints When Needed

```typescript
// Bad: no constraint, so T could be anything
function getLength<T>(value: T): number {
  // return value.length; // Error: Property 'length' does not exist on type 'T'
  return 0;
}

// Good: constrain T to types with length
function getLength<T extends { length: number }>(value: T): number {
  return value.length;
}
```

### Mistake 4: Assuming `T extends X` Means `T` IS `X`

```typescript
function process<T extends { name: string }>(obj: T) {
  // T could have many more properties than just name
  // Do not assume T is exactly { name: string }

  // obj.age; // Error: age might not exist
  obj.name;   // OK: guaranteed by constraint
}

// T might be { name: string; age: number; email: string }
// The constraint is a minimum, not an exact match
```

### Mistake 5: Generic Arrow Functions in TSX Files

```typescript
// In .tsx files, this looks like a JSX element
// const identity = <T>(value: T): T => value; // Syntax error in .tsx

// Fix 1: trailing comma
const identity = <T,>(value: T): T => value;

// Fix 2: extends constraint
const identity = <T extends unknown>(value: T): T => value;
```

### Mistake 6: Expecting Partial Type Argument Inference

```typescript
function create<T, K extends keyof T>(type: T, key: K): T[K] {
  return type[key];
}

// You cannot specify T and let K be inferred
// create<User>("name"); // Error: Expected 2 type arguments

// You must provide all or none
create<User, "name">({} as User, "name"); // all explicit
create({ name: "Alice", age: 30 }, "name"); // all inferred
```

## Best Practices

### 1. Let TypeScript Infer When Possible

```typescript
// Avoid specifying type arguments when inference works
const nums = [1, 2, 3].map((n) => n * 2);    // inference is sufficient
const first = [1, 2, 3].find((n) => n > 1);   // no need for <number>
```

### 2. Use Meaningful Type Parameter Names

```typescript
// For single-parameter generics, T is conventional
function identity<T>(value: T): T { return value; }

// For multiple parameters, use descriptive names
function transform<Input, Output>(value: Input, fn: (v: Input) => Output): Output {
  return fn(value);
}

// Common conventions:
// T, U, V for positional parameters
// K for keys, V for values
// E for element types
// R for return types
```

### 3. Constrain Generics to the Minimum Required Shape

```typescript
// Good: only requires what the function actually uses
function getName<T extends { name: string }>(obj: T): string {
  return obj.name;
}

// Bad: over-constrained to a specific type
function getName<T extends User>(obj: T): string {
  return obj.name;
}
```

### 4. Return the Generic Type, Not a Widened Version

```typescript
// Bad: returns string, losing the generic connection
function firstChar<T extends string>(value: T): string {
  return value[0];
}

// Good: preserves the input type relationship
function wrap<T>(value: T): { value: T } {
  return { value };
}

const wrapped = wrap("hello" as const);
// { value: "hello" } - literal type preserved
```

### 5. Use Generic Constraints Instead of Overloads When Possible

```typescript
// Overloads: verbose and repetitive
function first(arr: string[]): string | undefined;
function first(arr: number[]): number | undefined;
function first(arr: any[]): any {
  return arr[0];
}

// Generic: one declaration handles all types
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}
```

## Real-World Use Cases

### Type-Safe Event Emitter

```typescript
type EventMap = Record<string, any>;

class TypedEmitter<Events extends EventMap> {
  private handlers: {
    [K in keyof Events]?: ((payload: Events[K]) => void)[];
  } = {};

  on<K extends keyof Events>(event: K, handler: (payload: Events[K]) => void): void {
    if (!this.handlers[event]) {
      this.handlers[event] = [];
    }
    this.handlers[event]!.push(handler);
  }

  off<K extends keyof Events>(event: K, handler: (payload: Events[K]) => void): void {
    const list = this.handlers[event];
    if (list) {
      this.handlers[event] = list.filter((h) => h !== handler);
    }
  }

  emit<K extends keyof Events>(event: K, payload: Events[K]): void {
    this.handlers[event]?.forEach((handler) => handler(payload));
  }
}

// Define your application events
interface AppEvents {
  login: { userId: string; timestamp: number };
  logout: { userId: string };
  error: { message: string; code: number };
}

const emitter = new TypedEmitter<AppEvents>();

emitter.on("login", (payload) => {
  // payload is { userId: string; timestamp: number }
  console.log(`User ${payload.userId} logged in`);
});

emitter.emit("login", { userId: "abc", timestamp: Date.now() });
// emitter.emit("login", { userId: "abc" }); // Error: missing timestamp
// emitter.emit("unknown", {}); // Error: "unknown" is not in AppEvents
```

### Generic Repository Pattern

```typescript
interface Entity {
  id: string;
  createdAt: Date;
  updatedAt: Date;
}

interface Repository<T extends Entity> {
  findById(id: string): Promise<T | null>;
  findMany(filter: Partial<Omit<T, keyof Entity>>): Promise<T[]>;
  create(data: Omit<T, keyof Entity>): Promise<T>;
  update(id: string, data: Partial<Omit<T, keyof Entity>>): Promise<T>;
  delete(id: string): Promise<boolean>;
}

// Entity definitions
interface User extends Entity {
  name: string;
  email: string;
  role: "admin" | "user";
}

interface Post extends Entity {
  title: string;
  content: string;
  authorId: string;
  published: boolean;
}

// Generic implementation
class InMemoryRepository<T extends Entity> implements Repository<T> {
  private items = new Map<string, T>();

  async findById(id: string): Promise<T | null> {
    return this.items.get(id) ?? null;
  }

  async findMany(filter: Partial<Omit<T, keyof Entity>>): Promise<T[]> {
    return Array.from(this.items.values()).filter((item) =>
      Object.entries(filter).every(
        ([key, value]) => (item as any)[key] === value,
      ),
    );
  }

  async create(data: Omit<T, keyof Entity>): Promise<T> {
    const entity = {
      ...data,
      id: crypto.randomUUID(),
      createdAt: new Date(),
      updatedAt: new Date(),
    } as T;
    this.items.set(entity.id, entity);
    return entity;
  }

  async update(id: string, data: Partial<Omit<T, keyof Entity>>): Promise<T> {
    const existing = this.items.get(id);
    if (!existing) throw new Error(`Entity ${id} not found`);
    const updated = { ...existing, ...data, updatedAt: new Date() };
    this.items.set(id, updated);
    return updated;
  }

  async delete(id: string): Promise<boolean> {
    return this.items.delete(id);
  }
}

// Usage: type-safe for each entity
const userRepo = new InMemoryRepository<User>();
const postRepo = new InMemoryRepository<Post>();

// TypeScript knows exactly which fields are required
await userRepo.create({ name: "Alice", email: "alice@example.com", role: "admin" });
await postRepo.create({ title: "Hello", content: "World", authorId: "1", published: false });

// Type-safe filtering
const admins = await userRepo.findMany({ role: "admin" });
const published = await postRepo.findMany({ published: true });
```

### Generic Middleware Pipeline

```typescript
type Middleware<Context> = (
  ctx: Context,
  next: () => Promise<void>,
) => Promise<void>;

class Pipeline<Context> {
  private middlewares: Middleware<Context>[] = [];

  use(middleware: Middleware<Context>): this {
    this.middlewares.push(middleware);
    return this;
  }

  async execute(ctx: Context): Promise<void> {
    let index = 0;

    const next = async (): Promise<void> => {
      if (index < this.middlewares.length) {
        const middleware = this.middlewares[index++];
        await middleware(ctx, next);
      }
    };

    await next();
  }
}

// Application-specific context
interface RequestContext {
  path: string;
  method: string;
  headers: Record<string, string>;
  body?: unknown;
  user?: { id: string; role: string };
  response?: { status: number; data: unknown };
}

const pipeline = new Pipeline<RequestContext>();

// Middleware functions are fully typed
pipeline
  .use(async (ctx, next) => {
    console.log(`${ctx.method} ${ctx.path}`);
    const start = Date.now();
    await next();
    console.log(`Completed in ${Date.now() - start}ms`);
  })
  .use(async (ctx, next) => {
    const token = ctx.headers["authorization"];
    if (token) {
      ctx.user = { id: "user-1", role: "admin" };
    }
    await next();
  })
  .use(async (ctx, next) => {
    ctx.response = { status: 200, data: { message: "OK" } };
    await next();
  });

await pipeline.execute({
  path: "/api/users",
  method: "GET",
  headers: { authorization: "Bearer token" },
});
```

---

> **Key Takeaway:** Generics parameterize types so you can write code that works across many types while maintaining the relationship between inputs and outputs. Apply the golden rule: every type parameter should relate at least two values. Use constraints to express minimum requirements, and let TypeScript infer type arguments whenever possible.
