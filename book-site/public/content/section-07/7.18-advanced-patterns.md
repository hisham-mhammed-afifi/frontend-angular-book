# 7.18 Advanced Patterns

> Advanced TypeScript patterns go beyond basic typing to encode business rules, prevent impossible states, and make invalid code unrepresentable at compile time.

---

## Branded (Nominal) Types

TypeScript uses **structural typing**: two types with the same shape are interchangeable. This is usually fine, but sometimes structurally identical types represent entirely different things. A `UserId` and a `ProductId` might both be strings, but passing one where the other is expected is a bug.

Branded types simulate nominal typing by intersecting a base type with a phantom property that only exists at the type level.

### Basic Branding

```typescript
// The brand is a property that never exists at runtime.
// Using a unique symbol prevents accidental collisions.
declare const __brand: unique symbol;

type Brand<T, B extends string> = T & { readonly [__brand]: B };

type UserId = Brand<string, "UserId">;
type ProductId = Brand<string, "ProductId">;
type OrderId = Brand<string, "OrderId">;
```

Branded types cannot be created by accident. You need explicit constructor functions:

```typescript
function createUserId(id: string): UserId {
  // Validate if needed, then brand
  return id as UserId;
}

function createProductId(id: string): ProductId {
  return id as ProductId;
}

function getUser(id: UserId): void { /* ... */ }
function getProduct(id: ProductId): void { /* ... */ }

const userId = createUserId("user_123");
const productId = createProductId("prod_456");

getUser(userId);       // OK
getProduct(productId); // OK

getUser(productId);    // Error: ProductId is not assignable to UserId
getUser("raw_string"); // Error: string is not assignable to UserId
```

### Validated Branded Types

Branding becomes even more powerful when the constructor function enforces a validation invariant. Once the value is branded, you know it has passed validation:

```typescript
type Email = Brand<string, "Email">;
type PositiveNumber = Brand<number, "PositiveNumber">;
type NonEmptyString = Brand<string, "NonEmptyString">;

function validateEmail(input: string): Email {
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(input)) {
    throw new Error(`Invalid email: ${input}`);
  }
  return input as Email;
}

function toPositive(n: number): PositiveNumber {
  if (n <= 0) {
    throw new Error(`Expected positive number, got ${n}`);
  }
  return n as PositiveNumber;
}

function toNonEmpty(s: string): NonEmptyString {
  if (s.trim().length === 0) {
    throw new Error("String must not be empty");
  }
  return s as NonEmptyString;
}

// Downstream code can trust the brand
function sendEmail(to: Email, subject: NonEmptyString): void {
  // No need to re-validate. The types guarantee validity.
}
```

### Branded Types Still Act as Their Base Type

A `UserId` is still a `string`. You can pass it to any function that accepts `string`:

```typescript
const userId: UserId = createUserId("user_123");

// This works: UserId is assignable to string
console.log(userId.toUpperCase()); // "USER_123"
console.log(userId.length);       // 8

// But string is NOT assignable to UserId
const raw: string = "user_456";
// getUser(raw); // Error
```

The brand only restricts assignment **into** the branded type, not **out of** it.

## Exhaustive Discrimination

Discriminated unions combined with exhaustive checking ensure that every possible state is handled. This section goes deeper than the basics covered in 7.8.

### The `never` Exhaustiveness Check

```typescript
type Shape =
  | { kind: "circle"; radius: number }
  | { kind: "square"; side: number }
  | { kind: "triangle"; base: number; height: number };

function area(shape: Shape): number {
  switch (shape.kind) {
    case "circle":
      return Math.PI * shape.radius ** 2;
    case "square":
      return shape.side ** 2;
    case "triangle":
      return 0.5 * shape.base * shape.height;
    default:
      // If all cases are covered, shape is `never` here.
      // If someone adds a new variant and forgets to handle it,
      // this line will produce a compile error.
      const _exhaustive: never = shape;
      return _exhaustive;
  }
}
```

### Extracting an Exhaustive Helper

```typescript
function assertNever(value: never, message?: string): never {
  throw new Error(message ?? `Unexpected value: ${JSON.stringify(value)}`);
}

function area(shape: Shape): number {
  switch (shape.kind) {
    case "circle":
      return Math.PI * shape.radius ** 2;
    case "square":
      return shape.side ** 2;
    case "triangle":
      return 0.5 * shape.base * shape.height;
    default:
      return assertNever(shape, `Unknown shape: ${(shape as Shape).kind}`);
  }
}
```

Now if a new variant `"rectangle"` is added to `Shape`, every `switch` that uses `assertNever` will produce a compile error until the new case is handled.

### Exhaustive Object Mapping

An alternative to `switch` that is often cleaner:

```typescript
type Shape =
  | { kind: "circle"; radius: number }
  | { kind: "square"; side: number }
  | { kind: "triangle"; base: number; height: number };

// Record<Shape["kind"], ...> forces every variant to be present
const areaCalculators: Record<Shape["kind"], (shape: any) => number> = {
  circle: (s: Extract<Shape, { kind: "circle" }>) => Math.PI * s.radius ** 2,
  square: (s: Extract<Shape, { kind: "square" }>) => s.side ** 2,
  triangle: (s: Extract<Shape, { kind: "triangle" }>) => 0.5 * s.base * s.height,
};

function area(shape: Shape): number {
  return areaCalculators[shape.kind](shape);
}
```

Adding a new variant to `Shape["kind"]` immediately produces an error on the `areaCalculators` object because the new key is missing.

## Type-Safe State Machines

Discriminated unions can model finite state machines where the type system enforces valid transitions.

```typescript
// States
type ConnectionState =
  | { status: "disconnected" }
  | { status: "connecting"; attempt: number }
  | { status: "connected"; socket: WebSocket; connectedAt: number }
  | { status: "reconnecting"; attempt: number; lastError: string };

// Transition functions enforce valid state changes
function connect(state: { status: "disconnected" }): ConnectionState {
  return { status: "connecting", attempt: 1 };
}

function onConnected(
  state: { status: "connecting" },
  socket: WebSocket
): ConnectionState {
  return { status: "connected", socket, connectedAt: Date.now() };
}

function disconnect(
  state: { status: "connected" }
): ConnectionState {
  state.socket.close();
  return { status: "disconnected" };
}

function onError(
  state: { status: "connecting" | "connected" },
  error: string
): ConnectionState {
  return {
    status: "reconnecting",
    attempt: "attempt" in state ? state.attempt + 1 : 1,
    lastError: error,
  };
}

// The type system prevents invalid transitions:
// disconnect({ status: "disconnected" }); // Error: status "disconnected" not assignable
// onConnected({ status: "disconnected" }, ws); // Error
```

### Phantom State Tracking with Generics

For builder-style APIs, you can use generic type parameters to track which steps have been completed:

```typescript
type Step = "name" | "email" | "password";

type FormBuilder<Completed extends string = never> = {
  setName(name: string): FormBuilder<Completed | "name">;
  setEmail(email: string): FormBuilder<Completed | "email">;
  setPassword(password: string): FormBuilder<Completed | "password">;
  build(
    // This method only exists when all steps are completed
    ...args: [Completed] extends [Step] ? [] : [never]
  ): User;
};

// Usage (conceptual):
// builder.setName("Alice").setEmail("a@b.com").build();
// Error: "password" step not completed, so build() requires `never` argument
```

## Function Overloads

Function overloads provide multiple call signatures with precise return types based on input types.

### Basic Overloads

```typescript
function createElement(tag: "a"): HTMLAnchorElement;
function createElement(tag: "canvas"): HTMLCanvasElement;
function createElement(tag: "div"): HTMLDivElement;
function createElement(tag: "input"): HTMLInputElement;
function createElement(tag: string): HTMLElement;
function createElement(tag: string): HTMLElement {
  return document.createElement(tag);
}

const anchor = createElement("a");     // HTMLAnchorElement
const canvas = createElement("canvas"); // HTMLCanvasElement
const div = createElement("div");       // HTMLDivElement
const other = createElement("span");    // HTMLElement
```

### Overloads vs Conditional Return Types

You can sometimes replace overloads with conditional types for simpler signatures:

```typescript
// Overload approach (many signatures)
function parse(input: string, asNumber: true): number;
function parse(input: string, asNumber: false): string;
function parse(input: string, asNumber: boolean): string | number {
  return asNumber ? Number(input) : input;
}

// Conditional return type approach (single signature)
function parse2<T extends boolean>(
  input: string,
  asNumber: T
): T extends true ? number : string {
  return (asNumber ? Number(input) : input) as T extends true ? number : string;
}
```

Both provide the same developer experience. Overloads are clearer for distinct call shapes; conditional return types are better when the logic follows a simple branching pattern.

### When to Prefer Overloads

- When call signatures are fundamentally different (different parameter counts or types)
- When the implementation would be too complex as a single generic function
- When you want different JSDoc documentation per signature

### When to Prefer Generics/Conditional Types

- When the signatures follow a predictable pattern
- When you have many variants (overloads would be verbose)
- When you want a single source of truth

## The Newtype Pattern

The newtype pattern wraps a value in a container type that has a single field. Unlike branded types, newtypes can carry runtime information:

```typescript
// A wrapper that exists at runtime
class Meters {
  readonly value: number;
  private readonly _tag = "Meters" as const;

  constructor(value: number) {
    this.value = value;
  }
}

class Kilometers {
  readonly value: number;
  private readonly _tag = "Kilometers" as const;

  constructor(value: number) {
    this.value = value;
  }
}

function addDistances(a: Meters, b: Meters): Meters {
  return new Meters(a.value + b.value);
}

const a = new Meters(100);
const b = new Meters(200);
const c = new Kilometers(5);

addDistances(a, b); // OK
addDistances(a, c); // Error: Kilometers is not assignable to Meters
```

The private `_tag` field makes the types structurally distinct. Without it, TypeScript would treat `Meters` and `Kilometers` as interchangeable (since they have the same public shape).

**Tradeoff vs branded types:** Newtypes have runtime overhead (object allocation) but provide actual runtime identity. Branded types are zero-cost at runtime but rely on the type system alone.

## Opaque Types via Module Boundaries

You can create opaque types by exporting only the type and constructor, but not the branding mechanism:

```typescript
// money.ts
declare const __brand: unique symbol;
type Brand<T, B> = T & { readonly [__brand]: B };

// Export the type but not the constructor logic
export type USD = Brand<number, "USD">;
export type EUR = Brand<number, "EUR">;

// Only these functions can create branded values
export function usd(amount: number): USD {
  if (!Number.isFinite(amount)) throw new Error("Invalid amount");
  return Math.round(amount * 100) / 100 as unknown as USD;
}

export function eur(amount: number): EUR {
  if (!Number.isFinite(amount)) throw new Error("Invalid amount");
  return Math.round(amount * 100) / 100 as unknown as EUR;
}

// Type-safe arithmetic
export function addUSD(a: USD, b: USD): USD {
  return ((a as number) + (b as number)) as unknown as USD;
}
```

```typescript
// consumer.ts
import { USD, EUR, usd, eur, addUSD } from "./money";

const price = usd(29.99);
const tax = usd(2.40);
const total = addUSD(price, tax); // OK

const euroPrice = eur(25.00);
// addUSD(price, euroPrice); // Error: EUR not assignable to USD

// Consumers cannot forge branded values:
// const fake = 100 as USD; // Only works if they import the brand symbol (they can't)
```

## Type-Safe Event Systems

Combining template literal types, generics, and mapped types creates event systems where event names and payloads are linked at the type level:

```typescript
interface EventMap {
  "user:login": { userId: string; timestamp: number };
  "user:logout": { userId: string };
  "order:created": { orderId: string; total: number };
  "order:shipped": { orderId: string; trackingNumber: string };
}

class TypedEventEmitter<Events extends Record<string, unknown>> {
  private listeners = new Map<string, Set<Function>>();

  on<E extends keyof Events & string>(
    event: E,
    handler: (payload: Events[E]) => void
  ): () => void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    const handlers = this.listeners.get(event)!;
    handlers.add(handler);

    // Return unsubscribe function
    return () => handlers.delete(handler);
  }

  emit<E extends keyof Events & string>(
    event: E,
    payload: Events[E]
  ): void {
    this.listeners.get(event)?.forEach(fn => fn(payload));
  }
}

const emitter = new TypedEventEmitter<EventMap>();

// Full type safety on both event name and payload
emitter.on("user:login", (payload) => {
  console.log(payload.userId);    // string
  console.log(payload.timestamp); // number
});

emitter.emit("order:created", {
  orderId: "ord_123",
  total: 59.99,
}); // OK

emitter.emit("order:created", {
  orderId: "ord_123",
}); // Error: missing 'total'

emitter.emit("unknown:event", {}); // Error: not in EventMap
```

## Conditional Object Properties

Sometimes an object should have certain properties only when another property has a specific value. Discriminated unions are the primary tool, but you can also use conditional types for more dynamic patterns:

```typescript
// Discriminated union approach (preferred for known variants)
type Notification =
  | {
      type: "email";
      emailAddress: string;
      subject: string;
    }
  | {
      type: "sms";
      phoneNumber: string;
    }
  | {
      type: "push";
      deviceToken: string;
      badge?: number;
    };

function send(notification: Notification): void {
  switch (notification.type) {
    case "email":
      // notification.emailAddress and notification.subject are available
      break;
    case "sms":
      // notification.phoneNumber is available
      break;
    case "push":
      // notification.deviceToken and notification.badge are available
      break;
  }
}

// Impossible to create a malformed notification:
// send({ type: "email", phoneNumber: "123" }); // Error: phoneNumber does not exist on email variant
```

### Mutually Exclusive Properties

```typescript
// Only one of `width` or `flex` should be set, never both
type ColumnWidth =
  | { width: number; flex?: never }
  | { width?: never; flex: number };

interface Column {
  name: string;
  sizing: ColumnWidth;
}

const col1: Column = { name: "ID", sizing: { width: 100 } };     // OK
const col2: Column = { name: "Name", sizing: { flex: 2 } };       // OK
// const col3: Column = { name: "Bad", sizing: { width: 100, flex: 2 } }; // Error
```

The `?: never` pattern makes a property impossible to set. If you try to provide it, the value would need to be `never`, which is impossible.

## Type-Safe Dependency Injection Container

Generics and mapped types can create a compile-time-checked DI container:

```typescript
class Container<Registry extends Record<string, unknown> = {}> {
  private factories = new Map<string, () => unknown>();

  register<K extends string, V>(
    key: K,
    factory: () => V
  ): Container<Registry & Record<K, V>> {
    this.factories.set(key, factory);
    return this as unknown as Container<Registry & Record<K, V>>;
  }

  resolve<K extends keyof Registry & string>(key: K): Registry[K] {
    const factory = this.factories.get(key);
    if (!factory) {
      throw new Error(`No registration for: ${key}`);
    }
    return factory() as Registry[K];
  }
}

// Build the container with full type tracking
const container = new Container()
  .register("logger", () => ({ log: (msg: string) => console.log(msg) }))
  .register("config", () => ({ port: 3000, host: "localhost" }))
  .register("db", () => ({ query: (sql: string) => Promise.resolve([]) }));

// Fully typed resolution
const logger = container.resolve("logger");  // { log: (msg: string) => void }
const config = container.resolve("config");  // { port: number; host: string }
const db = container.resolve("db");          // { query: (sql: string) => Promise<never[]> }

// container.resolve("unknown"); // Error: "unknown" not in Registry
```

Each call to `register` returns a container with an expanded `Registry` type. By the time you call `resolve`, the compiler knows exactly which keys are available and what types they return.

## The `infer` Keyword in Advanced Patterns

The `infer` keyword enables extracting types from complex structures. Here are patterns beyond the basics:

### Extracting Function Parts

```typescript
type FirstParam<F> = F extends (first: infer P, ...args: any[]) => any ? P : never;
type LastParam<F> = F extends (...args: [...any[], infer L]) => any ? L : never;

type A = FirstParam<(a: string, b: number) => void>; // string
type B = LastParam<(a: string, b: number) => void>;  // number
```

### Extracting Promise Chains

```typescript
type DeepAwaited<T> =
  T extends Promise<infer U>
    ? DeepAwaited<U>
    : T;

type A = DeepAwaited<Promise<Promise<Promise<string>>>>; // string
```

### Extracting Array Element Types

```typescript
type ElementOf<T> = T extends readonly (infer E)[] ? E : never;

type A = ElementOf<string[]>;              // string
type B = ElementOf<readonly [1, "two", 3]>; // 1 | "two" | 3
```

### Composing `infer` with Template Literals

```typescript
type ParseQueryString<S extends string> =
  S extends `${infer Key}=${infer Value}&${infer Rest}`
    ? { [K in Key]: Value } & ParseQueryString<Rest>
    : S extends `${infer Key}=${infer Value}`
      ? { [K in Key]: Value }
      : {};

type Parsed = ParseQueryString<"name=alice&age=30&role=admin">;
// { name: "alice" } & { age: "30" } & { role: "admin" }
```

## Common Mistakes

### 1. Over-Engineering Types That Slow Down the Compiler

```typescript
// Bad Practice: deeply recursive type that computes at compile time
type DeepFlatten<T> = T extends readonly (infer U)[]
  ? DeepFlatten<U>
  : T;

// Fine for 3-4 levels, but passing a 50-level nested array type
// will freeze the editor and slow compilation to a crawl.
```

Advanced type patterns should serve productivity, not demonstrate cleverness. If a type takes more than a second to resolve in the editor, simplify it.

### 2. Branded Types Without Validation

```typescript
// Bad Practice: branding without checking the invariant
type PositiveNumber = number & { readonly __brand: "Positive" };

function positive(n: number): PositiveNumber {
  return n as PositiveNumber; // Caller can pass -1
}

// Good: validate before branding
function positive(n: number): PositiveNumber {
  if (n <= 0) throw new Error(`Expected positive, got ${n}`);
  return n as PositiveNumber;
}
```

A brand without validation is just a label. The value of branded types comes from enforcing an invariant at the boundary.

### 3. Using `any` in Overload Implementations

```typescript
// Bad Practice: using `any` loses all internal type safety
function process(input: string): string;
function process(input: number): number;
function process(input: any): any {
  // No type checking inside this body
  return input;
}

// Better: use a union or generic in the implementation
function process(input: string): string;
function process(input: number): number;
function process(input: string | number): string | number {
  if (typeof input === "string") return input.toUpperCase();
  return input * 2;
}
```

### 4. Forgetting That Discriminant Must Be a Literal Type

```typescript
// Bad Practice: discriminant is `string`, not a literal
interface Dog {
  type: string;  // This should be "dog"
  bark(): void;
}

interface Cat {
  type: string;  // This should be "cat"
  meow(): void;
}

type Animal = Dog | Cat;

function handle(animal: Animal) {
  if (animal.type === "dog") {
    // animal is still Animal, not Dog -- narrowing does not work
    // because `type` is `string`, not a literal
  }
}

// Good: use literal types
interface Dog { type: "dog"; bark(): void; }
interface Cat { type: "cat"; meow(): void; }
```

### 5. Mutually Exclusive Properties Without `never`

```typescript
// Bad Practice: optional properties allow both to be set
interface Layout {
  width?: number;
  flex?: number;
}

// This is valid but logically wrong:
const broken: Layout = { width: 100, flex: 2 };

// Good: use the never pattern
type Layout =
  | { width: number; flex?: never }
  | { width?: never; flex: number };
```

## Best Practices

### 1. Start Simple, Add Complexity When Needed

Do not reach for branded types, phantom types, or recursive conditional types on day one. Start with basic interfaces and unions. Introduce advanced patterns when you encounter a real bug or limitation that simpler types cannot prevent.

### 2. Keep Branded Type Constructors as Module Exports

Branded types are most effective when the `as` assertion is confined to a single function inside a module. Consumers should only interact with the branded type through the exported constructor:

```typescript
// Encapsulate the assertion
export function createUserId(id: string): UserId {
  if (!id.startsWith("user_")) throw new Error("Invalid user ID format");
  return id as UserId;
}
```

### 3. Use `assertNever` in Every Switch on a Discriminated Union

This is a low-cost, high-value pattern. Every time you add a new variant to a union, the compiler will point you to every place that needs updating:

```typescript
function assertNever(value: never, msg?: string): never {
  throw new Error(msg ?? `Unhandled value: ${JSON.stringify(value)}`);
}
```

### 4. Prefer Discriminated Unions Over Type Guards for State

Type guards work, but discriminated unions make impossible states unrepresentable:

```typescript
// Type guard approach: allows invalid combinations
interface State {
  loading: boolean;
  data: string | null;
  error: Error | null;
}
// loading: true, data: "hello", error: new Error("oops") -- all true at once?

// Discriminated union approach: each state is explicit
type State =
  | { status: "idle" }
  | { status: "loading" }
  | { status: "success"; data: string }
  | { status: "error"; error: Error };
```

### 5. Document the Invariant That a Brand Represents

```typescript
/**
 * A string that has been validated as a syntactically correct email address.
 * Created exclusively by `validateEmail()`.
 */
type Email = string & { readonly __brand: "Email" };
```

## Real-World Use Cases

### 1. Type-Safe Currency Handling

```typescript
declare const __brand: unique symbol;
type Brand<T, B> = T & { readonly [__brand]: B };

type USD = Brand<number, "USD">;
type EUR = Brand<number, "EUR">;
type GBP = Brand<number, "GBP">;

function usd(amount: number): USD {
  return Math.round(amount * 100) / 100 as unknown as USD;
}

function eur(amount: number): EUR {
  return Math.round(amount * 100) / 100 as unknown as EUR;
}

// Arithmetic operations preserve the brand
function addUSD(a: USD, b: USD): USD {
  return usd((a as unknown as number) + (b as unknown as number));
}

// Currency conversion must be explicit
interface ExchangeRate<From, To> {
  from: From;
  to: To;
  rate: number;
}

function convert(
  amount: USD,
  rate: ExchangeRate<"USD", "EUR">
): EUR {
  return eur((amount as unknown as number) * rate.rate);
}

const price = usd(49.99);
const shipping = usd(5.99);
const total = addUSD(price, shipping); // OK: USD + USD = USD

const euroRate: ExchangeRate<"USD", "EUR"> = { from: "USD", to: "EUR", rate: 0.92 };
const euroTotal = convert(total, euroRate); // OK: USD -> EUR

// addUSD(price, eur(10)); // Error: EUR not assignable to USD
```

### 2. Type-Safe API Response Handler

```typescript
type ApiResult<T> =
  | { status: "success"; data: T; statusCode: 200 | 201 }
  | { status: "error"; error: string; statusCode: 400 | 401 | 403 | 404 | 500 }
  | { status: "loading" };

function handleResult<T>(
  result: ApiResult<T>,
  handlers: {
    onSuccess: (data: T, code: 200 | 201) => void;
    onError: (error: string, code: 400 | 401 | 403 | 404 | 500) => void;
    onLoading: () => void;
  }
): void {
  switch (result.status) {
    case "success":
      handlers.onSuccess(result.data, result.statusCode);
      break;
    case "error":
      handlers.onError(result.error, result.statusCode);
      break;
    case "loading":
      handlers.onLoading();
      break;
    default:
      assertNever(result);
  }
}

// Usage
declare const result: ApiResult<{ id: string; name: string }>;

handleResult(result, {
  onSuccess(data, code) {
    // data is { id: string; name: string }
    // code is 200 | 201
    console.log(data.name);
  },
  onError(error, code) {
    if (code === 401) {
      redirectToLogin();
    }
  },
  onLoading() {
    showSpinner();
  },
});

function assertNever(value: never): never {
  throw new Error(`Unhandled: ${JSON.stringify(value)}`);
}
declare function redirectToLogin(): void;
declare function showSpinner(): void;
```

### 3. Type-Safe Configuration Validation Pipeline

```typescript
type Validator<T> = (value: unknown) => T;

function string(): Validator<string> {
  return (value) => {
    if (typeof value !== "string") throw new Error(`Expected string, got ${typeof value}`);
    return value;
  };
}

function number(): Validator<number> {
  return (value) => {
    if (typeof value !== "number") throw new Error(`Expected number, got ${typeof value}`);
    return value;
  };
}

function boolean(): Validator<boolean> {
  return (value) => {
    if (typeof value !== "boolean") throw new Error(`Expected boolean, got ${typeof value}`);
    return value;
  };
}

function object<T extends Record<string, Validator<unknown>>>(
  shape: T
): Validator<{ [K in keyof T]: T[K] extends Validator<infer V> ? V : never }> {
  return (value) => {
    if (typeof value !== "object" || value === null) {
      throw new Error("Expected object");
    }
    const result: Record<string, unknown> = {};
    for (const [key, validator] of Object.entries(shape)) {
      result[key] = validator((value as Record<string, unknown>)[key]);
    }
    return result as { [K in keyof T]: T[K] extends Validator<infer V> ? V : never };
  };
}

function optional<T>(validator: Validator<T>): Validator<T | undefined> {
  return (value) => {
    if (value === undefined) return undefined;
    return validator(value);
  };
}

// Define a schema
const configSchema = object({
  host: string(),
  port: number(),
  debug: boolean(),
  logLevel: optional(string()),
});

// Type is automatically inferred from the schema:
// {
//   host: string;
//   port: number;
//   debug: boolean;
//   logLevel: string | undefined;
// }

function loadConfig(raw: unknown) {
  const config = configSchema(raw);
  // config is fully typed with no manual type annotations needed
  console.log(config.host);     // string
  console.log(config.port);     // number
  console.log(config.logLevel); // string | undefined
  return config;
}
```

This pattern powers libraries like Zod and Valibot. The schema definition **is** the type definition: you write the validation logic once and the TypeScript type is inferred from it.

---

> **Key Takeaway:** Advanced TypeScript patterns encode domain rules into the type system so that invalid states, incorrect assignments, and unhandled cases become compile-time errors rather than runtime bugs.
