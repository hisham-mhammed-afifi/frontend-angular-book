# 7.4 Enums

> Enums define a set of named constants. TypeScript offers numeric enums, string enums, and const enums, but modern TypeScript increasingly favors union types and `as const` objects as lighter-weight alternatives.

---

## Numeric Enums

Numeric enums auto-increment from 0 (or from a custom starting value). Each member maps to a number.

```typescript
enum Direction {
  Up,     // 0
  Down,   // 1
  Left,   // 2
  Right,  // 3
}

const move: Direction = Direction.Up;
console.log(move);              // 0
console.log(Direction.Right);   // 3
```

### Custom Starting Values

```typescript
enum StatusCode {
  OK = 200,
  Created = 201,
  BadRequest = 400,
  Unauthorized = 401,
  NotFound = 404,
  InternalServerError = 500,
}

console.log(StatusCode.OK);        // 200
console.log(StatusCode.NotFound);  // 404
```

Auto-increment continues from the last explicit value:

```typescript
enum Priority {
  Low = 1,
  Medium,    // 2
  High,      // 3
  Critical,  // 4
}
```

### Reverse Mapping (Numeric Only)

Numeric enums create a bidirectional mapping: name to value AND value to name.

```typescript
enum Color {
  Red,    // 0
  Green,  // 1
  Blue,   // 2
}

console.log(Color.Red);   // 0
console.log(Color[0]);    // "Red"   <-- reverse mapping
console.log(Color[1]);    // "Green"

// The compiled JavaScript looks like:
// var Color;
// (function (Color) {
//   Color[Color["Red"] = 0] = "Red";
//   Color[Color["Green"] = 1] = "Green";
//   Color[Color["Blue"] = 2] = "Blue";
// })(Color || (Color = {}));
```

This bidirectional mapping is unique to numeric enums. String enums do not have it.

### The Numeric Enum Type Safety Problem

Numeric enums have a serious type safety flaw: TypeScript allows ANY number to be assigned to a numeric enum type.

```typescript
enum Status {
  Active,   // 0
  Inactive, // 1
}

let status: Status = Status.Active; // OK
status = 99; // No error! 99 is not a valid Status, but TypeScript allows it.

function setStatus(s: Status): void {
  console.log(s);
}

setStatus(42); // No error! This defeats the purpose of an enum.
```

This is one of the strongest arguments against numeric enums. String enums do not have this problem.

---

## String Enums

String enums require an explicit string value for every member. No auto-increment, no reverse mapping.

```typescript
enum LogLevel {
  Debug = "DEBUG",
  Info = "INFO",
  Warn = "WARN",
  Error = "ERROR",
}

console.log(LogLevel.Debug); // "DEBUG"
console.log(LogLevel.Error); // "ERROR"

// No reverse mapping:
// console.log(LogLevel["DEBUG"]); // Error: Element implicitly has an 'any' type
```

### Why String Enums Are Safer

```typescript
enum Theme {
  Light = "LIGHT",
  Dark = "DARK",
}

let theme: Theme = Theme.Light; // OK
// theme = "LIGHT";             // Error: Type '"LIGHT"' is not assignable to type 'Theme'
// theme = 42;                  // Error: Type '42' is not assignable to type 'Theme'

// Only actual enum members are accepted. This is proper type safety.
```

String enums are strict: only enum member references are assignable, not raw strings.

### Compiled Output

```typescript
enum Direction {
  Up = "UP",
  Down = "DOWN",
  Left = "LEFT",
  Right = "RIGHT",
}

// Compiles to:
// var Direction;
// (function (Direction) {
//   Direction["Up"] = "UP";
//   Direction["Down"] = "DOWN";
//   Direction["Left"] = "LEFT";
//   Direction["Right"] = "RIGHT";
// })(Direction || (Direction = {}));
```

This generates a real JavaScript object that exists at runtime. This is both an advantage (you can iterate over it) and a cost (it adds to bundle size).

---

## Heterogeneous Enums

TypeScript allows mixing number and string values. This is almost never useful and should be avoided.

```typescript
// BAD: heterogeneous enum
enum Mixed {
  No = 0,
  Yes = "YES",
}
// Confusing behavior, inconsistent types. Don't do this.
```

---

## Const Enums

`const enum` is inlined at compile time. No JavaScript object is generated. Each reference is replaced with its literal value.

```typescript
const enum HttpMethod {
  GET = "GET",
  POST = "POST",
  PUT = "PUT",
  DELETE = "DELETE",
}

const method = HttpMethod.GET;

// Compiles to:
// const method = "GET";
// No HttpMethod object exists at runtime.
```

### Benefits

- Zero runtime footprint. No extra JavaScript object.
- Smaller bundle size.
- Slightly faster (no object lookup).

### Drawbacks and Restrictions

**No runtime access:** You cannot iterate over members, use `Object.keys()`, or pass the enum as a value.

```typescript
const enum Color {
  Red = "RED",
  Green = "GREEN",
  Blue = "BLUE",
}

// Error: 'const' enums can only be used in property or index access expressions,
// calls to 'Object.keys', or 'Object.values' on an enum object.
// Object.keys(Color); // Not possible

// Cannot pass it as a function argument
// function printEnum(e: typeof Color) {} // Not meaningful
```

**Incompatible with `isolatedModules`:** Most modern bundlers (esbuild, swc, Vite) compile each file independently. They cannot inline values from a different file because they do not see the full program. Using `const enum` with `isolatedModules: true` produces an error for ambient const enums (from `.d.ts` files).

**Library publishing danger:** If you publish a library with `const enum`, consumers' compiled code will contain the inlined values from the version they compiled against. If you later change the enum values in a new library version, their code still has the old values baked in.

**Recommendation:** Avoid `const enum` in most modern projects. If you use a bundler with `isolatedModules: true` (which you should), plain enums or `as const` objects are safer alternatives.

## Enums as Types

Enum members can be used as types.

```typescript
enum Role {
  Admin = "ADMIN",
  Editor = "EDITOR",
  Viewer = "VIEWER",
}

// The enum itself is a type (union of all members)
function checkAccess(role: Role): boolean {
  return role === Role.Admin;
}

// Individual members are types too
function adminOnly(role: Role.Admin): void {
  console.log("Admin access granted");
}

adminOnly(Role.Admin); // OK
// adminOnly(Role.Editor); // Error
```

## Iterating Over Enums

Since enums generate runtime objects, you can iterate over them. But numeric enums include reverse mappings, making iteration tricky.

### String Enum Iteration

```typescript
enum Fruit {
  Apple = "APPLE",
  Banana = "BANANA",
  Cherry = "CHERRY",
}

// String enums: straightforward
const fruits = Object.values(Fruit);
console.log(fruits); // ["APPLE", "BANANA", "CHERRY"]

const keys = Object.keys(Fruit);
console.log(keys); // ["Apple", "Banana", "Cherry"]

const entries = Object.entries(Fruit);
console.log(entries);
// [["Apple", "APPLE"], ["Banana", "BANANA"], ["Cherry", "CHERRY"]]
```

### Numeric Enum Iteration (Gotcha)

```typescript
enum Color {
  Red,    // 0
  Green,  // 1
  Blue,   // 2
}

// Includes reverse mappings!
console.log(Object.keys(Color));
// ["0", "1", "2", "Red", "Green", "Blue"]

console.log(Object.values(Color));
// ["Red", "Green", "Blue", 0, 1, 2]

// Filter to get only the names:
const names = Object.keys(Color).filter((k) => isNaN(Number(k)));
// ["Red", "Green", "Blue"]

// Filter to get only the values:
const values = Object.values(Color).filter((v) => typeof v === "number");
// [0, 1, 2]
```

This is another reason string enums are preferred over numeric enums.

## Alternatives to Enums

Modern TypeScript offers lighter-weight patterns that achieve the same goals without the quirks of enums.

### Union Types (Type-Only, Zero Runtime)

```typescript
type Direction = "up" | "down" | "left" | "right";

function move(direction: Direction): void {
  console.log(`Moving ${direction}`);
}

move("up");    // OK
// move("diagonal"); // Error: not assignable to type 'Direction'
```

**Pros:** No generated JavaScript. Pure type-level construct. Works with `isolatedModules`. Supports exhaustive checks. No import needed at runtime.

**Cons:** No runtime object. Cannot iterate over members. No reverse lookup. Must repeat string values in code.

### as const Object (Runtime + Types)

```typescript
const Direction = {
  Up: "UP",
  Down: "DOWN",
  Left: "LEFT",
  Right: "RIGHT",
} as const;

type Direction = (typeof Direction)[keyof typeof Direction];
// "UP" | "DOWN" | "LEFT" | "RIGHT"

function move(direction: Direction): void {
  console.log(`Moving ${direction}`);
}

move(Direction.Up);   // OK: "UP"
move("DOWN");         // OK: "DOWN" is a valid Direction
// move("diagonal");  // Error

// Runtime iteration works:
console.log(Object.values(Direction)); // ["UP", "DOWN", "LEFT", "RIGHT"]
console.log(Object.keys(Direction));   // ["Up", "Down", "Left", "Right"]
```

**Pros:** Real runtime object (can iterate, pass around). No special TypeScript syntax (just plain JavaScript + `as const`). Works with `isolatedModules`. Tree-shakeable. No reverse mapping confusion.

**Cons:** Slightly more boilerplate to extract the type. Accepts raw string values (not just `Direction.Up`), which some people see as a positive.

### Comparison Table

| Feature | Numeric Enum | String Enum | Const Enum | Union Type | as const Object |
|---------|-------------|-------------|------------|------------|-----------------|
| Runtime object | Yes | Yes | No (inlined) | No | Yes |
| Type safety | Weak (any number) | Strong | Strong | Strong | Strong |
| Bundle size | Adds code | Adds code | Zero | Zero | Object only |
| isolatedModules | OK | OK | Problematic | OK | OK |
| Iteration | Tricky (reverse map) | Clean | Not possible | Not possible | Clean |
| Exhaustive switch | Yes | Yes | Yes | Yes | Yes |
| Library-safe | Yes | Yes | No | Yes | Yes |
| Tree-shakeable | No | No | N/A | N/A | Yes |

## Declaration Merging with Enums

Enums support declaration merging. Multiple declarations with the same name are combined. This is rarely useful and can be confusing.

```typescript
enum Animal {
  Dog = "DOG",
  Cat = "CAT",
}

enum Animal {
  Bird = "BIRD",
}

// Merged: Animal has Dog, Cat, and Bird
console.log(Animal.Dog);  // "DOG"
console.log(Animal.Bird); // "BIRD"
```

This does not work with `const enum` or with `as const` objects (which is another argument for the object pattern).

## Computed and Constant Members

Enum members can be constant (evaluated at compile time) or computed (evaluated at runtime).

```typescript
enum FileAccess {
  // Constant members (evaluated at compile time)
  None = 0,
  Read = 1 << 0,        // 1
  Write = 1 << 1,       // 2
  ReadWrite = Read | Write, // 3

  // Computed member (evaluated at runtime)
  UserSetting = "secret".length, // 6
}
```

Constant members can use:
- Numeric literals
- References to other constant enum members
- `+`, `-`, `~` unary operators
- `+`, `-`, `*`, `/`, `%`, `<<`, `>>`, `>>>`, `&`, `|`, `^` binary operators

### Bit Flags Pattern

Numeric enums are genuinely useful for bit flags.

```typescript
enum Permission {
  None = 0,
  Read = 1 << 0,        // 1  (binary: 001)
  Write = 1 << 1,       // 2  (binary: 010)
  Execute = 1 << 2,     // 4  (binary: 100)
  ReadWrite = Read | Write,          // 3  (binary: 011)
  All = Read | Write | Execute,     // 7  (binary: 111)
}

function hasPermission(userPerms: number, required: Permission): boolean {
  return (userPerms & required) === required;
}

const userPerms = Permission.Read | Permission.Write; // 3

console.log(hasPermission(userPerms, Permission.Read));     // true
console.log(hasPermission(userPerms, Permission.Write));    // true
console.log(hasPermission(userPerms, Permission.Execute));  // false
console.log(hasPermission(userPerms, Permission.All));      // false
```

This is one of the few cases where numeric enums are the best tool.

## Common Mistakes

- **Using numeric enums when string enums or unions would be safer.** Numeric enums accept any number, not just defined members. This silently breaks type safety. Use string enums or union types when you need strict value checking.

```typescript
// BAD: numeric enum accepts invalid values
enum Status { Active, Inactive }
const s: Status = 999; // No error!

// GOOD: string enum rejects invalid values
enum Status { Active = "ACTIVE", Inactive = "INACTIVE" }
// const s: Status = "RANDOM"; // Error
```

- **Using `const enum` in libraries or with bundlers.** `const enum` values are inlined at compile time. This breaks with `isolatedModules` (required by most bundlers), causes version mismatch issues in libraries, and prevents runtime access.

- **Iterating over numeric enums without filtering.** `Object.keys()` and `Object.values()` on numeric enums include reverse mappings. You get both names and numbers, leading to duplicated or corrupted data in UI rendering.

- **Using enums when a simple union type suffices.** If you only need type checking and do not need a runtime object, a union type is simpler, generates no code, and has no gotchas.

```typescript
// Over-engineered: enum just for type checking
enum Theme { Light = "LIGHT", Dark = "DARK" }

// Simpler: union type
type Theme = "light" | "dark";
```

- **Importing enums across files adds runtime dependencies.** Unlike types (which are erased), enum imports remain in the compiled JavaScript. This can prevent tree shaking and increase bundle size.

- **Mixing string and numeric members.** Heterogeneous enums confuse both developers and tooling. Keep all members the same type.

---

## Best Practices

- **Prefer union types for simple sets of values.** If you only need type checking and the values are strings, a union type like `type Status = "active" | "inactive"` is the simplest and most efficient option. No runtime code, no imports needed.

- **Use `as const` objects when you need both runtime values and types.** When you need to iterate over values, build a dropdown, or use values at runtime, an `as const` object provides all the benefits of an enum without the quirks.

```typescript
const LogLevel = {
  Debug: "debug",
  Info: "info",
  Warn: "warn",
  Error: "error",
} as const;

type LogLevel = (typeof LogLevel)[keyof typeof LogLevel];
```

- **Reserve numeric enums for bit flags.** The one scenario where numeric enums shine is bit manipulation with `|`, `&`, and `~` operators. For everything else, string-based approaches are safer.

- **If you do use enums, use string enums.** String enums provide proper type safety (no arbitrary number assignment), readable debug output, clean iteration, and predictable serialization.

- **Avoid `const enum` in any project using `isolatedModules`.** Since most modern toolchains (Vite, esbuild, Next.js, and others) require `isolatedModules: true`, `const enum` will cause problems. Use plain enums or `as const` objects instead.

- **Be explicit about the pattern in your team's style guide.** The enum-vs-union-vs-const-object choice should be a team decision, documented and enforced with linting rules.

## Real-World Use Cases

### 1. HTTP Status Codes (as const Object)

```typescript
const HttpStatus = {
  OK: 200,
  Created: 201,
  NoContent: 204,
  BadRequest: 400,
  Unauthorized: 401,
  Forbidden: 403,
  NotFound: 404,
  InternalServerError: 500,
  ServiceUnavailable: 503,
} as const;

type HttpStatus = (typeof HttpStatus)[keyof typeof HttpStatus];

function isSuccess(status: HttpStatus): boolean {
  return status >= 200 && status < 300;
}

function handleResponse(status: HttpStatus, body: unknown): void {
  if (isSuccess(status)) {
    console.log("Success:", body);
  } else if (status === HttpStatus.NotFound) {
    console.log("Resource not found");
  } else if (status === HttpStatus.Unauthorized) {
    console.log("Please log in");
  }
}

handleResponse(HttpStatus.OK, { data: "result" });
handleResponse(HttpStatus.NotFound, null);

// Build a status code dropdown from the runtime object
const statusOptions = Object.entries(HttpStatus).map(([name, code]) => ({
  label: name,
  value: code,
}));
```

### 2. Permission System with Bit Flags (Numeric Enum)

```typescript
enum Permission {
  None      = 0,
  Read      = 1 << 0,  // 1
  Write     = 1 << 1,  // 2
  Delete    = 1 << 2,  // 4
  Admin     = 1 << 3,  // 8
}

type PermissionSet = number;

function grant(current: PermissionSet, perm: Permission): PermissionSet {
  return current | perm;
}

function revoke(current: PermissionSet, perm: Permission): PermissionSet {
  return current & ~perm;
}

function has(current: PermissionSet, perm: Permission): boolean {
  return (current & perm) === perm;
}

function describe(perms: PermissionSet): string[] {
  const names: string[] = [];
  if (has(perms, Permission.Read)) names.push("Read");
  if (has(perms, Permission.Write)) names.push("Write");
  if (has(perms, Permission.Delete)) names.push("Delete");
  if (has(perms, Permission.Admin)) names.push("Admin");
  return names;
}

let userPerms: PermissionSet = Permission.None;
userPerms = grant(userPerms, Permission.Read);
userPerms = grant(userPerms, Permission.Write);

console.log(has(userPerms, Permission.Read));     // true
console.log(has(userPerms, Permission.Delete));   // false
console.log(describe(userPerms));                 // ["Read", "Write"]

userPerms = revoke(userPerms, Permission.Write);
console.log(describe(userPerms));                 // ["Read"]
```

### 3. Form Validation States (Union Type)

The simplest approach when you do not need a runtime object.

```typescript
type ValidationState = "idle" | "validating" | "valid" | "invalid";

interface FormField {
  value: string;
  state: ValidationState;
  error: string | null;
}

function validateEmail(field: FormField): FormField {
  if (field.value.length === 0) {
    return { ...field, state: "idle", error: null };
  }

  const isValid = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(field.value);
  return {
    ...field,
    state: isValid ? "valid" : "invalid",
    error: isValid ? null : "Invalid email format",
  };
}

function getFieldClass(state: ValidationState): string {
  switch (state) {
    case "idle":       return "field";
    case "validating": return "field field--loading";
    case "valid":      return "field field--success";
    case "invalid":    return "field field--error";
  }
  // Exhaustive: no default needed.
  // If a new state is added, TypeScript will error here.
}

const emailField: FormField = { value: "", state: "idle", error: null };
const validated = validateEmail({ ...emailField, value: "test@example.com" });
console.log(validated.state);  // "valid"
console.log(getFieldClass(validated.state)); // "field field--success"
```

---

> **Key Takeaway:** Prefer union types for pure type safety without runtime overhead, `as const` objects when you need runtime values and iteration, string enums when you want the classic enum pattern with proper type safety, and reserve numeric enums exclusively for bit flag operations. Avoid `const enum` in modern bundled projects.
