# 14.4 Injection Tokens

> `InjectionToken` creates unique DI tokens for non-class dependencies like configuration objects, feature flags, functions, and interfaces that have no runtime representation.

---

## Why Injection Tokens Exist

Angular's DI system identifies providers by a **token**. For class-based services, the class itself is the token:

```typescript
@Injectable({ providedIn: 'root' })
export class UserService {}

// The token IS the class reference
private userService = inject(UserService);
```

But not everything you want to inject is a class. Configuration objects, primitive values, callback functions, and TypeScript interfaces do not have a runtime class reference. You cannot write `inject(AppConfig)` when `AppConfig` is an interface because interfaces are erased at compile time.

`InjectionToken` solves this by creating a unique object that serves as the DI token.

---

## Creating an InjectionToken

### Basic Syntax

```typescript
import { InjectionToken } from '@angular/core';

export const API_BASE_URL = new InjectionToken<string>('API_BASE_URL');
```

The generic parameter `<string>` specifies the type of the injected value. The string argument `'API_BASE_URL'` is a description used in error messages and debugging tools. It does not need to be unique, but making it descriptive helps with debugging.

### Providing a Value

Register the token in a providers array:

```typescript
// app.config.ts
export const appConfig: ApplicationConfig = {
  providers: [
    { provide: API_BASE_URL, useValue: 'https://api.example.com' },
  ],
};
```

### Injecting the Value

```typescript
@Injectable({ providedIn: 'root' })
export class ApiService {
  private baseUrl = inject(API_BASE_URL);
  private http = inject(HttpClient);

  get<T>(path: string) {
    return this.http.get<T>(`${this.baseUrl}${path}`);
  }
}
```

## Tree-Shakable Tokens with `factory`

When you provide a `factory` function in the `InjectionToken` constructor, Angular registers the token at the root level by default and makes it tree-shakable.

```typescript
export const LOGGER = new InjectionToken<Logger>('Logger', {
  providedIn: 'root',
  factory: () => {
    const isDev = !inject(PLATFORM_ID).includes('server');
    return isDev ? new ConsoleLogger() : new NoopLogger();
  },
});
```

The `factory` function runs the first time the token is injected. It has access to `inject()` because Angular calls it within an injection context.

### Factory vs Explicit Providers

| Approach | Tree-Shakable | Registration |
|---|---|---|
| `new InjectionToken('X', { factory: () => ... })` | Yes | Automatic (no providers array needed) |
| `new InjectionToken('X')` + `{ provide: X, useValue: ... }` | No | Manual (must add to providers) |

Use the factory approach when the token has a sensible default. Use explicit providers when the value must be supplied externally (e.g., environment-specific configuration).

### `providedIn` Options

```typescript
// Root scope (default when factory is provided)
new InjectionToken('X', {
  providedIn: 'root',
  factory: () => 'default',
});

// Platform scope (shared across Angular apps on the same page)
new InjectionToken('X', {
  providedIn: 'platform',
  factory: () => 'shared',
});

// No automatic registration (must be provided manually)
new InjectionToken<string>('X');
```

## Common Token Patterns

### Configuration Object

```typescript
export interface AppConfig {
  apiUrl: string;
  maxRetries: number;
  enableAnalytics: boolean;
  defaultLocale: string;
}

export const APP_CONFIG = new InjectionToken<AppConfig>('APP_CONFIG');

// app.config.ts
export const appConfig: ApplicationConfig = {
  providers: [
    {
      provide: APP_CONFIG,
      useValue: {
        apiUrl: 'https://api.example.com',
        maxRetries: 3,
        enableAnalytics: true,
        defaultLocale: 'en-US',
      } satisfies AppConfig,
    },
  ],
};

// Usage
@Injectable({ providedIn: 'root' })
export class ApiService {
  private config = inject(APP_CONFIG);
  private http = inject(HttpClient);

  fetch<T>(path: string) {
    return this.http.get<T>(`${this.config.apiUrl}${path}`).pipe(
      retry(this.config.maxRetries),
    );
  }
}
```

### Feature Flags

```typescript
export interface FeatureFlags {
  darkMode: boolean;
  betaFeatures: boolean;
  newCheckout: boolean;
}

export const FEATURE_FLAGS = new InjectionToken<FeatureFlags>('FEATURE_FLAGS', {
  providedIn: 'root',
  factory: () => ({
    darkMode: false,
    betaFeatures: false,
    newCheckout: false,
  }),
});

// Override in tests or specific environments
TestBed.configureTestingModule({
  providers: [
    {
      provide: FEATURE_FLAGS,
      useValue: { darkMode: true, betaFeatures: true, newCheckout: true },
    },
  ],
});
```

### Callback Functions

```typescript
export type ErrorHandler = (error: Error, context: string) => void;

export const ERROR_HANDLER = new InjectionToken<ErrorHandler>('ERROR_HANDLER', {
  providedIn: 'root',
  factory: () => {
    return (error: Error, context: string) => {
      console.error(`[${context}]`, error.message);
    };
  },
});

// Usage
@Injectable({ providedIn: 'root' })
export class DataService {
  private handleError = inject(ERROR_HANDLER);
  private http = inject(HttpClient);

  loadUsers() {
    return this.http.get<User[]>('/api/users').pipe(
      catchError(err => {
        this.handleError(err, 'DataService.loadUsers');
        return of([]);
      }),
    );
  }
}
```

### Window and Document References

```typescript
export const WINDOW = new InjectionToken<Window>('Window', {
  providedIn: 'root',
  factory: () => window,
});

export const DOCUMENT_TOKEN = new InjectionToken<Document>('Document', {
  providedIn: 'root',
  factory: () => document,
});

// Usage: testable, SSR-safe
@Injectable({ providedIn: 'root' })
export class ScrollService {
  private win = inject(WINDOW);

  scrollToTop() {
    this.win.scrollTo({ top: 0, behavior: 'smooth' });
  }
}

// In tests, provide a mock
{ provide: WINDOW, useValue: { scrollTo: jasmine.createSpy() } }

// In SSR, provide a stub
{ provide: WINDOW, useValue: { scrollTo: () => {} } }
```

### Abstract Class as Token

When you need an interface-like token but want a runtime value for DI, use an abstract class:

```typescript
export abstract class StorageAdapter {
  abstract getItem(key: string): string | null;
  abstract setItem(key: string, value: string): void;
  abstract removeItem(key: string): void;
}

// Concrete implementation
@Injectable()
export class LocalStorageAdapter extends StorageAdapter {
  getItem(key: string) {
    return localStorage.getItem(key);
  }

  setItem(key: string, value: string) {
    localStorage.setItem(key, value);
  }

  removeItem(key: string) {
    localStorage.removeItem(key);
  }
}

// Provider
{ provide: StorageAdapter, useClass: LocalStorageAdapter }

// Injection: the abstract class IS the token
private storage = inject(StorageAdapter);
```

Abstract classes exist at runtime (unlike interfaces), so they can serve directly as DI tokens without `InjectionToken`.

## Factory Functions with Dependencies

Token factories can inject other tokens and services:

```typescript
export const API_CLIENT = new InjectionToken<ApiClient>('API_CLIENT', {
  providedIn: 'root',
  factory: () => {
    const http = inject(HttpClient);
    const config = inject(APP_CONFIG);
    const auth = inject(AuthService);

    return new ApiClient(http, config.apiUrl, auth);
  },
});
```

### Conditional Factory

```typescript
export const CACHE_STRATEGY = new InjectionToken<CacheStrategy>('CACHE_STRATEGY', {
  providedIn: 'root',
  factory: () => {
    const platformId = inject(PLATFORM_ID);

    if (isPlatformBrowser(platformId)) {
      return new IndexedDbCacheStrategy();
    }

    // SSR: use in-memory cache
    return new InMemoryCacheStrategy();
  },
});
```

## String Tokens and Why to Avoid Them

Before `InjectionToken`, Angular used plain strings as tokens. This still works but is dangerous:

```typescript
// Bad Practice: string tokens cause naming collisions
providers: [
  { provide: 'API_URL', useValue: 'https://api.example.com' },
]

// Injection
private apiUrl = inject<string>(new InjectionToken('API_URL')); // WRONG: different token

// This would require inject() with the exact same string
// but string tokens are not type-safe and collide easily
```

Two different libraries could both use `'API_URL'` as a token. The second registration silently overrides the first. `InjectionToken` prevents this because each `new InjectionToken(...)` creates a unique object reference, even if the descriptions match.

```typescript
const TOKEN_A = new InjectionToken<string>('config');
const TOKEN_B = new InjectionToken<string>('config');

TOKEN_A === TOKEN_B; // false: different objects, different tokens
```

## Overriding Token Values

### At the Application Level

```typescript
// Default factory
export const DATE_FORMAT = new InjectionToken<string>('DATE_FORMAT', {
  providedIn: 'root',
  factory: () => 'yyyy-MM-dd',
});

// Override in app.config.ts
export const appConfig: ApplicationConfig = {
  providers: [
    { provide: DATE_FORMAT, useValue: 'dd/MM/yyyy' },
  ],
};
```

The explicit provider in `appConfig` overrides the factory default.

### At the Component Level

```typescript
@Component({
  selector: 'app-european-dates',
  standalone: true,
  providers: [
    { provide: DATE_FORMAT, useValue: 'dd.MM.yyyy' },
  ],
  template: `<p>{{ today | date:dateFormat() }}</p>`,
})
export class EuropeanDatesComponent {
  protected dateFormat = inject(DATE_FORMAT);
  today = new Date();
}
```

### At the Route Level

```typescript
export const routes: Routes = [
  {
    path: 'admin',
    providers: [
      { provide: APP_CONFIG, useValue: { apiUrl: '/admin-api', maxRetries: 5, ... } },
    ],
    children: [ ... ],
  },
];
```

## Lightweight Injection Tokens for Libraries

When building a library, large service classes used as tokens can prevent tree-shaking even when the consumer does not use those services. The solution is a **lightweight token**: an `InjectionToken` or abstract class that carries no implementation weight.

### The Problem

```typescript
// Bad Practice for a library: heavy class used as token
@Injectable()
export class HeavyChartService {
  // 50KB of chart logic
  render(data: ChartData) { ... }
}

// Library component
@Component({
  selector: 'lib-widget',
  template: `...`,
})
export class WidgetComponent {
  // Optional dependency, but HeavyChartService is referenced as a token
  private chart = inject(HeavyChartService, { optional: true });
}
```

Even though `chart` is optional, the consumer's bundle includes the entire `HeavyChartService` class because it is referenced as a type token in the component code.

### The Solution: Lightweight Token

```typescript
// Lightweight abstract class (minimal size)
export abstract class ChartRenderer {
  abstract render(data: ChartData): void;
}

// Heavy implementation in a separate file
@Injectable()
export class CanvasChartRenderer extends ChartRenderer {
  // 50KB of chart logic
  render(data: ChartData) { ... }
}

// Library component uses the lightweight token
@Component({
  selector: 'lib-widget',
  template: `...`,
})
export class WidgetComponent {
  private chart = inject(ChartRenderer, { optional: true });
}

// Consumer provides the heavy implementation only when needed
providers: [
  { provide: ChartRenderer, useClass: CanvasChartRenderer },
]
```

If the consumer does not provide `CanvasChartRenderer`, the bundler tree-shakes the heavy class because it is never directly referenced.

## Tokens for Interfaces

TypeScript interfaces do not exist at runtime. You cannot use an interface as a DI token.

```typescript
// This does NOT work
export interface CacheConfig {
  maxSize: number;
  ttlMs: number;
}

// inject(CacheConfig) is impossible because CacheConfig is erased at runtime
```

Two solutions:

### Solution 1: InjectionToken (Preferred)

```typescript
export interface CacheConfig {
  maxSize: number;
  ttlMs: number;
}

export const CACHE_CONFIG = new InjectionToken<CacheConfig>('CACHE_CONFIG', {
  providedIn: 'root',
  factory: () => ({ maxSize: 100, ttlMs: 60_000 }),
});

// Usage
private cacheConfig = inject(CACHE_CONFIG);
```

### Solution 2: Abstract Class

```typescript
export abstract class CacheConfig {
  abstract maxSize: number;
  abstract ttlMs: number;
}

// Provide a concrete value
{ provide: CacheConfig, useValue: { maxSize: 100, ttlMs: 60_000 } }

// Usage
private cacheConfig = inject(CacheConfig);
```

The abstract class approach is simpler when the "interface" is used exclusively as a DI token. The `InjectionToken` approach is more explicit and works for any type, including primitives and functions.

## Tokens with Multi Providers

A single `InjectionToken` can collect values from multiple providers using `multi: true`:

```typescript
export type HttpInterceptorFactory = () => HttpInterceptorFn;

export const HTTP_PLUGIN = new InjectionToken<HttpInterceptorFactory>('HTTP_PLUGIN');

// Provide multiple plugins
providers: [
  { provide: HTTP_PLUGIN, useValue: () => loggingInterceptor, multi: true },
  { provide: HTTP_PLUGIN, useValue: () => cachingInterceptor, multi: true },
  { provide: HTTP_PLUGIN, useValue: () => retryInterceptor, multi: true },
]

// Injection: returns an ARRAY of all provided values
private plugins = inject(HTTP_PLUGIN); // type: HttpInterceptorFactory[]
```

Without `multi: true`, the last provider wins and previous ones are silently overwritten. With `multi: true`, all values are collected into an array.

See section 14.6 for a full deep dive on multi providers.

## Testing with Injection Tokens

### Overriding Tokens in Tests

```typescript
describe('ApiService', () => {
  let service: ApiService;

  beforeEach(() => {
    TestBed.configureTestingModule({
      providers: [
        provideHttpClient(),
        provideHttpClientTesting(),
        { provide: APP_CONFIG, useValue: { apiUrl: '/test-api', maxRetries: 0 } },
      ],
    });

    service = TestBed.inject(ApiService);
  });

  it('should use the test API URL', () => {
    service.fetch('/users').subscribe();

    const req = TestBed.inject(HttpTestingController).expectOne('/test-api/users');
    expect(req.request.method).toBe('GET');
  });
});
```

### Testing Factory Tokens

```typescript
describe('CACHE_STRATEGY', () => {
  it('should use IndexedDb in browser', () => {
    TestBed.configureTestingModule({
      providers: [
        { provide: PLATFORM_ID, useValue: 'browser' },
      ],
    });

    const strategy = TestBed.inject(CACHE_STRATEGY);
    expect(strategy).toBeInstanceOf(IndexedDbCacheStrategy);
  });

  it('should use InMemory in server', () => {
    TestBed.configureTestingModule({
      providers: [
        { provide: PLATFORM_ID, useValue: 'server' },
      ],
    });

    const strategy = TestBed.inject(CACHE_STRATEGY);
    expect(strategy).toBeInstanceOf(InMemoryCacheStrategy);
  });
});
```

## Common Mistakes

- **Creating multiple `InjectionToken` instances with the same description and expecting them to be the same token.** Each `new InjectionToken(...)` call creates a distinct object. Even if two tokens have identical description strings, they are different tokens. Always export a single `InjectionToken` instance and import it wherever needed.

```typescript
// Bad Practice: token created in two files
// file-a.ts
const MY_TOKEN = new InjectionToken<string>('MY_TOKEN');
// file-b.ts
const MY_TOKEN = new InjectionToken<string>('MY_TOKEN');
// These are DIFFERENT tokens. Providing on one, injecting from the other fails.

// Correct: single source of truth
// tokens.ts
export const MY_TOKEN = new InjectionToken<string>('MY_TOKEN');
// Both files import from tokens.ts
```

- **Using string tokens instead of `InjectionToken`.** String tokens have no collision protection and no type safety. Two libraries using `'config'` as a token silently overwrite each other. Always use `InjectionToken`.

- **Forgetting `multi: true` when multiple providers are intended.** Without `multi: true`, the last provider silently replaces all previous ones. If you expect an array of values, every provider for that token must include `multi: true`.

- **Providing a value for a tree-shakable token without realizing the factory is overridden.** When a token has a factory and you also add it to a `providers` array, the explicit provider takes precedence. The factory never runs. This is intentional but can be surprising if you expect the factory default.

- **Injecting an `InjectionToken<T>` and assuming it is never `null` when no provider exists.** If the token has no factory and no explicit provider, `inject()` throws `NullInjectorError`. If this is possible, use `{ optional: true }` and handle the `null` case.

---

## Best Practices

- **Use `InjectionToken` with a factory for configuration that has sensible defaults.** The factory makes the token self-contained and tree-shakable. Consumers only need to provide an override if the default is not appropriate.

- **Export tokens from a dedicated `tokens.ts` file.** Group related tokens together. This prevents accidental duplication and makes them easy to discover.

```
src/app/
  tokens/
    config.tokens.ts    // APP_CONFIG, FEATURE_FLAGS
    http.tokens.ts      // API_BASE_URL, HTTP_TIMEOUT
    storage.tokens.ts   // STORAGE_ADAPTER, CACHE_CONFIG
```

- **Use `satisfies` when providing object tokens.** This ensures the provided value matches the expected interface at compile time without losing the literal type.

```typescript
{ provide: APP_CONFIG, useValue: { apiUrl: '/api', maxRetries: 3 } satisfies AppConfig }
```

- **Prefer `InjectionToken` over abstract classes for simple values.** Abstract classes are heavier (they produce runtime code) and can be confusing when used purely as tokens. Reserve abstract classes for when you need polymorphic behavior.

- **Document the token's purpose and expected provider scope.** Since tokens are often defined far from where they are provided, a JSDoc comment clarifies the contract.

```typescript
/**
 * Base URL for all API requests.
 * Provide at the application level in app.config.ts.
 * Default: 'https://api.example.com'
 */
export const API_BASE_URL = new InjectionToken<string>('API_BASE_URL', {
  providedIn: 'root',
  factory: () => 'https://api.example.com',
});
```

## Real-World Use Cases

### Environment-Specific Configuration

```typescript
// tokens/environment.tokens.ts
export interface EnvironmentConfig {
  production: boolean;
  apiUrl: string;
  sentryDsn: string | null;
  stripePublicKey: string;
}

export const ENVIRONMENT = new InjectionToken<EnvironmentConfig>('ENVIRONMENT');

// environments/environment.ts
export const environment: EnvironmentConfig = {
  production: false,
  apiUrl: 'http://localhost:3000',
  sentryDsn: null,
  stripePublicKey: 'pk_test_...',
};

// environments/environment.prod.ts
export const environment: EnvironmentConfig = {
  production: true,
  apiUrl: 'https://api.myapp.com',
  sentryDsn: 'https://abc@sentry.io/123',
  stripePublicKey: 'pk_live_...',
};

// app.config.ts
import { environment } from '../environments/environment';

export const appConfig: ApplicationConfig = {
  providers: [
    { provide: ENVIRONMENT, useValue: environment },
  ],
};

// Usage
@Injectable({ providedIn: 'root' })
export class PaymentService {
  private env = inject(ENVIRONMENT);
  private http = inject(HttpClient);

  createCheckoutSession(items: CartItem[]) {
    return this.http.post(`${this.env.apiUrl}/checkout`, {
      items,
      stripeKey: this.env.stripePublicKey,
    });
  }
}
```

### Plugin Architecture with Tokens

```typescript
export interface AppPlugin {
  name: string;
  initialize(): void;
  destroy(): void;
}

export const APP_PLUGIN = new InjectionToken<AppPlugin>('APP_PLUGIN');

// analytics-plugin.ts
export function provideAnalyticsPlugin(): Provider {
  return {
    provide: APP_PLUGIN,
    multi: true,
    useFactory: () => {
      const env = inject(ENVIRONMENT);
      return {
        name: 'analytics',
        initialize() {
          if (env.production) {
            // Initialize analytics SDK
          }
        },
        destroy() {
          // Flush analytics queue
        },
      };
    },
  };
}

// error-tracking-plugin.ts
export function provideErrorTracking(): Provider {
  return {
    provide: APP_PLUGIN,
    multi: true,
    useFactory: () => {
      const env = inject(ENVIRONMENT);
      return {
        name: 'error-tracking',
        initialize() {
          if (env.sentryDsn) {
            // Initialize Sentry
          }
        },
        destroy() {},
      };
    },
  };
}

// app.config.ts
export const appConfig: ApplicationConfig = {
  providers: [
    { provide: ENVIRONMENT, useValue: environment },
    provideAnalyticsPlugin(),
    provideErrorTracking(),
  ],
};

// app.component.ts
@Component({ ... })
export class AppComponent {
  private plugins = inject(APP_PLUGIN, { optional: true }) ?? [];

  constructor() {
    this.plugins.forEach(plugin => plugin.initialize());
    inject(DestroyRef).onDestroy(() => {
      this.plugins.forEach(plugin => plugin.destroy());
    });
  }
}
```

### Theming System with Token Overrides

```typescript
export interface ThemeConfig {
  primaryColor: string;
  fontFamily: string;
  borderRadius: string;
  spacing: number;
}

export const THEME = new InjectionToken<ThemeConfig>('THEME', {
  providedIn: 'root',
  factory: () => ({
    primaryColor: '#1976d2',
    fontFamily: 'Roboto, sans-serif',
    borderRadius: '4px',
    spacing: 8,
  }),
});

// A section of the app with a different theme
@Component({
  selector: 'app-admin-section',
  standalone: true,
  providers: [
    {
      provide: THEME,
      useValue: {
        primaryColor: '#d32f2f',
        fontFamily: 'Inter, sans-serif',
        borderRadius: '8px',
        spacing: 12,
      } satisfies ThemeConfig,
    },
  ],
  template: `<ng-content />`,
})
export class AdminSectionComponent {}

// Any component can inject the theme
@Component({
  selector: 'app-button',
  standalone: true,
  template: `<button><ng-content /></button>`,
  host: {
    '[style.background-color]': 'theme.primaryColor',
    '[style.font-family]': 'theme.fontFamily',
    '[style.border-radius]': 'theme.borderRadius',
  },
})
export class ButtonComponent {
  protected theme = inject(THEME);
}
```

Inside `<app-admin-section>`, every `<app-button>` uses the admin theme. Outside, they use the root theme. The injector hierarchy handles the override automatically.

---

> **Key Takeaway:** Use `InjectionToken` with a `factory` for tree-shakable, type-safe tokens with sensible defaults; export each token from a single file to prevent duplicate instances; and override token values at any level in the injector hierarchy (application, route, or component) to customize behavior without modifying the consuming code.
