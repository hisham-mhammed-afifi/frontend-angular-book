# 14.9 Service Design Patterns

> Services are where business logic, state management, and external communication live. Knowing the right pattern for the right problem is the difference between a maintainable application and a tangled mess.

---

## Service Responsibilities

A well-designed Angular service has a clear, focused responsibility. Services fall into several broad categories:

| Category | Purpose | Example |
|----------|---------|---------|
| **Data access** | Communicate with APIs | `ProductApiService` |
| **State management** | Hold and expose application state | `CartStateService` |
| **Facade** | Orchestrate multiple services behind a simple API | `CheckoutFacade` |
| **Utility** | Provide reusable logic with no state | `DateFormatterService` |
| **Adapter** | Wrap third-party APIs into Angular-friendly interfaces | `StripeAdapterService` |
| **Guard/interceptor** | Cross-cutting concerns | `AuthInterceptor`, `RoleGuard` |

A single service should not try to be all of these at once.

## Pattern 1: Signal-Based State Service

The most important pattern in modern Angular. A service holds state as signals and exposes derived state as computed signals.

### Basic Structure

```typescript
import { Injectable, signal, computed } from '@angular/core';

export interface Todo {
  id: string;
  title: string;
  completed: boolean;
}

@Injectable({ providedIn: 'root' })
export class TodoStateService {
  // Private writable state
  private readonly todos = signal<Todo[]>([]);
  private readonly filter = signal<'all' | 'active' | 'completed'>('all');

  // Public read-only derived state
  readonly filteredTodos = computed(() => {
    const todos = this.todos();
    const filter = this.filter();
    switch (filter) {
      case 'active': return todos.filter(t => !t.completed);
      case 'completed': return todos.filter(t => t.completed);
      default: return todos;
    }
  });

  readonly activeCount = computed(() =>
    this.todos().filter(t => !t.completed).length
  );

  readonly completedCount = computed(() =>
    this.todos().filter(t => t.completed).length
  );

  // Public mutation methods
  addTodo(title: string) {
    this.todos.update(todos => [
      ...todos,
      { id: crypto.randomUUID(), title, completed: false },
    ]);
  }

  toggleTodo(id: string) {
    this.todos.update(todos =>
      todos.map(t => t.id === id ? { ...t, completed: !t.completed } : t)
    );
  }

  removeTodo(id: string) {
    this.todos.update(todos => todos.filter(t => t.id !== id));
  }

  setFilter(filter: 'all' | 'active' | 'completed') {
    this.filter.set(filter);
  }
}
```

### Why This Pattern Works

1. **Immutable updates**: `signal.update()` with spread operators ensures referential identity changes, which triggers change detection correctly.
2. **Derived state is automatic**: `computed()` recalculates only when its dependencies change. No manual synchronization.
3. **Encapsulation**: The writable signal is private. Components can only read state and call mutation methods.
4. **No subscriptions**: Signals are synchronous and do not require subscribe/unsubscribe lifecycle management.

### Usage in Components

```typescript
@Component({
  selector: 'app-todo-list',
  template: `
    <input #input (keyup.enter)="addTodo(input)" />
    <ul>
      @for (todo of state.filteredTodos(); track todo.id) {
        <li [class.done]="todo.completed" (click)="state.toggleTodo(todo.id)">
          {{ todo.title }}
        </li>
      }
    </ul>
    <p>{{ state.activeCount() }} items left</p>
  `,
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class TodoListComponent {
  protected state = inject(TodoStateService);

  addTodo(input: HTMLInputElement) {
    if (input.value.trim()) {
      this.state.addTodo(input.value.trim());
      input.value = '';
    }
  }
}
```

## Pattern 2: State Service with Async Loading

Real applications load data from APIs. The `resource` API (Angular 19+) integrates async loading with signals:

```typescript
import { Injectable, signal, computed, inject, resource } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { firstValueFrom } from 'rxjs';

export interface Product {
  id: number;
  name: string;
  price: number;
  category: string;
}

@Injectable({ providedIn: 'root' })
export class ProductStateService {
  private http = inject(HttpClient);

  // Filter signal drives the resource reload
  private readonly categoryFilter = signal<string>('all');

  // resource() ties async loading to signal dependencies
  private readonly productResource = resource({
    request: () => this.categoryFilter(),
    loader: async ({ request: category }) => {
      const url = category === 'all'
        ? '/api/products'
        : `/api/products?category=${category}`;
      return firstValueFrom(this.http.get<Product[]>(url));
    },
  });

  // Expose resource state as readable signals
  readonly products = this.productResource.value;
  readonly isLoading = this.productResource.isLoading;
  readonly error = this.productResource.error;

  // Derived state
  readonly productCount = computed(() => this.products()?.length ?? 0);
  readonly totalValue = computed(() =>
    (this.products() ?? []).reduce((sum, p) => sum + p.price, 0)
  );

  // Actions
  filterByCategory(category: string) {
    this.categoryFilter.set(category);
    // resource() automatically reloads because its request signal changed
  }

  reload() {
    this.productResource.reload();
  }
}
```

### `httpResource` Alternative

Angular 19.2+ introduced `httpResource` as a simpler API for HTTP-backed resources:

```typescript
import { Injectable, signal, computed, inject } from '@angular/core';
import { httpResource } from '@angular/common/http';

@Injectable({ providedIn: 'root' })
export class UserStateService {
  private readonly userId = signal<number | null>(null);

  private readonly userResource = httpResource<User>(() =>
    this.userId() ? `/api/users/${this.userId()}` : undefined
  );

  readonly user = this.userResource.value;
  readonly isLoading = this.userResource.isLoading;

  selectUser(id: number) {
    this.userId.set(id);
  }
}
```

## Pattern 3: Facade Service

A facade service sits between components and the core layer. It orchestrates multiple services, hiding complexity from the presentation layer.

### When to Use a Facade

- A component needs data or actions from 3+ services.
- Business workflows span multiple domains (user, cart, payment).
- You want to keep components thin and testable.

### Structure

```typescript
import { Injectable, inject, computed } from '@angular/core';

@Injectable({ providedIn: 'root' })
export class CheckoutFacade {
  private cart = inject(CartStateService);
  private user = inject(UserStateService);
  private payment = inject(PaymentService);
  private order = inject(OrderApiService);
  private notifications = inject(NotificationService);

  // Expose only what the checkout UI needs
  readonly cartItems = this.cart.items;
  readonly cartTotal = this.cart.total;
  readonly currentUser = this.user.currentUser;
  readonly isProcessing = this.payment.isProcessing;

  readonly canCheckout = computed(() =>
    this.cartItems().length > 0
    && this.currentUser() !== null
    && !this.isProcessing()
  );

  async checkout() {
    const user = this.currentUser();
    if (!user) throw new Error('User not authenticated');

    try {
      const paymentResult = await this.payment.charge(
        user.paymentMethod,
        this.cartTotal(),
      );

      const order = await this.order.create({
        userId: user.id,
        items: this.cartItems(),
        paymentId: paymentResult.id,
      });

      this.cart.clear();
      this.notifications.success(`Order ${order.id} placed successfully`);

      return order;
    } catch (error) {
      this.notifications.error('Checkout failed. Please try again.');
      throw error;
    }
  }
}
```

### Component Using the Facade

```typescript
@Component({
  selector: 'app-checkout',
  template: `
    <app-cart-summary [items]="facade.cartItems()" [total]="facade.cartTotal()" />
    <button
      (click)="facade.checkout()"
      [disabled]="!facade.canCheckout()"
    >
      @if (facade.isProcessing()) {
        Processing...
      } @else {
        Pay {{ facade.cartTotal() | currency }}
      }
    </button>
  `,
})
export class CheckoutComponent {
  protected facade = inject(CheckoutFacade);
}
```

The component knows nothing about `PaymentService`, `OrderApiService`, or `CartStateService`. It talks only to the facade.

### Facade vs Direct Injection

| Aspect | Direct Injection | Facade |
|--------|-----------------|--------|
| Component complexity | High (multiple injections, orchestration logic) | Low (single injection, simple API) |
| Testability | Must mock many services | Mock one facade |
| Reuse | Orchestration logic duplicated across components | Orchestration centralized |
| When to use | Simple components with 1-2 dependencies | Complex workflows spanning multiple services |

## Pattern 4: Adapter Service

An adapter wraps a third-party library or browser API into an Angular-friendly, injectable, and testable interface.

```typescript
// The adapter interface
export abstract class StorageAdapter {
  abstract getItem<T>(key: string): T | null;
  abstract setItem<T>(key: string, value: T): void;
  abstract removeItem(key: string): void;
  abstract clear(): void;
}

// LocalStorage implementation
@Injectable()
export class LocalStorageAdapter extends StorageAdapter {
  getItem<T>(key: string): T | null {
    const raw = localStorage.getItem(key);
    if (raw === null) return null;
    try {
      return JSON.parse(raw) as T;
    } catch {
      return null;
    }
  }

  setItem<T>(key: string, value: T): void {
    localStorage.setItem(key, JSON.stringify(value));
  }

  removeItem(key: string): void {
    localStorage.removeItem(key);
  }

  clear(): void {
    localStorage.clear();
  }
}

// In-memory implementation for SSR or testing
@Injectable()
export class InMemoryStorageAdapter extends StorageAdapter {
  private store = new Map<string, string>();

  getItem<T>(key: string): T | null {
    const raw = this.store.get(key);
    if (!raw) return null;
    return JSON.parse(raw) as T;
  }

  setItem<T>(key: string, value: T): void {
    this.store.set(key, JSON.stringify(value));
  }

  removeItem(key: string): void {
    this.store.delete(key);
  }

  clear(): void {
    this.store.clear();
  }
}

// Registration
export function provideStorage(): Provider[] {
  return [
    {
      provide: StorageAdapter,
      useFactory: () => {
        const platformId = inject(PLATFORM_ID);
        return isPlatformBrowser(platformId)
          ? new LocalStorageAdapter()
          : new InMemoryStorageAdapter();
      },
    },
  ];
}
```

### Using the Adapter

```typescript
@Injectable({ providedIn: 'root' })
export class UserPreferencesService {
  private storage = inject(StorageAdapter);

  private readonly PREFS_KEY = 'user-prefs';

  readonly preferences = signal<UserPreferences>(
    this.storage.getItem<UserPreferences>(this.PREFS_KEY) ?? DEFAULT_PREFS
  );

  updatePreferences(partial: Partial<UserPreferences>) {
    this.preferences.update(current => {
      const updated = { ...current, ...partial };
      this.storage.setItem(this.PREFS_KEY, updated);
      return updated;
    });
  }
}
```

## Pattern 5: Composition Over Inheritance

Avoid deep service inheritance hierarchies. Compose services from smaller, focused services.

### Bad: Inheritance Chain

```typescript
// BAD: Deep inheritance creates coupling and unclear responsibilities
@Injectable()
export class BaseDataService {
  protected http = inject(HttpClient);
  protected baseUrl = '';

  getAll<T>() { return this.http.get<T[]>(this.baseUrl); }
  getById<T>(id: string) { return this.http.get<T>(`${this.baseUrl}/${id}`); }
  create<T>(data: T) { return this.http.post<T>(this.baseUrl, data); }
  update<T>(id: string, data: T) { return this.http.put<T>(`${this.baseUrl}/${id}`, data); }
  delete(id: string) { return this.http.delete(`${this.baseUrl}/${id}`); }
}

@Injectable()
export class BaseEntityService extends BaseDataService {
  protected handleError(err: HttpErrorResponse) { /* ... */ }
  protected addCache(key: string, data: unknown) { /* ... */ }
}

@Injectable({ providedIn: 'root' })
export class ProductService extends BaseEntityService {
  override baseUrl = '/api/products';
  // Now tied to two base classes, hard to test, hard to change
}
```

### Good: Composition with Inject Functions

```typescript
// GOOD: Composable building blocks

// A reusable inject function for CRUD operations
function injectCrudApi<T>(baseUrl: string) {
  assertInInjectionContext(injectCrudApi);
  const http = inject(HttpClient);

  return {
    getAll: () => http.get<T[]>(baseUrl),
    getById: (id: string) => http.get<T>(`${baseUrl}/${id}`),
    create: (data: Partial<T>) => http.post<T>(baseUrl, data),
    update: (id: string, data: Partial<T>) => http.put<T>(`${baseUrl}/${id}`, data),
    remove: (id: string) => http.delete(`${baseUrl}/${id}`),
  };
}

// A reusable inject function for signal-based caching
function injectSignalCache<T>(initialValue: T) {
  assertInInjectionContext(injectSignalCache);
  const cache = signal<T>(initialValue);

  return {
    value: cache.asReadonly(),
    set: (value: T) => cache.set(value),
    update: (fn: (current: T) => T) => cache.update(fn),
    reset: () => cache.set(initialValue),
  };
}

// Composed service -- no inheritance needed
@Injectable({ providedIn: 'root' })
export class ProductService {
  private api = injectCrudApi<Product>('/api/products');
  private cache = injectSignalCache<Product[]>([]);

  readonly products = this.cache.value;

  async loadProducts() {
    const products = await firstValueFrom(this.api.getAll());
    this.cache.set(products);
  }

  async addProduct(data: Partial<Product>) {
    const product = await firstValueFrom(this.api.create(data));
    this.cache.update(list => [...list, product]);
    return product;
  }
}
```

Each building block is independent, testable, and reusable without inheritance chains.

## Pattern 6: Scoped Services

Not every service should be a singleton. Some services need per-component or per-route instances.

### Per-Component Service

```typescript
@Injectable() // No providedIn -- must be provided explicitly
export class FormStateService {
  private readonly fields = signal<Record<string, unknown>>({});
  private readonly errors = signal<Record<string, string>>({});
  private readonly dirty = signal(false);

  readonly isDirty = this.dirty.asReadonly();
  readonly hasErrors = computed(() =>
    Object.keys(this.errors()).length > 0
  );

  setField(name: string, value: unknown) {
    this.fields.update(f => ({ ...f, [name]: value }));
    this.dirty.set(true);
  }

  setError(name: string, message: string) {
    this.errors.update(e => ({ ...e, [name]: message }));
  }

  clearError(name: string) {
    this.errors.update(e => {
      const copy = { ...e };
      delete copy[name];
      return copy;
    });
  }

  reset() {
    this.fields.set({});
    this.errors.set({});
    this.dirty.set(false);
  }

  getValues() {
    return this.fields();
  }
}

// Each form component gets its own instance
@Component({
  selector: 'app-user-form',
  providers: [FormStateService],
  template: `
    <form>
      <!-- ... form fields ... -->
      <button [disabled]="!formState.isDirty() || formState.hasErrors()">
        Save
      </button>
    </form>
  `,
})
export class UserFormComponent {
  protected formState = inject(FormStateService);
}
```

### Per-Route Service

```typescript
// Route configuration creates a scoped injector
const routes: Routes = [
  {
    path: 'editor/:id',
    component: EditorComponent,
    providers: [
      EditorStateService,   // New instance per navigation to this route
      AutoSaveService,      // Scoped to the editor route
      UndoRedoService,      // Destroyed when navigating away
    ],
  },
];
```

When the user navigates away from `/editor/:id`, the route injector is destroyed, and all three services are cleaned up.

## Pattern 7: Notification / Event Bus Service

For cross-cutting communication between unrelated components:

```typescript
import { Injectable, signal, computed } from '@angular/core';

export interface Notification {
  id: string;
  type: 'success' | 'error' | 'warning' | 'info';
  message: string;
  timestamp: number;
}

@Injectable({ providedIn: 'root' })
export class NotificationService {
  private readonly notifications = signal<Notification[]>([]);

  readonly active = computed(() =>
    this.notifications().filter(n => Date.now() - n.timestamp < 5000)
  );

  readonly count = computed(() => this.active().length);

  success(message: string) { this.add('success', message); }
  error(message: string) { this.add('error', message); }
  warning(message: string) { this.add('warning', message); }
  info(message: string) { this.add('info', message); }

  dismiss(id: string) {
    this.notifications.update(list => list.filter(n => n.id !== id));
  }

  private add(type: Notification['type'], message: string) {
    const notification: Notification = {
      id: crypto.randomUUID(),
      type,
      message,
      timestamp: Date.now(),
    };

    this.notifications.update(list => [...list, notification]);

    // Auto-dismiss after 5 seconds
    setTimeout(() => this.dismiss(notification.id), 5000);
  }
}
```

> **Note:** For complex event-driven communication between many producers and consumers, consider an RxJS `Subject`-based approach for its filtering and transformation operators.

## Pattern 8: Configuration Service with Injection Tokens

Use injection tokens for configuration that varies between environments or deployment targets:

```typescript
import { InjectionToken, inject, makeEnvironmentProviders } from '@angular/core';

export interface AppConfig {
  apiUrl: string;
  wsUrl: string;
  features: {
    darkMode: boolean;
    analytics: boolean;
    experimentalSearch: boolean;
  };
  pagination: {
    defaultPageSize: number;
    maxPageSize: number;
  };
}

export const APP_CONFIG = new InjectionToken<AppConfig>('APP_CONFIG');

export function provideAppConfig(config: AppConfig) {
  return makeEnvironmentProviders([
    { provide: APP_CONFIG, useValue: Object.freeze(config) },
  ]);
}

// Usage at bootstrap
bootstrapApplication(AppComponent, {
  providers: [
    provideAppConfig({
      apiUrl: 'https://api.example.com',
      wsUrl: 'wss://ws.example.com',
      features: {
        darkMode: true,
        analytics: true,
        experimentalSearch: false,
      },
      pagination: {
        defaultPageSize: 25,
        maxPageSize: 100,
      },
    }),
  ],
});

// Consuming in a service
@Injectable({ providedIn: 'root' })
export class ApiService {
  private config = inject(APP_CONFIG);
  private http = inject(HttpClient);

  get<T>(path: string) {
    return this.http.get<T>(`${this.config.apiUrl}${path}`);
  }
}
```

## Service Scope Decision Guide

| Scenario | Scope | How to Provide |
|----------|-------|----------------|
| App-wide singleton (auth, router, config) | Root | `providedIn: 'root'` |
| Feature-scoped singleton (editor state, wizard state) | Route | Route `providers` array |
| Per-component instance (form state, local UI state) | Component | Component `providers` array |
| Per-dialog instance (modal state) | Component | Dialog component `providers` |
| Multi-instance utility (logger per context) | Component or `useFactory` | `providers` with factory |
| Platform-shared (across micro-frontends) | Platform | `providedIn: 'platform'` |

### Decision Flowchart

```
Is the state shared across the entire app?
  YES -> providedIn: 'root'
  NO  -> Is it shared across a feature (route subtree)?
           YES -> route-level providers
           NO  -> Is each component supposed to have its own state?
                    YES -> component-level providers
                    NO  -> Re-evaluate your design
```

## Common Mistakes

### Mistake 1: God Services

A single service that does everything:

```typescript
// BAD: One service handling API, state, caching, formatting, validation
@Injectable({ providedIn: 'root' })
export class ProductService {
  private products = signal<Product[]>([]);
  private http = inject(HttpClient);

  async loadProducts() { /* API call */ }
  formatPrice(price: number) { /* formatting */ }
  validateProduct(p: Product) { /* validation */ }
  sortProducts(by: string) { /* sorting */ }
  filterProducts(category: string) { /* filtering */ }
  exportToCsv() { /* export */ }
  cacheProducts() { /* caching */ }
  // ... 30 more methods
}

// GOOD: Split into focused services
// ProductApiService -- API communication only
// ProductStateService -- state management only
// ProductFormatterService -- display formatting
// ProductExportService -- export functionality
```

### Mistake 2: Exposing Writable Signals

```typescript
// BAD: Consumers can mutate state directly, bypassing business logic
@Injectable({ providedIn: 'root' })
export class CartService {
  readonly items = signal<CartItem[]>([]);  // Anyone can call items.set([])!
}

// GOOD: Expose read-only, mutate through methods
@Injectable({ providedIn: 'root' })
export class CartService {
  private readonly _items = signal<CartItem[]>([]);
  readonly items = this._items.asReadonly();

  addItem(item: CartItem) {
    // Business logic: check duplicates, validate quantity, etc.
    this._items.update(list => [...list, item]);
  }
}
```

### Mistake 3: Putting UI Logic in Services

```typescript
// BAD: Service knows about DOM and UI concerns
@Injectable({ providedIn: 'root' })
export class UserService {
  showUserProfile(user: User) {
    document.getElementById('profile')!.innerHTML = `<h1>${user.name}</h1>`;
    document.body.classList.add('profile-open');
  }
}

// GOOD: Service provides data, component handles presentation
@Injectable({ providedIn: 'root' })
export class UserService {
  private readonly selectedUser = signal<User | null>(null);
  readonly user = this.selectedUser.asReadonly();

  selectUser(user: User) {
    this.selectedUser.set(user);
  }
}
```

### Mistake 4: Circular Dependencies Between Services

```typescript
// BAD: ServiceA depends on ServiceB, ServiceB depends on ServiceA
@Injectable({ providedIn: 'root' })
export class AuthService {
  private userService = inject(UserService); // UserService also injects AuthService!
}

// GOOD: Extract shared logic into a third service, or use events/signals
@Injectable({ providedIn: 'root' })
export class AuthStateService {
  readonly isAuthenticated = signal(false);
  readonly currentUserId = signal<string | null>(null);
}

// Both AuthService and UserService depend on AuthStateService, not on each other
```

### Mistake 5: Forgetting That `providedIn: 'root'` Means Singleton

```typescript
// Unexpected: Two components share the same state when they should not
@Injectable({ providedIn: 'root' })  // Singleton!
export class FormStateService {
  readonly values = signal<Record<string, unknown>>({});
}

// Both forms share the same FormStateService instance
// <app-user-form />
// <app-address-form />

// FIX: Remove providedIn, provide at the component level
@Injectable()  // Each component gets its own instance
export class FormStateService { /* ... */ }

@Component({
  selector: 'app-user-form',
  providers: [FormStateService],  // Scoped instance
})
export class UserFormComponent {}
```

## Best Practices

### 1. Single Responsibility

Each service should have one reason to change. If you cannot describe a service's purpose in one sentence without using "and", split it.

### 2. Encapsulate State Mutations

Never expose writable signals or subjects directly. Always go through methods that enforce business rules.

### 3. Use `computed()` for Derived State

Never store derived data in a separate signal and manually keep it in sync. Use `computed()`:

```typescript
// BAD: Manual sync
private items = signal<Item[]>([]);
private total = signal(0);  // Must manually update when items change

// GOOD: Automatic derivation
private items = signal<Item[]>([]);
readonly total = computed(() => this.items().reduce((sum, i) => sum + i.price, 0));
```

### 4. Prefer `inject()` Functions for Reusable Logic

Instead of abstract base services, create composable `injectXxx()` functions that can be mixed and matched.

### 5. Design for Testability

Services that accept their dependencies through DI (not through global state or `fetch` directly) are easy to test:

```typescript
// Easy to test -- swap HttpClient in tests
@Injectable({ providedIn: 'root' })
export class ProductApi {
  private http = inject(HttpClient);
  getAll() { return this.http.get<Product[]>('/api/products'); }
}
```

### 6. Use `DestroyRef` for Cleanup in Scoped Services

```typescript
@Injectable()
export class WebSocketService {
  private destroyRef = inject(DestroyRef);
  private ws: WebSocket | null = null;

  constructor() {
    this.ws = new WebSocket('wss://example.com');
    this.destroyRef.onDestroy(() => {
      this.ws?.close();
    });
  }
}
```

## Real-World Use Cases

### Authentication State Service

A production-ready auth service managing tokens, user state, and auto-refresh:

```typescript
@Injectable({ providedIn: 'root' })
export class AuthService {
  private readonly user = signal<User | null>(null);
  private readonly token = signal<string | null>(null);
  private destroyRef = inject(DestroyRef);
  private http = inject(HttpClient);
  private storage = inject(StorageAdapter);
  private router = inject(Router);

  readonly currentUser = this.user.asReadonly();
  readonly isAuthenticated = computed(() => this.user() !== null);
  readonly isAdmin = computed(() => this.user()?.role === 'admin');

  constructor() {
    // Restore session from storage
    const savedToken = this.storage.getItem<string>('auth_token');
    if (savedToken) {
      this.token.set(savedToken);
      this.loadUserProfile();
    }
  }

  async login(credentials: { email: string; password: string }) {
    const response = await firstValueFrom(
      this.http.post<{ token: string; user: User }>('/api/auth/login', credentials)
    );

    this.token.set(response.token);
    this.user.set(response.user);
    this.storage.setItem('auth_token', response.token);
  }

  logout() {
    this.token.set(null);
    this.user.set(null);
    this.storage.removeItem('auth_token');
    this.router.navigate(['/login']);
  }

  private async loadUserProfile() {
    try {
      const user = await firstValueFrom(
        this.http.get<User>('/api/auth/me')
      );
      this.user.set(user);
    } catch {
      this.logout();
    }
  }
}
```

### Data Table Facade

A facade that coordinates pagination, sorting, filtering, and data loading:

```typescript
@Injectable()
export class DataTableFacade<T> {
  private http = inject(HttpClient);

  // Configuration
  private endpoint = '';

  // State
  private readonly page = signal(1);
  private readonly pageSize = signal(25);
  private readonly sortField = signal<string | null>(null);
  private readonly sortDirection = signal<'asc' | 'desc'>('asc');
  private readonly searchTerm = signal('');

  // Async data loading tied to all state signals
  private readonly dataResource = resource({
    request: () => ({
      page: this.page(),
      pageSize: this.pageSize(),
      sort: this.sortField(),
      direction: this.sortDirection(),
      search: this.searchTerm(),
    }),
    loader: async ({ request }) => {
      const params = new HttpParams()
        .set('page', request.page)
        .set('pageSize', request.pageSize)
        .set('search', request.search);

      if (request.sort) {
        params.set('sort', request.sort).set('direction', request.direction);
      }

      return firstValueFrom(
        this.http.get<{ items: T[]; total: number }>(this.endpoint, { params })
      );
    },
  });

  // Public API
  readonly items = computed(() => this.dataResource.value()?.items ?? []);
  readonly totalItems = computed(() => this.dataResource.value()?.total ?? 0);
  readonly totalPages = computed(() =>
    Math.ceil(this.totalItems() / this.pageSize())
  );
  readonly currentPage = this.page.asReadonly();
  readonly isLoading = this.dataResource.isLoading;

  configure(endpoint: string) {
    this.endpoint = endpoint;
  }

  goToPage(page: number) { this.page.set(page); }
  setPageSize(size: number) { this.pageSize.set(size); this.page.set(1); }
  sort(field: string) {
    if (this.sortField() === field) {
      this.sortDirection.update(d => d === 'asc' ? 'desc' : 'asc');
    } else {
      this.sortField.set(field);
      this.sortDirection.set('asc');
    }
  }
  search(term: string) { this.searchTerm.set(term); this.page.set(1); }
}
```

Usage:

```typescript
@Component({
  selector: 'app-user-table',
  providers: [DataTableFacade],  // Per-component instance
  template: `
    <input (input)="table.search($any($event.target).value)" placeholder="Search..." />
    <table>
      <thead>
        <tr>
          <th (click)="table.sort('name')">Name</th>
          <th (click)="table.sort('email')">Email</th>
        </tr>
      </thead>
      <tbody>
        @for (user of table.items(); track user.id) {
          <tr>
            <td>{{ user.name }}</td>
            <td>{{ user.email }}</td>
          </tr>
        }
      </tbody>
    </table>
    <app-pagination
      [currentPage]="table.currentPage()"
      [totalPages]="table.totalPages()"
      (pageChange)="table.goToPage($event)"
    />
  `,
})
export class UserTableComponent {
  protected table = inject<DataTableFacade<User>>(DataTableFacade);

  constructor() {
    this.table.configure('/api/users');
  }
}
```

> **Key Takeaway:** Match the service pattern to the problem: signal-based state services for reactive state, facades for orchestrating complex workflows, adapters for third-party isolation, composition functions for reusable logic, and scoped providers for per-component or per-route lifetimes.
