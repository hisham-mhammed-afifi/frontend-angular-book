# 15.3 Route Parameters and `input()` Binding

> Route parameters carry dynamic data in the URL. With `withComponentInputBinding()`, Angular automatically maps those parameters to signal inputs, eliminating most `ActivatedRoute` boilerplate.

---

## Types of Route Parameters

Angular supports several kinds of parameters that travel through the URL:

| Parameter Type | URL Example | Purpose |
|---------------|------------|---------|
| Path parameters | `/products/:id` -> `/products/42` | Identify a resource |
| Matrix parameters | `/products;color=red;size=lg` | Optional segment-level metadata |
| Query parameters | `/products?page=2&sort=name` | Global filtering, pagination |
| Fragment | `/docs#installation` | In-page anchoring |

This guide focuses on path and matrix parameters. Query parameters are covered in 15.4.

---

## Path Parameters

### Defining Path Parameters

Path parameters are declared in route configuration with a `:` prefix:

```typescript
const routes: Routes = [
  { path: 'products/:id', component: ProductDetailComponent },
  { path: 'users/:userId/posts/:postId', component: PostDetailComponent },
];
```

### Accessing with Signal Inputs (Modern Approach)

With `withComponentInputBinding()` enabled in `provideRouter`, path parameters automatically bind to component inputs that share the same name:

```typescript
// app.config.ts
provideRouter(routes, withComponentInputBinding())
```

```typescript
// product-detail.component.ts
import { Component, input, inject, effect } from '@angular/core';

@Component({
  selector: 'app-product-detail',
  template: `
    <h1>Product #{{ id() }}</h1>
  `,
})
export class ProductDetailComponent {
  // Automatically receives the :id path parameter as a string
  id = input.required<string>();
}
```

Angular matches the input name (`id`) to the route parameter name (`:id`) and sets it automatically.

### Multiple Path Parameters

```typescript
// Route: users/:userId/posts/:postId
@Component({
  template: `
    <p>User: {{ userId() }}, Post: {{ postId() }}</p>
  `,
})
export class PostDetailComponent {
  userId = input.required<string>();
  postId = input.required<string>();
}
```

> **Requirement:** `paramsInheritanceStrategy: 'always'` must be set in `withRouterConfig()` for child components to inherit parent route params automatically. Without it, `userId` would only be available in the component directly mapped to the `:userId` route segment.

## Type Transformation

All route parameters arrive as strings. If you need a number, boolean, or other type, use the `transform` option.

### Converting to Number

```typescript
import { Component, input, numberAttribute } from '@angular/core';

@Component({
  template: `<p>Product ID: {{ id() }}</p>`,
})
export class ProductDetailComponent {
  // Route param ':id' arrives as '42' (string), transformed to 42 (number)
  id = input.required({ transform: numberAttribute });
}
```

`numberAttribute` is Angular's built-in transform function. It converts the string to a number and returns `NaN` if conversion fails.

### Custom Transform Functions

```typescript
@Component({
  template: `<p>Slug: {{ slug() }}</p>`,
})
export class ArticleComponent {
  // Normalize the slug to lowercase
  slug = input.required({
    transform: (value: string) => value.toLowerCase().trim(),
  });
}
```

### Parsing Complex Parameters

```typescript
// Route: /calendar/:date  ->  /calendar/2026-03-08
@Component({
  template: `<p>Date: {{ date().toLocaleDateString() }}</p>`,
})
export class CalendarComponent {
  date = input.required({
    transform: (value: string) => new Date(value),
  });
}
```

### Transform with Validation

```typescript
@Component({ ... })
export class ProductComponent {
  id = input.required({
    transform: (value: string) => {
      const num = Number(value);
      if (Number.isNaN(num) || num <= 0) {
        throw new Error(`Invalid product ID: ${value}`);
      }
      return num;
    },
  });
}
```

## Reacting to Parameter Changes

When the user navigates from `/products/42` to `/products/99`, Angular does not destroy and recreate the component. Instead, the input signal updates. Use `computed()` or `effect()` to react.

### Derived State with `computed()`

```typescript
import { Component, input, computed, inject } from '@angular/core';

@Component({
  template: `
    @if (product(); as product) {
      <h1>{{ product.name }}</h1>
      <p>{{ product.description }}</p>
    } @else {
      <p>Loading...</p>
    }
  `,
})
export class ProductDetailComponent {
  private productService = inject(ProductService);

  id = input.required<string>();

  // resource() reloads when id() changes
  private productResource = resource({
    request: () => this.id(),
    loader: async ({ request: id }) => {
      return this.productService.getById(id);
    },
  });

  product = this.productResource.value;
}
```

### Side Effects with `effect()`

```typescript
@Component({ ... })
export class ProductDetailComponent {
  id = input.required<string>();
  private analytics = inject(AnalyticsService);

  constructor() {
    effect(() => {
      // Fires whenever id() changes (including the initial value)
      this.analytics.trackPageView(`/products/${this.id()}`);
    });
  }
}
```

## The `ActivatedRoute` Approach (Legacy)

Before `withComponentInputBinding()`, you had to inject `ActivatedRoute` and subscribe to its observables. You will encounter this in existing codebases.

### `paramMap` Observable

```typescript
import { Component, inject, OnInit, signal } from '@angular/core';
import { ActivatedRoute } from '@angular/router';

@Component({
  template: `<h1>Product #{{ productId() }}</h1>`,
})
export class ProductDetailComponent implements OnInit {
  private route = inject(ActivatedRoute);
  productId = signal('');

  ngOnInit() {
    this.route.paramMap.subscribe(params => {
      this.productId.set(params.get('id') ?? '');
    });
  }
}
```

### `paramMap` Methods

The `ParamMap` interface provides:

| Method | Returns | Purpose |
|--------|---------|---------|
| `get(name)` | `string \| null` | Get a single parameter value |
| `getAll(name)` | `string[]` | Get all values for a parameter (matrix params) |
| `has(name)` | `boolean` | Check if a parameter exists |
| `keys` | `string[]` | Get all parameter names |

### Snapshot vs Observable

```typescript
// Snapshot: reads parameters ONCE at component creation
const id = this.route.snapshot.paramMap.get('id');

// Observable: emits whenever parameters change
this.route.paramMap.subscribe(params => {
  const id = params.get('id');
});
```

**When snapshot fails:** If the route reuses the component (same route, different parameter), the snapshot value does not update. This happens when navigating from `/products/42` to `/products/99` because Angular reuses the `ProductDetailComponent` instance.

```typescript
// BAD: Snapshot does not update on same-component navigation
@Component({ ... })
export class ProductDetailComponent implements OnInit {
  ngOnInit() {
    const id = this.route.snapshot.paramMap.get('id');
    this.loadProduct(id!); // Only loads the first product, never updates
  }
}

// GOOD: Observable reacts to every parameter change
@Component({ ... })
export class ProductDetailComponent implements OnInit {
  ngOnInit() {
    this.route.paramMap.subscribe(params => {
      this.loadProduct(params.get('id')!);
    });
  }
}
```

With `withComponentInputBinding()` and signal inputs, this problem disappears because the input signal automatically updates.

## Matrix Parameters

Matrix parameters are optional parameters attached to a specific URL segment. They differ from query parameters in that they are segment-scoped rather than global.

### URL Structure

```
/products;color=red;size=lg/42;view=detailed
```

Here:
- `color=red` and `size=lg` are matrix parameters on the `products` segment.
- `view=detailed` is a matrix parameter on the `42` segment.

### Navigating with Matrix Parameters

```typescript
// Using RouterLink
<a [routerLink]="['/products', { color: 'red', size: 'lg' }]">
  Red Large Products
</a>
// Produces: /products;color=red;size=lg

// Using Router.navigate
this.router.navigate(['/products', { color: 'red', size: 'lg' }]);
```

### Reading Matrix Parameters

Matrix parameters are included in the `paramMap` alongside path parameters:

```typescript
// Route: { path: 'products', component: ProductListComponent }
// URL: /products;color=red;size=lg

@Component({ ... })
export class ProductListComponent {
  // With withComponentInputBinding(), matrix params bind to inputs too
  color = input<string>();
  size = input<string>();
}
```

Or with `ActivatedRoute`:

```typescript
ngOnInit() {
  this.route.paramMap.subscribe(params => {
    const color = params.get('color');  // 'red'
    const size = params.get('size');    // 'lg'
  });
}
```

### Matrix vs Query Parameters

| Aspect | Matrix Parameters | Query Parameters |
|--------|------------------|-----------------|
| Scope | Attached to a specific URL segment | Global to the entire URL |
| Syntax | `;key=value` | `?key=value&key2=value2` |
| Survives navigation | Only within their segment | Configurable via `queryParamsHandling` |
| Use case | Segment-specific metadata | Filters, pagination, global state |
| Common usage | Rare in modern apps | Very common |

## What Gets Bound with `withComponentInputBinding()`

The binding priority when multiple sources provide the same key:

```
1. Static route data          (highest priority)
2. Path parameters
3. Query parameters            (lowest priority)
```

### Complete Example

```typescript
// Route configuration
{
  path: 'products/:id',
  component: ProductDetailComponent,
  data: { source: 'catalog' },
  resolve: { product: productResolver },
}

// URL: /products/42?tab=reviews&source=search
```

```typescript
@Component({ ... })
export class ProductDetailComponent {
  // From path parameter :id
  id = input.required<string>();              // '42'

  // From static data (NOT from query param, because data has higher priority)
  source = input<string>();                    // 'catalog'

  // From query parameter
  tab = input<string>();                       // 'reviews'

  // From resolver
  product = input<Product>();                  // Resolved Product object
}
```

Note that `source` gets `'catalog'` from static `data`, not `'search'` from the query parameter, because static data has higher priority.

## Aliased Input Names

Sometimes the route parameter name and the desired property name differ:

```typescript
// Route: /users/:user_id
@Component({ ... })
export class UserComponent {
  // Map the route param 'user_id' to the component property 'userId'
  userId = input.required<string>({ alias: 'user_id' });
}
```

## Converting `ActivatedRoute` to Signals with `toSignal`

If you need to work with `ActivatedRoute` observables but want signal-based reactivity:

```typescript
import { Component, inject, computed } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { toSignal } from '@angular/core/rxjs-interop';
import { map } from 'rxjs';

@Component({ ... })
export class ProductComponent {
  private route = inject(ActivatedRoute);

  // Convert paramMap observable to a signal
  private params = toSignal(this.route.paramMap);

  // Derive individual parameters as signals
  readonly productId = computed(() => this.params()?.get('id') ?? '');

  // Convert route data observable to a signal
  private routeData = toSignal(this.route.data);
  readonly breadcrumb = computed(() => this.routeData()?.['breadcrumb'] ?? '');
}
```

This is useful when you need signal reactivity but cannot use `withComponentInputBinding()` (for example, in a library component that should not depend on router configuration).

## Building Reusable Router Signal Utilities

Custom inject functions can encapsulate common patterns:

### `injectRouteParam`

```typescript
import { assertInInjectionContext, inject, computed } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { toSignal } from '@angular/core/rxjs-interop';

export function injectRouteParam(name: string) {
  assertInInjectionContext(injectRouteParam);
  const route = inject(ActivatedRoute);
  const params = toSignal(route.paramMap);
  return computed(() => params()?.get(name) ?? null);
}

// Usage
@Component({ ... })
export class ProductComponent {
  productId = injectRouteParam('id');
  // productId is Signal<string | null>
}
```

### `injectRouteParams` (All Parameters)

```typescript
export function injectRouteParams() {
  assertInInjectionContext(injectRouteParams);
  const route = inject(ActivatedRoute);
  const params = toSignal(route.params, { initialValue: {} });
  return params;
}
```

### `injectRouteData`

```typescript
export function injectRouteData<T = unknown>(key: string) {
  assertInInjectionContext(injectRouteData);
  const route = inject(ActivatedRoute);
  const data = toSignal(route.data, { initialValue: {} });
  return computed(() => data()[key] as T);
}
```

## Resolver Data Binding

Resolvers load data before a route activates. With `withComponentInputBinding()`, resolver results bind to inputs automatically:

```typescript
// Resolver
const productResolver: ResolveFn<Product> = (route) => {
  const productService = inject(ProductService);
  const id = route.paramMap.get('id')!;
  return productService.getById(id);
};

// Route
{
  path: 'products/:id',
  component: ProductDetailComponent,
  resolve: { product: productResolver },
}

// Component
@Component({
  template: `
    <h1>{{ product().name }}</h1>
    <p>{{ product().description }}</p>
  `,
})
export class ProductDetailComponent {
  // The key 'product' in the resolve map matches this input name
  product = input.required<Product>();
}
```

The resolver runs before the component is created, so the input is guaranteed to have a value on initialization.

## Common Mistakes

### Mistake 1: Forgetting `withComponentInputBinding()` in Router Config

```typescript
// BAD: Input bindings silently do not work
provideRouter(routes);

@Component({ ... })
export class DetailComponent {
  id = input<string>();  // Always undefined!
}

// GOOD: Enable the feature
provideRouter(routes, withComponentInputBinding());
```

There is no error or warning. The inputs simply remain undefined.

### Mistake 2: Input Name Does Not Match Route Parameter Name

```typescript
// Route: { path: 'products/:productId', ... }

// BAD: Input name 'id' does not match route param 'productId'
id = input<string>();  // undefined

// GOOD: Name must match exactly
productId = input<string>();  // '42'

// OR: Use alias
id = input<string>({ alias: 'productId' });  // '42'
```

### Mistake 3: Expecting Numbers from Route Parameters

```typescript
// BAD: Route params are ALWAYS strings
id = input.required<number>();  // Actually receives '42' (string), not 42

// GOOD: Transform the value
id = input.required({ transform: numberAttribute });  // 42 (number)

// OR: Custom transform
id = input.required({ transform: (v: string) => parseInt(v, 10) });
```

### Mistake 4: Using Snapshot When Parameters Change

```typescript
// BAD: Does not update when navigating /products/42 -> /products/99
ngOnInit() {
  const id = this.route.snapshot.paramMap.get('id');
  this.loadProduct(id!);
}

// GOOD with signal input: automatically reactive
id = input.required<string>();
product = resource({
  request: () => this.id(),
  loader: ({ request: id }) => this.productService.getById(id),
});

// GOOD with ActivatedRoute observable: subscribe to changes
ngOnInit() {
  this.route.paramMap.subscribe(params => {
    this.loadProduct(params.get('id')!);
  });
}
```

### Mistake 5: Not Setting `paramsInheritanceStrategy: 'always'`

```typescript
// Route tree:
// /users/:userId/orders/:orderId

// BAD: Default 'emptyOnly' strategy -- child only sees :orderId
@Component({ ... })
export class OrderDetailComponent {
  userId = input<string>();   // undefined!
  orderId = input<string>();  // '100'
}

// FIX: Set paramsInheritanceStrategy in router config
provideRouter(
  routes,
  withComponentInputBinding(),
  withRouterConfig({ paramsInheritanceStrategy: 'always' }),
);
// Now userId = '5', orderId = '100'
```

### Mistake 6: Priority Confusion Between Data and Query Params

```typescript
// Route has data: { mode: 'edit' }
// URL has ?mode=view

// The input gets 'edit' (static data), NOT 'view' (query param)
mode = input<string>();  // 'edit'

// If you want the query param, either:
// 1. Rename the static data key to avoid conflict
// 2. Use ActivatedRoute.queryParamMap directly
```

## Best Practices

### 1. Always Enable `withComponentInputBinding()`

There is no performance cost and it eliminates boilerplate. Enable it in every application.

### 2. Use Signal Inputs for Route Parameters

Signal inputs with `withComponentInputBinding()` are reactive by default. No subscriptions, no cleanup, no lifecycle hooks needed.

### 3. Set `paramsInheritanceStrategy: 'always'`

Unless you have a specific reason to restrict parameter inheritance, `'always'` avoids confusion in nested routes.

### 4. Transform Parameters at the Input Level

Do not scatter `parseInt()` and `toLowerCase()` calls throughout your component. Use the `transform` option on the input declaration.

### 5. Use `input.required()` for Mandatory Parameters

If a route always has a `:id` parameter, declare it as `input.required<string>()`. This makes the contract explicit and prevents undefined checks.

### 6. Use `resource()` for Data Loading Based on Parameters

The `resource()` API automatically refetches when its `request` signal (the route parameter) changes, replacing the manual `switchMap` pattern.

## Real-World Use Cases

### Product Detail Page with Related Products

```typescript
@Component({
  template: `
    @if (product(); as prod) {
      <article>
        <h1>{{ prod.name }}</h1>
        <p class="price">{{ prod.price | currency }}</p>
        <div [innerHTML]="prod.description"></div>
      </article>

      <section>
        <h2>Related Products</h2>
        @for (related of relatedProducts(); track related.id) {
          <a [routerLink]="['/products', related.id]">
            {{ related.name }}
          </a>
        }
      </section>
    } @else if (isLoading()) {
      <app-spinner />
    }
  `,
})
export class ProductDetailComponent {
  private productService = inject(ProductService);

  // Route param automatically updates when clicking related products
  id = input.required({ transform: numberAttribute });

  private productResource = resource({
    request: () => this.id(),
    loader: async ({ request: id }) => this.productService.getById(id),
  });

  product = this.productResource.value;
  isLoading = this.productResource.isLoading;

  // Derived: load related products based on the current product
  private relatedResource = resource({
    request: () => this.product(),
    loader: async ({ request: product }) => {
      if (!product) return [];
      return this.productService.getRelated(product.category, product.id);
    },
  });

  relatedProducts = computed(() => this.relatedResource.value() ?? []);
}
```

Clicking a related product link navigates to `/products/99`, the `id` input updates from `42` to `99`, and both resources reload automatically.

### User Profile with Nested Tab Navigation

```typescript
// Route configuration
{
  path: 'users/:userId',
  component: UserProfileComponent,
  children: [
    { path: '', redirectTo: 'overview', pathMatch: 'full' },
    { path: 'overview', loadComponent: () => import('./user-overview.component') },
    { path: 'posts', loadComponent: () => import('./user-posts.component') },
    { path: 'settings', loadComponent: () => import('./user-settings.component') },
  ],
}
```

```typescript
// user-profile.component.ts
@Component({
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  template: `
    @if (user(); as u) {
      <header>
        <img [src]="u.avatar" [alt]="u.name" />
        <h1>{{ u.name }}</h1>
      </header>
      <nav>
        <a routerLink="overview" routerLinkActive="active">Overview</a>
        <a routerLink="posts" routerLinkActive="active">Posts</a>
        <a routerLink="settings" routerLinkActive="active">Settings</a>
      </nav>
      <router-outlet />
    }
  `,
})
export class UserProfileComponent {
  userId = input.required<string>();

  private userResource = resource({
    request: () => this.userId(),
    loader: async ({ request: id }) => inject(UserService).getUser(id),
  });

  user = this.userResource.value;
}
```

```typescript
// user-posts.component.ts (child route)
// With paramsInheritanceStrategy: 'always', this child inherits userId
@Component({
  template: `
    @for (post of posts(); track post.id) {
      <article>
        <h3>{{ post.title }}</h3>
        <p>{{ post.excerpt }}</p>
      </article>
    }
  `,
})
export class UserPostsComponent {
  // Inherited from parent route via paramsInheritanceStrategy: 'always'
  userId = input.required<string>();

  private postsResource = resource({
    request: () => this.userId(),
    loader: async ({ request: userId }) => inject(PostService).getByUser(userId),
  });

  posts = computed(() => this.postsResource.value() ?? []);
}
```

> **Key Takeaway:** Enable `withComponentInputBinding()` and use signal inputs to receive route parameters. Combined with `paramsInheritanceStrategy: 'always'` and the `transform` option, this eliminates nearly all `ActivatedRoute` boilerplate and gives you reactive, type-safe parameter access with zero subscriptions.
