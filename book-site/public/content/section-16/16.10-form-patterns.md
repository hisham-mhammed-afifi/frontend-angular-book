# 16.10 Form Patterns (Multi-Step, Nested, Autosave)

> Architectural patterns for building real-world forms: wizard flows, composable sub-forms, auto-saving drafts, and dynamic configuration-driven forms.

---

## Multi-Step Wizard Forms

Multi-step forms (wizards) break a large form into sequential steps. Each step collects a subset of data, with navigation between steps and validation per step.

### Architecture

The pattern has three parts:

1. **Container component**: Manages the overall form state, current step, and navigation
2. **Step components**: Each step is a standalone component rendering its portion of the form
3. **Shared FormGroup**: A single `FormGroup` holds all data, with each step working on a sub-group

### Implementation with Reactive Forms

```typescript
interface RegistrationData {
  // Step 1: Account
  account: {
    email: string;
    password: string;
    confirmPassword: string;
  };
  // Step 2: Profile
  profile: {
    firstName: string;
    lastName: string;
    dateOfBirth: string;
    phone: string;
  };
  // Step 3: Preferences
  preferences: {
    newsletter: boolean;
    theme: 'light' | 'dark';
    language: string;
  };
}
```

**Container component:**

```typescript
@Component({
  selector: 'app-registration-wizard',
  standalone: true,
  changeDetection: ChangeDetectionStrategy.OnPush,
  imports: [ReactiveFormsModule, AccountStepComponent, ProfileStepComponent, PreferencesStepComponent],
  template: `
    <div class="wizard">
      <nav class="steps-indicator">
        @for (step of steps; track step.index; let i = $index) {
          <button
            [class.active]="currentStep() === i"
            [class.completed]="i < currentStep()"
            [disabled]="i > currentStep()"
            (click)="goToStep(i)">
            {{ step.label }}
          </button>
        }
      </nav>

      <form [formGroup]="form" (ngSubmit)="onSubmit()">
        @switch (currentStep()) {
          @case (0) {
            <app-account-step [group]="accountGroup" />
          }
          @case (1) {
            <app-profile-step [group]="profileGroup" />
          }
          @case (2) {
            <app-preferences-step [group]="preferencesGroup" />
          }
        }

        <div class="wizard-actions">
          @if (currentStep() > 0) {
            <button type="button" (click)="previous()">Back</button>
          }
          @if (currentStep() < steps.length - 1) {
            <button type="button" (click)="next()">Next</button>
          } @else {
            <button type="submit" [disabled]="form.invalid">Submit</button>
          }
        </div>
      </form>
    </div>
  `,
})
export class RegistrationWizardComponent {
  private fb = inject(FormBuilder);

  steps = [
    { index: 0, label: 'Account' },
    { index: 1, label: 'Profile' },
    { index: 2, label: 'Preferences' },
  ];

  currentStep = signal(0);

  form = this.fb.group({
    account: this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(8)]],
      confirmPassword: ['', Validators.required],
    }),
    profile: this.fb.group({
      firstName: ['', Validators.required],
      lastName: ['', Validators.required],
      dateOfBirth: [''],
      phone: [''],
    }),
    preferences: this.fb.group({
      newsletter: [false],
      theme: ['light' as 'light' | 'dark'],
      language: ['en'],
    }),
  });

  get accountGroup(): FormGroup {
    return this.form.get('account') as FormGroup;
  }

  get profileGroup(): FormGroup {
    return this.form.get('profile') as FormGroup;
  }

  get preferencesGroup(): FormGroup {
    return this.form.get('preferences') as FormGroup;
  }

  next(): void {
    const currentGroup = this.getCurrentStepGroup();
    currentGroup.markAllAsTouched();
    if (currentGroup.valid) {
      this.currentStep.update(s => s + 1);
    }
  }

  previous(): void {
    this.currentStep.update(s => s - 1);
  }

  goToStep(index: number): void {
    // Only allow going back or to completed steps
    if (index <= this.currentStep()) {
      this.currentStep.set(index);
    }
  }

  onSubmit(): void {
    if (this.form.valid) {
      const data = this.form.getRawValue();
      console.log('Registration data:', data);
    }
  }

  private getCurrentStepGroup(): FormGroup {
    const keys = ['account', 'profile', 'preferences'];
    return this.form.get(keys[this.currentStep()]) as FormGroup;
  }
}
```

**Step component:**

### Multi-Step with Signal Forms

```typescript
@Component({
  selector: 'app-signal-wizard',
  standalone: true,
  imports: [FormField],
  template: `
    @switch (step()) {
      @case (0) {
        <input type="email" [formField]="regForm.account.email" placeholder="Email" />
        <input type="password" [formField]="regForm.account.password" placeholder="Password" />
      }
      @case (1) {
        <input [formField]="regForm.profile.firstName" placeholder="First Name" />
        <input [formField]="regForm.profile.lastName" placeholder="Last Name" />
      }
    }
    <button (click)="nextStep()">Next</button>
  `,
})
export class SignalWizardComponent {
  step = signal(0);

  model = signal({
    account: { email: '', password: '' },
    profile: { firstName: '', lastName: '' },
  });

  regForm = form(this.model, (path) => {
    required(path.account.email);
    email(path.account.email);
    required(path.account.password);
    minLength(path.account.password, 8);
    required(path.profile.firstName);
    required(path.profile.lastName);
  });

  nextStep(): void {
    this.step.update(s => s + 1);
  }
}
```

Signal Forms validate all fields in the schema regardless of which step is visible, but `hidden()` can be used to skip validation on non-visible steps if desired.

### Router-Based Wizard

For complex wizards, each step can be a separate route with a shared service holding form state:

```typescript
// wizard-state.service.ts
@Injectable()
export class WizardStateService {
  private form = inject(FormBuilder).group({
    step1: inject(FormBuilder).group({ name: ['', Validators.required] }),
    step2: inject(FormBuilder).group({ address: ['', Validators.required] }),
    step3: inject(FormBuilder).group({ payment: ['', Validators.required] }),
  });

  getStepGroup(step: number): FormGroup {
    const keys = ['step1', 'step2', 'step3'];
    return this.form.get(keys[step]) as FormGroup;
  }

  getFormValue(): any {
    return this.form.getRawValue();
  }

  isStepValid(step: number): boolean {
    return this.getStepGroup(step).valid;
  }
}

// Routes
export const wizardRoutes: Routes = [
  {
    path: 'wizard',
    providers: [WizardStateService], // Scoped to the wizard
    children: [
      { path: '', redirectTo: 'step-1', pathMatch: 'full' },
      { path: 'step-1', component: Step1Component },
      { path: 'step-2', component: Step2Component, canActivate: [stepGuard(1)] },
      { path: 'step-3', component: Step3Component, canActivate: [stepGuard(2)] },
    ],
  },
];

// Guard to prevent skipping steps
function stepGuard(requiredStep: number): CanActivateFn {
  return () => {
    const wizardState = inject(WizardStateService);
    return wizardState.isStepValid(requiredStep - 1);
  };
}
```

Each step component injects `WizardStateService` and binds to its sub-group. Navigation uses `router.navigate()`. The `providers` array on the parent route scopes the service to the wizard lifetime.

## Nested / Composable Sub-Forms

Reusable form fragments that can be plugged into any parent form. There are three approaches, each with different trade-offs.

### Approach 1: ControlContainer (Simplest)

The child component injects the parent's `FormGroup` via `ControlContainer`:

```typescript
@Component({
  selector: 'app-address-fields',
  standalone: true,
  imports: [ReactiveFormsModule],
  viewProviders: [
    {
      provide: ControlContainer,
      useExisting: FormGroupDirective,
    },
  ],
  template: `
    <div class="address-fields">
      <input formControlName="street" placeholder="Street" />
      <input formControlName="city" placeholder="City" />
      <input formControlName="state" placeholder="State" />
      <input formControlName="zip" placeholder="ZIP" />
    </div>
  `,
})
export class AddressFieldsComponent {}
```

**Parent usage:**

```typescript
@Component({
  selector: 'app-order-form',
  standalone: true,
  imports: [ReactiveFormsModule, AddressFieldsComponent],
  template: `
    <form [formGroup]="form">
      <h3>Shipping Address</h3>
      <div formGroupName="shipping">
        <app-address-fields />
      </div>

      <h3>Billing Address</h3>
      <div formGroupName="billing">
        <app-address-fields />
      </div>
    </form>
  `,
})
export class OrderFormComponent {
  form = inject(FormBuilder).group({
    shipping: inject(FormBuilder).group({
      street: ['', Validators.required],
      city: ['', Validators.required],
      state: [''],
      zip: ['', Validators.required],
    }),
    billing: inject(FormBuilder).group({
      street: ['', Validators.required],
      city: ['', Validators.required],
      state: [''],
      zip: ['', Validators.required],
    }),
  });
}
```

The `viewProviders` with `ControlContainer` tells the child to look up the form hierarchy to find its parent `FormGroup`. The `formGroupName="shipping"` wrapper in the parent scopes the child's `formControlName` directives to that sub-group.

**Pros:** Minimal boilerplate. The child component has no inputs and no form setup.
**Cons:** Tightly coupled to the parent's form structure. The parent must define the controls. Cannot be used standalone.

### Approach 2: Input-Bound FormGroup

Pass the sub-group as an input:

```typescript
@Component({
  selector: 'app-address-form',
  standalone: true,
  changeDetection: ChangeDetectionStrategy.OnPush,
  imports: [ReactiveFormsModule],
  template: `
    <fieldset [formGroup]="group()">
      <legend>{{ legend() }}</legend>
      <input formControlName="street" placeholder="Street" />
      <input formControlName="city" placeholder="City" />
      <input formControlName="state" placeholder="State" />
      <input formControlName="zip" placeholder="ZIP" />
    </fieldset>
  `,
})
export class AddressFormComponent {
  group = input.required<FormGroup>();
  legend = input('Address');
}
```

**Parent:**

```typescript
template: `
  <form [formGroup]="form">
    <app-address-form [group]="shippingGroup" legend="Shipping" />
    <app-address-form [group]="billingGroup" legend="Billing" />
  </form>
`
```

**Pros:** Explicit data flow. The child knows what it receives.
**Cons:** The parent still owns the `FormGroup` definition. Child cannot create its own controls.

### Approach 3: ControlValueAccessor (Most Encapsulated)

The child owns its own `FormGroup` and exposes a single value to the parent:

```typescript
@Component({
  selector: 'app-address-control',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <div [formGroup]="innerForm">
      <input formControlName="street" placeholder="Street" />
      <input formControlName="city" placeholder="City" />
      <input formControlName="state" placeholder="State" />
      <input formControlName="zip" placeholder="ZIP" />
    </div>
  `,
  providers: [
    {
      provide: NG_VALUE_ACCESSOR,
      useExisting: forwardRef(() => AddressControlComponent),
      multi: true,
    },
  ],
})
export class AddressControlComponent implements ControlValueAccessor, OnInit, OnDestroy {
  private destroyRef = inject(DestroyRef);

  innerForm = inject(FormBuilder).nonNullable.group({
    street: [''],
    city: [''],
    state: [''],
    zip: [''],
  });

  private onChange: (value: Address) => void = () => {};
  private onTouched: () => void = () => {};

  ngOnInit(): void {
    this.innerForm.valueChanges
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe(value => {
        this.onChange(value as Address);
      });
  }

  writeValue(value: Address | null): void {
    if (value) {
      this.innerForm.patchValue(value, { emitEvent: false });
    } else {
      this.innerForm.reset({}, { emitEvent: false });
    }
  }

  registerOnChange(fn: (value: Address) => void): void {
    this.onChange = fn;
  }

  registerOnTouched(fn: () => void): void {
    this.onTouched = fn;
  }

  setDisabledState(isDisabled: boolean): void {
    isDisabled ? this.innerForm.disable() : this.innerForm.enable();
  }

  ngOnDestroy(): void {}
}
```

**Parent usage:**

```typescript
@Component({
  template: `
    <form [formGroup]="form">
      <app-address-control formControlName="shippingAddress" />
      <app-address-control formControlName="billingAddress" />
    </form>
  `,
})
export class OrderComponent {
  form = inject(FormBuilder).group({
    shippingAddress: [{ street: '', city: '', state: '', zip: '' }],
    billingAddress: [{ street: '', city: '', state: '', zip: '' }],
  });
}
```

**Pros:** Fully encapsulated. The child owns its controls, validation, and structure. Parent treats it as a single value. Reusable anywhere.
**Cons:** More boilerplate. The parent cannot access individual sub-fields for fine-grained validation display.

### Comparison Table

| Feature | ControlContainer | Input FormGroup | CVA |
|---------|-----------------|-----------------|-----|
| Boilerplate | Low | Low | High |
| Encapsulation | None | Low | Full |
| Parent defines controls | Yes | Yes | No |
| Reusable standalone | No | Partially | Yes |
| Works with `formControlName` | No | No | Yes |
| Sub-field access from parent | Yes | Yes | No |
| Signal Forms equivalent | `FieldTree` input | `FieldTree` input | `FormValueControl` |

### Signal Forms: FieldTree Input

Signal Forms simplify nested forms by passing a branch of the field tree:

```typescript
@Component({

  standalone: true,
  imports: [FormField],
  template: `

    <input [formField]="address.city" placeholder="City" />
    <input [formField]="address.state" placeholder="State" />
    <input [formField]="address.zip" placeholder="ZIP" />

  <app-address-fields [address]="orderForm.shipping" />
  <app-address-fields [address]="orderForm.billing" />
`
```

No `ControlContainer`, no `viewProviders`, no CVA. Just pass the field tree branch as an input.

## Autosave Pattern

Automatically persist form data as the user types, with debouncing, error handling, and status indication.

### Basic Autosave

```typescript
@Component({
  selector: 'app-document-editor',
  standalone: true,
  changeDetection: ChangeDetectionStrategy.OnPush,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="form">
      <input formControlName="title" placeholder="Document title" />
      <textarea formControlName="content" rows="20"></textarea>
    </form>

    <div class="save-status">
      @switch (saveStatus()) {
        @case ('saved') { <span class="saved">All changes saved</span> }
        @case ('saving') { <span class="saving">Saving...</span> }
        @case ('unsaved') { <span class="unsaved">Unsaved changes</span> }
        @case ('error') { <span class="error">Save failed. Retrying...</span> }
      }
    </div>
  `,
})
export class DocumentEditorComponent implements OnInit {
  private destroyRef = inject(DestroyRef);
  private documentService = inject(DocumentService);
  private documentId = input.required<string>();

  saveStatus = signal<'saved' | 'saving' | 'unsaved' | 'error'>('saved');

  form = inject(FormBuilder).group({
    title: [''],
    content: [''],
  });

  ngOnInit(): void {
    // Load existing document
    this.documentService.get(this.documentId()).subscribe(doc => {
      this.form.patchValue(doc, { emitEvent: false }); // Don't trigger autosave
    });

    // Autosave pipeline
    this.form.valueChanges
      .pipe(
        tap(() => this.saveStatus.set('unsaved')),
        debounceTime(1500), // Wait 1.5s after last change
        tap(() => this.saveStatus.set('saving')),
        switchMap(value =>
          this.documentService.save(this.documentId(), value).pipe(
            retry({ count: 2, delay: 1000 }), // Retry twice on failure
            catchError(err => {
              this.saveStatus.set('error');
              return EMPTY;
            }),
          )
        ),
        takeUntilDestroyed(this.destroyRef),
      )
      .subscribe(() => {
        this.saveStatus.set('saved');
      });
  }
}
```

Key details:
- `emitEvent: false` when loading data prevents the load from triggering a save
- `debounceTime(1500)` waits until the user stops typing
- `switchMap` cancels in-flight save requests when new changes arrive
- `retry` handles transient network errors
- The status signal provides real-time feedback

### Autosave with Dirty Checking

Only save when the form has actually changed from the last saved state:

```typescript
ngOnInit(): void {
  let lastSavedValue = JSON.stringify(this.form.getRawValue());

  this.form.valueChanges
    .pipe(
      debounceTime(1500),
      filter(() => {
        const currentValue = JSON.stringify(this.form.getRawValue());
        if (currentValue === lastSavedValue) {
          return false; // No actual change, skip save
        }
        return true;
      }),
      tap(() => this.saveStatus.set('saving')),
      switchMap(value =>
        this.documentService.save(this.documentId(), value).pipe(
          tap(() => {
            lastSavedValue = JSON.stringify(value);
          }),
          catchError(() => {
            this.saveStatus.set('error');
            return EMPTY;
          }),
        )
      ),
      takeUntilDestroyed(this.destroyRef),
    )
    .subscribe(() => {
      this.saveStatus.set('saved');
    });
}
```

### Autosave with Local Storage Fallback

Persist drafts locally for offline support:

```typescript
@Injectable({ providedIn: 'root' })
export class DraftService {
  private prefix = 'draft_';

  saveDraft(key: string, value: unknown): void {
    try {
      localStorage.setItem(
        this.prefix + key,
        JSON.stringify({ value, timestamp: Date.now() })
      );
    } catch {
      // Storage full or unavailable
    }
  }

  loadDraft<T>(key: string): { value: T; timestamp: number } | null {
    const raw = localStorage.getItem(this.prefix + key);
    if (!raw) return null;
    try {
      return JSON.parse(raw);
    } catch {
      return null;
    }
  }

  clearDraft(key: string): void {
    localStorage.removeItem(this.prefix + key);
  }
}
```

Usage in a component:

```typescript
ngOnInit(): void {
  const draftService = inject(DraftService);
  const draftKey = `document-${this.documentId()}`;

  // Check for existing draft
  const draft = draftService.loadDraft<DocumentData>(draftKey);
  if (draft) {
    const serverData = await firstValueFrom(this.documentService.get(this.documentId()));
    // Use draft if it's newer than the server version
    if (draft.timestamp > serverData.updatedAt) {
      this.form.patchValue(draft.value, { emitEvent: false });
      this.saveStatus.set('unsaved');
    }
  }

  // Save to local storage immediately, debounce server saves
  this.form.valueChanges
    .pipe(takeUntilDestroyed(this.destroyRef))
    .subscribe(value => {
      draftService.saveDraft(draftKey, value);
    });

  // Server save with debounce
  this.form.valueChanges
    .pipe(
      debounceTime(2000),
      switchMap(value =>
        this.documentService.save(this.documentId(), value)
      ),
      takeUntilDestroyed(this.destroyRef),
    )
    .subscribe(() => {
      draftService.clearDraft(draftKey); // Clear local draft after server save
      this.saveStatus.set('saved');
    });
}
```

### Preventing Navigation with Unsaved Changes

Guard against leaving with unsaved data:

```typescript
// unsaved-changes.guard.ts
export interface HasUnsavedChanges {
  hasUnsavedChanges(): boolean;
}

export const unsavedChangesGuard: CanDeactivateFn<HasUnsavedChanges> = (component) => {
  if (component.hasUnsavedChanges()) {
    return confirm('You have unsaved changes. Leave anyway?');
  }
  return true;
};

// Route config
{
  path: 'editor/:id',
  component: DocumentEditorComponent,
  canDeactivate: [unsavedChangesGuard],
}

// Component implements the interface
export class DocumentEditorComponent implements HasUnsavedChanges {
  hasUnsavedChanges(): boolean {
    return this.form.dirty && this.saveStatus() !== 'saved';
  }
}
```

## Dynamic Configuration-Driven Forms

Generate forms from a configuration object or API response. Useful for admin panels, survey builders, and forms that change based on user roles or feature flags.

### Configuration Schema

```typescript
interface FormFieldConfig {
  key: string;
  label: string;
  type: 'text' | 'email' | 'number' | 'select' | 'checkbox' | 'textarea' | 'date';
  defaultValue?: any;
  validators?: {
    required?: boolean;
    minLength?: number;
    maxLength?: number;
    min?: number;
    max?: number;
    pattern?: string;
    email?: boolean;
  };
  options?: { value: string; label: string }[]; // For select
  placeholder?: string;
  helpText?: string;
  visible?: boolean;
}

interface FormConfig {
  title: string;
  fields: FormFieldConfig[];
  submitLabel?: string;
}
```

### Dynamic Form Component

```typescript
@Component({
  selector: 'app-dynamic-form',
  standalone: true,
  changeDetection: ChangeDetectionStrategy.OnPush,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="formGroup" (ngSubmit)="onSubmit()">
      @for (field of config().fields; track field.key) {
        @if (field.visible !== false) {
          <div class="form-field">
            <label [for]="field.key">{{ field.label }}</label>

            @switch (field.type) {
              @case ('text') {
                <input
                  [id]="field.key"
                  [formControlName]="field.key"
                  [placeholder]="field.placeholder ?? ''" />
              }
              @case ('email') {
                <input
                  [id]="field.key"
                  type="email"
                  [formControlName]="field.key"
                  [placeholder]="field.placeholder ?? ''" />
              }
              @case ('number') {
                <input
                  [id]="field.key"
                  type="number"
                  [formControlName]="field.key" />
              }
              @case ('textarea') {
                <textarea
                  [id]="field.key"
                  [formControlName]="field.key"
                  rows="4">
                </textarea>
              }
              @case ('select') {
                <select [id]="field.key" [formControlName]="field.key">
                  <option value="">Select...</option>
                  @for (opt of field.options; track opt.value) {
                    <option [value]="opt.value">{{ opt.label }}</option>
                  }
                </select>
              }
              @case ('checkbox') {
                <input
                  [id]="field.key"
                  type="checkbox"
                  [formControlName]="field.key" />
              }
              @case ('date') {
                <input
                  [id]="field.key"
                  type="date"
                  [formControlName]="field.key" />
              }
            }

            @if (field.helpText) {
              <small class="help">{{ field.helpText }}</small>
            }

            @if (formGroup.get(field.key)?.touched && formGroup.get(field.key)?.invalid) {
              <div class="errors">
                @if (formGroup.get(field.key)?.hasError('required')) {
                  <span>{{ field.label }} is required</span>
                }
                @if (formGroup.get(field.key)?.hasError('minlength')) {
                  <span>Minimum {{ field.validators?.minLength }} characters</span>
                }
                @if (formGroup.get(field.key)?.hasError('email')) {
                  <span>Invalid email</span>
                }
              </div>
            }
          </div>
        }
      }

      <button type="submit" [disabled]="formGroup.invalid">
        {{ config().submitLabel ?? 'Submit' }}
      </button>
    </form>
  `,
})
export class DynamicFormComponent implements OnInit {
  config = input.required<FormConfig>();
  submitted = output<Record<string, any>>();

  formGroup!: FormGroup;

  ngOnInit(): void {
    this.formGroup = this.buildForm(this.config());
  }

  private buildForm(config: FormConfig): FormGroup {
    const controls: Record<string, FormControl> = {};

    for (const field of config.fields) {
      const validators = this.buildValidators(field);
      controls[field.key] = new FormControl(
        field.defaultValue ?? (field.type === 'checkbox' ? false : ''),
        validators
      );
    }

    return new FormGroup(controls);
  }

  private buildValidators(field: FormFieldConfig): ValidatorFn[] {
    const validators: ValidatorFn[] = [];
    const v = field.validators;
    if (!v) return validators;

    if (v.required) validators.push(Validators.required);
    if (v.minLength) validators.push(Validators.minLength(v.minLength));
    if (v.maxLength) validators.push(Validators.maxLength(v.maxLength));
    if (v.min !== undefined) validators.push(Validators.min(v.min));
    if (v.max !== undefined) validators.push(Validators.max(v.max));
    if (v.pattern) validators.push(Validators.pattern(v.pattern));
    if (v.email) validators.push(Validators.email);

    return validators;
  }

  onSubmit(): void {
    if (this.formGroup.valid) {
      this.submitted.emit(this.formGroup.getRawValue());
    }
  }
}
```

## Copy/Mirror Pattern

Duplicate values between form sections, like "billing address same as shipping":

```typescript
@Component({
  selector: 'app-checkout',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="form">
      <h3>Shipping</h3>
      <div formGroupName="shipping">
        <input formControlName="street" placeholder="Street" />
        <input formControlName="city" placeholder="City" />
        <input formControlName="zip" placeholder="ZIP" />
      </div>

      <label>
        <input type="checkbox" [formControl]="sameAsShipping" />
        Billing same as shipping
      </label>

      <h3>Billing</h3>
      <div formGroupName="billing" [class.disabled]="sameAsShipping.value">
        <input formControlName="street" placeholder="Street" />
        <input formControlName="city" placeholder="City" />
        <input formControlName="zip" placeholder="ZIP" />
      </div>
    </form>
  `,
})
export class CheckoutComponent implements OnInit {
  private destroyRef = inject(DestroyRef);

  sameAsShipping = new FormControl(false);

  form = inject(FormBuilder).group({
    shipping: inject(FormBuilder).group({
      street: ['', Validators.required],
      city: ['', Validators.required],
      zip: ['', Validators.required],
    }),
    billing: inject(FormBuilder).group({
      street: ['', Validators.required],
      city: ['', Validators.required],
      zip: ['', Validators.required],
    }),
  });

  ngOnInit(): void {
    const shipping = this.form.get('shipping')!;
    const billing = this.form.get('billing')!;

    // When checkbox changes
    this.sameAsShipping.valueChanges
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe(same => {
        if (same) {
          billing.patchValue(shipping.value);
          billing.disable();
        } else {
          billing.enable();
        }
      });

    // When shipping changes while checkbox is checked, mirror to billing
    shipping.valueChanges
      .pipe(
        filter(() => !!this.sameAsShipping.value),
        takeUntilDestroyed(this.destroyRef),
      )
      .subscribe(value => {
        billing.patchValue(value, { emitEvent: false });
      });
  }
}
```

Disabling the billing group when mirroring has two benefits: it prevents user edits and `getRawValue()` still includes the billing data for submission.

## Conditional Fields Pattern

Show or hide fields based on other field values:

```typescript
@Component({
  selector: 'app-contact-form',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="form">
      <select formControlName="contactMethod">
        <option value="email">Email</option>
        <option value="phone">Phone</option>
        <option value="mail">Postal Mail</option>
      </select>

      @switch (form.get('contactMethod')?.value) {
        @case ('email') {
          <input formControlName="email" type="email" placeholder="Email address" />
        }
        @case ('phone') {
          <input formControlName="phone" type="tel" placeholder="Phone number" />
        }
        @case ('mail') {
          <div formGroupName="address">
            <input formControlName="street" placeholder="Street" />
            <input formControlName="city" placeholder="City" />
            <input formControlName="zip" placeholder="ZIP" />
          </div>
        }
      }
    </form>
  `,
})
export class ContactFormComponent implements OnInit {
  private destroyRef = inject(DestroyRef);

  form = inject(FormBuilder).group({
    contactMethod: ['email'],
    email: [''],
    phone: [''],
    address: inject(FormBuilder).group({
      street: [''],
      city: [''],
      zip: [''],
    }),
  });

  ngOnInit(): void {
    this.form.get('contactMethod')!.valueChanges
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe(method => this.updateValidation(method!));

    // Set initial validation
    this.updateValidation('email');
  }

  private updateValidation(method: string): void {
    const email = this.form.get('email')!;
    const phone = this.form.get('phone')!;
    const address = this.form.get('address')!;

    // Clear all validators first
    email.clearValidators();
    phone.clearValidators();
    address.get('street')!.clearValidators();
    address.get('city')!.clearValidators();

    // Apply validators based on selected method
    switch (method) {
      case 'email':
        email.setValidators([Validators.required, Validators.email]);
        break;
      case 'phone':
        phone.setValidators([Validators.required, Validators.pattern(/^\+?\d{7,15}$/)]);
        break;
      case 'mail':
        address.get('street')!.setValidators(Validators.required);
        address.get('city')!.setValidators(Validators.required);
        break;
    }

    // Recalculate validity
    email.updateValueAndValidity({ emitEvent: false });
    phone.updateValueAndValidity({ emitEvent: false });
    address.get('street')!.updateValueAndValidity({ emitEvent: false });
    address.get('city')!.updateValueAndValidity({ emitEvent: false });
  }
}
```

With Signal Forms, this is much cleaner using `applyWhen`:

```typescript

    required(path.email);
    email(path.email);
  });
  applyWhen(path, (ctx) => ctx.valueOf(path.contactMethod) === 'phone', (path) => {
    required(path.phone);
  });
  applyWhen(path, (ctx) => ctx.valueOf(path.contactMethod) === 'mail', (path) => {

  });
```

## Common Mistakes

### Mistake 1: Building one massive FormGroup in a single component

Forms with 20+ fields in one component are hard to maintain and slow to render:

```typescript
// BAD: 30 fields, all validators, all logic in one component
form = this.fb.group({
  // ... 30 controls with validators
});
// Template: 200+ lines of inputs and error messages

// GOOD: split into sub-components
// AccountFormComponent, AddressFormComponent, PaymentFormComponent
// Each ~5-7 fields, own validation display, own OnPush strategy
```

### Mistake 2: Not using emitEvent: false when loading data into a form with autosave

```typescript
// BAD: loading triggers autosave, which saves the just-loaded data
this.form.patchValue(serverData);
// The autosave pipeline sees a valueChanges emission and saves

// GOOD: suppress the event
this.form.patchValue(serverData, { emitEvent: false });
```

### Mistake 3: Losing form state on step navigation in a wizard

```typescript
// BAD: using *ngIf to conditionally render steps
// This destroys and recreates components, losing their state

// GOOD: use @switch (which keeps the FormGroup alive in the parent)
// Or use [hidden] to hide steps without destroying them
// Or keep a single FormGroup that persists across all steps
```

### Mistake 4: Forgetting to guard against navigation with unsaved changes

Users lose work when they accidentally navigate away. Always add a `canDeactivate` guard for forms with autosave.

### Mistake 5: Not disabling the submit button during async save

```typescript
// BAD: user can double-click submit
<button type="submit">Save</button>

// GOOD: disable during save
<button type="submit" [disabled]="saving()">
  {{ saving() ? 'Saving...' : 'Save' }}
</button>
```

## Best Practices

### 1. One FormGroup per page, split into sub-components

The parent owns the `FormGroup`, child components receive their sub-group. This maintains a single source of truth while keeping templates manageable.

### 2. Validate on step transition in wizards

Do not rely on the submit button alone. Validate each step when the user clicks "Next":

```typescript
next(): void {
  const currentGroup = this.getStepGroup(this.currentStep());
  currentGroup.markAllAsTouched();
  if (currentGroup.valid) {
    this.currentStep.update(s => s + 1);
  }
}
```

### 3. Use switchMap (not mergeMap) for autosave

`switchMap` cancels in-flight saves when new changes arrive. `mergeMap` would allow overlapping saves, potentially writing stale data.

### 4. Debounce autosave at 1-3 seconds

Less than 1 second causes too many network requests. More than 5 seconds risks data loss if the user closes the tab.

### 5. Store drafts locally for resilience

Use `localStorage` or `IndexedDB` as a first-pass save, with server sync as the durable save. This protects against network failures and accidental tab closures.

### 6. Use Signal Forms' `hidden()` for conditional fields

Instead of manually toggling validators with `setValidators`/`clearValidators`, use `hidden()` in the schema. Hidden fields skip validation automatically.

### 2. CMS Content Editor with Autosave

A blog post editor that autosaves every 2 seconds, persists drafts locally, shows save status, and warns before navigation:

```typescript
@Component({
  selector: 'app-post-editor',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="postForm">
      <input formControlName="title" placeholder="Post title" class="title-input" />
      <textarea formControlName="body" rows="30" class="body-editor"></textarea>

      <div class="toolbar">
        <span class="status">{{ saveStatus() }}</span>
        <select formControlName="status">
          <option value="draft">Draft</option>
          <option value="published">Published</option>
        </select>
      </div>
    </form>
  `,
})
export class PostEditorComponent implements OnInit, HasUnsavedChanges {
  private destroyRef = inject(DestroyRef);
  private postService = inject(PostService);
  private draftService = inject(DraftService);
  postId = input.required<string>();

  saveStatus = signal('Saved');

  postForm = inject(FormBuilder).group({
    title: ['', Validators.required],
    body: [''],
    status: ['draft' as 'draft' | 'published'],
  });

  hasUnsavedChanges(): boolean {
    return this.postForm.dirty && this.saveStatus() !== 'Saved';
  }

  ngOnInit(): void {
    const key = `post-${this.postId()}`;

    // Load from server, check for local draft
    this.postService.get(this.postId()).subscribe(post => {
      const draft = this.draftService.loadDraft<any>(key);
      const data = draft && draft.timestamp > post.updatedAt ? draft.value : post;
      this.postForm.patchValue(data, { emitEvent: false });
      if (draft && draft.timestamp > post.updatedAt) {
        this.saveStatus.set('Restored from draft');
      }
    });

    // Local draft save (immediate)
    this.postForm.valueChanges
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe(value => this.draftService.saveDraft(key, value));

    // Server save (debounced)
    this.postForm.valueChanges
      .pipe(
        tap(() => this.saveStatus.set('Unsaved changes')),
        debounceTime(2000),
        tap(() => this.saveStatus.set('Saving...')),
        switchMap(value =>
          this.postService.update(this.postId(), value).pipe(
            catchError(() => {
              this.saveStatus.set('Save failed');
              return EMPTY;
            }),
          )
        ),
        takeUntilDestroyed(this.destroyRef),
      )
      .subscribe(() => {
        this.saveStatus.set('Saved');
        this.postForm.markAsPristine();
        this.draftService.clearDraft(key);
      });
  }
}
```

### 3. Admin Config Form with Server-Driven Fields

An admin panel where the form fields are defined by the API, allowing non-developers to configure what data is collected:

```typescript
@Component({
  selector: 'app-admin-form-builder',
  standalone: true,
  imports: [DynamicFormComponent],

    @if (loading()) {
```
