# 17.3 Custom Pipes

> Custom pipes let you encapsulate reusable transformation logic in a declarative, testable unit that integrates directly with Angular's template engine and change detection system.

---

## Creating a Custom Pipe

A pipe is a class decorated with `@Pipe` that implements the `PipeTransform` interface. In Angular 21, pipes are standalone by default.

```typescript
import { Pipe, PipeTransform } from '@angular/core';

@Pipe({
  name: 'truncate',
  standalone: true
})
export class TruncatePipe implements PipeTransform {
  transform(value: string, limit: number = 50, trail: string = '...'): string {
    if (!value) return '';
    if (value.length <= limit) return value;
    return value.substring(0, limit) + trail;
  }
}
```

Usage in a component:

```typescript
import { Component } from '@angular/core';
import { TruncatePipe } from './truncate.pipe';

@Component({
  selector: 'app-article',
  standalone: true,
  imports: [TruncatePipe],
  template: `
    <p>{{ article.body | truncate:200 }}</p>
    <p>{{ article.summary | truncate:100:'---' }}</p>
  `
})
export class ArticleComponent {
  article = { body: '...', summary: '...' };
}
```

### Generating a Pipe with the CLI

```bash
ng generate pipe pipes/truncate
# Creates:
#   src/app/pipes/truncate.pipe.ts
#   src/app/pipes/truncate.pipe.spec.ts
```

The generated pipe is standalone by default in Angular 21.

## The @Pipe Decorator

The `@Pipe` decorator accepts a configuration object:

```typescript
@Pipe({
  name: 'myPipe',       // Required: the name used in templates
  standalone: true,      // Default in Angular 21; set false for NgModule pipes
  pure: true             // Default: true. Set false for impure pipes
})
```

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `name` | `string` | (required) | Template identifier used after `\|` |
| `standalone` | `boolean` | `true` | Whether the pipe can be imported directly |
| `pure` | `boolean` | `true` | Whether Angular uses reference-based change detection |

### Pipe Naming Conventions

- The `name` property uses **camelCase**: `'fileSize'`, `'timeAgo'`, `'safeHtml'`
- The class name uses **PascalCase** with a `Pipe` suffix: `FileSizePipe`, `TimeAgoPipe`, `SafeHtmlPipe`
- The file name uses **kebab-case**: `file-size.pipe.ts`, `time-ago.pipe.ts`

```typescript
// BAD: Inconsistent naming
@Pipe({ name: 'FileSize' })     // Should be camelCase
export class fileSizePipe {}     // Should be PascalCase with Pipe suffix

// GOOD: Consistent naming
@Pipe({ name: 'fileSize' })
export class FileSizePipe implements PipeTransform { ... }
```

## The PipeTransform Interface

Every pipe must implement `PipeTransform`, which requires a single `transform()` method:

```typescript
interface PipeTransform {
  transform(value: any, ...args: any[]): any;
}
```

The first argument is always the piped value. Additional arguments correspond to colon-separated parameters in the template.

### Typed Transform Methods

Always type your `transform()` method precisely. Avoid `any`:

```typescript
// BAD: Untyped
transform(value: any, ...args: any[]): any {
  return value.toUpperCase();
}

// GOOD: Fully typed
transform(value: string | null, maxLength: number = 100): string {
  if (!value) return '';
  return value.length > maxLength ? value.slice(0, maxLength) : value;
}
```

### Handling Null and Undefined

Templates often pass `null` or `undefined` to pipes, especially when data has not loaded yet. Always handle these cases:

```typescript
@Pipe({ name: 'initials', standalone: true })
export class InitialsPipe implements PipeTransform {
  transform(value: string | null | undefined): string {
    if (!value) return '';
    return value
      .split(' ')
      .map(word => word.charAt(0).toUpperCase())
      .join('');
  }
}
```

```html
{{ user?.fullName | initials }}
<!-- Outputs '' if user is null, 'JD' if 'John Doe' -->
```

## Injecting Services into Pipes

Pipes support dependency injection through the `inject()` function. This allows pipes to access services, configuration, or other injectables.

```typescript
import { Pipe, PipeTransform, inject } from '@angular/core';
import { TranslationService } from '../services/translation.service';

@Pipe({ name: 'translate', standalone: true })
export class TranslatePipe implements PipeTransform {
  private translationService = inject(TranslationService);

  transform(key: string, params?: Record<string, string>): string {
    return this.translationService.translate(key, params);
  }
}
```

```html
{{ 'greeting.hello' | translate }}
{{ 'items.count' | translate:countParams }}
```

### Injecting LOCALE_ID

```typescript
import { Pipe, PipeTransform, inject, LOCALE_ID } from '@angular/core';

@Pipe({ name: 'relativeTime', standalone: true })
export class RelativeTimePipe implements PipeTransform {
  private locale = inject(LOCALE_ID);

  transform(value: Date | string | number): string {
    const date = new Date(value);
    const rtf = new Intl.RelativeTimeFormat(this.locale, { numeric: 'auto' });
    const diff = date.getTime() - Date.now();
    const seconds = Math.round(diff / 1000);

    if (Math.abs(seconds) < 60) return rtf.format(seconds, 'second');
    const minutes = Math.round(seconds / 60);
    if (Math.abs(minutes) < 60) return rtf.format(minutes, 'minute');
    const hours = Math.round(minutes / 60);
    if (Math.abs(hours) < 24) return rtf.format(hours, 'hour');
    const days = Math.round(hours / 24);
    return rtf.format(days, 'day');
  }
}
```

```html
{{ comment.createdAt | relativeTime }}
<!-- Output: "3 hours ago", "yesterday", "in 2 days" -->
```

### Injecting DomSanitizer

```typescript
import { Pipe, PipeTransform, inject } from '@angular/core';
import { DomSanitizer, SafeHtml } from '@angular/platform-browser';

@Pipe({ name: 'safeHtml', standalone: true })
export class SafeHtmlPipe implements PipeTransform {
  private sanitizer = inject(DomSanitizer);

  transform(value: string | null): SafeHtml {
    if (!value) return '';
    return this.sanitizer.bypassSecurityTrustHtml(value);
  }
}
```

```html
<div [innerHTML]="richContent | safeHtml"></div>
```

**Warning:** `bypassSecurityTrustHtml` does not sanitize content. It bypasses Angular's sanitization entirely. Only use this with content you fully trust or have sanitized yourself (e.g., with DOMPurify). Using it with user-supplied content enables XSS attacks.

## Generic Pipes

Pipes can use TypeScript generics to preserve type information through the transformation:

```typescript
@Pipe({ name: 'defaultValue', standalone: true })
export class DefaultValuePipe implements PipeTransform {
  transform<T>(value: T | null | undefined, fallback: T): T {
    return value ?? fallback;
  }
}
```

```html
{{ user?.name | defaultValue:'Anonymous' }}
{{ user?.age | defaultValue:0 }}
```

### Typed Array Filtering

```typescript
@Pipe({ name: 'filterBy', standalone: true })
export class FilterByPipe implements PipeTransform {
  transform<T>(items: T[] | null, property: keyof T, value: unknown): T[] {
    if (!items) return [];
    return items.filter(item => item[property] === value);
  }
}
```

```html
@for (user of users | filterBy:'role':'admin'; track user.id) {
  <div>{{ user.name }}</div>
}
```

**Note:** `FilterByPipe` must be impure if the array is mutated in place rather than replaced, because pure pipes only detect reference changes. However, the Angular team explicitly discourages filter and sort pipes because impure pipes run every change detection cycle. The recommended approach is to filter in the component class using `computed()` signals:

```typescript
// Preferred: computed signal instead of filter pipe
role = signal('admin');
filteredUsers = computed(() =>
  this.users().filter(u => u.role === this.role())
);
```

## Common Custom Pipe Patterns

### File Size Pipe

```typescript
@Pipe({ name: 'fileSize', standalone: true })
export class FileSizePipe implements PipeTransform {
  private units = ['B', 'KB', 'MB', 'GB', 'TB'];

  transform(bytes: number | null, decimals: number = 1): string {
    if (bytes == null || bytes === 0) return '0 B';

    const isNegative = bytes < 0;
    bytes = Math.abs(bytes);

    const k = 1024;
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    const size = bytes / Math.pow(k, i);

    const formatted = size.toFixed(decimals);
    return `${isNegative ? '-' : ''}${formatted} ${this.units[i]}`;
  }
}
```

```html
{{ file.size | fileSize }}       <!-- "2.4 MB" -->
{{ file.size | fileSize:2 }}     <!-- "2.38 MB" -->
{{ 1024 | fileSize }}            <!-- "1.0 KB" -->
{{ 0 | fileSize }}               <!-- "0 B" -->
```

### Time Ago Pipe

```typescript
@Pipe({ name: 'timeAgo', standalone: true })
export class TimeAgoPipe implements PipeTransform {
  transform(value: Date | string | number | null): string {
    if (!value) return '';

    const date = new Date(value);
    const now = Date.now();
    const seconds = Math.floor((now - date.getTime()) / 1000);

    if (seconds < 0) return 'just now';
    if (seconds < 60) return `${seconds}s ago`;

    const minutes = Math.floor(seconds / 60);
    if (minutes < 60) return `${minutes}m ago`;

    const hours = Math.floor(minutes / 60);
    if (hours < 24) return `${hours}h ago`;

    const days = Math.floor(hours / 24);
    if (days < 30) return `${days}d ago`;

    const months = Math.floor(days / 30);
    if (months < 12) return `${months}mo ago`;

    const years = Math.floor(months / 12);
    return `${years}y ago`;
  }
}
```

```html
{{ post.createdAt | timeAgo }}
<!-- "5m ago", "3h ago", "2d ago" -->
```

**Important:** This pipe must be impure (`pure: false`) or paired with a timer to update as time passes. A pure pipe only re-evaluates when the input reference changes, and a timestamp does not change. See the impure approach below:

```typescript
@Pipe({ name: 'timeAgo', standalone: true, pure: false })
export class TimeAgoPipe implements PipeTransform {
  // Same transform logic as above
}
```

Or better, use a signal-based approach in the component:

```typescript
// Component approach (no impure pipe needed)
timeAgo = computed(() => {
  const now = this.ticker(); // a signal updated every minute
  return this.formatTimeAgo(this.post().createdAt, now);
});
```

### Pluralize Pipe

```typescript
@Pipe({ name: 'pluralize', standalone: true })
export class PluralizePipe implements PipeTransform {
  transform(
    count: number | null,
    singular: string,
    plural?: string
  ): string {
    if (count == null) return '';
    const word = count === 1 ? singular : (plural ?? singular + 's');
    return `${count} ${word}`;
  }
}
```

```html
{{ commentCount | pluralize:'comment' }}
<!-- 0 -> "0 comments", 1 -> "1 comment", 5 -> "5 comments" -->

{{ childCount | pluralize:'child':'children' }}
<!-- 1 -> "1 child", 3 -> "3 children" -->
```

### Excerpt Pipe (Smart Truncation)

```typescript
@Pipe({ name: 'excerpt', standalone: true })
export class ExcerptPipe implements PipeTransform {
  transform(
    value: string | null,
    maxLength: number = 150,
    boundary: 'word' | 'char' = 'word'
  ): string {
    if (!value) return '';
    if (value.length <= maxLength) return value;

    if (boundary === 'char') {
      return value.substring(0, maxLength).trim() + '...';
    }

    // Word boundary: find the last space before maxLength
    const truncated = value.substring(0, maxLength);
    const lastSpace = truncated.lastIndexOf(' ');
    if (lastSpace === -1) return truncated + '...';
    return truncated.substring(0, lastSpace).trim() + '...';
  }
}
```

```html
{{ article.body | excerpt:200:'word' }}
<!-- Truncates at the last word boundary before 200 chars -->
```

### Ordinal Pipe

```typescript
@Pipe({ name: 'ordinal', standalone: true })
export class OrdinalPipe implements PipeTransform {
  transform(value: number | null): string {
    if (value == null) return '';

    const abs = Math.abs(value);
    const lastDigit = abs % 10;
    const lastTwoDigits = abs % 100;

    if (lastTwoDigits >= 11 && lastTwoDigits <= 13) {
      return `${value}th`;
    }

    switch (lastDigit) {
      case 1: return `${value}st`;
      case 2: return `${value}nd`;
      case 3: return `${value}rd`;
      default: return `${value}th`;
    }
  }
}
```

```html
{{ position | ordinal }}
<!-- 1 -> "1st", 2 -> "2nd", 3 -> "3rd", 11 -> "11th", 21 -> "21st" -->
```

### Mask Pipe (for sensitive data)

```typescript
@Pipe({ name: 'mask', standalone: true })
export class MaskPipe implements PipeTransform {
  transform(
    value: string | null,
    visibleChars: number = 4,
    maskChar: string = '*'
  ): string {
    if (!value) return '';
    if (value.length <= visibleChars) return value;

    const visible = value.slice(-visibleChars);
    const masked = maskChar.repeat(value.length - visibleChars);
    return masked + visible;
  }
}
```

```html
{{ user.email | mask:5 }}
<!-- "john.doe@example.com" -> "***************e.com" -->

{{ user.phone | mask:4 }}
<!-- "+1234567890" -> "*******7890" -->

{{ creditCard | mask:4:'X' }}
<!-- "4111111111111111" -> "XXXXXXXXXXXX1111" -->
```

### Join Pipe

```typescript
@Pipe({ name: 'join', standalone: true })
export class JoinPipe implements PipeTransform {
  transform(value: string[] | null, separator: string = ', '): string {
    if (!value || value.length === 0) return '';
    return value.join(separator);
  }
}
```

```html
{{ user.roles | join }}
<!-- "admin, editor, viewer" -->

{{ tags | join:' | ' }}
<!-- "angular | typescript | frontend" -->
```

### Highlight Search Term Pipe

```typescript
@Pipe({ name: 'highlight', standalone: true })
export class HighlightPipe implements PipeTransform {
  transform(text: string | null, term: string | null): string {
    if (!text || !term || term.length === 0) return text ?? '';

    // Escape special regex characters in the search term
    const escaped = term.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const regex = new RegExp(`(${escaped})`, 'gi');
    return text.replace(regex, '<mark>$1</mark>');
  }
}
```

```html
<span [innerHTML]="result.title | highlight:searchQuery"></span>
```

## Testing Custom Pipes

Pipes are plain classes, making them straightforward to unit test. No `TestBed` is required for pure pipes without dependencies.

### Testing a Simple Pipe

```typescript
import { TruncatePipe } from './truncate.pipe';

describe('TruncatePipe', () => {
  let pipe: TruncatePipe;

  beforeEach(() => {
    pipe = new TruncatePipe();
  });

  it('should return empty string for null', () => {
    expect(pipe.transform(null)).toBe('');
  });

  it('should return the original string if under limit', () => {
    expect(pipe.transform('short', 10)).toBe('short');
  });

  it('should truncate and add trail', () => {
    expect(pipe.transform('hello world', 5)).toBe('hello...');
  });

  it('should use custom trail', () => {
    expect(pipe.transform('hello world', 5, '---')).toBe('hello---');
  });

  it('should use default limit of 50', () => {
    const longText = 'a'.repeat(100);
    const result = pipe.transform(longText);
    expect(result.length).toBe(53); // 50 chars + '...'
  });
});
```

### Testing a Pipe with Dependencies

When a pipe injects services, use `TestBed`:

```typescript
import { TestBed } from '@angular/core/testing';
import { LOCALE_ID } from '@angular/core';
import { RelativeTimePipe } from './relative-time.pipe';

describe('RelativeTimePipe', () => {
  let pipe: RelativeTimePipe;

  beforeEach(() => {
    TestBed.configureTestingModule({
      providers: [
        RelativeTimePipe,
        { provide: LOCALE_ID, useValue: 'en-US' }
      ]
    });
    pipe = TestBed.inject(RelativeTimePipe);
  });

  it('should show "just now" for recent dates', () => {
    const now = new Date();
    expect(pipe.transform(now)).toContain('second');
  });

  it('should show hours for dates within a day', () => {
    const threeHoursAgo = new Date(Date.now() - 3 * 60 * 60 * 1000);
    expect(pipe.transform(threeHoursAgo)).toContain('3 hours ago');
  });
});
```

### Testing a Pipe with a Mocked Service

```typescript
import { TestBed } from '@angular/core/testing';
import { TranslatePipe } from './translate.pipe';
import { TranslationService } from '../services/translation.service';

describe('TranslatePipe', () => {
  let pipe: TranslatePipe;
  let mockTranslationService: jasmine.SpyObj<TranslationService>;

  beforeEach(() => {
    mockTranslationService = jasmine.createSpyObj('TranslationService', ['translate']);

    TestBed.configureTestingModule({
      providers: [
        TranslatePipe,
        { provide: TranslationService, useValue: mockTranslationService }
      ]
    });

    pipe = TestBed.inject(TranslatePipe);
  });

  it('should call translation service with key', () => {
    mockTranslationService.translate.and.returnValue('Hello');
    expect(pipe.transform('greeting.hello')).toBe('Hello');
    expect(mockTranslationService.translate).toHaveBeenCalledWith('greeting.hello', undefined);
  });

  it('should pass params to translation service', () => {
    mockTranslationService.translate.and.returnValue('Hello, Alice');
    const params = { name: 'Alice' };
    pipe.transform('greeting.personal', params);
    expect(mockTranslationService.translate).toHaveBeenCalledWith('greeting.personal', params);
  });
});
```

## Pipe vs Computed Signal vs Method

When should you create a custom pipe versus using a `computed()` signal or a method?

| Approach | Reusability | Performance | Testability | Template Readability |
|----------|-------------|-------------|-------------|---------------------|
| Custom Pipe | High (importable) | Pure pipes are cached | Easy (plain class) | Very readable |
| `computed()` | Component-scoped | Signal-based caching | Requires component | Clean, no pipe syntax |
| Method | Component-scoped | No caching, runs every CD | Easy | Can clutter template |

**Use a custom pipe when:**
- The transformation is generic and reusable across components
- It takes a value and returns a transformed value (pure function)
- You want declarative template syntax

**Use a `computed()` signal when:**
- The transformation is specific to one component
- It depends on multiple reactive values (not just the template value)
- You need to compose transformations from several sources

**Use a method only when:**
- The logic is trivial and not performance-sensitive
- It is used in a single place

```typescript
// Pipe: reusable, cached, declarative
{{ price | fileSize }}

// Computed: component-specific, reactive
formattedSize = computed(() => this.formatFileSize(this.file().size));

// Method: avoid in templates if possible (runs every CD cycle)
{{ formatFileSize(file.size) }}
```

## Advanced: Pipe with State

While pipes are typically stateless, some use cases require state. These must be impure (`pure: false`):

```typescript
@Pipe({ name: 'debounceDisplay', standalone: true, pure: false })
export class DebounceDisplayPipe implements PipeTransform {
  private lastValue: string = '';
  private timeout: ReturnType<typeof setTimeout> | null = null;
  private displayValue: string = '';

  transform(value: string, delayMs: number = 300): string {
    if (value !== this.lastValue) {
      this.lastValue = value;
      if (this.timeout) clearTimeout(this.timeout);
      this.timeout = setTimeout(() => {
        this.displayValue = value;
      }, delayMs);
    }
    return this.displayValue;
  }
}
```

**Warning:** Stateful impure pipes run on every change detection cycle. They create performance overhead and should be used sparingly. In most cases, a signal-based approach in the component is better.

## Common Mistakes

- **Not handling null/undefined inputs.** Pipes receive `null` when data has not loaded or when chained after `AsyncPipe`. Always check for falsy values at the top of `transform()`.

- **Using `any` for the transform signature.** Untyped pipes hide bugs. Always provide explicit types for the value, parameters, and return type.

- **Creating impure pipes for filtering or sorting arrays.** Impure pipes run every change detection cycle. Use `computed()` signals instead for array transformations.

- **Forgetting to import the pipe in the component.** Standalone pipes must be added to the component's `imports` array. Otherwise, Angular throws a template error.

- **Naming the pipe the same as a built-in pipe.** If you name a custom pipe `'date'` or `'number'`, it shadows the built-in pipe within that component. Use unique, descriptive names.

- **Creating a pipe that mutates the input value.** Pipes should be pure functions that return a new value. Never modify the original array, object, or string passed to the pipe.

- **Using `bypassSecurityTrustHtml` without proper sanitization.** The `safeHtml` pipe pattern is common but dangerous. Only use it with content from trusted sources. Consider sanitizing with DOMPurify before bypassing Angular's sanitizer.

- **Calling `inject()` inside `transform()`.** The `inject()` function can only be called during construction (field initializers or constructor body). Calling it inside `transform()` throws a runtime error. Always assign injected dependencies as class fields.

---

## Best Practices

- **Keep pipes as pure functions.** The `transform()` method should depend only on its arguments and injected services, not on external mutable state.

- **Use `inject()` instead of constructor injection.** This is the preferred pattern in Angular 21 and works consistently in pipes, components, and services.

- **Make pipes standalone.** This is the default in Angular 21. Standalone pipes are tree-shakable and can be imported individually.

- **Write focused, single-purpose pipes.** A pipe should do one thing well. Create separate pipes rather than one pipe with mode parameters.

- **Test pipes as plain classes.** For pipes without dependencies, instantiate directly with `new`. For pipes with `inject()`, use `TestBed`.

- **Prefer computed signals over impure pipes.** If a pipe needs to be impure (for mutable data, timers, or state), reconsider whether a `computed()` or `effect()` in the component would be cleaner.

- **Export pipes from shared libraries** with barrel files for clean imports across your application.

## Real-World Use Cases

### File Upload Progress Display

```typescript
@Pipe({ name: 'uploadStatus', standalone: true })
export class UploadStatusPipe implements PipeTransform {
  transform(file: UploadFile): string {
    switch (file.status) {
      case 'pending': return 'Waiting...';
      case 'uploading': return `Uploading ${file.progress}%`;
      case 'complete': return 'Uploaded';
      case 'error': return `Failed: ${file.error}`;
      default: return '';
    }
  }
}

@Component({
  standalone: true,
  imports: [FileSizePipe, UploadStatusPipe],
  template: `
    @for (file of files(); track file.id) {
      <div class="upload-item">
        <span class="name">{{ file.name }}</span>
        <span class="size">{{ file.size | fileSize }}</span>
        <span class="status" [class]="file.status">
          {{ file | uploadStatus }}
        </span>
      </div>
    }
  `
})
export class FileUploadListComponent {
  files = input.required<UploadFile[]>();
}
```

### Configurable Number Abbreviation

```typescript
@Pipe({ name: 'abbreviateNumber', standalone: true })
export class AbbreviateNumberPipe implements PipeTransform {
  private suffixes = [
    { threshold: 1_000_000_000, suffix: 'B' },
    { threshold: 1_000_000, suffix: 'M' },
    { threshold: 1_000, suffix: 'K' }
  ];

  transform(value: number | null, decimals: number = 1): string {
    if (value == null) return '';
    if (value === 0) return '0';

    const isNegative = value < 0;
    const abs = Math.abs(value);

    for (const { threshold, suffix } of this.suffixes) {
      if (abs >= threshold) {
        const abbreviated = (abs / threshold).toFixed(decimals);
        return `${isNegative ? '-' : ''}${abbreviated}${suffix}`;
      }
    }

    return value.toString();
  }
}
```

```html
{{ followers | abbreviateNumber }}     <!-- 1500 -> "1.5K" -->
{{ revenue | abbreviateNumber:2 }}     <!-- 2500000 -> "2.50M" -->
{{ views | abbreviateNumber:0 }}       <!-- 3700 -> "4K" -->
```

### URL Slug Generator

```typescript
@Pipe({ name: 'slugify', standalone: true })
export class SlugifyPipe implements PipeTransform {
  transform(value: string | null): string {
    if (!value) return '';
    return value
      .toLowerCase()
      .normalize('NFD')                          // decompose accents
      .replace(/[\u0300-\u036f]/g, '')           // remove accent marks
      .replace(/[^a-z0-9\s-]/g, '')              // remove non-alphanumeric
      .replace(/[\s_]+/g, '-')                   // spaces/underscores to hyphens
      .replace(/-+/g, '-')                       // collapse multiple hyphens
      .replace(/^-|-$/g, '');                    // trim leading/trailing hyphens
  }
}
```

```html
<a [routerLink]="'/blog/' + (post.title | slugify)">
  {{ post.title }}
</a>
<!-- "Hello World! It's Angular" -> "/blog/hello-world-its-angular" -->
```

> **Key Takeaway:** Custom pipes encapsulate reusable, testable transformations with clean template syntax. Keep them pure, typed, null-safe, and standalone. For transformations that depend on mutable state or multiple reactive sources, prefer `computed()` signals over impure pipes.
