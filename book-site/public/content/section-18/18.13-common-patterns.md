# 18.13 Common Patterns (Autocomplete, Polling, Retry with Backoff)

> Battle-tested RxJS patterns that solve recurring problems in real Angular applications, from typeahead search to resilient HTTP retries.

## Pattern 1: Autocomplete / Typeahead Search

The autocomplete pattern is the canonical RxJS example. It combines debouncing, deduplication, cancellation, and error recovery.

### The Problem

A search input fires a request on every keystroke. Without proper handling:
- Typing "angular" fires 7 HTTP requests
- Responses arrive out of order (race condition)
- Empty queries waste bandwidth
- Rapid typing creates unnecessary server load

### The Solution

```typescript
@Component({
  selector: 'app-search',
  template: `
    <input
      [ngModel]="query()"
      (ngModelChange)="query.set($event)"
      placeholder="Search..."
    />
    @if (loading()) { <span class="spinner"></span> }
    @for (result of results(); track result.id) {
      <div class="result" (click)="select(result)">
        <strong>{{ result.name }}</strong>
        <span>{{ result.category }}</span>
      </div>
    }
    @if (results().length === 0 && query().length >= 2 && !loading()) {
      <div class="no-results">No results found</div>
    }
  `
})
export class SearchComponent {
  private readonly http = inject(HttpClient);

  readonly query = signal('');
  readonly loading = signal(false);

  readonly results = toSignal(
    toObservable(this.query).pipe(
      debounceTime(300),           // Wait 300ms after last keystroke
      distinctUntilChanged(),      // Skip if query hasn't changed
      tap(q => this.loading.set(q.length >= 2)),
      switchMap(query => {         // Cancel previous request
        if (query.length < 2) return of([]);
        return this.http.get<SearchResult[]>(`/api/search`, {
          params: { q: query }
        }).pipe(
          catchError(() => of([])), // Graceful error handling
          finalize(() => this.loading.set(false))
        );
      })
    ),
    { initialValue: [] }
  );

  select(result: SearchResult) {
    // Handle selection
  }
}
```

### Why Each Operator Matters

| Operator | Purpose | Without It |
|----------|---------|------------|
| `debounceTime(300)` | Waits for typing pause | 7 requests for "angular" |
| `distinctUntilChanged()` | Skips duplicate queries | Re-fires on same value after backspace |
| `switchMap()` | Cancels previous request | Race condition, stale results displayed |
| `catchError()` | Handles API errors | Entire stream dies on first error |
| `filter` / length check | Skips short queries | Searches for "a", "ab" |

### Edge Cases

**1. Minimum character threshold with clearing:**

```typescript
// When the user clears the input, immediately show empty results
// Don't wait for debounce
toObservable(this.query).pipe(
  switchMap(query => {
    if (query.length < 2) {
      return of([]); // Immediate clear, no debounce
    }
    return of(query).pipe(
      debounceTime(300),
      switchMap(q => this.http.get<SearchResult[]>(`/api/search?q=${q}`))
    );
  })
)
```

**2. Highlighting search terms in results:**

```typescript
readonly highlightedResults = computed(() => {
  const q = this.query().toLowerCase();
  return this.results().map(r => ({
    ...r,
    highlightedName: r.name.replace(
      new RegExp(`(${this.escapeRegex(q)})`, 'gi'),
      '<mark>$1</mark>'
    )
  }));
});
```

**3. Keyboard navigation:**

```typescript
readonly selectedIndex = signal(-1);

constructor() {
  // Reset selection when results change
  effect(() => {
    this.results(); // Track results
    this.selectedIndex.set(-1);
  });
}

onKeyDown(event: KeyboardEvent) {
  const results = this.results();
  switch (event.key) {
    case 'ArrowDown':
      event.preventDefault();
      this.selectedIndex.update(i => Math.min(i + 1, results.length - 1));
      break;
    case 'ArrowUp':
      event.preventDefault();
      this.selectedIndex.update(i => Math.max(i - 1, 0));
      break;
    case 'Enter':
      if (this.selectedIndex() >= 0) {
        this.select(results[this.selectedIndex()]);
      }
      break;
  }
}
```

## Pattern 2: Polling

Polling periodically fetches data from a server. Common for dashboards, status pages, and monitoring UIs.

### Basic Polling

```typescript
@Component({
  selector: 'app-status-dashboard',
  template: `
    <div class="status">
      <span>Server Status: {{ status() }}</span>
      <span>Last updated: {{ lastUpdated() | date:'HH:mm:ss' }}</span>
    </div>
  `
})
export class StatusDashboardComponent {
  private readonly http = inject(HttpClient);

  readonly lastUpdated = signal(new Date());

  readonly status = toSignal(
    interval(10_000).pipe(        // Every 10 seconds
      startWith(0),               // Trigger immediately on subscribe
      switchMap(() =>
        this.http.get<ServerStatus>('/api/status').pipe(
          catchError(() => of({ status: 'unknown' } as ServerStatus))
        )
      ),
      tap(() => this.lastUpdated.set(new Date())),
      map(res => res.status)
    ),
    { initialValue: 'loading' }
  );
}
```

### Smart Polling: Pause When Tab Is Hidden

Polling while the user is on another tab wastes bandwidth. Use the Page Visibility API to pause:

```typescript
@Component({
  selector: 'app-smart-dashboard',
  template: `<div>{{ data() | json }}</div>`
})
export class SmartDashboardComponent {
  private readonly http = inject(HttpClient);

  // Observable that emits true when tab is visible, false when hidden
  private readonly visibility$ = fromEvent(document, 'visibilitychange').pipe(
    map(() => !document.hidden),
    startWith(!document.hidden) // Current state on subscribe
  );

  readonly data = toSignal(
    this.visibility$.pipe(
      switchMap(visible => {
        if (!visible) return EMPTY; // Stop polling when hidden
        return interval(10_000).pipe(
          startWith(0),
          switchMap(() => this.http.get<DashboardData>('/api/dashboard').pipe(
            catchError(() => EMPTY) // Skip failed polls
          ))
        );
      })
    ),
    { initialValue: null }
  );
}
```

### Adaptive Polling: Slow Down on Errors

```typescript
private readonly poll$ = defer(() => this.http.get<Data>('/api/data')).pipe(
  repeatWhen(completed$ =>
    completed$.pipe(delay(5_000)) // 5s between successful polls
  ),
  retryWhen(errors$ =>
    errors$.pipe(
      scan((retryCount) => retryCount + 1, 0),
      delayWhen(retryCount =>
        timer(Math.min(retryCount * 5_000, 60_000)) // Back off up to 60s
      )
    )
  )
);

// Modern equivalent using repeat() and retry() config (RxJS 7+):
readonly data = toSignal(
  this.http.get<Data>('/api/data').pipe(
    retry({
      count: Infinity,
      delay: (error, retryCount) =>
        timer(Math.min(retryCount * 5_000, 60_000))
    }),
    repeat({
      delay: 5_000 // Repeat 5s after completion
    })
  ),
  { initialValue: null }
);
```

### Polling with User-Triggered Refresh

```typescript
@Component({
  selector: 'app-orders',
  template: `
    <button (click)="refresh$.next()">Refresh</button>
    @for (order of orders(); track order.id) {
      <div>{{ order.number }} - {{ order.status }}</div>
    }
  `
})
export class OrdersComponent {
  private readonly http = inject(HttpClient);
  readonly refresh$ = new Subject<void>();

  readonly orders = toSignal(
    merge(
      interval(30_000),   // Auto-poll every 30s
      this.refresh$       // Manual refresh trigger
    ).pipe(
      startWith(0),
      switchMap(() => this.http.get<Order[]>('/api/orders').pipe(
        catchError(() => of([]))
      ))
    ),
    { initialValue: [] }
  );
}
```

### `switchMap` vs `exhaustMap` for Polling

```typescript
// switchMap: If the previous request is still in flight when the timer fires,
// cancel it and start a new one. Good when you want the freshest data.
interval(5000).pipe(
  switchMap(() => this.http.get('/api/data'))
);

// exhaustMap: If the previous request is still in flight, skip this poll.
// Good when the endpoint is slow and you don't want overlapping requests.
interval(5000).pipe(
  exhaustMap(() => this.http.get('/api/data'))
);
```

## Pattern 3: Retry with Exponential Backoff

When an HTTP request fails, retrying immediately often fails again (server is overloaded). Exponential backoff increases the delay between retries, and jitter prevents all clients from retrying simultaneously.

### Basic Retry (RxJS 7+ Config)

```typescript
this.http.get<Data>('/api/data').pipe(
  retry({
    count: 3,                    // Max 3 retries
    delay: 1_000                 // Fixed 1 second delay
  })
);
```

### Exponential Backoff

```typescript
this.http.get<Data>('/api/data').pipe(
  retry({
    count: 4,
    delay: (error, retryCount) => {
      // 1s, 2s, 4s, 8s
      const delayMs = Math.pow(2, retryCount - 1) * 1_000;
      console.log(`Retry ${retryCount} in ${delayMs}ms`);
      return timer(delayMs);
    }
  })
);
```

### Exponential Backoff with Jitter

Jitter adds randomness to prevent the "thundering herd" problem where all clients retry at exactly the same moment.

```typescript
this.http.get<Data>('/api/data').pipe(
  retry({
    count: 4,
    delay: (error, retryCount) => {
      const baseDelay = Math.pow(2, retryCount - 1) * 1_000;
      const jitter = Math.random() * 1_000; // 0-1000ms random jitter
      const delayMs = baseDelay + jitter;
      return timer(delayMs);
    }
  })
);
```

### Conditional Retry

Not all errors should be retried. A 404 or 422 will never succeed on retry. Only retry transient errors (5xx, network errors, timeouts).

```typescript
this.http.get<Data>('/api/data').pipe(
  retry({
    count: 3,
    delay: (error: HttpErrorResponse, retryCount) => {
      // Don't retry client errors (4xx)
      if (error.status >= 400 && error.status < 500) {
        throw error; // Immediately propagate
      }

      // Retry server errors (5xx) and network errors (status 0)
      const delayMs = Math.pow(2, retryCount - 1) * 1_000;
      const jitter = Math.random() * 500;
      return timer(delayMs + jitter);
    }
  })
);
```

### Reusable Retry Operator

```typescript
// utils/retry-backoff.ts
import { MonoTypeOperatorFunction, retry, timer } from 'rxjs';
import { HttpErrorResponse } from '@angular/common/http';

interface RetryBackoffConfig {
  maxRetries?: number;
  initialDelay?: number;
  maxDelay?: number;
  shouldRetry?: (error: any) => boolean;
}

export function retryBackoff<T>(config: RetryBackoffConfig = {}): MonoTypeOperatorFunction<T> {
  const {
    maxRetries = 3,
    initialDelay = 1_000,
    maxDelay = 30_000,
    shouldRetry = (err) => {
      if (err instanceof HttpErrorResponse) {
        return err.status === 0 || err.status >= 500;
      }
      return true;
    }
  } = config;

  return retry({
    count: maxRetries,
    delay: (error, retryCount) => {
      if (!shouldRetry(error)) {
        throw error;
      }
      const delay = Math.min(
        Math.pow(2, retryCount - 1) * initialDelay,
        maxDelay
      );
      const jitter = Math.random() * initialDelay;
      return timer(delay + jitter);
    }
  });
}

// Usage
this.http.get<Data>('/api/data').pipe(
  retryBackoff({ maxRetries: 5, initialDelay: 500 })
);
```

### Retry in an HTTP Interceptor

Apply retry logic globally via an interceptor:

```typescript
// interceptors/retry.interceptor.ts
import { HttpInterceptorFn } from '@angular/common/http';

export const retryInterceptor: HttpInterceptorFn = (req, next) => {
  // Only retry GET requests (other methods are not idempotent)
  if (req.method !== 'GET') {
    return next(req);
  }

  return next(req).pipe(
    retryBackoff({
      maxRetries: 2,
      initialDelay: 1_000,
      shouldRetry: (err) => err.status === 0 || err.status >= 500
    })
  );
};

// app.config.ts
export const appConfig: ApplicationConfig = {
  providers: [
    provideHttpClient(
      withInterceptors([retryInterceptor])
    )
  ]
};
```

## Pattern 4: Optimistic Updates

Show the result immediately in the UI, then reconcile with the server response. If the server call fails, revert.

### Basic Optimistic Update

```typescript
@Component({
  selector: 'app-todo-list',
  template: `
    @for (todo of todos(); track todo.id) {
      <div [class.completed]="todo.completed">
        <input
          type="checkbox"
          [checked]="todo.completed"
          (change)="toggleTodo(todo)"
        />
        {{ todo.title }}
      </div>
    }
  `
})
export class TodoListComponent {
  private readonly http = inject(HttpClient);
  private readonly snackbar = inject(MatSnackBar);

  readonly todos = signal<Todo[]>([]);

  constructor() {
    // Load initial data
    this.http.get<Todo[]>('/api/todos').pipe(
      takeUntilDestroyed()
    ).subscribe(todos => this.todos.set(todos));
  }

  toggleTodo(todo: Todo) {
    const previousState = this.todos();

    // Step 1: Optimistically update the UI
    this.todos.update(todos =>
      todos.map(t =>
        t.id === todo.id ? { ...t, completed: !t.completed } : t
      )
    );

    // Step 2: Send the actual request
    this.http.patch<Todo>(`/api/todos/${todo.id}`, {
      completed: !todo.completed
    }).pipe(
      catchError(error => {
        // Step 3: Revert on failure
        this.todos.set(previousState);
        this.snackbar.open('Failed to update. Reverted.', 'Dismiss', {
          duration: 3000
        });
        return EMPTY;
      }),
      takeUntilDestroyed(inject(DestroyRef))
    ).subscribe();
  }
}
```

### Optimistic Delete with Undo

```typescript
deleteItem(item: Item) {
  const previousItems = this.items();

  // Optimistically remove
  this.items.update(items => items.filter(i => i.id !== item.id));

  // Show undo snackbar
  const snackbarRef = this.snackbar.open('Item deleted', 'Undo', {
    duration: 5000
  });

  snackbarRef.afterDismissed().pipe(
    take(1),
    switchMap(result => {
      if (result.dismissedByAction) {
        // User clicked Undo: revert
        this.items.set(previousItems);
        return EMPTY;
      }
      // Snackbar timed out: commit the delete
      return this.http.delete(`/api/items/${item.id}`).pipe(
        catchError(() => {
          this.items.set(previousItems);
          this.snackbar.open('Delete failed. Restored.', 'Dismiss');
          return EMPTY;
        })
      );
    }),
    takeUntilDestroyed(this.destroyRef)
  ).subscribe();
}
```

## Pattern 5: Infinite Scroll / Load More

Load data incrementally as the user scrolls.

### Using Intersection Observer

```typescript
@Component({
  selector: 'app-feed',
  template: `
    @for (item of items(); track item.id) {
      <app-feed-card [item]="item" />
    }
    @if (loading()) {
      <div class="loading-spinner">Loading...</div>
    }
    @if (hasMore()) {
      <div #sentinel class="sentinel"></div>
    }
  `
})
export class FeedComponent implements AfterViewInit {
  private readonly http = inject(HttpClient);
  private readonly destroyRef = inject(DestroyRef);

  @ViewChild('sentinel') sentinel!: ElementRef;

  readonly items = signal<FeedItem[]>([]);
  readonly loading = signal(false);
  readonly hasMore = signal(true);
  private page = 0;

  ngAfterViewInit() {
    // Create an Observable from IntersectionObserver
    const sentinel$ = new Observable<boolean>(subscriber => {
      const observer = new IntersectionObserver(
        entries => subscriber.next(entries[0].isIntersecting),
        { threshold: 0.1 }
      );
      observer.observe(this.sentinel.nativeElement);
      return () => observer.disconnect();
    });

    sentinel$.pipe(
      filter(visible => visible),        // Only when sentinel is visible
      filter(() => !this.loading()),     // Not already loading
      filter(() => this.hasMore()),      // More data available
      exhaustMap(() => {                 // Ignore scroll events during fetch
        this.loading.set(true);
        this.page++;
        return this.http.get<FeedItem[]>('/api/feed', {
          params: { page: this.page.toString(), limit: '20' }
        }).pipe(
          catchError(() => of([])),
          finalize(() => this.loading.set(false))
        );
      }),
      takeUntilDestroyed(this.destroyRef)
    ).subscribe(newItems => {
      if (newItems.length < 20) {
        this.hasMore.set(false);
      }
      this.items.update(current => [...current, ...newItems]);
    });
  }
}
```

## Pattern 6: Race Condition Prevention

Race conditions occur when multiple async operations compete and the wrong one wins.

### Problem: Stale Response Displayed

```typescript
// BAD: If user clicks "Product A" then quickly "Product B",
// and Product A's response arrives AFTER Product B's,
// the UI shows Product A (stale data)
loadProduct(id: string) {
  this.http.get<Product>(`/api/products/${id}`).subscribe(
    product => this.product.set(product) // Which response arrives last?
  );
}
```

### Solution: `switchMap` Cancels Previous Requests

```typescript
// GOOD: switchMap cancels the previous request when a new one starts
private readonly productId$ = new Subject<string>();

readonly product = toSignal(
  this.productId$.pipe(
    switchMap(id => this.http.get<Product>(`/api/products/${id}`).pipe(
      catchError(() => EMPTY)
    ))
  )
);

loadProduct(id: string) {
  this.productId$.next(id); // Previous request auto-cancelled
}
```

### Solution: Signal-Based with `rxResource`

```typescript
readonly selectedId = signal<string | null>(null);

readonly product = rxResource({
  params: () => {
    const id = this.selectedId();
    return id ? { id } : undefined; // undefined skips the request
  },
  stream: ({ params }) =>
    this.http.get<Product>(`/api/products/${params.id}`)
});
```

### Operator Selection Guide for Race Conditions

| Scenario | Operator | Reason |
|----------|----------|--------|
| Live search / typeahead | `switchMap` | Cancel stale requests; only latest matters |
| Form submit / save button | `exhaustMap` | Ignore clicks while save is in progress |
| Ordered writes (queue) | `concatMap` | Process one at a time, in order, no drops |
| Independent parallel requests | `mergeMap` | Fire all concurrently (limit with second arg) |
| Navigation / route change | `switchMap` | Cancel previous route's data loading |
| Delete button | `concatMap` or `exhaustMap` | Never cancel a delete |

### The `switchMap` Save Button Bug

This is one of the most common bugs in Angular applications. The user clicks "Save", the request fires. They click again before the response arrives. `switchMap` cancels the first request. The data is never saved, but the second request succeeds and the user sees a success message, believing both saves completed.

```typescript
// BAD: switchMap cancels the first save on double-click
clicks$.pipe(
  switchMap(() => this.http.put('/api/save', data))
).subscribe(() => this.showSuccess());

// GOOD: exhaustMap ignores clicks while save is in progress
clicks$.pipe(
  exhaustMap(() => this.http.put('/api/save', data))
).subscribe(() => this.showSuccess());
```

### Preventing Double Submissions

```typescript
// exhaustMap ignores new clicks while a request is in flight
readonly save$ = new Subject<FormData>();

constructor() {
  this.save$.pipe(
    exhaustMap(data =>
      this.http.post('/api/save', data).pipe(
        catchError(err => {
          this.error.set(err.message);
          return EMPTY;
        })
      )
    ),
    takeUntilDestroyed()
  ).subscribe(result => {
    this.success.set(true);
  });
}

onSubmit(formData: FormData) {
  this.save$.next(formData);
}
```

## Pattern 7: Caching with Invalidation

### Time-Based Cache

```typescript
@Injectable({ providedIn: 'root' })
export class CachedDataService {
  private readonly http = inject(HttpClient);

  // Cache for 60 seconds after last subscriber leaves
  readonly products$ = this.http.get<Product[]>('/api/products').pipe(
    share({
      connector: () => new ReplaySubject<Product[]>(1),
      resetOnRefCountZero: () => timer(60_000),
      resetOnComplete: false,
      resetOnError: () => timer(5_000)
    })
  );
}
```

### Event-Based Invalidation

```typescript
@Injectable({ providedIn: 'root' })
export class UserService {
  private readonly http = inject(HttpClient);
  private readonly invalidate$ = new Subject<void>();

  readonly users$ = this.invalidate$.pipe(
    startWith(undefined),
    switchMap(() => this.http.get<User[]>('/api/users')),
    shareReplay(1)
  );

  invalidateCache() {
    this.invalidate$.next();
  }

  createUser(user: Partial<User>) {
    return this.http.post<User>('/api/users', user).pipe(
      tap(() => this.invalidateCache()) // Auto-invalidate after mutation
    );
  }
}
```

### Stale-While-Revalidate

Show cached data immediately while fetching fresh data in the background:

```typescript
@Injectable({ providedIn: 'root' })
export class StaleWhileRevalidateService {
  private readonly http = inject(HttpClient);
  private cache = new Map<string, { data: any; timestamp: number }>();

  get<T>(url: string, maxAge = 30_000): Observable<T> {
    const cached = this.cache.get(url);
    const request$ = this.http.get<T>(url).pipe(
      tap(data => this.cache.set(url, { data, timestamp: Date.now() }))
    );

    if (cached && (Date.now() - cached.timestamp) < maxAge) {
      // Return cached data immediately, then fresh data
      return concat(
        of(cached.data as T),
        request$
      );
    }

    return request$;
  }
}
```

## Pattern 8: Bulk Operations with Progress

Process multiple items with concurrency control and progress tracking.

```typescript
@Component({
  selector: 'app-bulk-upload',
  template: `
    <progress [value]="progress()" max="100"></progress>
    <span>{{ completedCount() }} / {{ totalCount() }}</span>
    @if (errors().length > 0) {
      <div class="errors">{{ errors().length }} failed</div>
    }
  `
})
export class BulkUploadComponent {
  private readonly http = inject(HttpClient);
  private readonly destroyRef = inject(DestroyRef);

  readonly progress = signal(0);
  readonly completedCount = signal(0);
  readonly totalCount = signal(0);
  readonly errors = signal<string[]>([]);

  uploadFiles(files: File[]) {
    this.totalCount.set(files.length);
    this.completedCount.set(0);
    this.errors.set([]);

    from(files).pipe(
      mergeMap(file => {
        const formData = new FormData();
        formData.append('file', file);

        return this.http.post('/api/upload', formData).pipe(
          tap(() => {
            this.completedCount.update(c => c + 1);
            this.progress.set(
              Math.round((this.completedCount() / this.totalCount()) * 100)
            );
          }),
          catchError(err => {
            this.errors.update(errs => [...errs, `${file.name}: ${err.message}`]);
            this.completedCount.update(c => c + 1);
            this.progress.set(
              Math.round((this.completedCount() / this.totalCount()) * 100)
            );
            return EMPTY;
          })
        );
      }, 3), // Max 3 concurrent uploads
      takeUntilDestroyed(this.destroyRef)
    ).subscribe({
      complete: () => console.log('All uploads processed')
    });
  }
}
```

## Pattern 9: Coordinated Sequential Requests

When requests depend on each other (e.g., create entity, then upload file, then notify).

```typescript
createProjectWithAssets(project: ProjectData, files: File[]): Observable<Project> {
  return this.http.post<Project>('/api/projects', project).pipe(
    // Step 1 result feeds into Step 2
    switchMap(createdProject =>
      forkJoin(
        files.map(file => {
          const fd = new FormData();
          fd.append('file', file);
          return this.http.post(`/api/projects/${createdProject.id}/assets`, fd);
        })
      ).pipe(
        // If no files, skip the forkJoin (it would error on empty array)
        defaultIfEmpty([]),
        map(() => createdProject) // Return the created project
      )
    ),
    // Step 3: Notify team
    switchMap(project =>
      this.http.post('/api/notifications', {
        type: 'project_created',
        projectId: project.id
      }).pipe(
        map(() => project),
        catchError(() => of(project)) // Don't fail if notification fails
      )
    )
  );
}
```

## Common Mistakes

1. **Using `mergeMap` instead of `switchMap` for search.** `mergeMap` does not cancel previous requests. Old responses can overwrite newer ones. Always use `switchMap` for "latest wins" scenarios like search and navigation.

2. **Polling with `setInterval` instead of RxJS.** `setInterval` is not cancellable, not composable, and does not handle backpressure. Use `interval()` or `timer()` with `switchMap`/`exhaustMap`.

3. **Retrying non-idempotent requests.** Retrying a POST that creates a resource can create duplicates. Only retry GET requests and idempotent operations, or implement idempotency keys.

4. **Infinite retry without a max.** Setting `count: Infinity` without a circuit breaker wastes resources. Always set a reasonable max retry count or implement a circuit breaker pattern.

5. **Not adding jitter to exponential backoff.** Without jitter, all clients retry at exactly the same times, amplifying the load on a recovering server.

6. **Forgetting `startWith(0)` on polling.** `interval(5000)` emits its first value after 5 seconds. Without `startWith(0)`, the user sees nothing for 5 seconds on page load.

7. **Using `concatMap` for search requests.** `concatMap` queues requests instead of canceling them. If the user types fast, old searches pile up and execute sequentially, creating a long backlog.

8. **Optimistic updates without rollback.** Updating the UI optimistically but not handling the error case leaves the UI in an inconsistent state if the server rejects the change.

---

## Best Practices

1. **Use `switchMap` for "latest wins" patterns** (search, navigation, detail view). Use `exhaustMap` for "first wins" patterns (form submissions, button clicks). Use `mergeMap` for "process all" patterns (bulk operations). Use `concatMap` for "ordered queue" patterns (sequential operations).

2. **Always handle errors inside flattening operators** to prevent the outer stream from dying. Use `catchError(() => EMPTY)` to skip failed items or `catchError(() => of(fallback))` to provide defaults.

3. **Extract reusable patterns into custom operators.** `retryBackoff()`, `poll()`, and `cacheWithTTL()` are good candidates for shared utility libraries.

4. **Combine signals and RxJS strategically.** Use signals for synchronous state (UI flags, form values). Use RxJS for async operations (HTTP, WebSocket, timers). Bridge them with `toSignal()` and `toObservable()`.

5. **Add loading states for every async pattern.** Users need feedback. Track loading states as signals alongside data signals.

6. **Use `finalize()` for cleanup that must happen regardless of success or failure.** Loading spinners and progress indicators should always be cleaned up.

## Real-World Use Cases

### 1. Full Autocomplete with Recent Searches

```typescript
@Component({
  selector: 'app-smart-search',
  template: `
    <input
      [ngModel]="query()"
      (ngModelChange)="query.set($event)"
      (focus)="showDropdown.set(true)"
    />
    @if (showDropdown()) {
      <div class="dropdown">
        @if (query().length < 2 && recentSearches().length > 0) {
          <div class="section-header">Recent Searches</div>
          @for (recent of recentSearches(); track recent) {
            <div (click)="query.set(recent)">{{ recent }}</div>
          }
        }
        @for (result of results(); track result.id) {
          <div (click)="selectResult(result)">{{ result.name }}</div>
        }
      </div>
    }
  `
})
export class SmartSearchComponent {
  private readonly http = inject(HttpClient);

  readonly query = signal('');
  readonly showDropdown = signal(false);
  readonly recentSearches = signal<string[]>(
    JSON.parse(localStorage.getItem('recentSearches') || '[]')
  );

  readonly results = toSignal(
    toObservable(this.query).pipe(
      debounceTime(300),
      distinctUntilChanged(),
      switchMap(q => {
        if (q.length < 2) return of([]);
        return this.http.get<SearchResult[]>(`/api/search?q=${q}`).pipe(
          catchError(() => of([]))
        );
      })
    ),
    { initialValue: [] }
  );

  selectResult(result: SearchResult) {
    this.recentSearches.update(recent => {
      const updated = [result.name, ...recent.filter(r => r !== result.name)].slice(0, 5);
      localStorage.setItem('recentSearches', JSON.stringify(updated));
      return updated;
    });
    this.showDropdown.set(false);
  }
}
```

### 2. Real-Time Dashboard with Resilient Polling

```typescript
@Injectable({ providedIn: 'root' })
export class MetricsService {
  private readonly http = inject(HttpClient);

  private readonly visibility$ = fromEvent(document, 'visibilitychange').pipe(
    map(() => !document.hidden),
    startWith(true)
  );

  readonly metrics$ = this.visibility$.pipe(
    switchMap(visible => {
      if (!visible) return EMPTY;
      return timer(0, 15_000).pipe(
        exhaustMap(() =>
          this.http.get<Metrics>('/api/metrics').pipe(
            retryBackoff({ maxRetries: 2, initialDelay: 2_000 }),
            catchError(() => EMPTY) // Skip this poll cycle on failure
          )
        )
      );
    }),
    shareReplay({ bufferSize: 1, refCount: true })
  );
}
```

### 3. Multi-Step Wizard with Server Validation

```typescript
@Component({
  selector: 'app-wizard',
  template: `
    @switch (step()) {
      @case (1) { <app-step-one (next)="validateAndProceed($event)" /> }
      @case (2) { <app-step-two (next)="validateAndProceed($event)" (back)="step.set(1)" /> }
      @case (3) { <app-step-three (submit)="submit($event)" (back)="step.set(2)" /> }
    }
    @if (validating()) { <div class="overlay">Validating...</div> }
  `
})
export class WizardComponent {
  private readonly http = inject(HttpClient);
  private readonly destroyRef = inject(DestroyRef);

  readonly step = signal(1);
  readonly validating = signal(false);
  private readonly formData = signal<Record<string, any>>({});

  validateAndProceed(stepData: Record<string, any>) {
    this.validating.set(true);

    this.http.post<{ valid: boolean; errors?: string[] }>(
      `/api/wizard/validate/step${this.step()}`,
      stepData
    ).pipe(
      retryBackoff({ maxRetries: 1, initialDelay: 1_000 }),
      finalize(() => this.validating.set(false)),
      takeUntilDestroyed(this.destroyRef)
    ).subscribe(result => {
      if (result.valid) {
        this.formData.update(data => ({ ...data, ...stepData }));
        this.step.update(s => s + 1);
      }
    });
  }

  submit(finalData: Record<string, any>) {
    const allData = { ...this.formData(), ...finalData };
    this.http.post('/api/wizard/submit', allData).pipe(
      retryBackoff({ maxRetries: 2 }),
      takeUntilDestroyed(this.destroyRef)
    ).subscribe(() => {
      // Navigate to success page
    });
  }
}
```

> **Key Takeaway:** The right RxJS pattern depends on the concurrency behavior you need: `switchMap` for "latest wins" (search, navigation), `exhaustMap` for "ignore while busy" (form submission, polling slow endpoints), `mergeMap` for "process all in parallel" (bulk operations), and `concatMap` for "ordered queue" (sequential writes). Combine these with `retry`, `catchError`, and signals for production-ready Angular applications.
