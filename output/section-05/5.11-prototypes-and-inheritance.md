# 5.11 Prototypes and Inheritance

> JavaScript uses prototypal inheritance, where objects delegate property lookups to other objects through an internal prototype chain. Understanding this mechanism is essential for working with object-oriented patterns, debugging property resolution, and avoiding security pitfalls like prototype pollution.

---

## The Prototype Chain

Every object in JavaScript has a hidden internal slot called `[[Prototype]]` that holds a reference to another object (or `null`). When you access a property on an object, the engine:

1. Checks the object itself for the property.
2. If not found, follows the `[[Prototype]]` link to the next object.
3. Repeats until the property is found or `null` is reached.

```js
const animal = {
  type: "animal",
  speak() {
    return `${this.name} makes a sound`;
  }
};

const dog = Object.create(animal);
dog.name = "Rex";

console.log(dog.name);    // "Rex" (own property)
console.log(dog.type);    // "animal" (found on animal via [[Prototype]])
console.log(dog.speak()); // "Rex makes a sound" (found on animal, 'this' is dog)

// The chain: dog -> animal -> Object.prototype -> null
console.log(Object.getPrototypeOf(dog) === animal);              // true
console.log(Object.getPrototypeOf(animal) === Object.prototype); // true
console.log(Object.getPrototypeOf(Object.prototype) === null);   // true
```

## `[[Prototype]]` vs `__proto__` vs `.prototype`

These three concepts are constantly confused. They are different things.

| Concept | What It Is | Who Has It |
|---|---|---|
| `[[Prototype]]` | Internal slot on every object pointing to its prototype | Every object |
| `__proto__` | A getter/setter on `Object.prototype` that exposes `[[Prototype]]` | Accessible on most objects (legacy, Annex B) |
| `.prototype` | A regular property on constructor functions | Only functions intended to be called with `new` |

```js
function Dog(name) {
  this.name = name;
}

Dog.prototype.bark = function () {
  return `${this.name} barks`;
};

const rex = new Dog("Rex");

// .prototype is a property on the constructor function
// It becomes the [[Prototype]] of instances created with 'new'
console.log(Object.getPrototypeOf(rex) === Dog.prototype); // true

// Dog itself is a function; its [[Prototype]] is Function.prototype
console.log(Object.getPrototypeOf(Dog) === Function.prototype); // true

// __proto__ is the legacy accessor for [[Prototype]]
console.log(rex.__proto__ === Dog.prototype); // true (but don't use this)
```

Important: `.prototype` on a function is NOT the prototype of the function itself. It is the object that will become the `[[Prototype]]` of instances created with `new`.

### Accessing `[[Prototype]]`

| Method | Standard | Notes |
|---|---|---|
| `Object.getPrototypeOf(obj)` | Yes | Recommended for reading |
| `Reflect.getPrototypeOf(obj)` | Yes | Stricter: throws TypeError for non-objects |
| `Object.setPrototypeOf(obj, proto)` | Yes | Avoid (severe performance penalty) |
| `obj.__proto__` | Annex B (legacy) | Do not use in production |
| `Object.create(proto)` | Yes | Set prototype at creation time (preferred) |

## Setting Up Prototype Chains

### Constructor Functions

The traditional pre-class pattern.

```js
function Shape(color) {
  this.color = color;
}

Shape.prototype.describe = function () {
  return `A ${this.color} shape`;
};

function Circle(color, radius) {
  Shape.call(this, color); // call parent constructor
  this.radius = radius;
}

// Set up inheritance: Circle instances should delegate to Shape.prototype
Circle.prototype = Object.create(Shape.prototype);
Circle.prototype.constructor = Circle; // fix the constructor reference

Circle.prototype.area = function () {
  return Math.PI * this.radius ** 2;
};

const c = new Circle("red", 5);
console.log(c.describe()); // "A red shape" (inherited from Shape.prototype)
console.log(c.area());     // 78.539... (own method on Circle.prototype)
console.log(c instanceof Circle); // true
console.log(c instanceof Shape);  // true
```

### `Object.create()`

Creates a new object with a specified prototype. No constructor is called.

```js
const vehicleProto = {
  start() { return `${this.make} starting`; },
  stop() { return `${this.make} stopping`; }
};

const car = Object.create(vehicleProto);
car.make = "Toyota";
car.doors = 4;

console.log(car.start()); // "Toyota starting"
```

The second argument to `Object.create` accepts property descriptors:

```js
const car = Object.create(vehicleProto, {
  make: { value: "Toyota", writable: true, enumerable: true, configurable: true },
  doors: { value: 4, writable: false, enumerable: true, configurable: false }
});
```

### `class extends` (Under the Hood)

When you write `class Child extends Parent`, two prototype links are established:

```js
class Parent {
  greet() { return "hello from Parent"; }
  static create() { return new this(); }
}

class Child extends Parent {
  greet() { return "hello from Child"; }
}

// Link 1: Instance method inheritance
console.log(Object.getPrototypeOf(Child.prototype) === Parent.prototype); // true

// Link 2: Static method inheritance
console.log(Object.getPrototypeOf(Child) === Parent); // true

// So Child.create() works (inherited static method)
const c = Child.create();
console.log(c instanceof Child); // true
```

The full chain for an instance:

```
child instance -> Child.prototype -> Parent.prototype -> Object.prototype -> null
```

## `Object.setPrototypeOf` and Why to Avoid It

`Object.setPrototypeOf` changes the `[[Prototype]]` of an existing object. It works, but it is a performance disaster.

### Why It Is Slow

V8 uses hidden classes (called "shapes" or "maps" internally) to optimize property access. Each object tracks its property layout and prototype reference as part of its shape. When you mutate the prototype:

1. V8 must invalidate inline caches for the affected object.
2. The object transitions to a new hidden class.
3. Any code that touches this object may be deoptimized from optimized JIT code.
4. Inline caches that were monomorphic (fast) become megamorphic (slow).

The performance impact extends far beyond the `setPrototypeOf` call itself.

```js
// Bad Practice: mutating prototype after creation
const obj = {};
Object.setPrototypeOf(obj, someProto); // triggers deoptimization

// Good Practice: set prototype at creation time
const obj = Object.create(someProto);
```

### Circular Prototype Chains

The engine detects and prevents cycles:

```js
const a = {};
const b = Object.create(a);

Object.setPrototypeOf(a, b); // TypeError: Cyclic __proto__ value
```

## Property Shadowing

When you assign a property that already exists somewhere on the prototype chain, the behavior depends on the property's descriptor on the prototype.

### Case 1: Normal Writable Property (Expected Behavior)

```js
const proto = { count: 0 };
const obj = Object.create(proto);

obj.count = 5; // creates an OWN property on obj, shadowing proto.count

console.log(obj.count);   // 5 (own)
console.log(proto.count); // 0 (unchanged)
console.log(Object.hasOwn(obj, "count")); // true
```

### Case 2: Read-Only Property on the Prototype

```js
const proto = {};
Object.defineProperty(proto, "locked", {
  value: 42,
  writable: false // read-only
});

const obj = Object.create(proto);

obj.locked = 100; // silently fails in sloppy mode, TypeError in strict mode
console.log(obj.locked); // 42 (no own property was created!)
console.log(Object.hasOwn(obj, "locked")); // false
```

A read-only property on the prototype prevents creating a same-named own property via assignment. Use `Object.defineProperty` to work around this:

```js
Object.defineProperty(obj, "locked", { value: 100 }); // this works
console.log(obj.locked); // 100
```

### Case 3: Setter on the Prototype

```js
const proto = {
  _value: 0,
  set value(v) { this._value = v; },
  get value() { return this._value; }
};

const obj = Object.create(proto);

obj.value = 42; // calls the setter, does NOT create an own 'value' property
console.log(obj._value); // 42 (created on obj via the setter)
console.log(Object.hasOwn(obj, "value")); // false (no own 'value')
console.log(Object.hasOwn(obj, "_value")); // true
```

### The Implicit Shadowing Gotcha

```js
const proto = { count: 0 };
const obj = Object.create(proto);

// obj.count++ is actually: obj.count = obj.count + 1
// Reads 0 from proto, adds 1, assigns 1 as an OWN property on obj
obj.count++;

console.log(obj.count);   // 1 (own property)
console.log(proto.count); // 0 (unchanged)
```

This is subtle: you might think you are modifying shared state on the prototype, but you are actually creating instance-level state.

## Property Checking and Enumeration

### `in` vs `Object.hasOwn()` vs `hasOwnProperty()`

```js
const proto = { inherited: true };
const obj = Object.create(proto);
obj.own = true;

// 'in' checks the full prototype chain
console.log("own" in obj);       // true
console.log("inherited" in obj); // true
console.log("toString" in obj);  // true (from Object.prototype)

// hasOwnProperty checks only own properties (but can be overridden)
console.log(obj.hasOwnProperty("own"));       // true
console.log(obj.hasOwnProperty("inherited")); // false

// Object.hasOwn (ES2022) - safer, recommended
console.log(Object.hasOwn(obj, "own"));       // true
console.log(Object.hasOwn(obj, "inherited")); // false
```

Why `Object.hasOwn()` is preferred:
- It works on `Object.create(null)` objects (which lack `hasOwnProperty`).
- It cannot be overridden by the object itself.

### Enumeration Methods Comparison

| Method | Own Only | Enumerable Only | Includes Symbols |
|---|---|---|---|
| `for...in` | No (walks chain) | Yes | No |
| `Object.keys()` | Yes | Yes | No |
| `Object.getOwnPropertyNames()` | Yes | No | No |
| `Object.getOwnPropertySymbols()` | Yes | No | Yes (symbols only) |
| `Reflect.ownKeys()` | Yes | No | Yes |

```js
// Bad Practice: for...in without guard
for (const key in obj) {
  doSomething(obj[key]); // may include inherited enumerable properties
}

// Good Practice: Object.keys + for...of
for (const key of Object.keys(obj)) {
  doSomething(obj[key]); // only own enumerable properties
}
```

## `Object.create(null)` and the Dictionary Pattern

Creating an object with no prototype produces a "clean" dictionary with zero inherited properties.

```js
const dict = Object.create(null);
dict.key = "value";

console.log("key" in dict);       // true
console.log("toString" in dict);  // false (no prototype!)
console.log("constructor" in dict); // false

// Gotcha: built-in methods don't exist
// dict.hasOwnProperty("key"); // TypeError
// dict.toString();            // TypeError

// Use Object.hasOwn instead
console.log(Object.hasOwn(dict, "key")); // true
```

### `Object.create(null)` vs `Map`

| Feature | `Object.create(null)` | `Map` |
|---|---|---|
| Key types | Strings and Symbols | Any value |
| Size tracking | `Object.keys(dict).length` | `.size` |
| Iteration order | Mostly insertion (numeric keys first) | Guaranteed insertion order |
| Frequent add/delete | Slower | Faster |
| JSON serialization | Direct `JSON.stringify` | Requires conversion |
| Prototype pollution safe | Yes | Yes |

Use `Map` for dynamic key-value stores with frequent additions and deletions. Use `Object.create(null)` for static config objects or when you need JSON serialization.

## The `instanceof` Operator

`instanceof` checks whether a constructor's `.prototype` property appears anywhere in an object's prototype chain.

### The Algorithm

1. If the right-hand side has `Symbol.hasInstance`, call it and return the result.
2. Otherwise, walk the `[[Prototype]]` chain of the left operand. Compare each link with `Constructor.prototype`.
3. Return `true` if a match is found, `false` if `null` is reached.

```js
function Animal() {}
function Dog() {}

Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

const rex = new Dog();

console.log(rex instanceof Dog);    // true (Dog.prototype is in the chain)
console.log(rex instanceof Animal); // true (Animal.prototype is in the chain)
console.log(rex instanceof Object); // true (Object.prototype is in the chain)
```

### Gotcha: Reassigning `.prototype`

```js
function Foo() {}
const f = new Foo();

console.log(f instanceof Foo); // true

Foo.prototype = {}; // reassign to a new object

console.log(f instanceof Foo); // false! (the old prototype is no longer Foo.prototype)
```

### Gotcha: Cross-Realm (Iframes)

```js
// An array from a different iframe
const arr = iframe.contentWindow.eval("[]");

console.log(arr instanceof Array); // false! (different Array constructor)
console.log(Array.isArray(arr));   // true (the safe check)
```

Always use `Array.isArray()` instead of `instanceof Array`.

### Gotcha: `Object.create(null)`

```js
const obj = Object.create(null);
console.log(obj instanceof Object); // false (Object.prototype not in chain)
```

## `Symbol.hasInstance` and `Symbol.species`

### Customizing `instanceof` with `Symbol.hasInstance`

```js
class EvenNumber {
  static [Symbol.hasInstance](value) {
    return typeof value === "number" && value % 2 === 0;
  }
}

console.log(4 instanceof EvenNumber);  // true
console.log(3 instanceof EvenNumber);  // false
console.log("4" instanceof EvenNumber); // false
```

### Controlling Derived Objects with `Symbol.species`

When built-in methods on subclasses create new instances (like `Array.prototype.map`), `Symbol.species` determines which constructor to use.

```js
class TrackedArray extends Array {
  static get [Symbol.species]() {
    return Array; // map, filter, etc. return plain Arrays, not TrackedArrays
  }
}

const tracked = new TrackedArray(1, 2, 3);
const mapped = tracked.map((x) => x * 2);

console.log(tracked instanceof TrackedArray); // true
console.log(mapped instanceof TrackedArray);  // false (plain Array)
console.log(mapped instanceof Array);         // true
```

Without `Symbol.species`, `mapped` would be a `TrackedArray`, which might not be desired.

## Inheritance Patterns

### Prototype Delegation (Standard)

Objects link to other objects via `[[Prototype]]`. Methods are shared, not copied.

```js
const base = {
  init(name) {
    this.name = name;
    return this;
  },
  greet() {
    return `Hello, I'm ${this.name}`;
  }
};

const admin = Object.create(base);
admin.role = "admin";
admin.describe = function () {
  return `${this.greet()} (${this.role})`;
};

const user = Object.create(admin).init("Alice");
console.log(user.describe()); // "Hello, I'm Alice (admin)"
```

### Concatenative Inheritance (Copying)

Properties are copied directly from source objects. No prototype chain involved.

```js
const canFly = {
  fly() { return `${this.name} is flying`; }
};

const canSwim = {
  swim() { return `${this.name} is swimming`; }
};

function createDuck(name) {
  return Object.assign(
    { name },
    canFly,
    canSwim
  );
}

const duck = createDuck("Donald");
console.log(duck.fly());  // "Donald is flying"
console.log(duck.swim()); // "Donald is swimming"

// Methods are own properties (copied), not inherited
console.log(Object.hasOwn(duck, "fly"));  // true
console.log(Object.hasOwn(duck, "swim")); // true
```

### Functional Inheritance (Factory + Closure)

Provides true privacy through closures. No `new`, no `this` issues.

```js
function createAccount(owner, initialBalance) {
  let balance = initialBalance; // truly private

  return {
    getOwner() { return owner; },
    getBalance() { return balance; },
    deposit(amount) {
      if (amount <= 0) throw new Error("Amount must be positive");
      balance += amount;
      return balance;
    },
    withdraw(amount) {
      if (amount > balance) throw new Error("Insufficient funds");
      balance -= amount;
      return balance;
    }
  };
}

const acct = createAccount("Alice", 100);
acct.deposit(50);
console.log(acct.getBalance()); // 150
console.log(acct.balance);      // undefined (truly private)
```

## Mixins and Composition

JavaScript has single inheritance (one `[[Prototype]]` link). Mixins solve the need to compose behavior from multiple sources.

### Pattern 1: `Object.assign` onto the Prototype

```js
const Serializable = {
  serialize() {
    return JSON.stringify(this);
  }
};

const Timestamped = {
  touch() {
    this.updatedAt = Date.now();
    return this;
  }
};

class User {
  constructor(name) {
    this.name = name;
  }
}

Object.assign(User.prototype, Serializable, Timestamped);

const user = new User("Alice");
user.touch();
console.log(user.serialize()); // '{"name":"Alice","updatedAt":...}'
```

### Pattern 2: Subclass Factory (Class Expression Mixins)

Each mixin is a function that takes a base class and returns a new class extending it.

```js
const Serializable = (Base) =>
  class extends Base {
    serialize() {
      return JSON.stringify(this);
    }
  };

const Timestamped = (Base) =>
  class extends Base {
    touch() {
      this.updatedAt = Date.now();
      return this;
    }
  };

class BaseEntity {
  constructor(id) {
    this.id = id;
  }
}

// Compose mixins: read right to left
class User extends Timestamped(Serializable(BaseEntity)) {
  constructor(id, name) {
    super(id);
    this.name = name;
  }
}

const user = new User(1, "Alice");
user.touch();
console.log(user.serialize());
console.log(user instanceof BaseEntity); // true
```

This creates a linear chain: `User -> Timestamped(anon) -> Serializable(anon) -> BaseEntity`. It preserves `instanceof` for the base class.

### Pattern 3: Functional Mixins

```js
const withEventEmitter = (obj) => {
  const listeners = new Map(); // private via closure

  return Object.assign(obj, {
    on(event, fn) {
      if (!listeners.has(event)) listeners.set(event, []);
      listeners.get(event).push(fn);
      return obj;
    },
    emit(event, ...args) {
      const fns = listeners.get(event) || [];
      fns.forEach((fn) => fn(...args));
      return obj;
    },
    off(event, fn) {
      const fns = listeners.get(event);
      if (fns) {
        listeners.set(event, fns.filter((f) => f !== fn));
      }
      return obj;
    }
  });
};

const button = withEventEmitter({ label: "Submit" });
button.on("click", () => console.log("Clicked!"));
button.emit("click"); // "Clicked!"
```

## Prototype Pollution

Prototype pollution is a security vulnerability where an attacker injects properties into `Object.prototype`, causing those properties to appear on every object in the application.

### How It Happens

The most common attack vector is unsafe recursive merge or deep-copy functions that process user-controlled input.

```js
// Vulnerable merge function
function unsafeMerge(target, source) {
  for (const key in source) {
    if (typeof source[key] === "object" && source[key] !== null) {
      if (!target[key]) target[key] = {};
      unsafeMerge(target[key], source[key]);
    } else {
      target[key] = source[key];
    }
  }
  return target;
}

// Attacker sends this payload
const malicious = JSON.parse('{"__proto__": {"isAdmin": true}}');

const config = {};
unsafeMerge(config, malicious);

// Now EVERY object in the application has isAdmin === true
const user = {};
console.log(user.isAdmin); // true
```

### Impact

- Authentication bypass (`user.isAdmin` resolving to `true`)
- Denial of Service (triggering exceptions from unexpected properties)
- Remote Code Execution in Node.js (via `child_process` options)
- XSS via polluted DOM attributes

### Prevention

```js
// 1. Skip dangerous keys in merge functions
function safeMerge(target, source) {
  for (const key of Object.keys(source)) { // Object.keys skips __proto__
    if (key === "__proto__" || key === "constructor" || key === "prototype") {
      continue;
    }
    if (typeof source[key] === "object" && source[key] !== null) {
      if (!target[key]) target[key] = {};
      safeMerge(target[key], source[key]);
    } else {
      target[key] = source[key];
    }
  }
  return target;
}

// 2. Use Object.create(null) for dictionaries
const dict = Object.create(null);
// No __proto__ accessor, no inherited properties to pollute

// 3. Use Map for user-controlled key-value data
const userSettings = new Map();

// 4. Freeze Object.prototype (nuclear option)
Object.freeze(Object.prototype);
// Prevents any modification, but may break third-party libraries

// 5. Validate input schemas before merging
function validateNoProtoKeys(obj) {
  for (const key of Object.keys(obj)) {
    if (key === "__proto__" || key === "constructor" || key === "prototype") {
      throw new Error(`Forbidden key: ${key}`);
    }
    if (typeof obj[key] === "object" && obj[key] !== null) {
      validateNoProtoKeys(obj[key]);
    }
  }
}
```

## Prototype Methods vs Instance Methods

### Memory and Performance Comparison

```js
// Prototype method: shared across all instances
function Dog(name) {
  this.name = name;
}
Dog.prototype.bark = function () {
  return `${this.name} barks`;
};

// Instance method: new copy per instance
function Cat(name) {
  this.name = name;
  this.meow = function () {
    return `${this.name} meows`;
  };
}
```

| Aspect | Prototype Method | Instance Method |
|---|---|---|
| Memory | 1 copy shared on prototype | N copies for N instances |
| Creation speed | Faster (no method creation) | Slower |
| Invocation speed | Nearly identical (V8 inline caches) | Slightly faster (no chain lookup) |
| Overridable per instance | Yes (via shadowing) | Already per-instance |

For any meaningful number of instances, prototype methods are more memory-efficient. The invocation speed difference is negligible in practice due to V8's inline caching.

## Common Mistakes

- **Replacing the entire `.prototype` object**: Overwrites the `constructor` property and breaks `instanceof` for previously created instances.

```js
// Bad Practice
function Foo() {}
Foo.prototype = {
  method() {}
  // constructor is now Object, not Foo
};

// Good Practice
Foo.prototype.method = function () {};
// Or, if replacing:
Foo.prototype = Object.create(ParentProto);
Foo.prototype.constructor = Foo; // fix it
```

- **Confusing `.prototype` with `[[Prototype]]`**: `Dog.prototype` is not Dog's own prototype. It is the prototype that instances created with `new Dog()` will receive.

- **Using `for...in` without filtering inherited properties**: Always prefer `Object.keys()` or guard with `Object.hasOwn()`.

- **Using `__proto__` in production**: It is legacy (Annex B). Use `Object.getPrototypeOf()` and `Object.create()`.

- **Calling `Object.setPrototypeOf` in hot paths**: Destroys V8's hidden class optimizations. Always set prototypes at creation time.

- **Trusting `instanceof` across iframes**: Use `Array.isArray()`, `typeof`, or duck typing for cross-realm safety.

- **Modifying `Object.prototype`**: Adding enumerable properties to it breaks every `for...in` loop in the application.

- **Not guarding merge functions against `__proto__` keys**: This is the primary prototype pollution attack vector.

---

## Best Practices

- Set prototypes at creation time using `class extends`, `Object.create()`, or `new`. Never mutate prototypes after creation.
- Define methods on the prototype, data in the constructor. Methods are shared; data should be per-instance.
- Use `Object.hasOwn()` (ES2022) over `hasOwnProperty()` for own-property checks.
- Use `Object.keys()` / `Object.entries()` instead of `for...in` for object iteration.
- Prefer `class` syntax for readability and correct `extends` wiring.
- Use `Object.create(null)` or `Map` for dictionary/lookup patterns to avoid prototype collision and pollution.
- Validate and sanitize user input before merging into objects. Block `__proto__`, `constructor`, and `prototype` keys.
- Use `Array.isArray()` instead of `instanceof Array` for cross-realm safety.
- Prefer composition (mixins, functional patterns) over deep inheritance chains. Keep chains shallow.

## Real-World Use Cases

### Plugin System with Prototype Delegation

```js
function createPluginSystem() {
  const basePlugin = {
    init() { this.initialized = true; return this; },
    getName() { return this.name; },
    getVersion() { return this.version || "1.0.0"; }
  };

  const plugins = new Map();

  return {
    register(name, definition) {
      const plugin = Object.create(basePlugin);
      Object.assign(plugin, definition, { name });
      plugins.set(name, plugin);
      return plugin.init();
    },

    get(name) {
      return plugins.get(name);
    },

    list() {
      return [...plugins.keys()];
    }
  };
}

const system = createPluginSystem();

system.register("logger", {
  version: "2.1.0",
  log(msg) { console.log(`[${this.getName()}] ${msg}`); }
});

system.register("metrics", {
  track(event) { console.log(`Tracking: ${event}`); }
});

const logger = system.get("logger");
logger.log("System started");       // "[logger] System started"
console.log(logger.getVersion());    // "2.1.0" (own property shadows base)
console.log(logger.initialized);     // true (set by base init)
```

### Safe Configuration Merging

```js
function createConfig(defaults) {
  // Use null-prototype object to prevent pollution
  const config = Object.create(null);

  function set(key, value) {
    if (key === "__proto__" || key === "constructor" || key === "prototype") {
      throw new Error(`Forbidden config key: ${key}`);
    }
    config[key] = value;
  }

  function get(key) {
    return Object.hasOwn(config, key) ? config[key] : defaults?.[key];
  }

  function getAll() {
    return { ...defaults, ...config };
  }

  // Apply initial defaults safely
  if (defaults) {
    for (const key of Object.keys(defaults)) {
      set(key, defaults[key]);
    }
  }

  return { set, get, getAll };
}

const config = createConfig({ theme: "light", lang: "en" });
config.set("theme", "dark");
console.log(config.get("theme")); // "dark"
console.log(config.get("lang"));  // "en" (from defaults)
```

### Composable Behaviors with Subclass Factories

```js
const Cacheable = (Base) =>
  class extends Base {
    #cache = new Map();

    getCached(key, compute) {
      if (this.#cache.has(key)) return this.#cache.get(key);
      const value = compute();
      this.#cache.set(key, value);
      return value;
    }

    clearCache() {
      this.#cache.clear();
    }
  };

const Loggable = (Base) =>
  class extends Base {
    log(level, message) {
      console.log(`[${level.toUpperCase()}] ${this.constructor.name}: ${message}`);
    }
  };

class BaseService {
  constructor(name) {
    this.name = name;
  }
}

class UserService extends Loggable(Cacheable(BaseService)) {
  getUser(id) {
    return this.getCached(`user:${id}`, () => {
      this.log("info", `Fetching user ${id}`);
      return { id, name: `User ${id}` };
    });
  }
}

const service = new UserService("users");
service.getUser(1); // [INFO] UserService: Fetching user 1 -> { id: 1, name: "User 1" }
service.getUser(1); // cached, no log
```

---

> **Key Takeaway:** JavaScript's prototype chain is a delegation mechanism where objects link to other objects for property lookup. Master the distinction between `[[Prototype]]`, `__proto__`, and `.prototype`, understand property shadowing rules, guard against prototype pollution in any code that merges user input, and prefer composition patterns over deep inheritance hierarchies.
