# 5.8 Functions

> Functions are the primary unit of execution and organization in JavaScript. Understanding the nuances of declarations, expressions, closures, parameters, and composition patterns is essential for writing clean, predictable, and performant code.

---

## Function Declarations vs Function Expressions

These two forms look similar but behave differently due to hoisting.

### Function Declarations

A function declaration is fully hoisted. Both the name and the implementation are available before the line where the function appears.

```js
// This works because declarations are hoisted
console.log(add(2, 3)); // 5

function add(a, b) {
  return a + b;
}
```

### Function Expressions

A function expression assigns a function to a variable. Only the variable declaration is hoisted, not the assignment.

```js
// TypeError: multiply is not a function
console.log(multiply(2, 3));

const multiply = function (a, b) {
  return a * b;
};
```

With `var`, the variable is hoisted as `undefined`:

```js
console.log(typeof divide); // "undefined"
console.log(divide(10, 2)); // TypeError: divide is not a function

var divide = function (a, b) {
  return a / b;
};
```

### Named Function Expressions

The name of a named function expression is only accessible inside the function body itself.

```js
const factorial = function fact(n) {
  if (n <= 1) return 1;
  return n * fact(n - 1); // 'fact' is accessible here
};

console.log(factorial(5)); // 120
console.log(typeof fact);  // "undefined" (not accessible outside)
```

Named function expressions are useful for recursion and produce better stack traces during debugging.

### Function Declarations Inside Blocks

This is a major gotcha. Behavior differs between strict and sloppy mode, and across browsers.

```js
"use strict";

if (true) {
  function blockScoped() {
    return "inside block";
  }
  console.log(blockScoped()); // "inside block"
}

// In strict mode: blockScoped is block-scoped
console.log(typeof blockScoped); // "undefined"
```

```js
// Bad Practice: function declarations in blocks without strict mode
// Behavior is inconsistent across browsers
if (condition) {
  function doSomething() { /* ... */ }
}

// Good Practice: use function expressions instead
let doSomething;
if (condition) {
  doSomething = function () { /* ... */ };
}
```

## Arrow Functions

Arrow functions provide a concise syntax and differ from regular functions in several important ways.

### Syntax Variations

```js
// No parameters
const greet = () => "hello";

// Single parameter (parentheses optional)
const double = x => x * 2;

// Multiple parameters
const add = (a, b) => a + b;

// Multi-line body (requires braces and explicit return)
const calculate = (a, b) => {
  const sum = a + b;
  return sum * 2;
};

// Returning an object literal (wrap in parentheses)
const makeUser = (name) => ({ name, createdAt: Date.now() });
```

### Key Differences from Regular Functions

| Feature | Regular Function | Arrow Function |
|---|---|---|
| `this` binding | Dynamic (call site) | Lexical (definition site) |
| `arguments` object | Yes | No |
| Can be constructor (`new`) | Yes | No (throws TypeError) |
| `prototype` property | Yes | No |
| Can use `yield` (generator) | Yes | No |
| `super` binding | Own | Inherited from enclosing |
| `new.target` | Own | Inherited from enclosing |

### The `this` Binding Difference

This is the most important difference and the most common source of bugs.

```js
const timer = {
  seconds: 0,

  // Regular function: 'this' depends on how it is called
  startBroken: function () {
    setInterval(function () {
      this.seconds++; // 'this' is NOT the timer object
      console.log(this.seconds); // NaN
    }, 1000);
  },

  // Arrow function: 'this' is captured from the enclosing scope
  startFixed: function () {
    setInterval(() => {
      this.seconds++; // 'this' IS the timer object
      console.log(this.seconds);
    }, 1000);
  }
};
```

### When NOT to Use Arrow Functions

```js
// Bad Practice: arrow function as an object method
const user = {
  name: "Alice",
  greet: () => {
    return `Hello, ${this.name}`; // 'this' is NOT the user object
  }
};
console.log(user.greet()); // "Hello, undefined"

// Good Practice: use shorthand method syntax
const user2 = {
  name: "Alice",
  greet() {
    return `Hello, ${this.name}`;
  }
};

// Bad Practice: arrow function as event handler (when you need 'this')
button.addEventListener("click", () => {
  this.classList.toggle("active"); // 'this' is NOT the button
});

// Good Practice: regular function for DOM event handlers
button.addEventListener("click", function () {
  this.classList.toggle("active"); // 'this' is the button
});
```

### `call`, `apply`, and `bind` Have No Effect on Arrow `this`

```js
const arrow = () => this;
const obj = { value: 42 };

console.log(arrow.call(obj) === obj);  // false
console.log(arrow.apply(obj) === obj); // false
console.log(arrow.bind(obj)() === obj); // false

// The 'this' of an arrow function is permanently locked at definition time
```

## Parameters

### Rest Parameters

Rest parameters collect remaining arguments into a real array.

```js
function sum(first, ...rest) {
  return rest.reduce((total, n) => total + n, first);
}

console.log(sum(1, 2, 3, 4)); // 10
```

Rules:
- Must be the last parameter.
- Only one rest parameter allowed.
- It is a real `Array` (unlike `arguments`).

### The `arguments` Object

Available only in non-arrow functions. It is array-like but not an array.

```js
function legacy() {
  console.log(arguments.length); // Number of args passed
  console.log(arguments[0]);     // First argument

  // Not an array: no map, filter, etc.
  // Convert to array:
  const args = Array.from(arguments);
  // or
  const args2 = [...arguments];
}
```

Gotcha in sloppy mode: `arguments` and named parameters are linked.

```js
function sloppy(a) {
  arguments[0] = 99;
  console.log(a); // 99 (linked in sloppy mode!)
}

function strict(a) {
  "use strict";
  arguments[0] = 99;
  console.log(a); // original value (independent in strict mode)
}
```

Always prefer rest parameters over `arguments`:

```js
// Bad Practice
function legacy() {
  const args = Array.prototype.slice.call(arguments);
  return args.join(", ");
}

// Good Practice
function modern(...args) {
  return args.join(", ");
}
```

### Default Parameters

Default values are evaluated at call time, not at definition time.

```js
function createList(items = []) {
  items.push("default");
  return items;
}

// A fresh array is created each call (unlike Python!)
console.log(createList()); // ["default"]
console.log(createList()); // ["default"] (not ["default", "default"])
```

Left-to-right evaluation: later defaults can reference earlier parameters.

```js
function greet(name, greeting = "Hello", message = `${greeting}, ${name}!`) {
  return message;
}

console.log(greet("Alice"));           // "Hello, Alice!"
console.log(greet("Alice", "Hi"));     // "Hi, Alice!"
```

Temporal Dead Zone in defaults: you cannot reference a later parameter in an earlier default.

```js
// ReferenceError: Cannot access 'y' before initialization
function broken(x = y, y = 10) {
  return x + y;
}

// This works: x is already initialized when y's default is evaluated
function fixed(x = 10, y = x * 2) {
  return x + y; // 30
}
```

`undefined` triggers defaults, `null` does not:

```js
function test(value = "default") {
  return value;
}

test(undefined); // "default"
test(null);      // null
test(0);         // 0
test("");        // ""
test(false);     // false
```

### Impact on `Function.length`

Parameters with defaults (and everything after them) are excluded from `Function.length`. Rest parameters are also excluded.

```js
function a(x, y, z) {}
function b(x, y = 1, z) {}
function c(x, ...rest) {}
function d({ a, b } = {}) {}

console.log(a.length); // 3
console.log(b.length); // 1 (only x, before first default)
console.log(c.length); // 1 (rest excluded)
console.log(d.length); // 0 (destructured with default)
```

## `Function.name` Property

JavaScript infers function names from context. This affects stack traces and debugging.

```js
// Declaration
function declared() {}
console.log(declared.name); // "declared"

// Named expression
const expr = function myName() {};
console.log(expr.name); // "myName" (the expression name wins)

// Anonymous expression assigned to variable
const anon = function () {};
console.log(anon.name); // "anon" (inferred from variable)

// Arrow function
const arrow = () => {};
console.log(arrow.name); // "arrow" (inferred)

// Method shorthand
const obj = { method() {} };
console.log(obj.method.name); // "method"

// Bound function
function original() {}
const bound = original.bind(null);
console.log(bound.name); // "bound original"

// Getter/Setter
const descriptor = Object.getOwnPropertyDescriptor({
  get value() {},
  set value(_) {}
}, "value");
console.log(descriptor.get.name); // "get value"
console.log(descriptor.set.name); // "set value"

// new Function()
const dynamic = new Function("return 1");
console.log(dynamic.name); // "anonymous"
```

## Closures

A closure is formed when a function retains a reference to variables from its enclosing scope, even after that scope has finished executing.

### Basic Closure

```js
function createCounter(initial = 0) {
  let count = initial;

  return {
    increment() { return ++count; },
    decrement() { return --count; },
    getCount() { return count; }
  };
}

const counter = createCounter(10);
console.log(counter.increment()); // 11
console.log(counter.increment()); // 12
console.log(counter.getCount());  // 12

// 'count' is not accessible from outside
console.log(counter.count); // undefined
```

### Function Factories

```js
function createMultiplier(factor) {
  return (number) => number * factor;
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5));  // 10
console.log(triple(5));  // 15
```

### The Classic Loop Closure Problem

```js
// Bug: all callbacks log 3
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100);
}
// Output: 3, 3, 3

// Fix 1: use 'let' (block-scoped, new binding per iteration)
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100);
}
// Output: 0, 1, 2

// Fix 2: IIFE (pre-ES6 pattern)
for (var i = 0; i < 3; i++) {
  (function (j) {
    setTimeout(() => console.log(j), 100);
  })(i);
}
// Output: 0, 1, 2
```

### Closure Memory Implications

Closures keep their entire enclosing scope alive, which can prevent garbage collection of large objects.

```js
// Bad Practice: unintentional memory retention
function processData() {
  const hugeArray = new Array(1_000_000).fill("data");
  const label = "result";

  // This closure only needs 'label', but 'hugeArray' is also retained
  // because both variables share the same scope
  return function getLabel() {
    return label;
  };
}

// Better: extract only what the closure needs
function processData() {
  const hugeArray = new Array(1_000_000).fill("data");
  const label = "result";
  const result = processArray(hugeArray);

  // hugeArray can be garbage collected after this point
  return function getResult() {
    return { label, result };
  };
}
```

## IIFE (Immediately Invoked Function Expression)

An IIFE runs immediately upon definition. While less common in modern module-based code, it still has valid use cases.

### Syntax

```js
// Standard IIFE
(function () {
  const privateVar = "hidden";
  console.log(privateVar);
})();

// Arrow IIFE
(() => {
  const privateVar = "hidden";
  console.log(privateVar);
})();

// With arguments
(function (global) {
  global.myLib = { version: "1.0" };
})(window);

// Named IIFE (better stack traces)
(function init() {
  // initialization code
})();
```

### Modern Use Cases

IIFEs are still useful in specific scenarios:

```js
// 1. Async IIFE for top-level await in non-module scripts
(async () => {
  const response = await fetch("/api/data");
  const data = await response.json();
  console.log(data);
})();

// 2. Complex constant initialization
const config = (() => {
  const env = process.env.NODE_ENV;
  const base = { apiUrl: "https://api.example.com" };

  if (env === "development") {
    return { ...base, apiUrl: "http://localhost:3000", debug: true };
  }
  if (env === "staging") {
    return { ...base, apiUrl: "https://staging.api.example.com" };
  }
  return base;
})();

// 3. Switch/if as an expression
const label = ((status) => {
  switch (status) {
    case 200: return "OK";
    case 404: return "Not Found";
    case 500: return "Server Error";
    default: return "Unknown";
  }
})(response.status);
```

### Modern Alternative: Block Scoping

For simple variable isolation, use a block with `let`/`const` instead:

```js
// Old IIFE pattern
(function () {
  var temp = computeValue();
  doSomething(temp);
})();

// Modern equivalent
{
  const temp = computeValue();
  doSomething(temp);
}
```

## Higher-Order Functions

A higher-order function takes a function as an argument, returns a function, or both.

### Taking Functions as Arguments

```js
function repeat(n, action) {
  for (let i = 0; i < n; i++) {
    action(i);
  }
}

repeat(3, console.log); // 0, 1, 2

// Array methods are higher-order functions
const numbers = [1, 2, 3, 4, 5];
const evens = numbers.filter((n) => n % 2 === 0);
const doubled = numbers.map((n) => n * 2);
const sum = numbers.reduce((acc, n) => acc + n, 0);
```

### Returning Functions

```js
function withLogging(fn) {
  return function (...args) {
    console.log(`Calling ${fn.name} with`, args);
    const result = fn(...args);
    console.log(`${fn.name} returned`, result);
    return result;
  };
}

function add(a, b) {
  return a + b;
}

const loggedAdd = withLogging(add);
loggedAdd(2, 3);
// Calling add with [2, 3]
// add returned 5
```

### Debounce

A classic higher-order function that limits how often a function can execute.

```js
function debounce(fn, delay) {
  let timerId;

  return function (...args) {
    clearTimeout(timerId);
    timerId = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// Only executes 300ms after the user stops typing
const handleSearch = debounce((query) => {
  fetch(`/api/search?q=${query}`);
}, 300);

input.addEventListener("input", (e) => handleSearch(e.target.value));
```

### Memoize

Caches function results to avoid redundant computation.

```js
function memoize(fn) {
  const cache = new Map();

  return function (...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      return cache.get(key);
    }
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

const fastFib = memoize(fibonacci);
// Note: this only caches the top-level call, not recursive calls.
// For full memoization, the recursive calls must also use the memoized version.
```

Proper recursive memoization:

```js
function memoizedFibonacci() {
  const cache = new Map();

  function fib(n) {
    if (cache.has(n)) return cache.get(n);
    if (n <= 1) return n;
    const result = fib(n - 1) + fib(n - 2);
    cache.set(n, result);
    return result;
  }

  return fib;
}

const fib = memoizedFibonacci();
console.log(fib(50)); // 12586269025 (instant)
```

## Function Composition

### Pipe and Compose

```js
// pipe: left to right
const pipe = (...fns) => (value) =>
  fns.reduce((acc, fn) => fn(acc), value);

// compose: right to left
const compose = (...fns) => (value) =>
  fns.reduceRight((acc, fn) => fn(acc), value);
```

```js
const trim = (s) => s.trim();
const toLower = (s) => s.toLowerCase();
const split = (delimiter) => (s) => s.split(delimiter);

const processInput = pipe(
  trim,
  toLower,
  split(" ")
);

console.log(processInput("  Hello World  ")); // ["hello", "world"]
```

### Tap for Debugging Pipelines

```js
const tap = (fn) => (value) => {
  fn(value);
  return value;
};

const process = pipe(
  trim,
  tap((v) => console.log("After trim:", v)),
  toLower,
  tap((v) => console.log("After toLower:", v)),
  split(" ")
);
```

### Async Pipe

Standard pipe does not handle Promises. You need an async variant:

```js
const pipeAsync = (...fns) => (value) =>
  fns.reduce((promise, fn) => promise.then(fn), Promise.resolve(value));

const fetchAndProcess = pipeAsync(
  (url) => fetch(url),
  (response) => response.json(),
  (data) => data.results,
  (results) => results.filter((r) => r.active)
);

fetchAndProcess("/api/users").then(console.log);
```

## Currying and Partial Application

### Currying

Transforms a multi-argument function into a chain of single-argument functions.

```js
// Manual curry
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return (...moreArgs) => curried.apply(this, [...args, ...moreArgs]);
  };
}

const add = (a, b, c) => a + b + c;
const curriedAdd = curry(add);

console.log(curriedAdd(1)(2)(3)); // 6
console.log(curriedAdd(1, 2)(3)); // 6
console.log(curriedAdd(1)(2, 3)); // 6
console.log(curriedAdd(1, 2, 3)); // 6
```

Practical use with array methods:

```js
const multiply = curry((factor, number) => factor * number);
const double = multiply(2);
const triple = multiply(3);

console.log([1, 2, 3].map(double)); // [2, 4, 6]
console.log([1, 2, 3].map(triple)); // [3, 6, 9]
```

### Partial Application with `bind`

`Function.prototype.bind` is a built-in way to do partial application.

```js
function log(level, timestamp, message) {
  console.log(`[${level}] ${timestamp}: ${message}`);
}

// Pre-fill the level argument
const logError = log.bind(null, "ERROR");
const logInfo = log.bind(null, "INFO");

logError(Date.now(), "Something went wrong");
logInfo(Date.now(), "Server started");
```

Gotcha: curry relies on `Function.length`, which does not account for default or rest parameters:

```js
const fn = (a, b = 10, ...rest) => a + b;
console.log(fn.length); // 1

// Currying this function will not work as expected
// because curry thinks fn only needs 1 argument
```

## Recursion

### Basics and Base Cases

```js
function factorial(n) {
  if (n <= 1) return 1; // base case
  return n * factorial(n - 1); // recursive case
}

console.log(factorial(5)); // 120
```

### Tail Call Optimization

When the recursive call is the very last operation in the function (tail position), engines can theoretically reuse the stack frame.

```js
// NOT tail-recursive: multiplication happens AFTER the recursive call
function factorial(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1); // must multiply after recursion returns
}

// Tail-recursive: the recursive call IS the last operation
function factorialTail(n, accumulator = 1) {
  if (n <= 1) return accumulator;
  return factorialTail(n - 1, n * accumulator); // tail position
}
```

**Critical limitation:** Only Safari implements Tail Call Optimization (TCO) as of 2025. V8 (Chrome, Node.js) and SpiderMonkey (Firefox) do not. You cannot rely on TCO in cross-browser production code.

### The Trampoline Pattern

A trampoline converts recursion to iteration by returning thunks instead of directly recursing.

```js
function trampoline(fn) {
  return function (...args) {
    let result = fn(...args);
    while (typeof result === "function") {
      result = result();
    }
    return result;
  };
}

const factorial = trampoline(function fact(n, acc = 1) {
  if (n <= 1) return acc;
  return () => fact(n - 1, n * acc); // return thunk, don't recurse
});

console.log(factorial(100000)); // Infinity (but no stack overflow)
```

### Converting Recursion to Iteration

Any recursive algorithm can be rewritten with an explicit stack. This is always the most performant approach.

```js
// Recursive tree traversal
function traverseRecursive(node, callback) {
  callback(node);
  for (const child of node.children) {
    traverseRecursive(child, callback);
  }
}

// Iterative tree traversal (explicit stack)
function traverseIterative(root, callback) {
  const stack = [root];
  while (stack.length > 0) {
    const node = stack.pop();
    callback(node);
    // Push children in reverse order to maintain left-to-right order
    for (let i = node.children.length - 1; i >= 0; i--) {
      stack.push(node.children[i]);
    }
  }
}
```

## Callback Patterns

### Error-First Callbacks (Node.js Convention)

```js
function readConfig(path, callback) {
  fs.readFile(path, "utf8", (err, data) => {
    if (err) {
      callback(err, null);
      return; // Always return after error callback
    }

    try {
      const config = JSON.parse(data);
      callback(null, config);
    } catch (parseError) {
      callback(parseError, null);
    }
  });
}

// Usage
readConfig("config.json", (err, config) => {
  if (err) {
    console.error("Failed to read config:", err.message);
    return;
  }
  console.log(config);
});
```

### The Zalgo Problem: Sync vs Async Callbacks

A function should call its callback either always synchronously or always asynchronously. Mixing both creates unpredictable behavior.

```js
// Bad Practice: sometimes sync, sometimes async ("releasing Zalgo")
function getData(key, callback) {
  if (cache[key]) {
    callback(null, cache[key]); // synchronous!
  } else {
    fetchFromServer(key, (err, data) => {
      cache[key] = data;
      callback(err, data); // asynchronous!
    });
  }
}

// Good Practice: always async
function getData(key, callback) {
  if (cache[key]) {
    queueMicrotask(() => callback(null, cache[key]));
    return;
  }
  fetchFromServer(key, (err, data) => {
    cache[key] = data;
    callback(err, data);
  });
}
```

### `this` Loss When Passing Methods as Callbacks

```js
const logger = {
  prefix: "[APP]",
  log(message) {
    console.log(`${this.prefix} ${message}`);
  }
};

// Bug: 'this' is lost when passing method as callback
setTimeout(logger.log, 100, "hello"); // "undefined hello"

// Fix 1: arrow function wrapper
setTimeout((msg) => logger.log(msg), 100, "hello");

// Fix 2: bind
setTimeout(logger.log.bind(logger), 100, "hello");
```

## Generator Functions

Generator functions use `function*` syntax and can pause execution with `yield`.

### Basics

```js
function* idGenerator() {
  let id = 1;
  while (true) {
    yield id++;
  }
}

const gen = idGenerator();
console.log(gen.next()); // { value: 1, done: false }
console.log(gen.next()); // { value: 2, done: false }
console.log(gen.next()); // { value: 3, done: false }
```

### Generators Are Iterable

```js
function* range(start, end, step = 1) {
  for (let i = start; i < end; i += step) {
    yield i;
  }
}

// Works with for...of
for (const n of range(0, 10, 2)) {
  console.log(n); // 0, 2, 4, 6, 8
}

// Works with spread
console.log([...range(1, 6)]); // [1, 2, 3, 4, 5]

// Works with destructuring
const [a, b, c] = range(10, 20);
console.log(a, b, c); // 10, 11, 12
```

### Sending Values into Generators

```js
function* conversation() {
  const name = yield "What is your name?";
  const age = yield `Hello ${name}! How old are you?`;
  return `${name} is ${age} years old.`;
}

const gen = conversation();
console.log(gen.next());        // { value: "What is your name?", done: false }
console.log(gen.next("Alice")); // { value: "Hello Alice! How old are you?", done: false }
console.log(gen.next(30));      // { value: "Alice is 30 years old.", done: true }
```

### `yield*` Delegation

```js
function* inner() {
  yield "a";
  yield "b";
  return "inner done"; // This becomes the value of the yield* expression
}

function* outer() {
  yield 1;
  const result = yield* inner(); // Delegates to inner
  console.log("Inner returned:", result); // "inner done"
  yield 2;
}

console.log([...outer()]); // [1, "a", "b", 2]
// Note: "inner done" is NOT yielded, it's only the return value of yield*
```

### `return()` and `throw()` on Generators

```js
function* gen() {
  try {
    yield 1;
    yield 2;
    yield 3;
  } finally {
    console.log("Cleanup!");
  }
}

const g = gen();
console.log(g.next());    // { value: 1, done: false }
console.log(g.return(99)); // "Cleanup!" -> { value: 99, done: true }
console.log(g.next());    // { value: undefined, done: true }
```

### Gotcha: `return` Value in `for...of`

```js
function* gen() {
  yield 1;
  yield 2;
  return 3; // This value is IGNORED by for...of
}

for (const v of gen()) {
  console.log(v); // 1, 2 (NOT 3)
}

// But .next() sees it
const g = gen();
g.next(); // { value: 1, done: false }
g.next(); // { value: 2, done: false }
g.next(); // { value: 3, done: true }
```

## `new Function()` and `eval`

Both execute strings as code. Both should be avoided in almost all cases.

### `eval`

Executes code in the current scope. Can read and modify local variables.

```js
let x = 10;
eval("x = 20");
console.log(x); // 20
```

### `new Function()`

Creates a function that executes in the global scope only. Cannot access local variables.

```js
let x = 10;
const fn = new Function("return x");
// fn can only access global 'x', not the local one
```

### Why to Avoid Both

1. **Security**: code injection and XSS if user input is involved.
2. **Performance**: `eval` prevents engine optimizations in the surrounding scope.
3. **CSP violations**: Content Security Policy blocks `unsafe-eval` by default.
4. **Debugging**: dynamically generated code is harder to debug and trace.

### Alternatives

```js
// Bad Practice: eval for dynamic property access
eval(`obj.${propertyName}`);

// Good Practice: bracket notation
obj[propertyName];

// Bad Practice: eval for JSON parsing
eval("(" + jsonString + ")");

// Good Practice: JSON.parse
JSON.parse(jsonString);

// Bad Practice: eval for dynamic dispatch
eval(`handle${eventType}()`);

// Good Practice: lookup table
const handlers = {
  click: handleClick,
  hover: handleHover
};
handlers[eventType]?.();
```

## Common Mistakes

- **Forgetting `return` in block-body arrow functions**:

```js
// Bug: returns undefined
const double = (x) => { x * 2 };

// Fix: add return or use expression body
const double = (x) => x * 2;
```

- **Using arrow functions where dynamic `this` is needed**: Object methods, prototype methods, event handlers that need `this` to be the element.

- **Assuming `null` triggers default parameters**: Only `undefined` triggers defaults. `null`, `0`, `""`, and `false` do not.

- **Creating functions inside loops unnecessarily**:

```js
// Bad Practice: new function object each iteration
for (let i = 0; i < items.length; i++) {
  items[i].addEventListener("click", function () {
    console.log(i);
  });
}

// Better: hoist the handler if it doesn't need per-iteration state
function handleClick(event) {
  const index = items.indexOf(event.currentTarget);
  console.log(index);
}
for (let i = 0; i < items.length; i++) {
  items[i].addEventListener("click", handleClick);
}
```

- **Not handling `this` when passing methods as callbacks**: The `this` context is lost. Use `.bind()` or an arrow function wrapper.

- **Relying on TCO in cross-browser code**: Only Safari supports Proper Tail Calls. Use trampolines or iteration for deep recursion.

- **Mixing synchronous and asynchronous callbacks**: Always be consistent. Use `queueMicrotask()` or `setTimeout(fn, 0)` to make synchronous paths asynchronous.

## Best Practices

- Use function declarations for top-level named functions. Their hoisting makes code readable regardless of definition order.
- Use arrow functions for callbacks and inline functions where you want lexical `this`.
- Use shorthand method syntax (`method() {}`) for object methods.
- Keep functions small and single-purpose. If a function does more than one thing, split it.
- Limit parameters to three or fewer. Use an options object for functions with many arguments.

```js
// Bad Practice: too many positional parameters
function createUser(name, age, email, role, department, isActive) { /* ... */ }

// Good Practice: options object
function createUser({ name, age, email, role = "user", department = null, isActive = true }) {
  // ...
}
```

- Use early returns (guard clauses) to reduce nesting.

```js
// Bad Practice: deeply nested
function process(user) {
  if (user) {
    if (user.isActive) {
      if (user.hasPermission) {
        return doWork(user);
      }
    }
  }
  return null;
}

// Good Practice: guard clauses
function process(user) {
  if (!user) return null;
  if (!user.isActive) return null;
  if (!user.hasPermission) return null;
  return doWork(user);
}
```

- Always name functions used in stack traces (avoid anonymous functions in key positions).
- Prefer rest parameters over the `arguments` object.
- Use `structuredClone` or explicit copying when a function needs to avoid mutating its arguments.

## Real-World Use Cases

### Middleware Pattern

```js
function createApp() {
  const middlewares = [];

  return {
    use(fn) {
      middlewares.push(fn);
    },

    async handle(request) {
      let index = 0;

      async function next() {
        if (index < middlewares.length) {
          const middleware = middlewares[index++];
          await middleware(request, next);
        }
      }

      await next();
      return request;
    }
  };
}

const app = createApp();

app.use(async (req, next) => {
  req.startTime = Date.now();
  await next();
  req.duration = Date.now() - req.startTime;
});

app.use(async (req, next) => {
  console.log(`${req.method} ${req.url}`);
  await next();
});

app.use(async (req, next) => {
  req.response = { status: 200, body: "OK" };
});
```

### Retry with Exponential Backoff

```js
function withRetry(fn, { maxAttempts = 3, baseDelay = 1000 } = {}) {
  return async function (...args) {
    let lastError;

    for (let attempt = 1; attempt <= maxAttempts; attempt++) {
      try {
        return await fn.apply(this, args);
      } catch (error) {
        lastError = error;
        if (attempt < maxAttempts) {
          const delay = baseDelay * Math.pow(2, attempt - 1);
          const jitter = delay * Math.random() * 0.1;
          await new Promise((r) => setTimeout(r, delay + jitter));
        }
      }
    }

    throw lastError;
  };
}

const fetchWithRetry = withRetry(fetch, { maxAttempts: 3, baseDelay: 500 });
const response = await fetchWithRetry("/api/flaky-endpoint");
```

### Validation Pipeline

```js
function createValidator(...rules) {
  return function validate(value) {
    const errors = [];

    for (const rule of rules) {
      const error = rule(value);
      if (error) errors.push(error);
    }

    return {
      valid: errors.length === 0,
      errors
    };
  };
}

// Reusable validation rules (each is a function)
const required = (value) =>
  value == null || value === "" ? "Value is required" : null;

const minLength = (min) => (value) =>
  typeof value === "string" && value.length < min
    ? `Must be at least ${min} characters`
    : null;

const pattern = (regex, message) => (value) =>
  typeof value === "string" && !regex.test(value) ? message : null;

// Compose validators
const validateEmail = createValidator(
  required,
  minLength(5),
  pattern(/^[^\s@]+@[^\s@]+\.[^\s@]+$/, "Invalid email format")
);

console.log(validateEmail(""));
// { valid: false, errors: ["Value is required", "Must be at least 5 characters", "Invalid email format"] }

console.log(validateEmail("alice@example.com"));
// { valid: true, errors: [] }
```
