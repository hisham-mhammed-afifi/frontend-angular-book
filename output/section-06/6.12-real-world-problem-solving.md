## Debounce

Debounce delays the execution of a function until a specified time has passed since the last call. If the function is called again within that window, the timer resets. The function only fires once the caller stops calling.

**Use cases:** Search input (wait until user stops typing), window resize handler, form validation, auto-save.

### Basic Debounce

```javascript
function debounce(fn, delay) {
  let timerId = null;

  return function (...args) {
    clearTimeout(timerId);
    timerId = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

// Usage
const searchInput = document.querySelector("#search");
const handleSearch = debounce((event) => {
  console.log("Searching for:", event.target.value);
  // API call here
}, 300);

searchInput.addEventListener("input", handleSearch);
// Only fires 300ms after the user STOPS typing
```

### Debounce with Leading and Trailing Options

Lodash's debounce supports two modes:
- **Trailing (default):** Fires after the delay expires (the standard behavior above).
- **Leading:** Fires immediately on the first call, then suppresses until the delay expires.

```javascript
function debounce(fn, delay, options = {}) {
  let timerId = null;
  let lastArgs = null;
  const leading = options.leading ?? false;
  const trailing = options.trailing ?? true;

  function invoke() {
    if (lastArgs !== null && trailing) {
      fn.apply(this, lastArgs);
    }
    lastArgs = null;
    timerId = null;
  }

  const debounced = function (...args) {
    // If no timer is running and leading is true, fire immediately
    if (timerId === null && leading) {
      fn.apply(this, args);
    }

    lastArgs = args;
    clearTimeout(timerId);
    timerId = setTimeout(invoke, delay);
  };

  debounced.cancel = function () {
    clearTimeout(timerId);
    timerId = null;
    lastArgs = null;
  };

  debounced.flush = function () {
    if (timerId !== null) {
      clearTimeout(timerId);
      invoke();
    }
  };

  return debounced;
}

// Leading: fires on first click, ignores rapid clicks for 1s
const handleClick = debounce(
  () => console.log("Button clicked"),
  1000,
  { leading: true, trailing: false }
);

// Trailing (default): fires after user stops resizing for 200ms
const handleResize = debounce(
  () => console.log("Resize complete"),
  200
);
```

### Why `fn.apply(this, args)` Matters

Using `apply` preserves the `this` context of the original call site. Without it, `this` inside the debounced function would be `undefined` (strict mode) or `window`.

```javascript
const obj = {
  name: "Widget",
  // BAD: arrow function in setTimeout captures the wrong `this`
  // if debounce does not use apply
  logName: debounce(function () {
    console.log(this.name);
  }, 100),
};

obj.logName(); // "Widget" (correct because debounce uses fn.apply(this, args))
```

## Throttle

Throttle ensures a function runs at most once every `interval` milliseconds. Unlike debounce, throttle guarantees periodic execution during continuous events.

**Use cases:** Scroll handlers, mouse move tracking, progress bars, infinite scroll loading, game input polling.

### Basic Throttle

```javascript
function throttle(fn, interval) {
  let lastTime = 0;

  return function (...args) {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}

// Usage
window.addEventListener(
  "scroll",
  throttle(() => {
    console.log("Scroll position:", window.scrollY);
  }, 100)
);
// Fires at most once every 100ms while scrolling
```

### Throttle with Leading and Trailing

A more complete throttle fires on the leading edge and ensures one trailing call happens after the interval.

```javascript
function throttle(fn, interval, options = {}) {
  let lastTime = 0;
  let timerId = null;
  let lastArgs = null;
  const leading = options.leading ?? true;
  const trailing = options.trailing ?? true;

  const debounced = function (...args) {
    const now = Date.now();
    const remaining = interval - (now - lastTime);

    if (remaining <= 0) {
      // Enough time has passed
      if (timerId !== null) {
        clearTimeout(timerId);
        timerId = null;
      }
      lastTime = now;
      if (leading || lastTime !== 0) {
        fn.apply(this, args);
      }
    } else if (timerId === null && trailing) {
      // Schedule a trailing call
      lastArgs = args;
      timerId = setTimeout(() => {
        lastTime = Date.now();
        timerId = null;
        fn.apply(this, lastArgs);
        lastArgs = null;
      }, remaining);
    }
  };

  debounced.cancel = function () {
    clearTimeout(timerId);
    timerId = null;
    lastTime = 0;
    lastArgs = null;
  };

  return debounced;
}
```

### Throttle with requestAnimationFrame

For visual updates (DOM mutations, animations), throttle using `requestAnimationFrame` for smooth 60fps updates.

```javascript
function throttleRAF(fn) {
  let frameId = null;

  const throttled = function (...args) {
    if (frameId !== null) return;

    frameId = requestAnimationFrame(() => {
      fn.apply(this, args);
      frameId = null;
    });
  };

  throttled.cancel = function () {
    if (frameId !== null) {
      cancelAnimationFrame(frameId);
      frameId = null;
    }
  };

  return throttled;
}

// Smooth parallax on scroll
const updateParallax = throttleRAF(() => {
  const offset = window.scrollY * 0.5;
  document.querySelector(".parallax").style.transform =
    `translateY(${offset}px)`;
});

window.addEventListener("scroll", updateParallax);
```

### Debounce vs Throttle Summary

| Aspect | Debounce | Throttle |
|--------|----------|----------|
| When it fires | After inactivity | At regular intervals |
| Guarantees execution during continuous events | No | Yes |
| Best for | Final value (search, save) | Continuous feedback (scroll, resize) |
| Typical delay | 200-500ms | 16-200ms |

## Deep Clone

Creating an independent copy of a nested object where changes to the clone do not affect the original.

### The Problem with Shallow Copy

```javascript
const original = {
  name: "Alice",
  scores: [90, 85, 92],
  address: { city: "NYC", zip: "10001" },
};

// Shallow copies share nested references
const shallow = { ...original };
shallow.scores.push(100);
console.log(original.scores); // [90, 85, 92, 100] <- MUTATED!

const shallow2 = Object.assign({}, original);
shallow2.address.city = "LA";
console.log(original.address.city); // "LA" <- MUTATED!
```

### structuredClone (Modern Standard)

The built-in `structuredClone()` is the recommended approach for deep cloning in modern JavaScript (supported in all browsers and Node 17+).

```javascript
const original = {
  name: "Alice",
  scores: [90, 85, 92],
  address: { city: "NYC", zip: "10001" },
  created: new Date("2025-01-01"),
  pattern: /test/gi,
  data: new Map([["key", "value"]]),
  tags: new Set(["js", "ts"]),
};

const clone = structuredClone(original);

clone.scores.push(100);
clone.address.city = "LA";

console.log(original.scores);       // [90, 85, 92] (safe)
console.log(original.address.city); // "NYC" (safe)
console.log(clone.created instanceof Date); // true (preserved)
console.log(clone.data instanceof Map);     // true (preserved)
console.log(clone.tags instanceof Set);     // true (preserved)
```

**What structuredClone handles:**
- Nested objects and arrays
- Date, RegExp, Map, Set, ArrayBuffer, typed arrays
- Circular references
- Error objects

**What structuredClone does NOT handle:**
- Functions (throws `DataCloneError`)
- DOM nodes (throws `DataCloneError`)
- Class prototype chains (returns plain objects)
- Symbols as property keys (ignored)

### JSON.parse(JSON.stringify()) and Its Pitfalls

The old approach. It works for plain data but silently corrupts many types.

```javascript
const original = {
  name: "Alice",
  created: new Date("2025-01-01"),
  pattern: /test/gi,
  callback: () => "hello",
  value: undefined,
  sym: Symbol("id"),
  nan: NaN,
  inf: Infinity,
};

const clone = JSON.parse(JSON.stringify(original));

console.log(clone.created);   // "2025-01-01T00:00:00.000Z" (string, not Date!)
console.log(clone.pattern);   // {} (empty object, not RegExp!)
console.log(clone.callback);  // undefined (function silently dropped!)
console.log(clone.value);     // missing (undefined properties dropped!)
console.log(clone.sym);       // missing (Symbol properties dropped!)
console.log(clone.nan);       // null (NaN becomes null!)
console.log(clone.inf);       // null (Infinity becomes null!)

// Also throws on circular references:
const circular = { a: 1 };
circular.self = circular;
// JSON.stringify(circular) -> TypeError: Converting circular structure to JSON
```

### Manual Deep Clone (Interview Implementation)

When you need to handle functions or class instances, or for interview purposes:

```javascript
function deepClone(value, seen = new WeakMap()) {
  // Primitives and null
  if (value === null || typeof value !== "object") {
    return value;
  }

  // Handle circular references
  if (seen.has(value)) {
    return seen.get(value);
  }

  // Handle Date
  if (value instanceof Date) {
    return new Date(value.getTime());
  }

  // Handle RegExp
  if (value instanceof RegExp) {
    return new RegExp(value.source, value.flags);
  }

  // Handle Map
  if (value instanceof Map) {
    const mapClone = new Map();
    seen.set(value, mapClone);
    for (const [k, v] of value) {
      mapClone.set(deepClone(k, seen), deepClone(v, seen));
    }
    return mapClone;
  }

  // Handle Set
  if (value instanceof Set) {
    const setClone = new Set();
    seen.set(value, setClone);
    for (const v of value) {
      setClone.add(deepClone(v, seen));
    }
    return setClone;
  }

  // Handle Array
  if (Array.isArray(value)) {
    const arrClone = [];
    seen.set(value, arrClone);
    for (let i = 0; i < value.length; i++) {
      arrClone[i] = deepClone(value[i], seen);
    }
    return arrClone;
  }

  // Handle plain objects
  const objClone = Object.create(Object.getPrototypeOf(value));
  seen.set(value, objClone);

  for (const key of Reflect.ownKeys(value)) {
    const descriptor = Object.getOwnPropertyDescriptor(value, key);
    if (descriptor.value !== undefined) {
      descriptor.value = deepClone(descriptor.value, seen);
    }
    Object.defineProperty(objClone, key, descriptor);
  }

  return objClone;
}

// Test with circular reference
const obj = { a: 1, b: { c: 2 } };
obj.self = obj;
const cloned = deepClone(obj);
console.log(cloned.self === cloned); // true (circular ref preserved)
console.log(cloned.b === obj.b);     // false (deep copy)
```

### Cloning Comparison

| Feature | Spread / Object.assign | JSON roundtrip | structuredClone | Manual deepClone |
|---------|----------------------|----------------|-----------------|-----------------|
| Shallow copy | Yes | No | No | No |
| Nested objects | No | Yes | Yes | Yes |
| Date | No | Becomes string | Yes | Yes |
| RegExp | No | Becomes `{}` | Yes | Yes |
| Map / Set | No | Becomes `{}` | Yes | Yes |
| Functions | Reference only | Dropped | Throws | Customizable |
| Circular refs | No | Throws | Yes | Yes (with WeakMap) |
| Prototype chain | No | Lost | Lost | Preservable |
| Performance | Fastest | Slow for large data | Fast (native) | Depends on impl |

**Rule of thumb:** Use `structuredClone()` for data. Use manual deep clone only when you need prototype preservation or function cloning.

## Flatten

### Flatten an Array

Convert a nested array into a single-level array.

#### Built-in: Array.prototype.flat()

```javascript
const nested = [1, [2, 3], [4, [5, [6]]]];

console.log(nested.flat());         // [1, 2, 3, 4, [5, [6]]]  depth 1 (default)
console.log(nested.flat(2));        // [1, 2, 3, 4, 5, [6]]     depth 2
console.log(nested.flat(Infinity)); // [1, 2, 3, 4, 5, 6]       fully flat
```

#### Manual Implementation (Recursive)

```javascript
function flattenArray(arr, depth = Infinity) {
  const result = [];

  for (const item of arr) {
    if (Array.isArray(item) && depth > 0) {
      result.push(...flattenArray(item, depth - 1));
    } else {
      result.push(item);
    }
  }

  return result;
}

console.log(flattenArray([1, [2, [3, [4]]]]));    // [1, 2, 3, 4]
console.log(flattenArray([1, [2, [3, [4]]]], 1)); // [1, 2, [3, [4]]]
```

#### Manual Implementation (Iterative with Stack)

Avoids stack overflow for deeply nested arrays.

```javascript
function flattenIterative(arr) {
  const stack = [...arr];
  const result = [];

  while (stack.length > 0) {
    const item = stack.pop();
    if (Array.isArray(item)) {
      stack.push(...item);
    } else {
      result.push(item);
    }
  }

  return result.reverse(); // reverse because we used pop (LIFO)
}

console.log(flattenIterative([1, [2, [3, [4]]]])); // [1, 2, 3, 4]
```

#### Using reduce

```javascript
function flattenReduce(arr, depth = Infinity) {
  return arr.reduce((acc, item) => {
    if (Array.isArray(item) && depth > 0) {
      return acc.concat(flattenReduce(item, depth - 1));
    }
    return acc.concat(item);
  }, []);
}
```

### Flatten a Nested Object

Convert a deeply nested object into a single-level object with dot-notation keys.

```javascript
function flattenObject(obj, prefix = "", result = {}) {
  for (const [key, value] of Object.entries(obj)) {
    const newKey = prefix ? `${prefix}.${key}` : key;

    if (
      value !== null &&
      typeof value === "object" &&
      !Array.isArray(value) &&
      !(value instanceof Date)
    ) {
      flattenObject(value, newKey, result);
    } else {
      result[newKey] = value;
    }
  }

  return result;
}

const nested = {
  name: "Alice",
  address: {
    street: "123 Main St",
    city: "NYC",
    geo: { lat: 40.7, lng: -74.0 },
  },
  scores: [90, 85],
};

console.log(flattenObject(nested));
// {
//   "name": "Alice",
//   "address.street": "123 Main St",
//   "address.city": "NYC",
//   "address.geo.lat": 40.7,
//   "address.geo.lng": -74.0,
//   "scores": [90, 85]
// }
```

### Unflatten an Object

Reverse the process: dot-notation keys back to nested object.

```javascript
function unflattenObject(obj) {
  const result = {};

  for (const [key, value] of Object.entries(obj)) {
    const parts = key.split(".");
    let current = result;

    for (let i = 0; i < parts.length - 1; i++) {
      if (!(parts[i] in current)) {
        // Check if next part is a number (array index)
        current[parts[i]] = /^\d+$/.test(parts[i + 1]) ? [] : {};
      }
      current = current[parts[i]];
    }

    current[parts[parts.length - 1]] = value;
  }

  return result;
}

const flat = {
  "name": "Alice",
  "address.street": "123 Main St",
  "address.city": "NYC",
  "address.geo.lat": 40.7,
};

console.log(unflattenObject(flat));
// {
//   name: "Alice",
//   address: { street: "123 Main St", city: "NYC", geo: { lat: 40.7 } }
// }
```

## Event Emitter

An event emitter implements the publish-subscribe (pub/sub) pattern. Components can emit events and other components can subscribe to those events without direct coupling.

### Basic Event Emitter

```javascript
class EventEmitter {
  constructor() {
    this.events = new Map();
  }

  // Subscribe to an event. Returns an unsubscribe function.
  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event).push(listener);

    // Return unsubscribe function
    return () => this.off(event, listener);
  }

  // Subscribe to an event, but only fire once
  once(event, listener) {
    const wrapper = (...args) => {
      listener.apply(this, args);
      this.off(event, wrapper);
    };
    wrapper._original = listener; // for off() comparison
    return this.on(event, wrapper);
  }

  // Unsubscribe a specific listener
  off(event, listener) {
    if (!this.events.has(event)) return;

    const listeners = this.events.get(event).filter(
      (l) => l !== listener && l._original !== listener
    );

    if (listeners.length === 0) {
      this.events.delete(event);
    } else {
      this.events.set(event, listeners);
    }
  }

  // Emit an event with data
  emit(event, ...args) {
    if (!this.events.has(event)) return false;

    // Create a copy to avoid issues if a listener calls off() during iteration
    const listeners = [...this.events.get(event)];
    for (const listener of listeners) {
      listener.apply(this, args);
    }
    return true;
  }

  // Remove all listeners for an event (or all events)
  removeAllListeners(event) {
    if (event) {
      this.events.delete(event);
    } else {
      this.events.clear();
    }
  }

  // Get the count of listeners for an event
  listenerCount(event) {
    return this.events.has(event) ? this.events.get(event).length : 0;
  }
}
```

### Usage

```javascript
const emitter = new EventEmitter();

// Subscribe
const unsubscribe = emitter.on("message", (data) => {
  console.log("Received:", data);
});

emitter.on("message", (data) => {
  console.log("Also received:", data);
});

// Once listener
emitter.once("connect", () => {
  console.log("Connected! (only fires once)");
});

// Emit
emitter.emit("message", { text: "Hello" });
// Received: { text: "Hello" }
// Also received: { text: "Hello" }

emitter.emit("connect");
// Connected! (only fires once)

emitter.emit("connect");
// (nothing, once listener already removed)

// Unsubscribe using returned function
unsubscribe();

emitter.emit("message", { text: "After unsub" });
// Also received: { text: "After unsub" }
// (first listener was removed)
```

### Type-Safe Event Emitter

Using a pattern that enforces event names and payload types at call sites.

```javascript
// Using JSDoc for type safety without TypeScript
/**
 * @template {Record<string, any[]>} Events
 */
class TypedEventEmitter {
  constructor() {
    /** @type {Map<string, Function[]>} */
    this.events = new Map();
  }

  /**
   * @template {keyof Events} K
   * @param {K} event
   * @param {(...args: Events[K]) => void} listener
   */
  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event).push(listener);
    return () => this.off(event, listener);
  }

  /**
   * @template {keyof Events} K
   * @param {K} event
   * @param {...Events[K]} args
   */
  emit(event, ...args) {
    const listeners = this.events.get(event) || [];
    for (const listener of [...listeners]) {
      listener(...args);
    }
  }

  off(event, listener) {
    if (!this.events.has(event)) return;
    this.events.set(
      event,
      this.events.get(event).filter((l) => l !== listener)
    );
  }
}
```

## Curry and Partial Application

Currying transforms a function that takes multiple arguments into a chain of functions that each take a single argument.

### Basic Curry

```javascript
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    }
    return function (...moreArgs) {
      return curried.apply(this, [...args, ...moreArgs]);
    };
  };
}

function add(a, b, c) {
  return a + b + c;
}

const curriedAdd = curry(add);

console.log(curriedAdd(1)(2)(3));     // 6
console.log(curriedAdd(1, 2)(3));     // 6
console.log(curriedAdd(1)(2, 3));     // 6
console.log(curriedAdd(1, 2, 3));     // 6
```

### Infinite Curry (sum(1)(2)(3)...())

A variation where the function accumulates values until called with no arguments.

```javascript
function sum(initial) {
  let total = initial;

  function inner(next) {
    if (next === undefined) return total;
    total += next;
    return inner;
  }

  return inner;
}

console.log(sum(1)(2)(3)());          // 6
console.log(sum(10)(20)(30)(40)());   // 100

// Alternative using valueOf for implicit conversion
function sumAuto(initial) {
  function inner(next) {
    return sumAuto(initial + next);
  }
  inner.valueOf = () => initial;
  inner.toString = () => String(initial);
  return inner;
}

console.log(+sumAuto(1)(2)(3));       // 6 (unary + triggers valueOf)
```

### Partial Application

Fix some arguments of a function, producing a new function with fewer parameters.

```javascript
function partial(fn, ...presetArgs) {
  return function (...laterArgs) {
    return fn.apply(this, [...presetArgs, ...laterArgs]);
  };
}

function log(level, timestamp, message) {
  console.log(`[${level}] ${timestamp}: ${message}`);
}

const warn = partial(log, "WARN");
const error = partial(log, "ERROR");

warn("2025-03-01", "Disk space low");
// [WARN] 2025-03-01: Disk space low

error("2025-03-01", "Connection failed");
// [ERROR] 2025-03-01: Connection failed
```

## Memoize

Cache the results of expensive function calls and return the cached result when the same inputs occur again.

```javascript
function memoize(fn, keyResolver) {
  const cache = new Map();

  const memoized = function (...args) {
    const key = keyResolver ? keyResolver(...args) : args[0];

    if (cache.has(key)) {
      return cache.get(key);
    }

    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };

  memoized.cache = cache;
  return memoized;
}

// Simple usage
const factorial = memoize(function f(n) {
  if (n <= 1) return 1;
  return n * f(n - 1); // Note: inner calls bypass memoize
});

// To memoize recursive calls, reassign the reference
let fib = function (n) {
  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
};
fib = memoize(fib);

console.log(fib(40)); // 102334155 (instant, not 2^40 calls)

// Multi-argument with custom key resolver
const expensiveLookup = memoize(
  (userId, orgId) => {
    console.log("Fetching...");
    return { userId, orgId, data: "..." };
  },
  (userId, orgId) => `${userId}:${orgId}`
);

expensiveLookup(1, 100); // "Fetching..."
expensiveLookup(1, 100); // (cached, no log)
expensiveLookup(1, 200); // "Fetching..." (different key)
```

### Memoize with LRU Cache

Prevent unbounded memory growth by evicting the least recently used entry.

```javascript
function memoizeLRU(fn, maxSize = 100) {
  const cache = new Map();

  return function (...args) {
    const key = JSON.stringify(args);

    if (cache.has(key)) {
      const value = cache.get(key);
      // Move to end (most recently used)
      cache.delete(key);
      cache.set(key, value);
      return value;
    }

    const result = fn.apply(this, args);
    cache.set(key, result);

    if (cache.size > maxSize) {
      // Delete the oldest (first) entry
      const firstKey = cache.keys().next().value;
      cache.delete(firstKey);
    }

    return result;
  };
}
```

## Common Mistakes

- **Debouncing scroll handlers instead of throttling.** Debounce waits until scrolling stops, so the UI gets no updates while the user scrolls. Throttle provides smooth periodic updates. Use debounce for "done" events (search, resize), throttle for "ongoing" events (scroll, drag).

- **Not preserving `this` context in debounce/throttle.** Using an arrow function for the returned wrapper prevents `fn.apply(this, args)` from forwarding `this` correctly. Always use a regular function expression for the wrapper.

```javascript
// BAD: arrow function loses `this`
function debounce(fn, delay) {
  let timerId;
  return (...args) => {  // arrow function captures outer `this`
    clearTimeout(timerId);
    timerId = setTimeout(() => fn.apply(this, args), delay); // `this` is wrong
  };
}

// GOOD: regular function preserves call-site `this`
function debounce(fn, delay) {
  let timerId;
  return function (...args) {  // regular function
    clearTimeout(timerId);
    timerId = setTimeout(() => fn.apply(this, args), delay); // `this` is correct
  };
}
```

- **Using JSON.parse(JSON.stringify()) for deep cloning.** It silently drops functions, converts Date to strings, turns RegExp to empty objects, and throws on circular references. Use `structuredClone()` for data cloning.

- **Mutating the original during flatten.** Using methods like `splice()` on the input array while iterating modifies the source. Always build a new result array.

- **Not handling the once() listener correctly in Event Emitter.** If the wrapper function is not properly tracked, calling `off()` with the original listener reference fails to remove it. Store a reference to the original on the wrapper (`wrapper._original = listener`).

- **Unbounded memoization cache.** Memoizing without eviction can leak memory over time in long-running applications. Add an LRU limit or use `WeakMap` when keys are objects.

- **Forgetting to handle edge cases in curry.** Functions with `fn.length === 0` (rest params, default params) break arity detection. `fn.length` only counts parameters before the first default or rest parameter.

---

## Best Practices

- **Use `structuredClone()` as the default deep clone.** It handles the vast majority of cases correctly and performantly. Fall back to manual cloning only when you need function or prototype preservation.

- **Always return an unsubscribe/cancel function.** Debounce should expose `.cancel()`. Event emitter's `on()` should return an unsubscribe function. This makes cleanup easy and prevents memory leaks.

- **Throttle visual updates with requestAnimationFrame.** For any DOM manipulation triggered by scroll/resize, `requestAnimationFrame` is the right primitive. It syncs with the browser's paint cycle and avoids layout thrashing.

- **Test debounce and throttle with fake timers.** Use `jest.useFakeTimers()` or `sinon.useFakeTimers()` to control time and test that the right number of calls happen at the right times.

- **Prefer `Array.prototype.flat(Infinity)` over manual flatten.** The built-in is optimized by the engine and handles edge cases. Only implement manual flatten for interview situations or environments without the built-in.

- **Use WeakMap for circular reference tracking in deep clone.** A regular Map would prevent garbage collection of the cloned objects. WeakMap lets the garbage collector clean up when references are no longer held.

### 1. Search-As-You-Type with Debounce

```javascript
class SearchComponent {
  constructor(inputEl, resultsEl) {
    this.inputEl = inputEl;
    this.resultsEl = resultsEl;
    this.abortController = null;

    this.search = debounce(this.fetchResults.bind(this), 300);
    this.inputEl.addEventListener("input", (e) => this.search(e.target.value));
  }

  async fetchResults(query) {
    if (query.length < 2) {
      this.resultsEl.innerHTML = "";
      return;
    }

    // Cancel previous in-flight request
    if (this.abortController) {
      this.abortController.abort();
    }
    this.abortController = new AbortController();

    try {
      const response = await fetch(`/api/search?q=${encodeURIComponent(query)}`, {
        signal: this.abortController.signal,
      });
      const data = await response.json();
      this.renderResults(data.results);
    } catch (err) {
      if (err.name !== "AbortError") {
        console.error("Search failed:", err);
      }
    }
  }

  renderResults(results) {
    this.resultsEl.innerHTML = results
      .map((r) => `<div class="result">${r.title}</div>`)
      .join("");
  }
}
```

### 2. Application Event Bus

A centralized event bus for decoupled communication between modules.

```javascript
class EventBus extends EventEmitter {
  constructor() {
    super();
    this.history = [];
    this.maxHistory = 100;
  }

  emit(event, ...args) {
    this.history.push({
      event,
      args,
      timestamp: Date.now(),
    });

    if (this.history.length > this.maxHistory) {
      this.history.shift();
    }

    return super.emit(event, ...args);
  }

  // Replay missed events for late subscribers
  replay(event, listener) {
    const past = this.history.filter((h) => h.event === event);
    for (const entry of past) {
      listener(...entry.args);
    }
    return this.on(event, listener);
  }

  getHistory(event) {
    if (event) {
      return this.history.filter((h) => h.event === event);
    }
    return [...this.history];
  }
}

const bus = new EventBus();

// Module A publishes user events
bus.emit("user:login", { id: 1, name: "Alice" });
bus.emit("user:action", { type: "click", target: "buy-btn" });

// Module B subscribes later and replays missed events
bus.replay("user:login", (user) => {
  console.log("User logged in:", user.name);
});
// Immediately logs: "User logged in: Alice"
```

### 3. Configuration Merger with Deep Clone and Flatten

Merge user config with defaults, flatten for storage, unflatten for usage.

```javascript
function mergeDeep(target, source) {
  const result = structuredClone(target);

  for (const [key, value] of Object.entries(source)) {
    if (
      value !== null &&
      typeof value === "object" &&
      !Array.isArray(value) &&
      key in result &&
      typeof result[key] === "object"
    ) {
      result[key] = mergeDeep(result[key], value);
    } else {
      result[key] = structuredClone(value);
    }
  }

  return result;
}

const defaults = {
  theme: { mode: "light", fontSize: 14, colors: { primary: "#007bff" } },
  editor: { tabSize: 2, wordWrap: true },
  notifications: { email: true, push: false },
};

const userPrefs = {
  theme: { mode: "dark", colors: { primary: "#00ff88" } },
  notifications: { push: true },
};

const merged = mergeDeep(defaults, userPrefs);
console.log(merged);
// {
//   theme: { mode: "dark", fontSize: 14, colors: { primary: "#00ff88" } },
//   editor: { tabSize: 2, wordWrap: true },
//   notifications: { email: true, push: true }
// }

// Flatten for localStorage
const flat = flattenObject(merged);
localStorage.setItem("config", JSON.stringify(flat));

// Unflatten when reading back
const stored = JSON.parse(localStorage.getItem("config"));
const config = unflattenObject(stored);
```
