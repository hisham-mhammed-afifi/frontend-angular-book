# 7.12 Utility Types

> Utility types are TypeScript's built-in generic types that transform existing types into new ones. They eliminate boilerplate by letting you derive variations of your types (optional fields, readonly properties, subsets, exclusions) without rewriting definitions.

---

## Why Utility Types Matter

Without utility types, you would need to manually create every variation of a type:

```typescript
// Without utility types: manual duplication
interface User {
  id: string;
  name: string;
  email: string;
  password: string;
}

// You want a type for updates (all optional)
interface UserUpdate {
  id?: string;
  name?: string;
  email?: string;
  password?: string;
}

// You want a type without password
interface PublicUser {
  id: string;
  name: string;
  email: string;
}
```

With utility types, you derive these from the source:

```typescript
type UserUpdate = Partial<User>;
type PublicUser = Omit<User, "password">;
```

One source of truth, zero duplication.

## Property Modifier Utility Types

These types change property modifiers (optional, readonly) across all properties.

### `Partial<T>`

Makes every property optional.

```typescript
type Partial<T> = {
  [K in keyof T]?: T[K];
};
```

```typescript
interface User {
  id: string;
  name: string;
  email: string;
}

type UserUpdate = Partial<User>;
// { id?: string; name?: string; email?: string; }

function updateUser(id: string, changes: Partial<User>): User {
  const existing = getUser(id);
  return { ...existing, ...changes };
}

updateUser("1", { name: "Bob" });           // only name
updateUser("1", { name: "Bob", email: "b@example.com" }); // name and email
updateUser("1", {});                         // nothing (valid)
```

### `Required<T>`

Makes every property required (removes `?` modifiers).

```typescript
type Required<T> = {
  [K in keyof T]-?: T[K];
};
```

```typescript
interface Config {
  host?: string;
  port?: number;
  ssl?: boolean;
}

type ResolvedConfig = Required<Config>;
// { host: string; port: number; ssl: boolean; }

function startServer(config: Required<Config>) {
  // All properties guaranteed to exist
  console.log(`Starting on ${config.host}:${config.port}`);
}

// Must provide all properties
startServer({ host: "localhost", port: 3000, ssl: false });
```

### `Readonly<T>`

Makes every property readonly.

```typescript
type Readonly<T> = {
  readonly [K in keyof T]: T[K];
};
```

```typescript
interface State {
  count: number;
  items: string[];
}

type FrozenState = Readonly<State>;

function processState(state: FrozenState) {
  // state.count = 10; // Error: Cannot assign to 'count' because it is read-only
  console.log(state.count);

  // Note: Readonly is shallow. Array methods still work:
  // state.items.push("new"); // No compile error! Runtime mutation.
}
```

`Readonly` is shallow. For deep immutability, you need a custom `DeepReadonly` (covered later).

## Property Selection Utility Types

These types create new types by including or excluding specific properties.

### `Pick<T, K>`

Constructs a type with only the specified properties.

```typescript
type Pick<T, K extends keyof T> = {
  [P in K]: T[P];
};
```

```typescript
interface User {
  id: string;
  name: string;
  email: string;
  password: string;
  createdAt: Date;
}

type UserPreview = Pick<User, "id" | "name">;
// { id: string; name: string; }

type LoginCredentials = Pick<User, "email" | "password">;
// { email: string; password: string; }
```

### `Omit<T, K>`

Constructs a type with all properties except the specified ones.

```typescript
type Omit<T, K extends string | number | symbol> = Pick<T, Exclude<keyof T, K>>;
```

```typescript
type PublicUser = Omit<User, "password">;
// { id: string; name: string; email: string; createdAt: Date; }

type CreateUserInput = Omit<User, "id" | "createdAt">;
// { name: string; email: string; password: string; }
```

Note: `Omit`'s key parameter does not require `K extends keyof T`. You can omit keys that do not exist, which is by design but can mask typos:

```typescript
type Test = Omit<User, "pasword">; // No error! "pasword" is just ignored
// Still has all 5 properties because "pasword" does not match "password"
```

### `Pick` vs `Omit`: When to Use Which

| Scenario | Use |
|----------|-----|
| You want a few specific fields | `Pick` |
| You want everything except a few fields | `Omit` |
| The included set is smaller | `Pick` |
| The excluded set is smaller | `Omit` |

```typescript
// Pick when you want few fields
type Coordinates = Pick<FullLocation, "lat" | "lng">;

// Omit when you want to remove few fields
type SafeUser = Omit<User, "password" | "ssn">;
```

## Object Construction Utility Types

### `Record<K, V>`

Creates an object type with keys of type `K` and values of type `V`.

```typescript
type Record<K extends string | number | symbol, T> = {
  [P in K]: T;
};
```

```typescript
// Simple key-value map
type StringMap = Record<string, string>;

// Exhaustive mapping from a union
type Status = "active" | "inactive" | "pending";
type StatusInfo = Record<Status, { label: string; color: string }>;

const statusMap: StatusInfo = {
  active: { label: "Active", color: "green" },
  inactive: { label: "Inactive", color: "gray" },
  pending: { label: "Pending", color: "yellow" },
  // Missing any status is a compile error
};
```

```typescript
// Record vs index signature
// These are equivalent:
type A = Record<string, number>;
type B = { [key: string]: number };

// But Record is required for union keys:
type Scores = Record<"math" | "science" | "english", number>;
// { math: number; science: number; english: number }
```

## Union Filtering Utility Types

These types operate on union types, not object types.

### `Exclude<T, U>`

Removes members from a union that are assignable to `U`.

```typescript
type Exclude<T, U> = T extends U ? never : T;
```

```typescript
type Status = "active" | "inactive" | "pending" | "deleted";

type ActiveStatuses = Exclude<Status, "deleted">;
// "active" | "inactive" | "pending"

type NonStringPrimitive = Exclude<string | number | boolean, string>;
// number | boolean
```

### `Extract<T, U>`

Keeps only members from a union that are assignable to `U`.

```typescript
type Extract<T, U> = T extends U ? T : never;
```

```typescript
type Status = "active" | "inactive" | "pending" | "deleted";

type DangerStatuses = Extract<Status, "deleted" | "inactive">;
// "deleted" | "inactive"

type StringOrNumber = Extract<string | number | boolean | null, string | number>;
// string | number
```

### `NonNullable<T>`

Removes `null` and `undefined` from a union.

```typescript
type NonNullable<T> = Exclude<T, null | undefined>;
```

```typescript
type MaybeString = string | null | undefined;

type DefiniteString = NonNullable<MaybeString>;
// string

// Practical use with mapped types
type User = {
  id: string;
  name: string;
  nickname: string | null;
};

function getDisplayName(user: User): string {
  const name: NonNullable<User["nickname"]> | string = user.nickname ?? user.name;
  return name;
}
```

## Function Utility Types

These types extract type information from function types.

### `ReturnType<T>`

Extracts the return type of a function type.

```typescript
type ReturnType<T extends (...args: any) => any> = T extends (...args: any) => infer R ? R : any;
```

```typescript
function createUser(name: string, email: string) {
  return { id: crypto.randomUUID(), name, email, createdAt: new Date() };
}

type User = ReturnType<typeof createUser>;
// { id: string; name: string; email: string; createdAt: Date }

// Useful when you do not have an explicit type definition
// but want to reference what a function returns
```

### `Parameters<T>`

Extracts the parameter types as a tuple.

```typescript
type Parameters<T extends (...args: any) => any> = T extends (...args: infer P) => any ? P : never;
```

```typescript
function search(query: string, page: number, limit: number): SearchResult[] {
  // ...
  return [];
}

type SearchParams = Parameters<typeof search>;
// [query: string, page: number, limit: number]

// Useful for wrapping functions
function loggedSearch(...args: Parameters<typeof search>): SearchResult[] {
  console.log("Searching:", args);
  return search(...args);
}
```

### `ConstructorParameters<T>`

Extracts constructor parameter types.

```typescript
class HttpClient {
  constructor(
    private baseUrl: string,
    private timeout: number,
    private headers?: Record<string, string>,
  ) {}
}

type HttpClientArgs = ConstructorParameters<typeof HttpClient>;
// [baseUrl: string, timeout: number, headers?: Record<string, string>]
```

### `InstanceType<T>`

Extracts the instance type of a constructor.

```typescript
class User {
  name: string;
  constructor(name: string) {
    this.name = name;
  }
}

type UserInstance = InstanceType<typeof User>;
// User

// Useful with generic factory functions
function create<T extends new (...args: any[]) => any>(
  ctor: T,
  ...args: ConstructorParameters<T>
): InstanceType<T> {
  return new ctor(...args);
}

const user = create(User, "Alice"); // User
```

### `Awaited<T>`

Recursively unwraps `Promise` types. Mimics the behavior of `await`.

```typescript
type A = Awaited<Promise<string>>;           // string
type B = Awaited<Promise<Promise<number>>>;  // number
type C = Awaited<boolean>;                   // boolean (not a Promise)
type D = Awaited<string | Promise<number>>;  // string | number
```

```typescript
// Practical: typing the result of Promise.all
async function fetchData() {
  const [users, posts] = await Promise.all([fetchUsers(), fetchPosts()]);
  // users is Awaited<ReturnType<typeof fetchUsers>>
  // posts is Awaited<ReturnType<typeof fetchPosts>>
}
```

## String Manipulation Utility Types

These transform string literal types.

```typescript
type Uppercase<S extends string>;
type Lowercase<S extends string>;
type Capitalize<S extends string>;
type Uncapitalize<S extends string>;
```

```typescript
type Shouting = Uppercase<"hello world">;   // "HELLO WORLD"
type Whisper = Lowercase<"HELLO">;          // "hello"
type Title = Capitalize<"hello">;           // "Hello"
type Lower = Uncapitalize<"Hello">;         // "hello"

// Distribute over unions
type Events = "click" | "focus" | "blur";
type HandlerNames = `on${Capitalize<Events>}`;
// "onClick" | "onFocus" | "onBlur"
```

## `ThisType<T>`

Controls what `this` refers to inside an object literal. Used with `noImplicitThis`.

```typescript
type ObjectDescriptor<D, M> = {
  data?: D;
  methods?: M & ThisType<D & M>;
};

function makeObject<D, M>(desc: ObjectDescriptor<D, M>): D & M {
  const data: object = desc.data || {};
  const methods: object = desc.methods || {};
  return { ...data, ...methods } as D & M;
}

const obj = makeObject({
  data: { x: 0, y: 0 },
  methods: {
    moveBy(dx: number, dy: number) {
      // 'this' is typed as { x: number; y: number; moveBy: ... }
      this.x += dx;
      this.y += dy;
    },
  },
});

obj.moveBy(5, 10);
```

## `NoInfer<T>`

Added in TypeScript 5.4. Prevents inference from a specific position, forcing TypeScript to infer the type from other arguments.

```typescript
function createFSM<T extends string>(config: {
  initial: NoInfer<T>;
  states: T[];
}) {
  // ...
}

// Without NoInfer, T would be inferred from both 'initial' and 'states'
// With NoInfer on 'initial', T is inferred only from 'states'

createFSM({
  initial: "idle",
  states: ["idle", "loading", "success", "error"],
});

// If initial has a typo:
// createFSM({
//   initial: "idel", // Error: not assignable to "idle" | "loading" | "success" | "error"
//   states: ["idle", "loading", "success", "error"],
// });
```

## Composing Utility Types

The real power emerges when you combine utility types.

### Partial + Omit for Update Payloads

```typescript
interface User {
  id: string;
  name: string;
  email: string;
  role: "admin" | "user";
  createdAt: Date;
}

// Update payload: all fields optional, but id and createdAt cannot be changed
type UserUpdate = Partial<Omit<User, "id" | "createdAt">>;
// { name?: string; email?: string; role?: "admin" | "user"; }
```

### Pick + Required for Specific Required Fields

```typescript
interface FormData {
  name?: string;
  email?: string;
  phone?: string;
  address?: string;
}

// Name and email are required, rest stays optional
type RequiredContactInfo = Required<Pick<FormData, "name" | "email">> &
  Omit<FormData, "name" | "email">;
// { name: string; email: string; phone?: string; address?: string; }
```

### Record + Extract for Subset Mappings

```typescript
type AllEvents = "click" | "focus" | "blur" | "submit" | "reset";
type FormEvents = Extract<AllEvents, "submit" | "reset" | "focus" | "blur">;

type FormHandlers = Record<FormEvents, (event: Event) => void>;
// {
//   submit: (event: Event) => void;
//   focus: (event: Event) => void;
//   blur: (event: Event) => void;
//   reset: (event: Event) => void;
// }
```

### Readonly + Pick for Immutable Subsets

```typescript
type ImmutableId = Readonly<Pick<User, "id" | "createdAt">>;
// { readonly id: string; readonly createdAt: Date; }

type MutableUser = Omit<User, "id" | "createdAt"> & ImmutableId;
// name, email, role are mutable; id, createdAt are readonly
```

## Building Custom Utility Types

### `DeepPartial<T>`

Recursively makes all nested properties optional.

```typescript
type DeepPartial<T> = {
  [K in keyof T]?: T[K] extends object
    ? T[K] extends Function
      ? T[K]
      : DeepPartial<T[K]>
    : T[K];
};

interface Config {
  database: {
    host: string;
    port: number;
    credentials: {
      user: string;
      password: string;
    };
  };
  cache: {
    enabled: boolean;
    ttl: number;
  };
}

type ConfigOverrides = DeepPartial<Config>;
// Can override any nested property without providing the full structure

const overrides: ConfigOverrides = {
  database: {
    credentials: {
      password: "new-password",
      // host, port, user all omitted
    },
  },
  // cache entirely omitted
};
```

### `DeepReadonly<T>`

Recursively makes all nested properties readonly.

```typescript
type DeepReadonly<T> = {
  readonly [K in keyof T]: T[K] extends Function
    ? T[K]
    : T[K] extends object
      ? DeepReadonly<T[K]>
      : T[K];
};

type FrozenConfig = DeepReadonly<Config>;
// All nested properties are readonly
// frozenConfig.database.host = "x"; // Error at every level
```

### `PickByType<T, V>`

Pick properties whose values match a specific type.

```typescript
type PickByType<T, V> = {
  [K in keyof T as T[K] extends V ? K : never]: T[K];
};

interface User {
  id: number;
  name: string;
  email: string;
  age: number;
  isActive: boolean;
}

type StringFields = PickByType<User, string>;
// { name: string; email: string }

type NumberFields = PickByType<User, number>;
// { id: number; age: number }
```

### `OmitByType<T, V>`

Remove properties whose values match a specific type.

```typescript
type OmitByType<T, V> = {
  [K in keyof T as T[K] extends V ? never : K]: T[K];
};

type NonStringFields = OmitByType<User, string>;
// { id: number; age: number; isActive: boolean }
```

### `Mutable<T>`

Remove readonly from all properties (opposite of `Readonly`).

```typescript
type Mutable<T> = {
  -readonly [K in keyof T]: T[K];
};

type MutableUser = Mutable<Readonly<User>>;
// All properties are writable again
```

## Common Mistakes

### Mistake 1: Using `Omit` with Typos in Key Names

```typescript
interface User {
  id: string;
  name: string;
  password: string;
}

// Typo in key name: no error!
type Safe = Omit<User, "pasword">;
// Still has password because "pasword" does not match

// Fix: use a helper that enforces valid keys
type StrictOmit<T, K extends keyof T> = Omit<T, K>;

// type Safe = StrictOmit<User, "pasword">; // Error: "pasword" is not in keyof User
type Safe = StrictOmit<User, "password">; // Works correctly
```

### Mistake 2: Expecting `Readonly` to Be Deep

```typescript
const state: Readonly<{ items: string[]; config: { debug: boolean } }> = {
  items: ["a"],
  config: { debug: false },
};

// state.items = []; // Error: readonly
state.items.push("b"); // No error! Array is still mutable
state.config.debug = true; // No error! Nested object is mutable

// Fix: use DeepReadonly for true immutability
```

### Mistake 3: Using `Partial` When Only Some Fields Should Be Optional

```typescript
interface CreateUserInput {
  name: string;
  email: string;
  role?: "admin" | "user";
}

// Bad: makes ALL fields optional, including required ones
type UpdateInput = Partial<CreateUserInput>;
// Now name and email are optional too

// Good: be specific about what is optional
type UpdateInput = Partial<Pick<CreateUserInput, "name" | "email">> &
  Pick<CreateUserInput, "role">;
```

### Mistake 4: Confusing `Extract`/`Exclude` with `Pick`/`Omit`

```typescript
// Extract/Exclude operate on UNION TYPES
type Status = "active" | "inactive" | "deleted";
type Active = Extract<Status, "active" | "inactive">; // "active" | "inactive"
type Safe = Exclude<Status, "deleted">;                // "active" | "inactive"

// Pick/Omit operate on OBJECT TYPES
type User = { id: string; name: string; password: string };
type Public = Pick<User, "id" | "name">;    // { id: string; name: string }
type Safe2 = Omit<User, "password">;        // { id: string; name: string }

// Do not mix them up:
// Pick<Status, "active"> // Error: Status is not an object type
// Exclude<User, "password"> // Does not do what you expect
```

### Mistake 5: Forgetting That Utility Types Are Shallow

```typescript
type Config = {
  db: { host: string; port: number };
  cache: { enabled: boolean };
};

type PartialConfig = Partial<Config>;
// { db?: { host: string; port: number }; cache?: { enabled: boolean }; }
// db itself is optional, but if provided, ALL of db's properties are required

const config: PartialConfig = {
  db: { host: "localhost" }, // Error: port is required within db
};
```

## Best Practices

### 1. Derive Types From a Single Source

```typescript
// Define the base type once
interface User {
  id: string;
  name: string;
  email: string;
  password: string;
  role: "admin" | "user";
  createdAt: Date;
  updatedAt: Date;
}

// Derive everything else
type CreateUserInput = Omit<User, "id" | "createdAt" | "updatedAt">;
type UpdateUserInput = Partial<Omit<User, "id" | "createdAt" | "updatedAt">>;
type PublicUser = Omit<User, "password">;
type UserPreview = Pick<User, "id" | "name">;
```

### 2. Create Domain-Specific Utility Types

```typescript
// Project-wide utility for entity operations
type CreateInput<T> = Omit<T, "id" | "createdAt" | "updatedAt">;
type UpdateInput<T> = Partial<CreateInput<T>>;
type PublicView<T> = T extends { password: string } ? Omit<T, "password"> : T;

// Clean, consistent across all entities
type CreateUser = CreateInput<User>;
type UpdateUser = UpdateInput<User>;
type SafeUser = PublicView<User>;
```

### 3. Use `Record` for Exhaustive Mappings

```typescript
type Permission = "read" | "write" | "delete";

// Record ensures every permission is mapped
const permissionLabels: Record<Permission, string> = {
  read: "Can read resources",
  write: "Can create and modify",
  delete: "Can delete resources",
  // Forgetting any permission is a compile error
};
```

### 4. Prefer `satisfies` with `Record` for Validated Maps

```typescript
const config = {
  development: { apiUrl: "http://localhost:3000", debug: true },
  staging: { apiUrl: "https://staging.api.com", debug: true },
  production: { apiUrl: "https://api.com", debug: false },
} as const satisfies Record<string, { apiUrl: string; debug: boolean }>;

// Keys are known precisely, values are literal types, and structure is validated
```

### 5. Compose Utility Types Instead of Writing Complex Mapped Types

```typescript
// Instead of writing a custom mapped type:
type ReadonlyPartialUser = {
  readonly [K in keyof User]?: User[K];
};

// Compose built-in utilities:
type ReadonlyPartialUser = Readonly<Partial<User>>;
```

## Real-World Use Cases

### API Layer with Consistent Type Derivation

```typescript
interface BaseEntity {
  id: string;
  createdAt: string;
  updatedAt: string;
}

interface User extends BaseEntity {
  name: string;
  email: string;
  role: "admin" | "editor" | "viewer";
}

interface Post extends BaseEntity {
  title: string;
  content: string;
  authorId: string;
  published: boolean;
  tags: string[];
}

// Generic CRUD types
type CreateDTO<T extends BaseEntity> = Omit<T, keyof BaseEntity>;
type UpdateDTO<T extends BaseEntity> = Partial<CreateDTO<T>>;
type ListItem<T extends BaseEntity> = Pick<T, "id"> & Partial<Omit<T, "id">>;

// Derived automatically
type CreateUserDTO = CreateDTO<User>;
// { name: string; email: string; role: "admin" | "editor" | "viewer" }

type UpdatePostDTO = UpdateDTO<Post>;
// { title?: string; content?: string; authorId?: string; published?: boolean; tags?: string[] }

// API functions with consistent typing
async function create<T extends BaseEntity>(
  endpoint: string,
  data: CreateDTO<T>,
): Promise<T> {
  const res = await fetch(endpoint, {
    method: "POST",
    body: JSON.stringify(data),
  });
  return res.json();
}

async function update<T extends BaseEntity>(
  endpoint: string,
  id: string,
  data: UpdateDTO<T>,
): Promise<T> {
  const res = await fetch(`${endpoint}/${id}`, {
    method: "PATCH",
    body: JSON.stringify(data),
  });
  return res.json();
}
```

### Form State Management

```typescript
interface FormField<T> {
  value: T;
  error: string | null;
  touched: boolean;
  dirty: boolean;
}

type FormState<T extends Record<string, any>> = {
  [K in keyof T]: FormField<T[K]>;
};

type FormValues<T extends Record<string, any>> = {
  [K in keyof T]: T[K];
};

type FormErrors<T extends Record<string, any>> = Partial<
  Record<keyof T, string>
>;

// Usage
interface LoginForm {
  email: string;
  password: string;
  rememberMe: boolean;
}

type LoginFormState = FormState<LoginForm>;
// {
//   email: FormField<string>;
//   password: FormField<string>;
//   rememberMe: FormField<boolean>;
// }

function getValues<T extends Record<string, any>>(
  state: FormState<T>,
): FormValues<T> {
  const values = {} as FormValues<T>;
  for (const key in state) {
    values[key] = state[key].value;
  }
  return values;
}

function getErrors<T extends Record<string, any>>(
  state: FormState<T>,
): FormErrors<T> {
  const errors: FormErrors<T> = {};
  for (const key in state) {
    if (state[key].error) {
      errors[key] = state[key].error!;
    }
  }
  return errors;
}
```

### Permission System with Mapped Types

```typescript
type Resource = "user" | "post" | "comment" | "settings";
type Action = "create" | "read" | "update" | "delete";

type PermissionKey = `${Resource}:${Action}`;

type PermissionMap = Record<PermissionKey, boolean>;

type Role = "admin" | "editor" | "viewer";
type RolePermissions = Record<Role, Partial<PermissionMap>>;

const rolePermissions: RolePermissions = {
  admin: {
    "user:create": true,
    "user:read": true,
    "user:update": true,
    "user:delete": true,
    "post:create": true,
    "post:read": true,
    "post:update": true,
    "post:delete": true,
    "comment:create": true,
    "comment:read": true,
    "comment:update": true,
    "comment:delete": true,
    "settings:read": true,
    "settings:update": true,
  },
  editor: {
    "post:create": true,
    "post:read": true,
    "post:update": true,
    "comment:create": true,
    "comment:read": true,
    "comment:update": true,
    "comment:delete": true,
  },
  viewer: {
    "post:read": true,
    "comment:read": true,
  },
};

function hasPermission(
  role: Role,
  resource: Resource,
  action: Action,
): boolean {
  const key: PermissionKey = `${resource}:${action}`;
  return rolePermissions[role][key] === true;
}
```

> **Key Takeaway:** Utility types let you derive new types from existing ones without duplication. Master `Partial`, `Required`, `Pick`, `Omit`, `Record`, `Exclude`, `Extract`, `ReturnType`, and `Parameters` as your core toolkit, then compose them and build custom utilities like `DeepPartial` and `PickByType` for project-specific needs.
