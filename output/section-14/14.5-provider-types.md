# 14.5 Provider Types

> Angular's DI system supports four provider strategies (`useClass`, `useValue`, `useFactory`, `useExisting`) that control how dependency instances are created and delivered.

---

## The Provider Object

Every provider is a mapping from a token (what you ask for) to a recipe (how Angular creates it). The full syntax is an object with a `provide` property and one of the four strategy properties.

```typescript
{
  provide: SomeToken,     // the DI token (class, InjectionToken, or string)
  useClass: SomeClass,    // OR
  useValue: someValue,    // OR
  useFactory: someFactory, // OR
  useExisting: AnotherToken,
}
```

### Shorthand Syntax

When the token and the class are the same, Angular accepts a shorthand:

```typescript
// Shorthand
providers: [UserService]

// Expands to
providers: [{ provide: UserService, useClass: UserService }]
```

---

## `useClass`

Tells Angular to create an instance of the specified class when the token is requested. Angular calls `new` on the class and resolves its own dependencies through DI.

### Basic Usage

```typescript
providers: [
  { provide: Logger, useClass: Logger },
]
```

### Substituting an Implementation

The real power of `useClass` is providing a different class than the token:

```typescript
// The token
export abstract class Logger {
  abstract log(message: string): void;
  abstract error(message: string, error?: Error): void;
}

// Production implementation
@Injectable()
export class RemoteLogger extends Logger {
  private http = inject(HttpClient);

  log(message: string) {
    this.http.post('/api/logs', { level: 'info', message }).subscribe();
  }

  error(message: string, error?: Error) {
    this.http.post('/api/logs', { level: 'error', message, stack: error?.stack }).subscribe();
  }
}

// Development implementation
@Injectable()
export class ConsoleLogger extends Logger {
  log(message: string) {
    console.log(`[LOG] ${message}`);
  }

  error(message: string, error?: Error) {
    console.error(`[ERROR] ${message}`, error);
  }
}

// app.config.ts
import { environment } from '../environments/environment';

export const appConfig: ApplicationConfig = {
  providers: [
    {
      provide: Logger,
      useClass: environment.production ? RemoteLogger : ConsoleLogger,
    },
  ],
};

// Any component or service injects Logger and gets the right implementation
@Injectable({ providedIn: 'root' })
export class OrderService {
  private logger = inject(Logger);

  placeOrder(order: Order) {
    this.logger.log(`Placing order ${order.id}`);
    // ...
  }
}
```

### `useClass` Creates a New Instance

Every `useClass` entry creates its own instance. This matters when aliasing:

```typescript
// Bad Practice: useClass creates TWO separate instances
providers: [
  NewLogger,
  { provide: OldLogger, useClass: NewLogger }, // SECOND instance of NewLogger
]
```

If you want `OldLogger` and `NewLogger` to resolve to the SAME instance, use `useExisting` instead.

### Tree-Shakable `useClass` with `@Injectable`

```typescript
@Injectable({
  providedIn: 'root',
  useClass: environment.production ? ProdAnalytics : DevAnalytics,
})
export class AnalyticsService {
  // This is the token. Angular creates ProdAnalytics or DevAnalytics.
}
```

This is equivalent to providing `useClass` in the providers array but with tree-shaking support.

## `useValue`

Provides a static value directly. Angular does not call `new` or invoke any constructor. It returns the exact value you provide.

### Primitives and Objects

```typescript
export const API_URL = new InjectionToken<string>('API_URL');
export const MAX_RETRIES = new InjectionToken<number>('MAX_RETRIES');
export const FEATURE_FLAGS = new InjectionToken<Record<string, boolean>>('FEATURE_FLAGS');

providers: [
  { provide: API_URL, useValue: 'https://api.example.com' },
  { provide: MAX_RETRIES, useValue: 3 },
  { provide: FEATURE_FLAGS, useValue: { darkMode: true, betaSearch: false } },
]
```

### Functions

```typescript
export type DateFormatter = (date: Date) => string;
export const DATE_FORMATTER = new InjectionToken<DateFormatter>('DATE_FORMATTER');

providers: [
  {
    provide: DATE_FORMATTER,
    useValue: (date: Date) => date.toISOString().split('T')[0],
  },
]

// Usage
@Component({ ... })
export class InvoiceComponent {
  private formatDate = inject(DATE_FORMATTER);

  formattedDate = computed(() => this.formatDate(this.invoice().createdAt));
}
```

### Pre-Constructed Instances

```typescript
const loggerInstance = new ConsoleLogger();
loggerInstance.setLevel('debug');

providers: [
  { provide: Logger, useValue: loggerInstance },
]
```

`useValue` returns the exact same object reference every time. It does not clone or copy.

### Testing with `useValue`

`useValue` is the most common provider type in tests because it lets you supply mocks without creating real services:

```typescript
const mockUserService = {
  getAll: () => [{ id: '1', name: 'Test User' }],
  getById: (id: string) => ({ id, name: 'Test User' }),
};

TestBed.configureTestingModule({
  providers: [
    { provide: UserService, useValue: mockUserService },
  ],
});
```

## `useFactory`

Provides a factory function that Angular calls to create the value. The factory runs within an injection context, so `inject()` works inside it.

### Basic Factory

```typescript
providers: [
  {
    provide: CacheService,
    useFactory: () => {
      const config = inject(APP_CONFIG);
      return new CacheService(config.cacheMaxSize, config.cacheTtl);
    },
  },
]
```

### Conditional Logic in Factories

```typescript
providers: [
  {
    provide: StorageService,
    useFactory: () => {
      const platformId = inject(PLATFORM_ID);

      if (isPlatformBrowser(platformId)) {
        return new BrowserStorageService();
      }

      return new InMemoryStorageService();
    },
  },
]
```

### Factory with `deps` (Legacy Pattern)

Before `inject()` was available in factory functions, the `deps` array was used to specify dependencies. The values are injected as positional arguments to the factory function:

```typescript
// Legacy pattern: still works but inject() is preferred
providers: [
  {
    provide: ApiClient,
    useFactory: (http: HttpClient, config: AppConfig) => {
      return new ApiClient(http, config.apiUrl);
    },
    deps: [HttpClient, APP_CONFIG],
  },
]
```

The modern equivalent using `inject()`:

```typescript
// Modern pattern: cleaner, no deps array needed
providers: [
  {
    provide: ApiClient,
    useFactory: () => {
      const http = inject(HttpClient);
      const config = inject(APP_CONFIG);
      return new ApiClient(http, config.apiUrl);
    },
  },
]
```

### Optional Dependencies in `deps`

When using `deps`, wrap optional tokens to avoid errors:

```typescript
// Legacy: optional dependency in deps
import { Optional } from '@angular/core';

providers: [
  {
    provide: LoggerService,
    useFactory: (analytics?: AnalyticsService) => {
      return new LoggerService(analytics);
    },
    deps: [[new Optional(), AnalyticsService]],
  },
]

// Modern: cleaner with inject()
providers: [
  {
    provide: LoggerService,
    useFactory: () => {
      const analytics = inject(AnalyticsService, { optional: true });
      return new LoggerService(analytics);
    },
  },
]
```

### Async Initialization Patterns

Factories must return synchronously. You cannot `await` inside a factory. For async initialization, return the service and let it initialize lazily:

```typescript
// Bad Practice: async factory
{
  provide: ConfigService,
  useFactory: async () => {
    const response = await fetch('/config.json'); // WRONG: factories must be sync
    return new ConfigService(await response.json());
  },
}

// Correct: return the service, let it load internally
{
  provide: ConfigService,
  useFactory: () => {
    const http = inject(HttpClient);
    const service = new ConfigService();
    service.load(http); // triggers async loading internally
    return service;
  },
}
```

For true async initialization before the application starts, use `APP_INITIALIZER` (covered in section 14.6).

## `useExisting`

Creates an alias. When the token is requested, Angular resolves the aliased token and returns that same instance. No new instance is created.

### Basic Aliasing

```typescript
@Injectable({ providedIn: 'root' })
export class LoggerService {
  log(msg: string) { console.log(msg); }
  warn(msg: string) { console.warn(msg); }
  error(msg: string) { console.error(msg); }
}

providers: [
  // OldLogger resolves to the SAME instance as LoggerService
  { provide: OldLogger, useExisting: LoggerService },
]
```

Both `inject(LoggerService)` and `inject(OldLogger)` return the exact same object.

### Interface Segregation with `useExisting`

When a service has many methods but a consumer only needs a subset, create a narrow interface token that aliases the full service:

```typescript
// Full service
@Injectable({ providedIn: 'root' })
export class UserService {
  getAll(): User[] { ... }
  getById(id: string): User | undefined { ... }
  create(data: CreateUserDto): User { ... }
  update(id: string, data: UpdateUserDto): User { ... }
  delete(id: string): void { ... }
}

// Narrow read-only token
export abstract class UserReader {
  abstract getAll(): User[];
  abstract getById(id: string): User | undefined;
}

// Alias: UserReader resolves to the same UserService instance
providers: [
  { provide: UserReader, useExisting: UserService },
]

// Components that only read users inject the narrow token
@Component({ ... })
export class UserListComponent {
  private users = inject(UserReader); // only getAll() and getById() are visible
}

// Components that manage users inject the full service
@Component({ ... })
export class UserAdminComponent {
  private users = inject(UserService); // all methods available
}
```

This follows the Interface Segregation Principle: consumers depend only on the methods they need.

### `useExisting` vs `useClass`: The Critical Difference

```typescript
@Injectable({ providedIn: 'root' })
export class AuthService {
  currentUser = signal<User | null>(null);
}

// useExisting: SAME instance (correct for aliasing)
providers: [
  { provide: OldAuthService, useExisting: AuthService },
]
// inject(OldAuthService).currentUser === inject(AuthService).currentUser  // true

// useClass: DIFFERENT instance (wrong for aliasing!)
providers: [
  { provide: OldAuthService, useClass: AuthService },
]
// inject(OldAuthService).currentUser !== inject(AuthService).currentUser  // true: different signals!
```

If you want two tokens to resolve to the same instance, always use `useExisting`.

---

## Comparison Table

| Strategy | Creates New Instance | Use When |
|---|---|---|
| `useClass` | Yes (per provider scope) | Substituting a class with a different implementation |
| `useValue` | No (returns the exact value) | Static values, config objects, mock objects in tests |
| `useFactory` | Yes (factory runs once per scope) | Dynamic creation based on runtime conditions or other dependencies |
| `useExisting` | No (aliases to existing instance) | Creating an alias, narrowing an interface, backward compatibility |

## The `provide` Function Pattern

Angular's own APIs (like `provideRouter`, `provideHttpClient`) use a pattern of exporting functions that return provider arrays. This is the recommended approach for library authors and for grouping related providers.

```typescript
export interface ToastConfig {
  position: 'top-right' | 'top-left' | 'bottom-right' | 'bottom-left';
  duration: number;
  maxVisible: number;
}

const TOAST_CONFIG = new InjectionToken<ToastConfig>('TOAST_CONFIG');

export function provideToast(config: Partial<ToastConfig> = {}): Provider[] {
  const defaults: ToastConfig = {
    position: 'top-right',
    duration: 5000,
    maxVisible: 3,
  };

  return [
    {
      provide: TOAST_CONFIG,
      useValue: { ...defaults, ...config },
    },
    ToastService,
  ];
}

// Consumer
export const appConfig: ApplicationConfig = {
  providers: [
    provideToast({ position: 'bottom-right', duration: 3000 }),
  ],
};
```

Benefits:
- Type-safe configuration.
- Encapsulated internal tokens (the consumer never sees `TOAST_CONFIG`).
- Follows the same pattern as Angular's own `provideRouter()`.

## Provider Scope and Lifetime

The scope of a provider determines when the factory or constructor runs and how many instances exist.

| Registration Location | Scope | Instances |
|---|---|---|
| `@Injectable({ providedIn: 'root' })` | Application | 1 singleton |
| `ApplicationConfig.providers` | Application | 1 singleton |
| Route `providers` | Route (lazy boundary) | 1 per route activation |
| Component `providers` | Component | 1 per component instance |
| Component `viewProviders` | Component view | 1 per component instance (hidden from projected content) |

A `useFactory` at the component level runs every time a new component instance is created. A `useFactory` at the root level runs only once, the first time the token is injected.

## Common Mistakes

- **Using `useClass` when you mean `useExisting`.** This is the single most common DI mistake. `useClass` always creates a new instance. If you want two tokens to point to the same singleton, use `useExisting`. Using `useClass` for aliasing creates a second instance with separate state.

```typescript
// Bad Practice: creates TWO LoggerService instances
providers: [
  LoggerService,
  { provide: OldLogger, useClass: LoggerService }, // new instance!
]

// Correct: both tokens resolve to the SAME instance
providers: [
  LoggerService,
  { provide: OldLogger, useExisting: LoggerService },
]
```

- **Passing a class reference to `useValue` instead of an instance.** `useValue` returns exactly what you give it. Passing a class reference returns the class constructor, not an instance of the class.

```typescript
// Bad Practice: useValue with a class reference
{ provide: Logger, useValue: ConsoleLogger }
// inject(Logger) returns the ConsoleLogger CLASS, not an instance

// Correct: pass an instance
{ provide: Logger, useValue: new ConsoleLogger() }

// Or use useClass to let Angular instantiate it
{ provide: Logger, useClass: ConsoleLogger }
```

- **Trying to use `await` in a `useFactory`.** Factories must return synchronously. Angular does not support async factory functions. If you need async initialization, use `APP_INITIALIZER` or initialize the service lazily after construction.

- **Forgetting that `useFactory` with `inject()` requires an injection context.** The factory function runs in an injection context, so `inject()` works. But if you extract the factory into a separate file and call it from a non-injection context (like a test setup), it fails with `NG0203`.

- **Providing the same token at multiple levels without understanding shadowing.** A component-level `useValue` overrides a root-level `useFactory` for that token within the component and its children. This is intentional but can be confusing if you expect the root factory to run.

---

## Best Practices

- **Use `useClass` for swapping implementations.** Production vs development loggers, real vs mock API clients, platform-specific storage.

- **Use `useValue` for configuration objects, constants, and test mocks.** Pair with `InjectionToken<T>` for type safety.

- **Use `useFactory` when the value depends on runtime conditions or other injected services.** Prefer `inject()` inside the factory over the `deps` array.

- **Use `useExisting` exclusively for aliasing.** When you need two tokens to resolve to the same instance, `useExisting` is the only correct choice.

- **Export `provideXxx()` functions from libraries.** This pattern encapsulates internal tokens, provides type-safe configuration, and follows Angular conventions.

- **Use `satisfies` with `useValue` objects.** This catches type errors at compile time without losing the literal type:

```typescript
{ provide: APP_CONFIG, useValue: { apiUrl: '/api', retries: 3 } satisfies AppConfig }
```

## Real-World Use Cases

### Platform-Adaptive Storage Service

```typescript
export abstract class StorageService {
  abstract get(key: string): string | null;
  abstract set(key: string, value: string): void;
  abstract remove(key: string): void;
}

@Injectable()
export class BrowserStorageService extends StorageService {
  get(key: string) { return localStorage.getItem(key); }
  set(key: string, value: string) { localStorage.setItem(key, value); }
  remove(key: string) { localStorage.removeItem(key); }
}

@Injectable()
export class ServerStorageService extends StorageService {
  private store = new Map<string, string>();
  get(key: string) { return this.store.get(key) ?? null; }
  set(key: string, value: string) { this.store.set(key, value); }
  remove(key: string) { this.store.delete(key); }
}

// app.config.ts
export const appConfig: ApplicationConfig = {
  providers: [
    {
      provide: StorageService,
      useFactory: () => {
        return isPlatformBrowser(inject(PLATFORM_ID))
          ? new BrowserStorageService()
          : new ServerStorageService();
      },
    },
  ],
};
```

### User-Aware API Client via Factory

```typescript
export const API_CLIENT = new InjectionToken<ApiClient>('API_CLIENT');

export function provideApiClient(): Provider {
  return {
    provide: API_CLIENT,
    useFactory: () => {
      const http = inject(HttpClient);
      const auth = inject(AuthService);
      const config = inject(APP_CONFIG);

      const client = new ApiClient(http, config.apiUrl);

      // Automatically attach auth headers
      effect(() => {
        const token = auth.accessToken();
        client.setAuthToken(token);
      });

      return client;
    },
  };
}

// app.config.ts
providers: [provideApiClient()]
```

### Narrowing a Service Interface for Read-Only Access

```typescript
@Injectable({ providedIn: 'root' })
export class ProductCatalogService {
  private _products = signal<Product[]>([]);
  readonly products = this._products.asReadonly();

  getById(id: string) {
    return this._products().find(p => p.id === id);
  }

  // Admin-only operations
  addProduct(product: Product) {
    this._products.update(list => [...list, product]);
  }

  updateProduct(id: string, updates: Partial<Product>) {
    this._products.update(list =>
      list.map(p => p.id === id ? { ...p, ...updates } : p)
    );
  }

  removeProduct(id: string) {
    this._products.update(list => list.filter(p => p.id !== id));
  }
}

// Read-only interface token
export abstract class ProductReader {
  abstract readonly products: Signal<readonly Product[]>;
  abstract getById(id: string): Product | undefined;
}

// Alias the full service as the narrow interface
export function provideProductReader(): Provider {
  return { provide: ProductReader, useExisting: ProductCatalogService };
}

// Public-facing components use the narrow interface
@Component({ ... })
export class ProductListComponent {
  private catalog = inject(ProductReader);
  // Only products and getById() are available
  // addProduct, updateProduct, removeProduct are hidden
}

// Admin components use the full service
@Component({ ... })
export class ProductAdminComponent {
  private catalog = inject(ProductCatalogService);
  // All methods available
}
```

---

> **Key Takeaway:** Choose `useClass` for implementation swaps, `useValue` for static values and test mocks, `useFactory` for runtime-dependent creation with `inject()`, and `useExisting` for aliasing tokens to the same instance; the most critical distinction is that `useClass` always creates a new instance while `useExisting` reuses an existing one.
