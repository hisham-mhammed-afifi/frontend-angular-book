# 15.5 Child Routes and Nested Outlets

> Child routes let you compose page layouts from nested components, each with its own `<router-outlet>`. Named outlets enable parallel views in the same template. Together, they make complex layouts like dashboards, tabbed interfaces, and master-detail views possible.

---

## How Child Routes Work

A child route is a route defined inside the `children` array of a parent route. The child component renders inside the parent component's `<router-outlet>`, not the root outlet.

```
URL: /settings/profile

Root <router-outlet>
  └── SettingsComponent (matches 'settings')
        └── <router-outlet>  (inside SettingsComponent template)
              └── ProfileComponent (matches 'profile')
```

Each level of nesting requires its own `<router-outlet>`.

---

## Basic Child Route Configuration

```typescript
// app.routes.ts
export const routes: Routes = [
  {
    path: 'settings',
    component: SettingsComponent,
    children: [
      { path: '', redirectTo: 'profile', pathMatch: 'full' },
      { path: 'profile', component: ProfileComponent },
      { path: 'security', component: SecurityComponent },
      { path: 'notifications', component: NotificationsComponent },
    ],
  },
];
```

```typescript
// settings.component.ts
import { Component } from '@angular/core';
import { RouterOutlet, RouterLink, RouterLinkActive } from '@angular/router';

@Component({
  selector: 'app-settings',
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  template: `
    <h1>Settings</h1>
    <nav>
      <a routerLink="profile" routerLinkActive="active">Profile</a>
      <a routerLink="security" routerLinkActive="active">Security</a>
      <a routerLink="notifications" routerLinkActive="active">Notifications</a>
    </nav>
    <section class="settings-content">
      <router-outlet />
    </section>
  `,
})
export class SettingsComponent {}
```

The `<router-outlet />` in `SettingsComponent` is where `ProfileComponent`, `SecurityComponent`, or `NotificationsComponent` renders depending on the URL.

### Relative Links

Inside a child context, use relative paths in `routerLink`:

```typescript
// These are relative to the current route (/settings)
<a routerLink="profile">Profile</a>       <!-- /settings/profile -->
<a routerLink="security">Security</a>     <!-- /settings/security -->

// Absolute paths start with /
<a routerLink="/settings/profile">Profile</a>  <!-- Same result, but absolute -->

// Navigate to sibling
<a routerLink="../other">Other</a>  <!-- Goes up one level, then to 'other' -->
```

## Deeply Nested Routes

Routes can be nested multiple levels deep. Each level adds a new `<router-outlet>`.

```typescript
const routes: Routes = [
  {
    path: 'admin',
    loadComponent: () => import('./admin/admin-layout.component'),
    children: [
      { path: '', redirectTo: 'dashboard', pathMatch: 'full' },
      {
        path: 'dashboard',
        loadComponent: () => import('./admin/dashboard.component'),
      },
      {
        path: 'users',
        loadComponent: () => import('./admin/users/users-layout.component'),
        children: [
          {
            path: '',
            loadComponent: () => import('./admin/users/user-list.component'),
          },
          {
            path: ':userId',
            loadComponent: () => import('./admin/users/user-detail.component'),
            children: [
              { path: '', redirectTo: 'overview', pathMatch: 'full' },
              {
                path: 'overview',
                loadComponent: () => import('./admin/users/user-overview.component'),
              },
              {
                path: 'activity',
                loadComponent: () => import('./admin/users/user-activity.component'),
              },
            ],
          },
        ],
      },
    ],
  },
];
```

URL structure:
```
/admin                          -> AdminLayout > Dashboard
/admin/dashboard                -> AdminLayout > Dashboard
/admin/users                    -> AdminLayout > UsersLayout > UserList
/admin/users/42                 -> AdminLayout > UsersLayout > UserDetail > UserOverview
/admin/users/42/overview        -> AdminLayout > UsersLayout > UserDetail > UserOverview
/admin/users/42/activity        -> AdminLayout > UsersLayout > UserDetail > UserActivity
```

Each component in the chain must have a `<router-outlet>`:

```typescript
// admin-layout.component.ts
@Component({
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  template: `
    <div class="admin-shell">
      <aside class="admin-sidebar">
        <a routerLink="dashboard" routerLinkActive="active">Dashboard</a>
        <a routerLink="users" routerLinkActive="active">Users</a>
      </aside>
      <main class="admin-content">
        <router-outlet />
      </main>
    </div>
  `,
})
export class AdminLayoutComponent {}
```

```typescript
// users-layout.component.ts
@Component({
  imports: [RouterOutlet],
  template: `
    <div class="users-section">
      <router-outlet />
    </div>
  `,
})
export class UsersLayoutComponent {}
```

```typescript
// user-detail.component.ts
@Component({
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  template: `
    <div class="user-detail">
      <h2>User {{ userId() }}</h2>
      <nav>
        <a routerLink="overview" routerLinkActive="active">Overview</a>
        <a routerLink="activity" routerLinkActive="active">Activity</a>
      </nav>
      <router-outlet />
    </div>
  `,
})
export class UserDetailComponent {
  userId = input.required<string>();
}
```

## Layout Routes Pattern

A common architecture uses parent routes to define different layouts (navigation, sidebar, footer) for different sections of the app:

```typescript
export const routes: Routes = [
  // Public layout: simple header, no sidebar
  {
    path: '',
    loadComponent: () => import('./layouts/public-layout.component'),
    children: [
      { path: '', loadComponent: () => import('./pages/home.component') },
      { path: 'about', loadComponent: () => import('./pages/about.component') },
      { path: 'login', loadComponent: () => import('./auth/login.component') },
    ],
  },
  // App layout: sidebar, top nav, auth required
  {
    path: '',
    loadComponent: () => import('./layouts/app-layout.component'),
    canActivate: [authGuard],
    children: [
      { path: 'dashboard', loadComponent: () => import('./dashboard/dashboard.component') },
      { path: 'projects', loadComponent: () => import('./projects/project-list.component') },
      { path: 'settings', loadComponent: () => import('./settings/settings.component') },
    ],
  },
  { path: '**', loadComponent: () => import('./pages/not-found.component') },
];
```

```typescript
// public-layout.component.ts
@Component({
  imports: [RouterOutlet],
  template: `
    <header class="public-header">
      <a routerLink="/">MyApp</a>
      <a routerLink="/login">Sign In</a>
    </header>
    <main>
      <router-outlet />
    </main>
    <footer>Public footer</footer>
  `,
})
export class PublicLayoutComponent {}
```

```typescript
// app-layout.component.ts
@Component({
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  template: `
    <div class="app-shell">
      <aside class="sidebar">
        <a routerLink="/dashboard" routerLinkActive="active">Dashboard</a>
        <a routerLink="/projects" routerLinkActive="active">Projects</a>
        <a routerLink="/settings" routerLinkActive="active">Settings</a>
      </aside>
      <main class="main-content">
        <router-outlet />
      </main>
    </div>
  `,
})
export class AppLayoutComponent {}
```

Both layout routes have `path: ''`, so the router uses the first matching route. The public layout matches unauthenticated paths (`/`, `/about`, `/login`). The app layout matches authenticated paths (`/dashboard`, `/projects`). Route order matters.

## Named Outlets (Auxiliary Routes)

Named outlets let you render multiple independent route views simultaneously in the same template.

### Defining Named Outlets

```typescript
@Component({
  imports: [RouterOutlet],
  template: `
    <div class="main-area">
      <router-outlet />                         <!-- Primary outlet -->
    </div>
    <aside class="side-panel">
      <router-outlet name="panel" />             <!-- Named outlet: 'panel' -->
    </aside>
    <div class="chat-widget">
      <router-outlet name="chat" />              <!-- Named outlet: 'chat' -->
    </div>
  `,
})
export class ShellComponent {}
```

### Configuring Routes for Named Outlets

```typescript
const routes: Routes = [
  {
    path: 'app',
    component: ShellComponent,
    children: [
      // Primary outlet routes
      { path: 'dashboard', component: DashboardComponent },
      { path: 'projects', component: ProjectsComponent },

      // Named outlet routes
      { path: 'help', component: HelpPanelComponent, outlet: 'panel' },
      { path: 'notifications', component: NotificationsPanelComponent, outlet: 'panel' },
      { path: 'support', component: ChatSupportComponent, outlet: 'chat' },
    ],
  },
];
```

### Navigating to Named Outlets

```typescript
// Template navigation
<a [routerLink]="['/app', { outlets: { panel: 'help' } }]">
  Open Help Panel
</a>

// Show both primary and named outlet content
<a [routerLink]="['/app', { outlets: { primary: 'dashboard', panel: 'help' } }]">
  Dashboard + Help
</a>

// Close a named outlet
<a [routerLink]="['/app', { outlets: { panel: null } }]">
  Close Panel
</a>
```

### URL Format

Named outlets produce URLs with parentheses:

```
/app/dashboard(panel:help)
/app/dashboard(panel:help//chat:support)
/app/projects(panel:notifications)
```

The format is: `primaryPath(outletName:outletPath//otherOutlet:otherPath)`

### Programmatic Navigation

```typescript
// Open a named outlet
this.router.navigate(['/app', { outlets: { panel: 'help' } }]);

// Close a named outlet
this.router.navigate(['/app', { outlets: { panel: null } }]);

// Update multiple outlets at once
this.router.navigate(['/app', {
  outlets: {
    primary: 'projects',
    panel: 'notifications',
    chat: 'support',
  },
}]);
```

## `routerOutletData` (Angular 19+)

The `routerOutletData` input lets you pass contextual data from a parent component to whatever component the router renders in that outlet, without modifying route configuration:

```typescript
// Parent component
@Component({
  imports: [RouterOutlet],
  template: `
    <router-outlet [routerOutletData]="{ theme: 'dark', compact: true }" />
  `,
})
export class DashboardLayoutComponent {}
```

```typescript
// Any child component rendered in that outlet
import { Component, inject, Signal } from '@angular/core';
import { ROUTER_OUTLET_DATA } from '@angular/router';

interface OutletConfig {
  theme: string;
  compact: boolean;
}

@Component({
  template: `
    <div [class]="data().theme">
      @if (data().compact) {
        <app-compact-view />
      } @else {
        <app-full-view />
      }
    </div>
  `,
})
export class WidgetComponent {
  data = inject<Signal<OutletConfig>>(ROUTER_OUTLET_DATA);
}
```

### Dynamic Outlet Data

The data can be a signal, making it reactive:

```typescript
@Component({
  imports: [RouterOutlet],
  template: `
    <button (click)="toggleCompact()">Toggle Compact</button>
    <router-outlet [routerOutletData]="outletConfig()" />
  `,
})
export class LayoutComponent {
  private compact = signal(false);

  outletConfig = computed(() => ({
    theme: 'light',
    compact: this.compact(),
  }));

  toggleCompact() {
    this.compact.update(v => !v);
  }
}
```

The routed component's `ROUTER_OUTLET_DATA` injection automatically reflects the updated value because it is a `Signal`.

## RouterOutlet Events

The `<router-outlet>` emits lifecycle events you can listen to:

```typescript
@Component({
  imports: [RouterOutlet],
  template: `
    <router-outlet
      (activate)="onActivate($event)"
      (deactivate)="onDeactivate($event)"
    />
  `,
})
export class ShellComponent {
  onActivate(component: unknown) {
    console.log('Route activated:', component);
    // component is the instance of the routed component

    // Useful for scrolling to top
    window.scrollTo(0, 0);
  }

  onDeactivate(component: unknown) {
    console.log('Route deactivated:', component);
  }
}
```

| Event | Fires When | `$event` Value |
|-------|-----------|----------------|
| `activate` | A new component is rendered in the outlet | The component instance |
| `deactivate` | The current component is removed from the outlet | The destroyed component instance |
| `attach` | A previously detached component is reattached (RouteReuseStrategy) | The component instance |
| `detach` | A component is detached instead of destroyed (RouteReuseStrategy) | The component instance |

## Componentless Routes for Grouping

A route without a `component` or `loadComponent` does not render anything. It groups child routes under shared configuration:

```typescript
const routes: Routes = [
  {
    path: 'admin',
    // No component -- children render in the PARENT'S outlet
    canActivate: [adminGuard],
    providers: [AdminService],
    children: [
      { path: 'users', loadComponent: () => import('./admin/users.component') },
      { path: 'logs', loadComponent: () => import('./admin/logs.component') },
      { path: 'config', loadComponent: () => import('./admin/config.component') },
    ],
  },
];
```

Key difference: with a componentless route, child components render in the nearest ancestor's `<router-outlet>`, not in the componentless route's outlet (because it has none).

### Componentless vs Component Parent

```
WITH component parent:
Root outlet -> AdminLayout -> [admin outlet] -> UsersComponent

WITHOUT component (componentless):
Root outlet -> UsersComponent (renders directly in root outlet)
```

Use componentless routes when you need shared guards or providers but do not need a wrapper layout.

## Empty-Path Routes

An empty-path route (`path: ''`) matches the parent's URL without adding a segment:

```typescript
const routes: Routes = [
  {
    path: 'products',
    component: ProductLayoutComponent,
    children: [
      { path: '', component: ProductListComponent },      // /products
      { path: ':id', component: ProductDetailComponent },  // /products/42
    ],
  },
];
```

Navigating to `/products` renders `ProductLayoutComponent` with `ProductListComponent` inside its outlet.

### Default Child with Redirect

```typescript
{
  path: 'settings',
  component: SettingsLayoutComponent,
  children: [
    { path: '', redirectTo: 'general', pathMatch: 'full' },
    { path: 'general', component: GeneralSettingsComponent },
    { path: 'appearance', component: AppearanceComponent },
  ],
}
// /settings -> redirects to /settings/general
```

## Sharing Data Between Parent and Child Routes

### Through Route Data and Resolvers

```typescript
const routes: Routes = [
  {
    path: 'projects/:projectId',
    component: ProjectLayoutComponent,
    resolve: { project: projectResolver },
    children: [
      { path: '', redirectTo: 'overview', pathMatch: 'full' },
      { path: 'overview', component: ProjectOverviewComponent },
      { path: 'tasks', component: ProjectTasksComponent },
      { path: 'members', component: ProjectMembersComponent },
    ],
  },
];
```

With `paramsInheritanceStrategy: 'always'` and `withComponentInputBinding()`, every child component receives both `projectId` and `project`:

```typescript
// project-tasks.component.ts
@Component({
  template: `
    <h2>Tasks for {{ project()?.name }}</h2>
  `,
})
export class ProjectTasksComponent {
  projectId = input.required<string>();   // Inherited from parent
  project = input.required<Project>();    // Inherited from parent resolver
}
```

### Through the Parent Component and Services

```typescript
// project-layout.component.ts
@Component({
  providers: [ProjectContextService],
  imports: [RouterOutlet, RouterLink, RouterLinkActive],
  template: `
    <nav>
      <a routerLink="overview" routerLinkActive="active">Overview</a>
      <a routerLink="tasks" routerLinkActive="active">Tasks</a>
      <a routerLink="members" routerLinkActive="active">Members</a>
    </nav>
    <router-outlet />
  `,
})
export class ProjectLayoutComponent {
  projectId = input.required<string>();
  private context = inject(ProjectContextService);

  constructor() {
    effect(() => {
      this.context.setProjectId(this.projectId());
    });
  }
}

// Child component
@Component({ ... })
export class ProjectTasksComponent {
  private context = inject(ProjectContextService);
  tasks = this.context.tasks; // Signal from the scoped service
}
```

### Through `routerOutletData`

```typescript
// Parent passes data to whatever child renders
@Component({
  imports: [RouterOutlet],
  template: `
    <router-outlet [routerOutletData]="{ projectId: projectId(), permissions }" />
  `,
})
export class ProjectLayoutComponent {
  projectId = input.required<string>();
  permissions = inject(PermissionService).getPermissions();
}

// Child reads it
@Component({ ... })
export class ProjectTasksComponent {
  private outletData = inject<Signal<{ projectId: string; permissions: Permissions }>>(
    ROUTER_OUTLET_DATA
  );

  projectId = computed(() => this.outletData().projectId);
}
```

## Common Mistakes

### Mistake 1: Forgetting `<router-outlet>` in the Parent Component

```typescript
// BAD: Parent has children but no outlet for them to render in
@Component({
  template: `
    <h1>Settings</h1>
    <nav>
      <a routerLink="profile">Profile</a>
    </nav>
    <!-- Where does ProfileComponent render? Nowhere! -->
  `,
})
export class SettingsComponent {}

// GOOD: Include a router-outlet
@Component({
  imports: [RouterOutlet, RouterLink],
  template: `
    <h1>Settings</h1>
    <nav>
      <a routerLink="profile">Profile</a>
    </nav>
    <router-outlet />
  `,
})
export class SettingsComponent {}
```

### Mistake 2: Using Absolute Paths in Child Route Links

```typescript
// BAD: Absolute path breaks if the parent route path changes
<a routerLink="/settings/profile">Profile</a>

// GOOD: Relative path is maintainable
<a routerLink="profile">Profile</a>
```

If the parent route changes from `settings` to `preferences`, relative links continue to work. Absolute links break.

### Mistake 3: No Default Child Route

```typescript
// BAD: /account shows the layout with an empty outlet
{
  path: 'account',
  component: AccountLayoutComponent,
  children: [
    { path: 'profile', component: ProfileComponent },
    { path: 'billing', component: BillingComponent },
  ],
}

// GOOD: Redirect or provide a default child
{
  path: 'account',
  component: AccountLayoutComponent,
  children: [
    { path: '', redirectTo: 'profile', pathMatch: 'full' },
    { path: 'profile', component: ProfileComponent },
    { path: 'billing', component: BillingComponent },
  ],
}
```

### Mistake 4: Not Importing `RouterOutlet` in Standalone Components

```typescript
// BAD: Outlet element is ignored, children render nowhere
@Component({
  template: `<router-outlet />`, // Treated as unknown element
})
export class LayoutComponent {}

// GOOD: Import the directive
@Component({
  imports: [RouterOutlet],
  template: `<router-outlet />`,
})
export class LayoutComponent {}
```

### Mistake 5: Over-Nesting Routes

```
/app/admin/section/subsection/detail/tab/subtab
```

Five levels of nesting means five `<router-outlet>` components, five layout wrappers, and a maintenance nightmare. If nesting goes beyond three levels, consider flattening the structure.

### Mistake 6: Confusing Named Outlet Syntax

```typescript
// BAD: Putting the outlet name in the path
<a routerLink="/app/panel/help">Help</a>  // This is a primary route!

// GOOD: Use the outlets syntax
<a [routerLink]="['/app', { outlets: { panel: 'help' } }]">Help</a>
```

## Best Practices

### 1. Use Layout Routes for Shared Chrome

Group routes under layout components that provide navigation, sidebars, and footers. This keeps individual page components focused on content.

### 2. Always Provide a Default Child

Every route with children should either redirect `''` to a default child or provide a component for the empty path.

### 3. Use Relative `routerLink` Paths

Relative paths (`routerLink="profile"`) are more resilient to refactoring than absolute paths (`routerLink="/settings/profile"`).

### 4. Limit Nesting Depth to Three Levels

Beyond three levels of nesting, the route tree becomes hard to understand and maintain. Flatten by using componentless routes or restructuring the URL scheme.

### 5. Use `routerOutletData` for Layout-to-Child Communication

When a layout component needs to pass configuration to its children, `routerOutletData` is cleaner than shared services for simple data.

### 6. Use Named Outlets Sparingly

Named outlets add URL complexity (parenthesized segments). They are useful for persistent side panels or dialogs but should not be the primary navigation mechanism. For most secondary content, consider using component state or a service-driven overlay instead.

### Multi-Section App with Auth Layout Split

```typescript
export const routes: Routes = [
  // Auth pages: minimal layout, no sidebar
  {
    path: '',
    loadComponent: () => import('./layouts/auth-layout.component'),
    children: [
      { path: 'login', loadComponent: () => import('./auth/login.component') },
      { path: 'register', loadComponent: () => import('./auth/register.component') },
      { path: 'forgot-password', loadComponent: () => import('./auth/forgot-password.component') },
    ],
  },
  // Main app: full layout with sidebar
  {
    path: '',
    loadComponent: () => import('./layouts/main-layout.component'),
    canActivate: [authGuard],
    children: [
      { path: '', redirectTo: 'dashboard', pathMatch: 'full' },
      { path: 'dashboard', loadComponent: () => import('./pages/dashboard.component') },
      {
        path: 'projects',
        loadComponent: () => import('./projects/project-layout.component'),
        children: [
          { path: '', loadComponent: () => import('./projects/project-list.component') },
          {
            path: ':id',
            loadComponent: () => import('./projects/project-detail.component'),
            children: [
              { path: '', redirectTo: 'overview', pathMatch: 'full' },
              { path: 'overview', loadComponent: () => import('./projects/overview.component') },
              { path: 'settings', loadComponent: () => import('./projects/settings.component') },
            ],
          },
        ],
      },
    ],
  },
  { path: '**', loadComponent: () => import('./pages/not-found.component') },
];
```

This is a production-ready route tree demonstrating:
- Two layouts for different authentication contexts.
- Lazy loading at every level.
- Up to three levels of nesting (main > projects > project detail > tabs).
- Default redirects at each child level.

---

> **Key Takeaway:** Child routes and nested `<router-outlet>` elements let you compose complex layouts from independent, routed components. Use layout routes for shared UI chrome, provide default child routes at every level, prefer relative `routerLink` paths, and limit nesting depth to keep the route tree maintainable.
