# 15.8 Functional Route Resolvers

> Route resolvers fetch data before a route activates, guaranteeing that the component receives its data on initialization. Functional resolvers (`ResolveFn`) are the modern replacement for class-based resolvers.

---

## What a Resolver Does

A resolver runs after guards succeed but before the component renders. The router waits for the resolver to complete, then passes the resolved data to the component.

```
Navigation Request
      |
  canMatch guards
      |
  canActivate guards
      |
  *** Resolvers run here ***
      |
  Component renders (data available immediately)
```

Without a resolver, the component must handle loading states:

```typescript
// Without resolver: component must manage loading
@Component({
  template: `
    @if (isLoading()) {
      <app-spinner />
    } @else if (product()) {
      <h1>{{ product()!.name }}</h1>
    } @else {
      <p>Product not found</p>
    }
  `,
})
export class ProductComponent {
  private productService = inject(ProductService);
  id = input.required<string>();

  isLoading = signal(true);
  product = signal<Product | null>(null);

  constructor() {
    effect(() => {
      this.isLoading.set(true);
      this.productService.getById(this.id()).then(p => {
        this.product.set(p);
        this.isLoading.set(false);
      });
    });
  }
}
```

With a resolver, the component receives data immediately:

```typescript
// With resolver: data is ready when the component renders
@Component({
  template: `<h1>{{ product().name }}</h1>`,
})
export class ProductComponent {
  product = input.required<Product>(); // Guaranteed to have a value
}
```

## `ResolveFn` Signature

```typescript
type ResolveFn<T> = (
  route: ActivatedRouteSnapshot,
  state: RouterStateSnapshot,
) => MaybeAsync<T | RedirectCommand>;
```

The function receives route and state snapshots, and can return:
- A value of type `T` (synchronous)
- A `Promise<T>` (async)
- An `Observable<T>` (the router uses the first emitted value)
- A `RedirectCommand` (to redirect instead of resolving)

---

## Basic Resolver

```typescript
import { inject } from '@angular/core';
import { ResolveFn } from '@angular/router';
import { ProductService } from './product.service';
import { Product } from './product.model';

export const productResolver: ResolveFn<Product> = (route) => {
  const productService = inject(ProductService);
  const id = route.paramMap.get('id')!;
  return productService.getById(id);
};
```

### Route Configuration

```typescript
const routes: Routes = [
  {
    path: 'products/:id',
    loadComponent: () => import('./product-detail.component'),
    resolve: { product: productResolver },
    //         ^key      ^resolver function
  },
];
```

The `resolve` map uses string keys. Each key maps to a resolver function. The resolved values are available under those keys.

### Accessing Resolved Data

#### With `withComponentInputBinding()` (Recommended)

The resolved data binds to an input whose name matches the resolve key:

```typescript
@Component({
  template: `<h1>{{ product().name }}</h1>`,
})
export class ProductDetailComponent {
  // Input name 'product' matches the resolve key 'product'
  product = input.required<Product>();
}
```

#### With `ActivatedRoute` (Legacy)

```typescript
@Component({ ... })
export class ProductDetailComponent {
  private route = inject(ActivatedRoute);

  product = toSignal(
    this.route.data.pipe(map(data => data['product'] as Product))
  );
}
```

## Multiple Resolvers

A route can have multiple resolvers. They run in parallel:

```typescript
const routes: Routes = [
  {
    path: 'products/:id',
    loadComponent: () => import('./product-detail.component'),
    resolve: {
      product: productResolver,
      reviews: reviewsResolver,
      relatedProducts: relatedResolver,
    },
  },
];

// Resolvers
export const productResolver: ResolveFn<Product> = (route) => {
  return inject(ProductService).getById(route.paramMap.get('id')!);
};

export const reviewsResolver: ResolveFn<Review[]> = (route) => {
  return inject(ReviewService).getByProductId(route.paramMap.get('id')!);
};

export const relatedResolver: ResolveFn<Product[]> = (route) => {
  return inject(ProductService).getRelated(route.paramMap.get('id')!);
};

// Component receives all three
@Component({ ... })
export class ProductDetailComponent {
  product = input.required<Product>();
  reviews = input.required<Review[]>();
  relatedProducts = input.required<Product[]>();
}
```

All three resolvers execute concurrently. The router waits for all of them to complete before rendering the component.

## Resolver Return Types

### Synchronous

```typescript
export const staticDataResolver: ResolveFn<Config> = () => {
  return { theme: 'dark', language: 'en' };
};
```

### Promise (Async/Await)

```typescript
export const userResolver: ResolveFn<User> = async (route) => {
  const userService = inject(UserService);
  const id = route.paramMap.get('id')!;
  return await userService.fetchUser(id);
};
```

### Observable

```typescript
import { firstValueFrom } from 'rxjs';

export const userResolver: ResolveFn<User> = (route) => {
  const http = inject(HttpClient);
  const id = route.paramMap.get('id')!;
  return http.get<User>(`/api/users/${id}`);
  // Router subscribes and uses the first emitted value
};
```

For `Observable` returns, the router subscribes, takes the first emitted value, and unsubscribes. If the observable never emits, navigation hangs.

## Error Handling in Resolvers

### Redirecting on Error with `RedirectCommand`

```typescript
import { RedirectCommand, Router } from '@angular/router';

export const productResolver: ResolveFn<Product> = async (route) => {
  const productService = inject(ProductService);
  const router = inject(Router);
  const id = route.paramMap.get('id')!;

  try {
    return await productService.getById(id);
  } catch (error) {
    // Redirect to a not-found page instead of crashing
    return new RedirectCommand(router.parseUrl('/not-found'));
  }
};
```

### Redirecting Based on Data

```typescript
export const projectResolver: ResolveFn<Project> = async (route) => {
  const projectService = inject(ProjectService);
  const router = inject(Router);
  const id = route.paramMap.get('id')!;

  const project = await projectService.getById(id);

  if (project.status === 'archived') {
    return new RedirectCommand(
      router.parseUrl(`/projects/${id}/archived`),
      { replaceUrl: true },
    );
  }

  return project;
};
```

### Catching HTTP Errors

```typescript
import { catchError, of } from 'rxjs';

export const userResolver: ResolveFn<User | null> = (route) => {
  const http = inject(HttpClient);
  const id = route.paramMap.get('id')!;

  return http.get<User>(`/api/users/${id}`).pipe(
    catchError(() => of(null)), // Return null on error, let the component handle it
  );
};
```

The component then checks for null:

```typescript
@Component({
  template: `
    @if (user()) {
      <h1>{{ user()!.name }}</h1>
    } @else {
      <p>User not found.</p>
    }
  `,
})
export class UserComponent {
  user = input.required<User | null>();
}
```

## Resolver Factories

Like guard factories, you can create parameterized resolvers:

```typescript
export function resolveEntity<T>(
  serviceToken: Type<{ getById(id: string): Promise<T> }>,
  paramName: string = 'id',
): ResolveFn<T> {
  return (route) => {
    const service = inject(serviceToken);
    const id = route.paramMap.get(paramName)!;
    return service.getById(id);
  };
}

// Usage
const routes: Routes = [
  {
    path: 'products/:id',
    loadComponent: () => import('./product.component'),
    resolve: { product: resolveEntity(ProductService) },
  },
  {
    path: 'users/:userId',
    loadComponent: () => import('./user.component'),
    resolve: { user: resolveEntity(UserService, 'userId') },
  },
];
```

### Resolver with Configuration

```typescript
interface PaginatedResult<T> {
  items: T[];
  total: number;
}

export function resolveList<T>(
  endpoint: string,
  defaultPageSize: number = 25,
): ResolveFn<PaginatedResult<T>> {
  return (route) => {
    const http = inject(HttpClient);
    const page = Number(route.queryParamMap.get('page')) || 1;
    const pageSize = Number(route.queryParamMap.get('pageSize')) || defaultPageSize;

    return http.get<PaginatedResult<T>>(endpoint, {
      params: { page: page.toString(), pageSize: pageSize.toString() },
    });
  };
}

// Usage
{
  path: 'users',
  loadComponent: () => import('./user-list.component'),
  resolve: { users: resolveList<User>('/api/users', 50) },
}
```

## Resolver Re-Execution

By default, resolvers re-run when path parameters change. The `runGuardsAndResolvers` property controls when:

```typescript
{
  path: 'search',
  component: SearchComponent,
  resolve: { results: searchResolver },
  runGuardsAndResolvers: 'paramsOrQueryParamsChange',
}
```

| Value | Resolvers Re-Run When |
|-------|----------------------|
| `'paramsChange'` (default) | Path parameters change |
| `'pathParamsChange'` | Path params change (excludes matrix params) |
| `'paramsOrQueryParamsChange'` | Path params or query params change |
| `'pathParamsOrQueryParamsChange'` | Path params or query params change (excludes matrix) |
| `'always'` | Every navigation, even if nothing changes |

### Search Page Example

```typescript
export const searchResolver: ResolveFn<SearchResult> = (route) => {
  const searchService = inject(SearchService);
  const q = route.queryParamMap.get('q') ?? '';
  const page = Number(route.queryParamMap.get('page')) || 1;
  const sort = route.queryParamMap.get('sort') ?? 'relevance';

  return searchService.search({ q, page, sort });
};

{
  path: 'search',
  component: SearchComponent,
  resolve: { results: searchResolver },
  // Re-run resolver when query params change (page, sort, q)
  runGuardsAndResolvers: 'paramsOrQueryParamsChange',
}
```

## Dynamic Page Title with Resolvers

The `title` property on a route can be a `ResolveFn<string>`:

```typescript
const routes: Routes = [
  {
    path: 'products/:id',
    loadComponent: () => import('./product-detail.component'),
    resolve: { product: productResolver },
    title: productTitleResolver,
  },
];

export const productTitleResolver: ResolveFn<string> = (route) => {
  const productService = inject(ProductService);
  const id = route.paramMap.get('id')!;
  return productService.getById(id).then(p => `${p.name} - MyApp`);
};
```

Angular automatically sets `document.title` to the resolved value.

> **Note:** If you already have a `product` resolver, you might be fetching the product twice. To avoid this, either combine the resolvers or use a `TitleStrategy` that reads from resolved data.

### TitleStrategy Using Resolved Data

```typescript
@Injectable({ providedIn: 'root' })
export class AppTitleStrategy extends TitleStrategy {
  private title = inject(Title);

  override updateTitle(snapshot: RouterStateSnapshot) {
    const title = this.buildTitle(snapshot);
    if (title) {
      this.title.setTitle(`${title} | MyApp`);
    } else {
      this.title.setTitle('MyApp');
    }
  }
}
```

## Resolvers vs Component-Level Data Loading

| Aspect | Resolver | Component Loading |
|--------|---------|-------------------|
| Data available at | Component initialization | After component renders |
| Loading indicator | Navigation delay (no spinner in component) | Spinner/skeleton in component |
| Error handling | Redirect before component loads | Error state in component |
| UX perception | "Slow navigation" | "Fast navigation, then loading" |
| SSR compatibility | Excellent (data ready for initial render) | Requires careful handling |
| Complexity | Extra resolver function | Loading/error logic in component |

### When to Use Resolvers

- The component cannot render meaningfully without the data.
- SSR requires data in the initial HTML.
- You want to redirect to an error page if data is unavailable.
- The data fetch is fast (under 300ms).

### When to Use Component-Level Loading

- The data takes a long time to fetch (show skeleton UI instead of blocking).
- Only part of the page needs the fetched data.
- You want immediate navigation feedback.
- The `resource()` API handles your needs with built-in loading/error states.

### The Hybrid Approach

Use a resolver for critical data and `resource()` for secondary data:

```typescript
// Resolver fetches the essential product data
export const productResolver: ResolveFn<Product> = (route) => {
  return inject(ProductService).getById(route.paramMap.get('id')!);
};

// Component uses resource() for secondary data
@Component({
  template: `
    <h1>{{ product().name }}</h1>
    <p>{{ product().description }}</p>

    @if (reviews.isLoading()) {
      <app-review-skeleton />
    } @else {
      <app-reviews [reviews]="reviews.value() ?? []" />
    }
  `,
})
export class ProductDetailComponent {
  // From resolver (guaranteed available)
  product = input.required<Product>();

  // Loaded in component (shows skeleton while loading)
  private reviewsResource = resource({
    request: () => this.product().id,
    loader: ({ request: id }) => inject(ReviewService).getByProductId(id),
  });

  reviews = this.reviewsResource;
}
```

## Caching in Resolvers

Resolvers run on every navigation to the route. For expensive API calls, add caching:

### Simple In-Memory Cache

```typescript
export const categoryResolver: ResolveFn<Category[]> = () => {
  const categoryService = inject(CategoryService);

  // CategoryService internally caches the result
  return categoryService.getAll();
};

@Injectable({ providedIn: 'root' })
export class CategoryService {
  private http = inject(HttpClient);
  private cache = signal<Category[] | null>(null);

  async getAll(): Promise<Category[]> {
    if (this.cache()) {
      return this.cache()!;
    }

    const categories = await firstValueFrom(
      this.http.get<Category[]>('/api/categories')
    );
    this.cache.set(categories);
    return categories;
  }

  invalidateCache() {
    this.cache.set(null);
  }
}
```

### ShareReplay for Observable-Based Caching

```typescript
@Injectable({ providedIn: 'root' })
export class ConfigService {
  private http = inject(HttpClient);

  private config$ = this.http.get<AppConfig>('/api/config').pipe(
    shareReplay(1),
  );

  getConfig() {
    return this.config$;
  }
}

export const configResolver: ResolveFn<AppConfig> = () => {
  return inject(ConfigService).getConfig();
};
```

## Resolver with Parent Route Data

Resolvers can access parent route parameters:

```typescript
// Parent route: /teams/:teamId
// Child route: /teams/:teamId/members

export const teamMembersResolver: ResolveFn<Member[]> = (route) => {
  const memberService = inject(MemberService);

  // Access the parent's parameter
  const teamId = route.parent?.paramMap.get('teamId');
  if (!teamId) {
    return [];
  }

  return memberService.getByTeamId(teamId);
};
```

With `paramsInheritanceStrategy: 'always'`, child resolvers can read parent params directly from their own `route`:

```typescript
// With paramsInheritanceStrategy: 'always'
export const teamMembersResolver: ResolveFn<Member[]> = (route) => {
  const teamId = route.paramMap.get('teamId')!; // Inherited from parent
  return inject(MemberService).getByTeamId(teamId);
};
```

## Testing Resolvers

### Basic Test

```typescript
import { TestBed } from '@angular/core/testing';
import { productResolver } from './product.resolver';
import { ProductService } from './product.service';
import { ActivatedRouteSnapshot, RouterStateSnapshot } from '@angular/router';

describe('productResolver', () => {
  let productServiceSpy: jasmine.SpyObj<ProductService>;

  beforeEach(() => {
    productServiceSpy = jasmine.createSpyObj('ProductService', ['getById']);
    TestBed.configureTestingModule({
      providers: [{ provide: ProductService, useValue: productServiceSpy }],
    });
  });

  it('should resolve the product by route param id', async () => {
    const mockProduct: Product = { id: '42', name: 'Widget', price: 9.99 };
    productServiceSpy.getById.and.resolveTo(mockProduct);

    const route = { paramMap: new Map([['id', '42']]) } as unknown as ActivatedRouteSnapshot;

    const result = await TestBed.runInInjectionContext(() =>
      productResolver(route, {} as RouterStateSnapshot)
    );

    expect(result).toEqual(mockProduct);
    expect(productServiceSpy.getById).toHaveBeenCalledWith('42');
  });

  it('should redirect when product not found', async () => {
    productServiceSpy.getById.and.rejectWith(new Error('Not found'));

    const route = { paramMap: new Map([['id', '999']]) } as unknown as ActivatedRouteSnapshot;

    const result = await TestBed.runInInjectionContext(() =>
      productResolver(route, {} as RouterStateSnapshot)
    );

    expect(result).toBeInstanceOf(RedirectCommand);
  });
});
```

## Common Mistakes

### Mistake 1: Slow Resolvers Blocking Navigation

```typescript
// BAD: Fetching everything in the resolver blocks navigation for seconds
export const dashboardResolver: ResolveFn<DashboardData> = async () => {
  const analytics = await inject(AnalyticsService).getFullReport(); // 3 seconds
  const users = await inject(UserService).getAll();                 // 2 seconds
  const logs = await inject(LogService).getRecent();                // 1 second
  return { analytics, users, logs }; // Total: 6 seconds of blocked navigation!
};

// GOOD: Resolve only critical data, load the rest in the component
export const dashboardResolver: ResolveFn<QuickStats> = () => {
  return inject(AnalyticsService).getQuickStats(); // 200ms
};

// Component loads the rest with resource()
```

### Mistake 2: Not Handling Errors

```typescript
// BAD: Unhandled errors crash the navigation
export const productResolver: ResolveFn<Product> = (route) => {
  return inject(ProductService).getById(route.paramMap.get('id')!);
  // If the API returns 404, navigation crashes with an unhandled error
};

// GOOD: Catch errors and redirect or return a fallback
export const productResolver: ResolveFn<Product> = async (route) => {
  try {
    return await inject(ProductService).getById(route.paramMap.get('id')!);
  } catch {
    return new RedirectCommand(inject(Router).parseUrl('/not-found'));
  }
};
```

### Mistake 3: Observable That Never Completes

```typescript
// BAD: BehaviorSubject never completes, navigation hangs
export const userResolver: ResolveFn<User> = () => {
  return inject(UserStore).currentUser$; // BehaviorSubject -- never completes
};

// GOOD: Take the first emission
export const userResolver: ResolveFn<User> = () => {
  return inject(UserStore).currentUser$.pipe(take(1));
};

// OR: Use firstValueFrom for a promise
export const userResolver: ResolveFn<User> = () => {
  return firstValueFrom(inject(UserStore).currentUser$);
};
```

### Mistake 4: Duplicating Fetch Logic Between Resolver and Title

```typescript
// BAD: Two separate API calls for the same product
resolve: { product: productResolver },
title: (route) => {
  return inject(ProductService).getById(route.paramMap.get('id')!).then(p => p.name);
  // This is a SECOND API call for the same product!
},

// GOOD: Use TitleStrategy to read from already-resolved data
// Or set the title from the component after it receives the resolved product
```

### Mistake 5: Using Resolvers for Non-Critical Data

```typescript
// BAD: Resolving recommendations blocks navigation for secondary content
resolve: {
  product: productResolver,
  reviews: reviewsResolver,           // User can wait for these
  recommendations: recommendationsResolver, // And these
  // Navigation blocked until ALL three complete
},

// GOOD: Resolve only the product, load the rest in the component
resolve: { product: productResolver },
// reviews and recommendations loaded via resource() with skeleton UI
```

## Best Practices

### 1. Keep Resolvers Fast

If a resolver takes more than 300ms, the user perceives the app as unresponsive. Resolve only critical data that the component cannot render without.

### 2. Handle Errors with `RedirectCommand`

Never let resolver errors crash silently. Redirect to error pages or return fallback data.

### 3. Use Resolver Factories for Repetitive Patterns

If multiple routes resolve entities by ID, create a generic `resolveEntity()` factory.

### 4. Combine Resolvers with Component Loading

Use resolvers for essential data (fast, critical) and `resource()` for secondary data (slower, non-blocking).

### 5. Set `runGuardsAndResolvers` for Query-Driven Routes

If your resolver reads query parameters, change `runGuardsAndResolvers` to `'paramsOrQueryParamsChange'` so the resolver re-runs when query params change.

### 6. Cache When Appropriate

If the resolved data changes infrequently, cache it in the service to avoid redundant API calls on back/forward navigation.

## Real-World Use Cases

### Article Page with SEO-Ready Data

```typescript
export const articleResolver: ResolveFn<Article> = async (route) => {
  const cms = inject(CmsService);
  const router = inject(Router);
  const slug = route.paramMap.get('slug')!;

  try {
    const article = await cms.getBySlug(slug);
    if (article.status !== 'published') {
      return new RedirectCommand(router.parseUrl('/not-found'));
    }
    return article;
  } catch {
    return new RedirectCommand(router.parseUrl('/not-found'));
  }
};

export const articleTitleResolver: ResolveFn<string> = async (route) => {
  const cms = inject(CmsService);
  const slug = route.paramMap.get('slug')!;
  const article = await cms.getBySlug(slug);
  return article.title;
};

const routes: Routes = [
  {
    path: 'blog/:slug',
    loadComponent: () => import('./article.component'),
    resolve: { article: articleResolver },
    title: articleTitleResolver,
  },
];
```

### Order Confirmation with Validation

```typescript
export const orderResolver: ResolveFn<Order> = async (route) => {
  const orderService = inject(OrderService);
  const authService = inject(AuthService);
  const router = inject(Router);
  const orderId = route.paramMap.get('orderId')!;

  try {
    const order = await orderService.getById(orderId);

    // Verify the order belongs to the current user
    if (order.userId !== authService.currentUserId()) {
      return new RedirectCommand(router.parseUrl('/unauthorized'));
    }

    return order;
  } catch {
    return new RedirectCommand(router.parseUrl('/orders'));
  }
};

{
  path: 'orders/:orderId',
  loadComponent: () => import('./order-detail.component'),
  resolve: { order: orderResolver },
  canActivate: [authGuard],
}
```

### Dashboard with Hybrid Loading

```typescript
// Resolve: fast, critical data
export const dashboardResolver: ResolveFn<DashboardSummary> = () => {
  return inject(DashboardService).getSummary();
};

// Route
{
  path: 'dashboard',
  loadComponent: () => import('./dashboard.component'),
  resolve: { summary: dashboardResolver },
}

// Component: resolver data + deferred secondary data
@Component({
  template: `
    <app-stats-bar [stats]="summary()" />

    @defer (on idle) {
      <app-activity-feed />
    } @placeholder {
      <app-feed-skeleton />
    }

    @defer (on viewport) {
      <app-charts [period]="period()" />
    } @placeholder {
      <div class="chart-placeholder" style="height: 400px"></div>
    }
  `,
})
export class DashboardComponent {
  summary = input.required<DashboardSummary>(); // From resolver
  period = signal<'week' | 'month' | 'year'>('month');
}
```

---

> **Key Takeaway:** Use functional resolvers for critical data that the component cannot render without, keep them fast (under 300ms), handle errors with `RedirectCommand`, and combine them with component-level `resource()` and `@defer` for secondary data to avoid blocking navigation unnecessarily.
