# 16.6 Dynamic Forms with `FormArray`

> `FormArray` manages an ordered list of form controls that can be added, removed, reordered, and validated at runtime, making it the core building block for dynamic forms like editable tables, repeatable field groups, and multi-item entry screens.

---

## FormArray Fundamentals

`FormArray` is a reactive forms class that holds an ordered collection of controls (any mix of `FormControl`, `FormGroup`, or nested `FormArray`). Controls are accessed by numeric index, not by name.

```typescript
import { FormArray, FormControl, FormGroup, Validators } from '@angular/forms';

const skills = new FormArray([
  new FormControl('Angular'),
  new FormControl('TypeScript'),
  new FormControl('RxJS')
]);

skills.value;       // ['Angular', 'TypeScript', 'RxJS']
skills.length;      // 3
skills.at(0).value; // 'Angular'
```

---

## FormArray API

### Adding Controls

```typescript
// Append to end
skills.push(new FormControl('CSS'));

// Insert at specific index
skills.insert(1, new FormControl('JavaScript'));

// Result: ['Angular', 'JavaScript', 'TypeScript', 'RxJS', 'CSS']
```

### Removing Controls

```typescript
// Remove by index
skills.removeAt(2);

// Remove all controls
skills.clear();
```

### Accessing Controls

```typescript
// By index
const first = skills.at(0);

// All controls
skills.controls;  // FormControl[]

// Length
skills.length;    // number
```

### Replacing a Control

```typescript
// Replace the control at index 1
skills.setControl(1, new FormControl('Rust'));
```

### Reordering Controls

Angular does not provide a built-in `move()` method. Implement it manually:

```typescript
function moveItemInFormArray(
  formArray: FormArray,
  fromIndex: number,
  toIndex: number
): void {
  const control = formArray.at(fromIndex);
  formArray.removeAt(fromIndex);
  formArray.insert(toIndex, control);
}
```

## Template Binding with formArrayName

Use `formArrayName` to bind a `FormArray` to the template. Inside, use `[formControlName]="$index"` for simple controls or `[formGroupName]="$index"` for groups.

### Simple FormArray (Array of FormControls)

```typescript
@Component({
  selector: 'app-tags',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="form">
      <div formArrayName="tags">
        <h3>Tags</h3>
        @for (tag of tags.controls; track $index) {
          <div class="tag-row">
            <input [formControlName]="$index" />
            <button type="button" (click)="removeTag($index)">Remove</button>
          </div>
        }
      </div>
      <button type="button" (click)="addTag()">Add Tag</button>
    </form>
  `
})
export class TagsComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    tags: this.fb.array<string>(['angular', 'typescript'])
  });

  get tags(): FormArray {
    return this.form.get('tags') as FormArray;
  }

  addTag(): void {
    this.tags.push(this.fb.control(''));
  }

  removeTag(index: number): void {
    this.tags.removeAt(index);
  }
}
```

### FormArray of FormGroups

Each item in the array is a `FormGroup` with multiple fields:

```typescript
@Component({
```

## Factory Pattern for Row Creation

Extract the row creation logic into a private method. This keeps the component clean and ensures every row is created consistently:

```typescript
@Component({ /* ... */ })
export class InvoiceComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    invoiceNumber: ['', Validators.required],
    lineItems: this.fb.array<FormGroup>([this.createLineItem()])
  });

  get lineItems(): FormArray<FormGroup> {
    return this.form.get('lineItems') as FormArray<FormGroup>;
  }

  private createLineItem(data?: Partial<LineItem>): FormGroup {
    return this.fb.group({
      description: [data?.description ?? '', Validators.required],
      quantity: [data?.quantity ?? 1, [Validators.required, Validators.min(1)]],
      unitPrice: [data?.unitPrice ?? 0, [Validators.required, Validators.min(0)]],
      taxRate: [data?.taxRate ?? 0, [Validators.min(0), Validators.max(100)]]
    });
  }

  addLineItem(): void {
    this.lineItems.push(this.createLineItem());
  }

  removeLineItem(index: number): void {
    this.lineItems.removeAt(index);
  }
}

interface LineItem {
  description: string;
  quantity: number;
  unitPrice: number;
  taxRate: number;
}
```

## Populating FormArray from API Data

Load an array of items from an API and build the form dynamically:

```typescript
@Component({ /* ... */ })
export class EditOrderComponent {
  private fb = inject(NonNullableFormBuilder);
  private orderService = inject(OrderService);

  form = this.fb.group({
    orderId: [''],
    items: this.fb.array<FormGroup>([])
  });

  get items(): FormArray<FormGroup> {
    return this.form.get('items') as FormArray<FormGroup>;
  }

  constructor() {
    const orderId = inject(ActivatedRoute).snapshot.paramMap.get('id')!;

    this.orderService.getById(orderId).subscribe(order => {
      // Populate scalar fields
      this.form.patchValue({ orderId: order.id });

      // Clear existing items and rebuild from API data
      this.items.clear();
      order.items.forEach(item => {
        this.items.push(this.createItemGroup(item));
      });

      this.form.markAsPristine();
    });
  }

  private createItemGroup(item?: OrderItem): FormGroup {
    return this.fb.group({
      productId: [item?.productId ?? '', Validators.required],
      quantity: [item?.quantity ?? 1, [Validators.required, Validators.min(1)]],
      price: [item?.price ?? 0, Validators.required]
    });
  }
}
```

**Important:** Use `clear()` before repopulating to avoid stale rows. `patchValue()` does not add or remove array items; it only updates existing ones.

## Nested FormArrays

FormArrays can be nested inside FormGroups inside FormArrays, creating deeply nested dynamic structures:

```typescript
// A form for a quiz with multiple questions, each having multiple answer options
@Component({
  selector: 'app-quiz-builder',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="form">
      <div formArrayName="questions">
        @for (question of questions.controls; track $index; let qi = $index) {
          <div [formGroupName]="qi" class="question-block">
            <h3>Question {{ qi + 1 }}</h3>
            <input formControlName="text" placeholder="Question text" />

            <div formArrayName="options">
              @for (option of getOptions(qi).controls; track $index; let oi = $index) {
                <div class="option-row">
                  <input [formControlName]="oi" [placeholder]="'Option ' + (oi + 1)" />
                  <button type="button" (click)="removeOption(qi, oi)">x</button>
                </div>
              }
            </div>
            <button type="button" (click)="addOption(qi)">Add Option</button>
            <button type="button" (click)="removeQuestion(qi)">Remove Question</button>
          </div>
        }
      </div>
      <button type="button" (click)="addQuestion()">Add Question</button>
    </form>
  `
})
export class QuizBuilderComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    questions: this.fb.array<FormGroup>([])
  });

  get questions(): FormArray<FormGroup> {
    return this.form.get('questions') as FormArray<FormGroup>;
  }

  getOptions(questionIndex: number): FormArray {
    return this.questions.at(questionIndex).get('options') as FormArray;
  }

  addQuestion(): void {
    this.questions.push(this.fb.group({
      text: ['', Validators.required],
      options: this.fb.array([
        this.fb.control('', Validators.required),
        this.fb.control('', Validators.required)
      ])
    }));
  }

  removeQuestion(index: number): void {
    this.questions.removeAt(index);
  }

  addOption(questionIndex: number): void {
    this.getOptions(questionIndex).push(this.fb.control('', Validators.required));
  }

  removeOption(questionIndex: number, optionIndex: number): void {
    this.getOptions(questionIndex).removeAt(optionIndex);
  }
}
```

## FormArray Validation

### Array-Level Validators

Apply validators to the `FormArray` itself to enforce constraints on the collection:

```typescript
// Minimum number of items
function minArrayLength(min: number): ValidatorFn {
  return (control: AbstractControl): ValidationErrors | null => {
    const arr = control as FormArray;
    return arr.length >= min ? null : {
      minArrayLength: { required: min, actual: arr.length }
    };
  };
}

// Maximum number of items
function maxArrayLength(max: number): ValidatorFn {
  return (control: AbstractControl): ValidationErrors | null => {
    const arr = control as FormArray;
    return arr.length <= max ? null : {
      maxArrayLength: { allowed: max, actual: arr.length }
    };
  };
}

// No duplicate values
function noDuplicates(): ValidatorFn {
  return (control: AbstractControl): ValidationErrors | null => {
    const arr = control as FormArray;
    const values = arr.controls.map(c => c.value);
    const hasDuplicates = new Set(values).size !== values.length;
    return hasDuplicates ? { duplicates: true } : null;
  };
}
```

Apply:

```typescript
const tags = this.fb.array<string>(
  [],
  { validators: [minArrayLength(1), maxArrayLength(10), noDuplicates()] }
);
```

### Using Validators.required on FormArray

`Validators.required` on a `FormArray` checks that the array is not empty (length > 0):

```typescript
const items = this.fb.array([], Validators.required);
items.valid;   // false (empty array)
items.push(this.fb.control('item'));
items.valid;   // true
```

### Aggregate Validity

A `FormArray` is invalid if:
- Any of its own validators fail.
- Any child control is invalid.

```typescript
const items = this.fb.array([
  this.fb.control('', Validators.required),   // invalid (empty)
  this.fb.control('Valid Item')                // valid
]);

items.valid;  // false (first child is invalid)
```

## Editable Table Pattern

Build an inline-editable data table using a `FormArray` of `FormGroup`s:

```typescript
@Component({
  selector: 'app-editable-table',
  standalone: true,
  imports: [ReactiveFormsModule, CurrencyPipe],
  template: `
    <form [formGroup]="form">
      <table>
        <thead>
          <tr>
            <th>Product</th>
            <th>Qty</th>
            <th>Price</th>
            <th>Subtotal</th>
            <th></th>
          </tr>
        </thead>
        <tbody formArrayName="rows">
          @for (row of rows.controls; track $index) {
            <tr [formGroupName]="$index">
              <td>
                <input formControlName="product" />
                @if (row.get('product')?.hasError('required') && row.get('product')?.touched) {
                  <span class="error">Required</span>
                }
              </td>
              <td>
                <input formControlName="qty" type="number" style="width: 60px" />
              </td>
              <td>
                <input formControlName="price" type="number" style="width: 80px" />
              </td>
              <td class="subtotal">
                {{ subtotal($index) | currency }}
              </td>
              <td>
                <button type="button" (click)="removeRow($index)"
                        [disabled]="rows.length <= 1">
                  Delete
                </button>
              </td>
            </tr>
          }
        </tbody>
        <tfoot>
          <tr>
            <td colspan="3" class="total-label">Total:</td>
            <td class="total-value">{{ total() | currency }}</td>
            <td></td>
          </tr>
        </tfoot>
      </table>

      <button type="button" (click)="addRow()">Add Row</button>
      <button type="submit" [disabled]="form.invalid" (click)="submit()">Save</button>
    </form>
  `
})
export class EditableTableComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    rows: this.fb.array<FormGroup>(
      [this.createRow()],
      { validators: minArrayLength(1) }
    )
  });

  get rows(): FormArray<FormGroup> {
    return this.form.get('rows') as FormArray<FormGroup>;
  }

  private createRow(): FormGroup {
    return this.fb.group({
      product: ['', Validators.required],
      qty: [1, [Validators.required, Validators.min(1)]],
      price: [0, [Validators.required, Validators.min(0)]]
    });
  }

  addRow(): void {
    this.rows.push(this.createRow());
  }

  removeRow(index: number): void {
    if (this.rows.length > 1) {
      this.rows.removeAt(index);
    }
  }

  subtotal(index: number): number {
    const row = this.rows.at(index);
    return (row.get('qty')?.value ?? 0) * (row.get('price')?.value ?? 0);
  }

  total(): number {
    return this.rows.controls.reduce(
      (sum, _, i) => sum + this.subtotal(i), 0
    );
  }

  submit(): void {
    if (this.form.valid) {
      console.log(this.form.getRawValue());
    }
  }
}
```

## Drag-and-Drop Reordering

Integrate with Angular CDK's `cdkDrag` and `cdkDropList` to reorder `FormArray` items:

```typescript
import { CdkDragDrop, CdkDropList, CdkDrag } from '@angular/cdk/drag-drop';

@Component({
  selector: 'app-sortable-list',
  standalone: true,
  imports: [ReactiveFormsModule, CdkDropList, CdkDrag],
  template: `
    <form [formGroup]="form">
      <div
        formArrayName="items"
        cdkDropList
        (cdkDropListDropped)="onDrop($event)"
      >
        @for (item of items.controls; track $index) {
          <div [formGroupName]="$index" cdkDrag class="draggable-item">
            <span cdkDragHandle class="drag-handle">&#9776;</span>
            <input formControlName="label" />
            <button type="button" (click)="removeItem($index)">Remove</button>
          </div>
        }
      </div>
      <button type="button" (click)="addItem()">Add Item</button>
    </form>
  `
})
export class SortableListComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    items: this.fb.array<FormGroup>([])
  });

  get items(): FormArray<FormGroup> {
    return this.form.get('items') as FormArray<FormGroup>;
  }

  addItem(): void {
    this.items.push(this.fb.group({
      label: ['', Validators.required]
    }));
  }

  removeItem(index: number): void {
    this.items.removeAt(index);
  }

  onDrop(event: CdkDragDrop<FormGroup[]>): void {
    // Move the control in the FormArray
    const control = this.items.at(event.previousIndex);
    this.items.removeAt(event.previousIndex);
    this.items.insert(event.currentIndex, control);
  }
}
```

**Note:** You must move the control in the `FormArray` to keep it synchronized with the DOM. If you only use CDK's `moveItemInArray` on the `controls` array directly, the form model and template bindings can desync.

## Tracking Changes in FormArray

### valueChanges on the Array

```typescript
this.items.valueChanges
  .pipe(takeUntilDestroyed())
  .subscribe(values => {
    // values is the full array value
    console.log('Items changed:', values);
    this.recalculateTotal(values);
  });
```

### valueChanges on Individual Items

```typescript
this.items.at(0).valueChanges
  .pipe(takeUntilDestroyed())
  .subscribe(value => {
    console.log('First item changed:', value);
  });
```

### Watching for Structural Changes

`valueChanges` fires when items are added, removed, or their values change. To distinguish structural changes from value changes, compare lengths:

```typescript
let previousLength = this.items.length;

this.items.valueChanges
  .pipe(takeUntilDestroyed())
  .subscribe(() => {
    if (this.items.length !== previousLength) {
      console.log('Items added or removed');
      previousLength = this.items.length;
    }
  });
```

## Performance Considerations

### Track by Index

Always use `track $index` in `@for` loops over `FormArray` controls:

```html
@for (item of items.controls; track $index) {
  <div [formGroupName]="$index">...</div>
}
```

Without tracking, Angular recreates all DOM elements when any control is added or removed, causing input focus loss and poor performance.

### Avoid Calling Methods in Template Expressions

```html
<!-- BAD: creates new objects on every change detection cycle -->
@for (item of getItems(); track $index) { ... }

<!-- GOOD: reference the getter directly -->
@for (item of items.controls; track $index) { ... }
```

### Batch Operations

When populating a large `FormArray` from API data, disable events during the build and re-enable afterward:

```typescript
loadItems(data: Item[]): void {
  // Clear without emitting
  while (this.items.length > 0) {
    this.items.removeAt(0, { emitEvent: false });
  }

  // Add all items without emitting
  data.forEach(item => {
    this.items.push(this.createItemGroup(item), { emitEvent: false });
  });

  // Emit once at the end
  this.items.updateValueAndValidity();
}
```

## Sub-Component Pattern

Extract each row into its own component for cleaner code:

```typescript
// row.component.ts
@Component({
  selector: 'app-item-row',
  standalone: true,
  imports: [ReactiveFormsModule],
  viewProviders: [{
    provide: ControlContainer,
    useExisting: FormGroupDirective
  }],
  template: `
    <div [formGroupName]="index()">
      <input formControlName="name" placeholder="Name" />
      <input formControlName="qty" type="number" placeholder="Qty" />
      <button type="button" (click)="remove.emit()">Remove</button>
    </div>
  `
})
export class ItemRowComponent {
  index = input.required<number>();
  remove = output<void>();
}
```

```typescript
// parent.component.ts
@Component({
  selector: 'app-item-list',
  standalone: true,
  imports: [ReactiveFormsModule, ItemRowComponent],
  template: `
    <form [formGroup]="form">
      <div formArrayName="items">
        @for (item of items.controls; track $index) {
          <app-item-row
            [index]="$index"
            (remove)="removeItem($index)"
          />
        }
      </div>
      <button type="button" (click)="addItem()">Add</button>
    </form>
  `
})
export class ItemListComponent {
  // ...
}
```

The key is `viewProviders` with `ControlContainer` and `FormGroupDirective`. This gives the child component access to the parent's `FormGroup` context so `formGroupName` works correctly.

## Common Mistakes

- **Using `patchValue` to add items.** `patchValue` only updates existing controls. It does not add or remove items in a `FormArray`. To add items, use `push()` or `insert()`. To remove, use `removeAt()` or `clear()`.

- **Forgetting `track $index` in the loop.** Without tracking, Angular destroys and recreates all row DOM elements when the array changes. This causes focus loss and poor performance.

- **Accessing `FormArray` without a typed getter.** Writing `this.form.get('items') as FormArray` everywhere is noisy. Create a getter: `get items(): FormArray<FormGroup>`.

- **Directly mutating `controls` array.** Never do `this.items.controls.splice(...)`. Use the `FormArray` API (`push`, `removeAt`, `insert`, `clear`) so Angular tracks the changes and updates validity.

- **Not disabling events during batch operations.** Adding 100 items in a loop fires `valueChanges` 100 times. Pass `{ emitEvent: false }` to each operation and call `updateValueAndValidity()` once at the end.

- **Missing `formArrayName` in the template.** The `formArrayName` directive must wrap the loop. Without it, `[formControlName]="$index"` or `[formGroupName]="$index"` does not know which `FormArray` to bind to.

---

## Best Practices

- **Use a factory method for row creation.** Extract the `FormGroup` creation into a `private createRow()` method. This guarantees every row has the same structure and validators.

- **Apply array-level validators for collection constraints.** Use custom validators like `minArrayLength`, `maxArrayLength`, and `noDuplicates` directly on the `FormArray`.

- **Prevent removing the last item.** For forms that require at least one entry, disable the remove button when `items.length <= 1` and validate with `minArrayLength(1)`.

- **Use `emitEvent: false` for bulk loading.** When populating from API data, suppress events until all items are added, then emit once.

- **Extract rows into sub-components for complex forms.** Use `viewProviders` with `ControlContainer` to give child components access to the parent form context.

- **Synchronize CDK drag-drop with the FormArray.** When reordering, remove the control and re-insert it at the new index. Do not manipulate the `controls` array directly.

```typescript
            <button type="button" (click)="removeQuestion(qi)">Delete Question</button>
          </fieldset>
        }
      </div>

      <button type="button" (click)="addQuestion()">Add Question</button>
      <button type="submit" [disabled]="form.invalid">Publish</button>
    </form>
  `
})
export class SurveyBuilderComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    surveyTitle: ['', Validators.required],
    questions: this.fb.array<FormGroup>([], minArrayLength(1))
  });

  get questions(): FormArray<FormGroup> {
    return this.form.get('questions') as FormArray<FormGroup>;
  }

  getChoices(qi: number): FormArray {
    return this.questions.at(qi).get('choices') as FormArray;
  }

  addQuestion(): void {
    this.questions.push(this.fb.group({
      text: ['', Validators.required],
      type: ['text'],
      required: [false],
      choices: this.fb.array<string>(['', ''])
    }));
  }

  removeQuestion(qi: number): void {
    this.questions.removeAt(qi);
  }

  addChoice(qi: number): void {
    this.getChoices(qi).push(this.fb.control('', Validators.required));
  }

  removeChoice(qi: number, ci: number): void {
    if (this.getChoices(qi).length > 2) {
      this.getChoices(qi).removeAt(ci);
    }
  }

  onTypeChange(qi: number): void {
    const type = this.questions.at(qi).get('type')?.value;
    const choices = this.getChoices(qi);
    if (type === 'choice' && choices.length === 0) {
      choices.push(this.fb.control(''));
      choices.push(this.fb.control(''));
    }
  }

  publish(): void {
    console.log(this.form.getRawValue());
  }
}
```

### 3. Bulk CSV Import with Editable Preview

```typescript
@Component({
  selector: 'app-csv-import',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <input type="file" (change)="onFileSelected($event)" accept=".csv" />

    @if (rows.length > 0) {
      <form [formGroup]="form" (ngSubmit)="importAll()">
        <table>
          <thead>
            <tr>
              <th>Name</th>
              <th>Email</th>
              <th>Role</th>
              <th>Valid</th>
            </tr>
          </thead>
          <tbody formArrayName="rows">
            @for (row of rows.controls; track $index) {
              <tr [formGroupName]="$index" [class.invalid-row]="row.invalid">
                <td><input formControlName="name" /></td>
                <td><input formControlName="email" /></td>
                <td>
                  <select formControlName="role">
                    <option value="user">User</option>
                    <option value="admin">Admin</option>
                  </select>
                </td>
                <td>{{ row.valid ? 'Yes' : 'No' }}</td>
              </tr>
            }
          </tbody>
        </table>

        <p>{{ validCount() }} of {{ rows.length }} rows valid</p>
        <button type="submit" [disabled]="form.invalid">Import All</button>
      </form>
    }
  `
})
export class CsvImportComponent {
  private fb = inject(NonNullableFormBuilder);

  form = this.fb.group({
    rows: this.fb.array<FormGroup>([])
  });

  get rows(): FormArray<FormGroup> {
    return this.form.get('rows') as FormArray<FormGroup>;
  }

  validCount(): number {
    return this.rows.controls.filter(c => c.valid).length;
  }

  onFileSelected(event: Event): void {
    const file = (event.target as HTMLInputElement).files?.[0];
    if (!file) return;

    const reader = new FileReader();
    reader.onload = () => {
      const csv = reader.result as string;
      const lines = csv.split('\n').slice(1); // skip header

      this.rows.clear();
      lines
        .filter(line => line.trim())
        .forEach(line => {
          const [name, email, role] = line.split(',').map(s => s.trim());
          this.rows.push(this.fb.group({
            name: [name, Validators.required],
            email: [email, [Validators.required, Validators.email]],
            role: [role || 'user']
          }), { emitEvent: false });
        });

      this.rows.updateValueAndValidity();
    };
    reader.readAsText(file);
  }

  importAll(): void {
    console.log('Importing:', this.form.getRawValue());
  }
}
```

> **Key Takeaway:** Use `FormArray` with a factory method for consistent row creation, always use `track $index` in template loops, apply array-level validators for collection constraints, and suppress events with `{ emitEvent: false }` during bulk operations to keep dynamic forms performant and maintainable.
