# 18.11 Memory Management (`takeUntilDestroyed`, `DestroyRef`)

> Proper subscription cleanup prevents memory leaks, stale callbacks, and ghost HTTP requests in Angular applications.

---

## Why Memory Management Matters in Angular

Every RxJS subscription that outlives its component creates a memory leak. The component is destroyed, but the subscription callback still holds a reference to the component instance, preventing garbage collection. Worse, the callback continues executing, updating UI that no longer exists or triggering side effects on stale data.

```typescript
// MEMORY LEAK: This subscription lives forever
@Component({
  selector: 'app-ticker',
  template: `{{ price }}`
})
export class TickerComponent implements OnInit {
  price = 0;

  ngOnInit() {
    // This interval NEVER stops, even after the component is destroyed
    interval(1000).pipe(
      switchMap(() => this.http.get<number>('/api/price'))
    ).subscribe(p => {
      this.price = p; // Updating a destroyed component's property
    });
  }
}
```

The subscription holds a closure over `this` (the component instance). The garbage collector cannot reclaim the component because the active subscription still references it. Navigate away and back 10 times, and you have 10 zombie subscriptions running simultaneously.

---

## The Unsubscription Strategies

Angular offers multiple approaches to subscription cleanup. Here they are ordered from most recommended to least.

### Strategy 1: Declarative Templates (`async` pipe / `toSignal`)

The best subscription is one you never manage. Let Angular handle it.

**`async` pipe (legacy but still valid):**

```html
<!-- Angular subscribes on render, unsubscribes on destroy automatically -->
<div>{{ data$ | async }}</div>

<!-- With @if for null handling -->
@if (data$ | async; as data) {
  <div>{{ data.name }}</div>
}
```

**`toSignal()` (preferred in Angular 21):**

```typescript
@Component({
  selector: 'app-user',
  template: `
    @if (user(); as user) {
      <h1>{{ user.name }}</h1>
    }
  `
})
export class UserComponent {
  private readonly userService = inject(UserService);

  // toSignal subscribes internally, unsubscribes when the component's
  // DestroyRef fires. No manual cleanup needed.
  readonly user = toSignal(this.userService.currentUser$);
}
```

`toSignal()` internally uses `takeUntilDestroyed` via the injection context. When the component is destroyed, the subscription is automatically cleaned up.

**`toSignal()` options that affect cleanup:**

```typescript
// manualCleanup: true -- disables automatic DestroyRef teardown
// Use when the Observable completes on its own (e.g., take(5))
// No injection context required when this is set
const data = toSignal(source$.pipe(take(5)), { manualCleanup: true });

// requireSync: true -- enforces synchronous first emission
// The signal type excludes undefined (no initial value needed)
// Useful for BehaviorSubject-backed Observables
const count = toSignal(countSubject$, { requireSync: true });
// Type: Signal<number> instead of Signal<number | undefined>
```

**When to use:** Whenever the Observable's purpose is to provide data for the template. This covers the majority of use cases.

### Strategy 2: `takeUntilDestroyed()`

For imperative subscriptions that must exist (side effects, logging, coordination between streams), `takeUntilDestroyed()` is the modern standard.

```typescript
import { takeUntilDestroyed } from '@angular/core/rxjs-interop';

@Component({
  selector: 'app-tracker',
  template: `...`
})
export class TrackerComponent {
  constructor() {
    // Works in the constructor because injection context is available
    interval(5000).pipe(
      takeUntilDestroyed(),
      switchMap(() => this.analyticsService.flush())
    ).subscribe();
  }

  private readonly analyticsService = inject(AnalyticsService);
}
```

### Strategy 3: `DestroyRef.onDestroy()`

For non-RxJS cleanup (event listeners, timers, third-party library teardown), use `DestroyRef` directly.

```typescript
@Component({
  selector: 'app-chart',
  template: `<div #chartContainer></div>`
})
export class ChartComponent {
  private readonly destroyRef = inject(DestroyRef);
  private chart: ThirdPartyChart | null = null;

  ngAfterViewInit() {
    this.chart = new ThirdPartyChart(this.chartContainer.nativeElement);

    this.destroyRef.onDestroy(() => {
      this.chart?.dispose(); // Clean up non-Angular resource
      this.chart = null;
    });
  }
}
```

### Strategy 4: Manual Unsubscription

Sometimes you need fine-grained control over when a subscription ends, not just on component destruction.

```typescript
@Component({
  selector: 'app-recorder',
  template: `
    <button (click)="startRecording()">Start</button>
    <button (click)="stopRecording()">Stop</button>
  `
})
export class RecorderComponent {
  private recordingSub: Subscription | null = null;

  startRecording() {
    this.recordingSub = this.audioService.stream$.subscribe(
      chunk => this.buffer.push(chunk)
    );
  }

  stopRecording() {
    this.recordingSub?.unsubscribe();
    this.recordingSub = null;
    this.processBuffer();
  }
}
```

## `takeUntilDestroyed()` In Depth

### How It Works Internally

`takeUntilDestroyed()` does three things:

1. Obtains a `DestroyRef` from the current injection context (or from the parameter you pass)
2. Registers an `onDestroy` callback on that `DestroyRef`
3. When `onDestroy` fires, completes the Observable chain (like `takeUntil` with a Subject that emits)

Simplified internal implementation:

```typescript
// Conceptual implementation (not actual source code)
function takeUntilDestroyed(destroyRef?: DestroyRef) {
  // If no DestroyRef provided, get it from the current injection context
  if (!destroyRef) {
    destroyRef = inject(DestroyRef);
  }

  return <T>(source: Observable<T>) =>
    new Observable<T>(subscriber => {
      // Subscribe to source
      const sub = source.subscribe(subscriber);

      // Register cleanup
      destroyRef!.onDestroy(() => {
        sub.unsubscribe();
      });

      return sub;
    });
}
```

### Injection Context Requirement

`takeUntilDestroyed()` without a parameter requires an **injection context**. This context exists in:

- Constructor body
- Field initializers
- Factory functions passed to `inject()`
- `runInInjectionContext()` calls

It does **NOT** exist in:

- `ngOnInit()`
- `ngAfterViewInit()`
- Event handlers
- Callback functions
- `setTimeout` / `setInterval` callbacks

```typescript
@Component({ selector: 'app-demo', template: `...` })
export class DemoComponent implements OnInit {
  private readonly destroyRef = inject(DestroyRef);

  // WORKS: Field initializer has injection context
  private readonly data$ = inject(DataService).stream$.pipe(
    takeUntilDestroyed()
  );

  constructor() {
    // WORKS: Constructor has injection context
    someObservable$.pipe(
      takeUntilDestroyed()
    ).subscribe();
  }

  ngOnInit() {
    // DOES NOT WORK: No injection context in ngOnInit
    // Error: NG0203 - inject() must be called from an injection context
    someObservable$.pipe(
      takeUntilDestroyed() // THROWS!
    ).subscribe();

    // FIX: Pass DestroyRef explicitly
    someObservable$.pipe(
      takeUntilDestroyed(this.destroyRef) // WORKS
    ).subscribe();
  }

  onClick() {
    // DOES NOT WORK: No injection context in event handlers
    someObservable$.pipe(
      takeUntilDestroyed() // THROWS!
    ).subscribe();

    // FIX: Pass DestroyRef explicitly
    someObservable$.pipe(
      takeUntilDestroyed(this.destroyRef) // WORKS
    ).subscribe();
  }
}
```

**The pattern:** Capture `DestroyRef` as a field, then pass it explicitly whenever you are outside the injection context.

**Alternative trick:** Capture the operator itself as a field initializer, then reuse it anywhere:

```typescript
export class MyComponent {
  // Captured in injection context (field initializer)
  private readonly untilDestroyed = takeUntilDestroyed();

  ngOnInit() {
    // Works without passing DestroyRef because the operator was already created
    someObservable$.pipe(this.untilDestroyed).subscribe();
  }
}
```

### `takeUntilDestroyed` in Services

Services provided at the root level live for the entire application lifecycle. `takeUntilDestroyed` in a root service is effectively a no-op because the service is never destroyed.

```typescript
// Root service: takeUntilDestroyed does nothing useful here
@Injectable({ providedIn: 'root' })
export class GlobalService {
  constructor() {
    // This subscription lives until the app is closed anyway
    interval(60000).pipe(
      takeUntilDestroyed() // Redundant for root services
    ).subscribe(() => this.heartbeat());
  }
}
```

However, for services provided at the component level, `takeUntilDestroyed` is valuable:

```typescript
// Component-provided service: destroyed with the component
@Injectable()
export class FormValidationService {
  private readonly destroyRef = inject(DestroyRef);

  constructor() {
    // This subscription ends when the component (and this service) is destroyed
    this.formChanges$.pipe(
      takeUntilDestroyed()
    ).subscribe(changes => this.validate(changes));
  }
}

@Component({
  selector: 'app-form',
  providers: [FormValidationService], // Scoped to this component
  template: `...`
})
export class FormComponent {
  private readonly validation = inject(FormValidationService);
}
```

### `takeUntilDestroyed` Placement in the Pipe

`takeUntilDestroyed()` should generally be the **last operator** before `subscribe()`. Placing it earlier can cause inner subscriptions from flattening operators to leak.

```typescript
// BAD: takeUntilDestroyed before switchMap
// The inner HTTP subscription can outlive the component
source$.pipe(
  takeUntilDestroyed(),
  switchMap(id => this.http.get(`/api/items/${id}`)) // Inner sub not cleaned up!
).subscribe();

// GOOD: takeUntilDestroyed after all operators
source$.pipe(
  switchMap(id => this.http.get(`/api/items/${id}`)),
  takeUntilDestroyed()
).subscribe();
```

The same rule applies to the legacy `takeUntil(destroy$)` pattern.

## `DestroyRef` In Depth

### The API

`DestroyRef` is an injectable token that represents the destroy lifecycle of its context (component, directive, service, or pipe).

```typescript
abstract class DestroyRef {
  abstract onDestroy(callback: () => void): () => void;
}
```

The `onDestroy` method:
- Takes a callback that runs when the context is destroyed
- Returns a cleanup function that unregisters the callback (useful if you need to cancel the registration)

```typescript
@Component({ selector: 'app-demo', template: `...` })
export class DemoComponent {
  private readonly destroyRef = inject(DestroyRef);

  startPolling() {
    const intervalId = setInterval(() => this.poll(), 5000);

    // Register cleanup
    const unregister = this.destroyRef.onDestroy(() => {
      clearInterval(intervalId);
    });

    // Optional: cancel the registration itself (e.g., if polling stops early)
    // unregister();
  }
}
```

### `DestroyRef` Hierarchy

`DestroyRef` is tied to the injector hierarchy. Each provider scope gets its own `DestroyRef`:

```
AppModule Injector (root)
  -> DestroyRef (fires on app shutdown)

LazyLoadedModule Injector
  -> DestroyRef (fires when the module's route is destroyed)

ComponentA Injector
  -> DestroyRef (fires on component destroy)
  -> ServiceX (provided in ComponentA)
    -> inject(DestroyRef) gets ComponentA's DestroyRef
```

```typescript
// Service scoped to a route's lazy-loaded module
@Injectable()
export class FeatureStateService {
  private readonly destroyRef = inject(DestroyRef);

  constructor() {
    // This subscription lives as long as the lazy-loaded route
    this.store.pipe(
      select(featureState),
      takeUntilDestroyed()
    ).subscribe(state => this.cache.set(state));
  }
}
```

### Using `DestroyRef` to Replace `ngOnDestroy`

Before `DestroyRef`, cleanup required implementing the `OnDestroy` interface. `DestroyRef` is more flexible because:

1. You can register cleanup at the point of creation, not in a separate lifecycle method
2. Multiple independent cleanup registrations are cleaner than one monolithic `ngOnDestroy`
3. Helper functions can accept `DestroyRef` without needing access to the component class

```typescript
// BEFORE: Monolithic ngOnDestroy
@Component({ selector: 'app-old', template: `...` })
export class OldComponent implements OnInit, OnDestroy {
  private sub1!: Subscription;
  private sub2!: Subscription;
  private resizeListener!: () => void;

  ngOnInit() {
    this.sub1 = this.service.data$.subscribe(/*...*/);
    this.sub2 = this.events.click$.subscribe(/*...*/);
    this.resizeListener = () => this.onResize();
    window.addEventListener('resize', this.resizeListener);
  }

  ngOnDestroy() {
    this.sub1.unsubscribe();
    this.sub2.unsubscribe();
    window.removeEventListener('resize', this.resizeListener);
  }
}

// AFTER: Colocated cleanup with DestroyRef
@Component({ selector: 'app-new', template: `...` })
export class NewComponent {
  private readonly destroyRef = inject(DestroyRef);

  constructor() {
    // Each subscription manages its own cleanup
    this.service.data$.pipe(
      takeUntilDestroyed()
    ).subscribe(/*...*/);

    this.events.click$.pipe(
      takeUntilDestroyed()
    ).subscribe(/*...*/);

    // Non-RxJS cleanup, colocated with the setup
    const resizeListener = () => this.onResize();
    window.addEventListener('resize', resizeListener);
    this.destroyRef.onDestroy(() => {
      window.removeEventListener('resize', resizeListener);
    });
  }
}
```

## The Legacy `takeUntil` + Subject Pattern

Before Angular 16, the standard pattern used a Subject to signal destruction:

```typescript
// LEGACY PATTERN: Still works, but takeUntilDestroyed is better
@Component({ selector: 'app-legacy', template: `...` })
export class LegacyComponent implements OnInit, OnDestroy {
  private readonly destroy$ = new Subject<void>();

  ngOnInit() {
    this.service.data$.pipe(
      takeUntil(this.destroy$)
    ).subscribe(data => this.process(data));

    this.events.stream$.pipe(
      takeUntil(this.destroy$)
    ).subscribe(event => this.handle(event));
  }

  ngOnDestroy() {
    this.destroy$.next();
    this.destroy$.complete();
  }
}
```

### Why `takeUntilDestroyed` Is Better

| Aspect | `takeUntil` + Subject | `takeUntilDestroyed` |
|--------|----------------------|---------------------|
| Boilerplate | 3 pieces: field, `next()`, `complete()` | Zero (or 1 field for `DestroyRef`) |
| Forgetting cleanup | Easy to forget `ngOnDestroy` | Impossible to forget |
| Lint enforcement | Requires custom lint rules | Built into Angular |
| Works in services | Requires manual `OnDestroy` impl | Works automatically |
| Injection context | Not needed | Required (or pass `DestroyRef`) |

## Common Memory Leak Patterns

### Leak 1: Subscribing in `ngOnInit` Without Cleanup

```typescript
// BAD: Subscription never cleaned up
ngOnInit() {
  this.route.params.subscribe(params => {
    this.loadItem(params['id']);
  });
}

// GOOD: takeUntilDestroyed with explicit DestroyRef
private readonly destroyRef = inject(DestroyRef);

ngOnInit() {
  this.route.params.pipe(
    takeUntilDestroyed(this.destroyRef)
  ).subscribe(params => {
    this.loadItem(params['id']);
  });
}

// BETTER: Declarative with toSignal (no subscribe at all)
private readonly route = inject(ActivatedRoute);
private readonly params = toSignal(this.route.params);
// Use params() in the template or in computed/effect
```

### Leak 2: `fromEvent` Without Cleanup

```typescript
// BAD: DOM event listener leaks
ngAfterViewInit() {
  fromEvent(this.button.nativeElement, 'click').subscribe(() => {
    this.doSomething();
  });
}

// GOOD: Cleaned up on destroy
private readonly destroyRef = inject(DestroyRef);

ngAfterViewInit() {
  fromEvent(this.button.nativeElement, 'click').pipe(
    takeUntilDestroyed(this.destroyRef)
  ).subscribe(() => {
    this.doSomething();
  });
}

// BETTER: Use Angular's event binding
// <button (click)="doSomething()">Click</button>
```

### Leak 3: `interval` / `timer` Without Cleanup

```typescript
// BAD: Interval runs forever
ngOnInit() {
  interval(5000).subscribe(() => this.refresh());
}

// GOOD
private readonly destroyRef = inject(DestroyRef);

ngOnInit() {
  interval(5000).pipe(
    takeUntilDestroyed(this.destroyRef)
  ).subscribe(() => this.refresh());
}
```

### Leak 4: Nested Subscriptions

```typescript
// BAD: Inner subscription leaks even if outer is cleaned up
ngOnInit() {
  this.route.params.pipe(
    takeUntilDestroyed(this.destroyRef)
  ).subscribe(params => {
    // This inner subscription is NOT cleaned up by takeUntilDestroyed!
    this.http.get(`/api/items/${params['id']}`).subscribe(item => {
      this.item = item;
    });
  });
}

// GOOD: Use flattening operators
ngOnInit() {
  this.route.params.pipe(
    switchMap(params => this.http.get(`/api/items/${params['id']}`)),
    takeUntilDestroyed(this.destroyRef)
  ).subscribe(item => {
    this.item = item;
  });
}
```

### Leak 5: `shareReplay` Without `refCount`

Covered in detail in 18.10. In short: `shareReplay(1)` defaults to `refCount: false`, meaning the source subscription persists even after all subscribers leave.

```typescript
// BAD in component-scoped context
readonly data$ = this.http.get('/api/data').pipe(
  shareReplay(1) // refCount: false by default
);

// GOOD
readonly data$ = this.http.get('/api/data').pipe(
  shareReplay({ bufferSize: 1, refCount: true })
);
```

### Leak 6: Forgotten `Subscription.add()`

```typescript
// BAD: Only the first subscription is tracked
private sub = new Subscription();

ngOnInit() {
  this.sub = this.a$.subscribe(); // Tracked
  this.b$.subscribe();            // Forgotten! Leaks.
}

ngOnDestroy() {
  this.sub.unsubscribe(); // Only unsubscribes a$
}

// Less bad: Use Subscription.add() to track multiple
private sub = new Subscription();

ngOnInit() {
  this.sub.add(this.a$.subscribe());
  this.sub.add(this.b$.subscribe());
}

ngOnDestroy() {
  this.sub.unsubscribe(); // Unsubscribes both
}

// GOOD: Just use takeUntilDestroyed and avoid tracking entirely
constructor() {
  this.a$.pipe(takeUntilDestroyed()).subscribe();
  this.b$.pipe(takeUntilDestroyed()).subscribe();
}
```

## Streams That Do NOT Need Cleanup

Not every subscription needs manual cleanup. Some Observables complete on their own:

| Source | Completes? | Needs Cleanup? |
|--------|-----------|----------------|
| `HttpClient.get/post/put/delete` | Yes (after response) | No* |
| `of()`, `from([])` | Yes (immediately) | No |
| `firstValueFrom()` / `lastValueFrom()` | Yes (Promise resolves) | No |
| `take(1)` | Yes (after 1 emission) | No |
| `first()` | Yes (after 1 emission) | No |
| `interval()` | No | **Yes** |
| `fromEvent()` | No | **Yes** |
| `Subject` / `BehaviorSubject` | Not until `.complete()` | **Yes** |
| `ActivatedRoute.params` | No | **Yes** |
| `Router.events` | No | **Yes** |
| `FormControl.valueChanges` | No | **Yes** |

*HTTP requests technically do not leak because they complete, but unsubscribing cancels the in-flight request, which is desirable for abandoned navigations.

```typescript
// HTTP: Technically safe, but better to cancel on destroy
this.http.get('/api/data').subscribe(data => this.data = data);

// HTTP: Better, cancels the request if component is destroyed mid-flight
this.http.get('/api/data').pipe(
  takeUntilDestroyed(this.destroyRef)
).subscribe(data => this.data = data);

// HTTP: Best, no subscription management at all
readonly data = toSignal(this.http.get('/api/data'));
```

## Building Custom Cleanup Utilities

### Utility: Auto-Cleanup for Non-RxJS Resources

```typescript
// Reusable utility for registering cleanup in any injection context
function onCleanup(callback: () => void): void {
  inject(DestroyRef).onDestroy(callback);
}

// Usage in a component
@Component({ selector: 'app-map', template: `...` })
export class MapComponent {
  constructor() {
    const map = new MapLibrary(this.container);
    onCleanup(() => map.destroy());

    const ws = new WebSocket('wss://example.com');
    onCleanup(() => ws.close());
  }
}
```

### Utility: Polling with Automatic Cleanup

```typescript
function poll<T>(
  request: () => Observable<T>,
  intervalMs: number
): Observable<T> {
  const destroyRef = inject(DestroyRef);

  return interval(intervalMs).pipe(
    startWith(0),
    switchMap(() => request()),
    takeUntilDestroyed(destroyRef)
  );
}

// Usage
@Component({ selector: 'app-status', template: `{{ status() }}` })
export class StatusComponent {
  private readonly http = inject(HttpClient);

  readonly status = toSignal(
    poll(() => this.http.get<string>('/api/status'), 10000),
    { initialValue: 'loading' }
  );
}
```

### Utility: Event Listener with Cleanup

```typescript
function useEventListener(
  target: EventTarget,
  event: string,
  handler: (e: Event) => void,
  options?: AddEventListenerOptions
): void {
  const destroyRef = inject(DestroyRef);

  target.addEventListener(event, handler, options);
  destroyRef.onDestroy(() => {
    target.removeEventListener(event, handler, options);
  });
}

// Usage
@Component({ selector: 'app-scroll', template: `...` })
export class ScrollComponent {
  constructor() {
    useEventListener(window, 'scroll', (e) => {
      this.handleScroll(e);
    }, { passive: true });
  }
}
```

## Decision Matrix: Choosing the Right Strategy

| Scenario | Strategy |
|----------|----------|
| Data for the template | `toSignal()` or `async` pipe |
| Side effect in constructor | `takeUntilDestroyed()` |
| Side effect in `ngOnInit` | `takeUntilDestroyed(this.destroyRef)` |
| Side effect triggered by user action | `takeUntilDestroyed(this.destroyRef)` in handler |
| Subscription that should stop before destroy | Manual `unsubscribe()` |
| Non-RxJS resource cleanup | `DestroyRef.onDestroy()` |
| Third-party library teardown | `DestroyRef.onDestroy()` |
| Root singleton service | Usually nothing needed |
| Component-provided service | `takeUntilDestroyed()` in constructor |

---

## Common Mistakes

1. **Calling `takeUntilDestroyed()` without parameters outside an injection context.** This throws `NG0203`. The fix is to inject `DestroyRef` as a field and pass it explicitly: `takeUntilDestroyed(this.destroyRef)`.

2. **Placing `takeUntilDestroyed` before flattening operators.** Inner subscriptions from `switchMap`, `mergeMap`, etc. can outlive the component if `takeUntilDestroyed` is placed before them. Always place it last.

3. **Assuming HTTP calls do not need cleanup.** While they complete and do not technically leak, unsubscribing cancels in-flight requests. Without cleanup, a slow API call can trigger state updates on an already-destroyed component.

4. **Using `takeUntilDestroyed` in root-level services.** Root services live for the app's lifetime. `takeUntilDestroyed` is useless there because the `DestroyRef` never fires (until the app shuts down).

5. **Forgetting about `FormControl.valueChanges` and `statusChanges`.** These are infinite Observables that do not complete. They need cleanup just like `interval()` or `fromEvent()`.

6. **Mixing `ngOnDestroy` and `DestroyRef.onDestroy` without understanding execution order.** `DestroyRef.onDestroy` callbacks fire during the same lifecycle phase as `ngOnDestroy`, but the relative order depends on registration order, not declaration order.

---

## Best Practices

1. **Default to `toSignal()` for template data.** It handles subscription, unsubscription, and change detection automatically.

2. **Use `takeUntilDestroyed()` for imperative side effects.** Reserve `subscribe()` for operations that produce side effects (logging, navigation, state mutations).

3. **Capture `DestroyRef` as a class field.** This lets you use `takeUntilDestroyed(this.destroyRef)` anywhere in the component, not just the constructor.

4. **Colocate setup and teardown.** Register cleanup at the point of creation using `DestroyRef.onDestroy()`, not in a separate `ngOnDestroy` method.

5. **Lint for unmanaged subscriptions.** ESLint rules like `rxjs/no-ignored-subscribe` or `@angular-eslint/prefer-on-push-component-change-detection` help catch leaks at build time.

6. **Cancel HTTP requests on navigation.** Even though they complete, canceling abandoned requests saves bandwidth and prevents stale updates.

## Real-World Use Cases

### 1. Dashboard with Multiple Data Streams

```typescript
@Component({
  selector: 'app-dashboard',
  template: `
    <app-user-count [count]="userCount()" />
    <app-revenue-chart [data]="revenue()" />
    <app-alerts [alerts]="alerts()" />
  `
})
export class DashboardComponent {
  private readonly dashboardService = inject(DashboardService);

  // All three: no manual cleanup needed
  readonly userCount = toSignal(this.dashboardService.userCount$, { initialValue: 0 });
  readonly revenue = toSignal(this.dashboardService.revenue$, { initialValue: [] });
  readonly alerts = toSignal(this.dashboardService.alerts$, { initialValue: [] });
}
```

### 2. Real-Time Notifications with Side Effects

```typescript
@Component({
  selector: 'app-notifications',
  template: `
    @for (n of notifications(); track n.id) {
      <app-notification-card [notification]="n" />
    }
  `
})
export class NotificationsComponent {
  private readonly ws = inject(WebSocketService);
  private readonly snackbar = inject(MatSnackBar);
  private readonly destroyRef = inject(DestroyRef);

  readonly notifications = toSignal(this.ws.notifications$, { initialValue: [] });

  constructor() {
    // Side effect: show snackbar for urgent notifications
    this.ws.notifications$.pipe(
      filter(notifications => notifications.some(n => n.priority === 'urgent')),
      takeUntilDestroyed()
    ).subscribe(notifications => {
      const urgent = notifications.filter(n => n.priority === 'urgent');
      this.snackbar.open(`${urgent.length} urgent notification(s)`, 'View', {
        duration: 5000
      });
    });
  }
}
```

### 3. Form Auto-Save with Debounce

```typescript
@Component({
  selector: 'app-editor',
  template: `
    <form [formGroup]="form">
      <input formControlName="title" />
      <textarea formControlName="content"></textarea>
      <span>{{ saveStatus() }}</span>
    </form>
  `
})
export class EditorComponent {
  private readonly http = inject(HttpClient);
  private readonly destroyRef = inject(DestroyRef);

  readonly form = inject(FormBuilder).group({
    title: [''],
    content: ['']
  });

  readonly saveStatus = signal('saved');

  constructor() {
    this.form.valueChanges.pipe(
      debounceTime(2000),
      distinctUntilChanged((a, b) => JSON.stringify(a) === JSON.stringify(b)),
      tap(() => this.saveStatus.set('saving...')),
      switchMap(value => this.http.put('/api/drafts/current', value)),
      tap(() => this.saveStatus.set('saved')),
      takeUntilDestroyed()
    ).subscribe();
  }
}
```

> **Key Takeaway:** Use `toSignal()` for template data, `takeUntilDestroyed()` for imperative side effects, and `DestroyRef.onDestroy()` for non-RxJS cleanup. Always place `takeUntilDestroyed` as the last operator before `subscribe`, and remember it requires an injection context or an explicit `DestroyRef` parameter.
